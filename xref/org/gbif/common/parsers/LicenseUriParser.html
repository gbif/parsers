14  /**
15   * A license parser giving URIs, compared to {@link License} which is for our GBIF enumeration.
16   *
17   * Singleton implementation of the dictionary that uses the file /dictionaries/parse/license_uri.txt to lookup a
18   * License by its URI or its acronym/title, e.g. a lookup by "CC-BY 4.0" returns http://creativecommons.org/licenses/by/4.0/.
19   * <br/>
20   * Note a lookup by license acronym/title without a version number defaults to the latest version of that license,
21   * e.g. a lookup by "CC-BY" returns http://creativecommons.org/licenses/by/4.0/.
22   */
23  public class LicenseUriParser extends FileBasedDictionaryParser<URI> {
24  
25    private static final String COMMENT_MARKER = "#";
26    private static final String LICENSE_FILEPATH = "/dictionaries/parse/license_uri.tsv";
27    // allows us to remove the protocol part for http:// and https://
28    private static final Pattern REMOVE_HTTP_PATTERN = Pattern.compile("^https?:\\/\\/", Pattern.CASE_INSENSITIVE);
29    private static final Pattern NORMALIZER = Pattern.compile("[^\\p{IsAlphabetic}\\p{N}Â©]+");
30    protected final ASCIIParser asciiParser = ASCIIParser.getInstance();
31    private static LicenseUriParser singletonObject = null;
32  
33    private LicenseUriParser() {
34      super(false);
35      init(LicenseUriParser.class.getResourceAsStream(LICENSE_FILEPATH), COMMENT_MARKER);
36    }
37  
38    /**
39     * @param source To build the dictionary from
40     */
41    @Override
42    public void init(Iterator<KeyValue<String, URI>> source) {
43      while (source.hasNext()) {
44        KeyValue<String, URI> kvp = source.next();
45        add(kvp.getKey(), kvp.getValue());
46        // Also adds the value, to save defining all of them.
47        add(kvp.getValue().toString(), kvp.getValue());
48      }
49    }
50  
51    @Override
52    protected String normalize(String value) {
53      if (Strings.isNullOrEmpty(value)) {
54        return null;
55      }
56      ParseResult<String> ascii = asciiParser.parse(value);
57      String noHttp = REMOVE_HTTP_PATTERN.matcher(ascii.getPayload()).replaceAll("").toLowerCase();
58      return super.normalize(NORMALIZER.matcher(noHttp).replaceAll(""));
59    }
60  
61    public static LicenseUriParser getInstance() {
62      synchronized (LicenseUriParser.class) {
63        if (singletonObject == null) {
64          singletonObject = new LicenseUriParser();
65        }
66      }
67      return singletonObject;
68    }
69  
70    @Override
71    protected URI fromDictFile(String value) {
72      try {
73        return URI.create(value);
74      } catch (RuntimeException e) {
75        return null;
76      }
77    }
78  }
