66  /**
67   * Numerical DateParser based on threetenbp (JSR310 backport) library which also means it is almost
68   * ready for Java 8.
69   *
70   * <p>Months are in numerical value starting at 1 for January.
71   *
72   * <p>Note that LocalDateTime and LocalDate are TimeZone agnostic.
73   *
74   * <p>Be aware that LocalDate and LocalDateTime doesn't map correctly to Date object for all dates
75   * before the Gregorian cut-off date (1582-10-15). To transform a such date use GregorianCalendar by
76   * setting the date according to the TemporalAccessor you got back from that class.
77   *
78   * <p>Thread-Safe after creation.
79   */
80  class ThreeTenNumericalDateParser implements TemporalParser {

86    // ISO 8601 specifies a Unicode minus (CHAR_MINUS), with a hyphen (CHAR_HYPHEN) as an alternative.
87    static final char CHAR_HYPHEN = '\u002d'; // Unicode hyphen, U+002d, char '-'
88    static final char CHAR_MINUS = '\u2212'; // Unicode minus, U+2212, char '−'
89    static final String HYPHEN = String.valueOf(CHAR_HYPHEN);
90    static final String MINUS = String.valueOf(CHAR_MINUS);

94    // DateTimeFormatter includes some ISO parsers but just to make it explicit we define our own
95    private static final DateTimeFormatter ISO_PARSER = (new DateTimeFormatterBuilder()
96      .appendValue(ChronoField.YEAR, 2, 4, SignStyle.NEVER)
97      .optionalStart().appendLiteral('-')
98      .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
99      .optionalStart().appendLiteral('-')
100     .appendValue(ChronoField.DAY_OF_MONTH, 1, 2, SignStyle.NEVER))
101     .optionalEnd()
102     .optionalEnd()
103     .toFormatter().withResolverStyle(ResolverStyle.STRICT);

105   /*
106    * Brackets [] represent optional sections of the pattern. (And subsequent patters don't make parts optional, if an earlier pattern already matched that.)
107    * Unfortunately, it's not possible to specify an arbitrary decimal for seconds in one pattern.

109   // separator is a CHAR_HYPHEN
110   private static final List<DateTimeParser> BASE_PARSER_LIST =
111       Collections.unmodifiableList(
112           DateTimeParserBuilder.newParserListBuilder()
113               .appendDateTimeParser("uuuuMMdd", YMD, LocalDate::from)
114               .appendDateTimeParser("uuuu-M-d", YMD, LocalDate::from, HYPHEN, MINUS + ".")
115 
116               // Either no fractional seconds, milliseconds or microseconds. T or space.
117               .appendDateTimeParser("uuuu-M-d' 'HH[[:]mm[[:]ss[.S]]]", YMDT, LocalDateTime::from, HYPHEN, MINUS + ".")
118               .appendDateTimeParser("uuuu-M-d' 'HH[:]mm[[:]ss.SS]", YMDT, LocalDateTime::from)
119               .appendDateTimeParser("uuuu-M-d' 'HH[:]mm[[:]ss.SSS]", YMDT, LocalDateTime::from)
120               .appendDateTimeParser("uuuu-M-d' 'HH[:]mm[[:]ss.SSSSSS]", YMDT, LocalDateTime::from)
121               .appendDateTimeParser("uuuu-M-d' 'HH[:]mm[[:]ss.SSSSSSS]", YMDT, LocalDateTime::from)
122               .appendDateTimeParser("uuuu-M-d' 'HH[:]mm[[:]ss[.SSS]]X", YMDTZ, ZonedDateTime::from, HYPHEN, MINUS)
123               .appendDateTimeParser("uuuu-M-d'T'HH[[:]mm[[:]ss[.S]]]", YMDT, LocalDateTime::from)
124               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[[:]ss.SS]", YMDT, LocalDateTime::from)
125               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[[:]ss.SSS]", YMDT, LocalDateTime::from)
126               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[[:]ss.SSSSSS]", YMDT, LocalDateTime::from)
127               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[[:]ss.SSSSSSS]", YMDT, LocalDateTime::from)
128               // T, but with a time zone, accepting Z, +00, +0000 and +00:00 for UTC and - or − for negative offsets.
129               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[[:]ss[.SSS]]X", YMDTZ, ZonedDateTime::from, HYPHEN, MINUS)
130               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[:]ss.SSSSSSX", YMDTZ, ZonedDateTime::from, HYPHEN, MINUS)
131               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[[:]ss[.SSS]]xxx", YMDTZ, ZonedDateTime::from, HYPHEN, MINUS)
132               .appendDateTimeParser("uuuu-M-d'T'HH[:]mm[:]ss.SSSSSSxxx", YMDTZ, ZonedDateTime::from, HYPHEN, MINUS)
133 
134               .appendDateTimeParser("uuuu-M", YM, YearMonth::from)
135               .appendDateTimeParser("uuuu", Y, Year::from)
136               .appendDateTimeParser("uuuu/MM/dd", YMD, LocalDate::from)
137               .appendDateTimeParser("uuuu/M/d", YMD, LocalDate::from)
138               .appendDateTimeParser("uuuu年MM月dd日", HAN, LocalDate::from)
139               .appendDateTimeParser("uuuu年M月d日", HAN, LocalDate::from)
140               .appendDateTimeParser("YYYY-'W'ww", YW, LocalDate::from) // ISO "week years", 2018-W43.
141               .appendDateTimeParser("uuuu-DDD", YD, LocalDate::from) // Week days, 2018-296.
142               .build()
143       );

146   private static final List<DateTimeMultiParser> MULTIPARSER_PARSER_LIST =
147     Arrays.asList(
148       DateTimeParserBuilder.newMultiParserListBuilder()
149         // Dot-formats are used by some European languages (German, Danish, Swedish etc), but not by
150         // the USA, so we can prefer the d.M.uuuu pattern.

152         .appendDateTimeParser("M.d.uuuu", MDY, LocalDate::from)
153         .build(),
154       // These are mostly derived of the difference between the common DMY format with slashes and US MDY format
155       DateTimeParserBuilder.newMultiParserListBuilder()
156         .appendDateTimeParser("d/M/uuuu'T'HH[[:]mm[[:]ss[.SSS]]]", DMYT, LocalDateTime::from, "/", HYPHEN + MINUS)
157         .appendDateTimeParser("M/d/uuuu'T'HH[[:]mm[[:]ss[.SSS]]]", MDYT, LocalDateTime::from, "/", HYPHEN + MINUS)
158         .build(),
159       DateTimeParserBuilder.newMultiParserListBuilder()
160         .appendDateTimeParser("d/M/uuuu'T'HH[[:]mm[[:]ss[.SSS]]][X]", DMYT, ZonedDateTime::from, "/", HYPHEN + MINUS)
161         .appendDateTimeParser("M/d/uuuu'T'HH[[:]mm[[:]ss[.SSS]]][X]", MDYT, ZonedDateTime::from, "/", HYPHEN + MINUS)
162         .build(),
163       DateTimeParserBuilder.newMultiParserListBuilder()
164         .appendDateTimeParser("d/M/uuuu' 'HH[[:]mm[[:]ss[.SSS]]]", DMYT, LocalDateTime::from, "/", HYPHEN + MINUS)
165         .appendDateTimeParser("M/d/uuuu' 'HH[[:]mm[[:]ss[.SSS]]]", MDYT, LocalDateTime::from, "/", HYPHEN + MINUS)
166         .build(),
167       DateTimeParserBuilder.newMultiParserListBuilder()
168         .appendDateTimeParser("d/M/uuuu' 'HH[[:]mm[[:]ss[.SSS]]][X]", DMYT, ZonedDateTime::from, "/", HYPHEN + MINUS)
169         .appendDateTimeParser("M/d/uuuu' 'HH[[:]mm[[:]ss[.SSS]]][X]", MDYT, ZonedDateTime::from, "/", HYPHEN + MINUS)
170         .build(),
171       DateTimeParserBuilder.newMultiParserListBuilder()
172         .appendDateTimeParser("d/M/uuuu", DMY, LocalDate::from, "/", HYPHEN + MINUS)
173         .appendDateTimeParser("M/d/uuuu", MDY, LocalDate::from, "/", HYPHEN + MINUS)
174         .build(),
175       // Dates with no separator.
176       DateTimeParserBuilder.newMultiParserListBuilder()
177         .appendDateTimeParser("ddMMuuuu", DMY, LocalDate::from)
178         .appendDateTimeParser("MMdduuuu", MDY, LocalDate::from)
179         .build(),
180       // Backslashes and underscores are not officially recommended by any country, but are sometimes used
181       DateTimeParserBuilder.newMultiParserListBuilder()
182         .appendDateTimeParser("d\\M\\uuuu", DMY, LocalDate::from, "\\", "_")
183         .appendDateTimeParser("M\\d\\uuuu", MDY, LocalDate::from, "\\", "_")

186   static {
187     for (DateTimeParser parser : BASE_PARSER_LIST) {
188       FORMATTERS_BY_ORDERING.putIfAbsent(parser.getOrdering(), new ArrayList<>());
189       FORMATTERS_BY_ORDERING.get(parser.getOrdering()).add(parser);
190     }
191 
192     for (DateTimeMultiParser parserAmbiguity : MULTIPARSER_PARSER_LIST) {
193       for (DateTimeParser parser : parserAmbiguity.getAllParsers()) {
194         FORMATTERS_BY_ORDERING.putIfAbsent(parser.getOrdering(), new ArrayList<>());
195         FORMATTERS_BY_ORDERING.get(parser.getOrdering()).add(parser);
196       }
197     }

229   private ThreeTenNumericalDateParser(Year baseYear) {
230     PreconditionUtils.checkState(baseYear.getValue() <= LocalDate.now().getYear(),
231       "Base year is less or equals to the current year");
232 
233     Map<DateComponentOrdering, List<DateTimeParser>> formattersByOrdering = new HashMap<>(FORMATTERS_BY_ORDERING);
234 
235     List<DateTimeMultiParser> multiParserList = new ArrayList<>(MULTIPARSER_PARSER_LIST);
236     multiParserList.addAll(new ArrayList<>(
237         Arrays.asList(
238             DateTimeParserBuilder.newMultiParserListBuilder()
239                 .preferredDateTimeParser("d.M.uu", DMY, LocalDate::from, baseYear)
240                 .appendDateTimeParser("M.d.uu", MDY, LocalDate::from, baseYear)
241                 .build(),
242             DateTimeParserBuilder.newMultiParserListBuilder()
243                 .appendDateTimeParser("d/M/uu", DMY, LocalDate::from, "/", HYPHEN + MINUS, baseYear)
244                 .appendDateTimeParser("M/d/uu", MDY, LocalDate::from, "/", HYPHEN + MINUS, baseYear)
245                 .build(),
246             DateTimeParserBuilder.newMultiParserListBuilder()
247                 .appendDateTimeParser("ddMMuu", DMY, LocalDate::from, baseYear)
248                 .appendDateTimeParser("MMdduu", MDY, LocalDate::from, baseYear)
249                 .build(),
250             DateTimeParserBuilder.newMultiParserListBuilder()
251                 .appendDateTimeParser("d\\M\\uu", DMY, LocalDate::from, "\\", "_", baseYear)
252                 .appendDateTimeParser("M\\d\\uu", MDY, LocalDate::from, "\\", "_", baseYear)
253                 .build()
254         )));
255 
256     for (DateTimeMultiParser multiParser : multiParserList) {
257       for (DateTimeParser parser : multiParser.getAllParsers()) {
258         formattersByOrdering.putIfAbsent(parser.getOrdering(), new ArrayList<>());
259         formattersByOrdering.get(parser.getOrdering()).add(parser);
260       }
261     }
262 
263     this.activeMultiParserList = Collections.unmodifiableList(multiParserList);
264     this.activeFormattersByOrdering = Collections.unmodifiableMap(formattersByOrdering);
265   }

269     return parse(input, ISO_ETC);

272   @Override
273   public ParseResult<TemporalAccessor> parse(String input, @Nullable DateComponentOrdering ordering) {
274 
275     if (StringUtils.isBlank(input)) {
276       return ParseResult.fail();
277     }
278     // make sure ordering is never null
279     if (ordering == null) {
280       ordering = ISO_ETC;
281     }
282 
283     // If ordering is given, BASE_PARSER_LIST is ignored.
284     List<DateTimeParser> parserList =
285         activeFormattersByOrdering.getOrDefault(ordering, BASE_PARSER_LIST);
286 
287     // First attempt: find a match with definite confidence
288     TemporalAccessor parsedTemporalAccessor;
289     for (DateTimeParser parser : parserList) {
290       parsedTemporalAccessor = parser.parse(input);
291       if (parsedTemporalAccessor != null) {
292         return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, parsedTemporalAccessor);
293       }
294     }
295 
296     // if a format ordering was provided we already tried all possible format
297     if (ordering != ISO_ETC) {
298       return ParseResult.fail();
299     }
300 
301     // Second attempt: find one or multiple matches in the list of DateTimeMultiParser
302     int numberOfPossiblyAmbiguousMatch = 0;
303     TemporalAccessor lastParsedSuccess = null;
304     TemporalAccessor lastParsedPreferred = null;
305     Set<TemporalAccessor> otherParsed = new HashSet<>();
306     // Checking if result is the same due to duplicated format in multiple orderings
307     // like "d/M/uu" is defined in DMY and DMYT, it will generate same dates.
308     List<TemporalAccessor> verificationDuplication = new ArrayList<>();
309     // Are the results all equal (representing the same TemporalAccessor), used if there is no
310     // preferred result defined
311     boolean lastParsedSuccessOtherResultsEqual = false;
312     DateTimeMultiParser.MultipleParseResult result = null;
313 
314     // here we do not stop when we find a match, we try them all to check for a possible ambiguity
315     for (DateTimeMultiParser parserAmbiguity : activeMultiParserList) {
316       result = parserAmbiguity.parse(input);
317 
318       if (result.getNumberParsed() > 0) {
319         lastParsedSuccess = result.getResult();
320         if (!verificationDuplication.contains(lastParsedSuccess)) {
321           numberOfPossiblyAmbiguousMatch += result.getNumberParsed();
322           verificationDuplication.add(lastParsedSuccess);
323         }
324 
325         // make sure to log in case lastParsedSuccessOtherResultsEqual already true
326         if (lastParsedSuccessOtherResultsEqual) {
327           LOGGER.warn("Issue with DateTimeMultiParser configuration: Input {} produces more results even " +
328             "if lastParsedSuccessOtherResultsEqual is set to true.", input);
329         }
330         lastParsedSuccessOtherResultsEqual = false;
331 
332         if (result.getPreferredResult() != null) {
333           if (lastParsedPreferred != null) {
334             LOGGER.warn("Issue with DateTimeMultiParser configuration: Input {} produces 2 preferred results", input);
335           }
336           lastParsedPreferred = result.getPreferredResult();
337         } else {
338           //if we have no PreferredResult but all results represent the same TemporalAccessor
339           if (result.getOtherResults() != null && result.getOtherResults().size() > 1) {
340             lastParsedSuccessOtherResultsEqual = allEquals(result.getOtherResults());
341             otherParsed.addAll(result.getOtherResults());
342           }
343         }
344       }
345     }
346 
347     //if there is only one pattern that can be applied it is not ambiguous
348     if (numberOfPossiblyAmbiguousMatch == 1) {
349       return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, lastParsedSuccess);
350     } else if (numberOfPossiblyAmbiguousMatch > 1) {
351       // if all the possible ambiguities are equal, there is no ambiguity
352       if (lastParsedSuccessOtherResultsEqual) {
353         return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, lastParsedSuccess);
354       }
355 
356       // check if we have result from the preferred parser
357       if (lastParsedPreferred != null) {
358         return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, lastParsedPreferred);
359       }
360     }
361 
362     LOGGER.debug("Number of matches for {} : {}", input, numberOfPossiblyAmbiguousMatch);
363     if (result == null) {
364       return ParseResult.fail();
365     } else {
366       return new ParseResult<>(ParseResult.STATUS.FAIL, ParseResult.CONFIDENCE.POSSIBLE, null, new ArrayList<>(otherParsed), null);
367     }
368   }

370   /**
371    * Parse a date, and if given an ambiguous date, like 2/3/2000, try {@code orderings} to parse date,
372    * and return the first successful result.
373    * <p>
374    * NOTE, this behaviour <strong>differs</strong> from <code>parse(String input, DateComponentOrdering ordering)</code>
375    */
376   @Override
377   public ParseResult<TemporalAccessor> parse(String input, DateComponentOrdering[] orderings) {
378     ParseResult<TemporalAccessor> result = parse(input);
379     if (result.getStatus() == ParseResult.STATUS.FAIL
380       && result.getConfidence() == ParseResult.CONFIDENCE.POSSIBLE
381       && result.getAlternativePayloads().size() > 1) {
382       for (DateComponentOrdering ordering : orderings) {
383         result = parse(input, ordering);
384         if (result.isSuccessful()) {
385           return result;
386         }
387       }
388     }
389     return result;
390   }

392   @Override
393   public ParseResult<TemporalAccessor> parse(
394     @Nullable String year, @Nullable String month, @Nullable String day) {
395 
396     // avoid possible misinterpretation when month is not provided (but day is)
397     if (StringUtils.isBlank(month) && StringUtils.isNotBlank(day)) {
398       return ParseResult.fail();
399     }
400 
401     String date = Stream.of(StringUtils.trimToNull(year), StringUtils.trimToNull(month), StringUtils.trimToNull(day))
402         .filter(Objects::nonNull)
403         .collect(Collectors.joining(String.valueOf(CHAR_HYPHEN)));
404 
405     TemporalAccessor tp = tryParse(date, ISO_PARSER, null);
406 
407     if (tp != null) {
408       return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, tp);
409     }
410     return ParseResult.fail();
411   }
412 
413   @Override
414   public ParseResult<TemporalAccessor> parse(@Nullable Integer year, @Nullable Integer month, @Nullable Integer day) {
415 
416     // avoid possible misinterpretation when month is not provided (but day is)
417     if (month == null && day != null) {
418       return ParseResult.fail();
419     }
420 
421     String date = Stream.of(year, month, day)
422         .filter(Objects::nonNull)
423         .map(String::valueOf)
424         .collect(Collectors.joining(String.valueOf(CHAR_HYPHEN)));
425 
426     TemporalAccessor tp = tryParse(date, ISO_PARSER, null);
427 
428     if (tp != null) {
429       return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, tp);
430     }
431     return ParseResult.fail();
432   }

434   /**
435    * Utility private method to avoid throwing a runtime exception when the formatter can not parse the String.
436    * TODO normalizer is call too often maybe this class should not take it and only try to parse
437    *
438    * @param input
439    * @param formatter
440    * @param normalizer
441    * @return
442    */
443   private static TemporalAccessor tryParse(String input, DateTimeFormatter formatter, DateTimeSeparatorNormalizer normalizer) {
444 
445     if (normalizer != null) {
446       input = normalizer.normalize(input);
447     }
448     try {
449       return formatter.parseBest(input, ZonedDateTime::from, LocalDateTime::from, LocalDate::from,
450         YearMonth::from, Year::from);
451     } catch (DateTimeParseException dpe) {
452     }
453     return null;
454   }

456   /**
457    * Check if all the TemporalAccessor of the list are equal.
458    * <p>
459    * Return false for an empty list; true for a list with a single element.
460    */
461   private static boolean allEquals(List<TemporalAccessor> taList) {
462     if (taList == null || taList.isEmpty()) {
463       return false;
464     }
465 
466     TemporalAccessor reference = taList.get(1);
467     for (TemporalAccessor ta : taList) {
468       if (!ta.equals(reference)) {
469         return false;
470       }
471     }
472     return true;
473   }
