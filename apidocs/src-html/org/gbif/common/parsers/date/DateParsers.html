001package org.gbif.common.parsers.date;
002
003
004import java.time.format.DateTimeFormatter;
005import java.time.format.DateTimeFormatterBuilder;
006import java.time.format.ResolverStyle;
007import java.time.format.SignStyle;
008import java.time.temporal.ChronoField;
009
010/**
011 * Factories for date parsing related instances.
012 */
013public class DateParsers {
014
015  /**
016   * {@link DateTimeFormatter} for ISO Year-MonthOfYear-DayOfMonth (4 digits year)
017   * This formatter does NOT handle time and timezone.
018   */
019  public static final DateTimeFormatter ISO_LOCAL_DATE =
020          new DateTimeFormatterBuilder()
021                  .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
022                  .appendLiteral('-')
023                  .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
024                  .appendLiteral('-')
025                  .appendValue(ChronoField.DAY_OF_MONTH, 1, 2, SignStyle.NEVER)
026                  .toFormatter().withResolverStyle(ResolverStyle.STRICT);
027
028  /**
029   * {@link DateTimeFormatter} for ISO Year (4 digits)
030   */
031  public static final DateTimeFormatter ISO_YEAR =
032          new DateTimeFormatterBuilder()
033                  .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
034                  .toFormatter().withResolverStyle(ResolverStyle.STRICT);
035
036  /**
037   * {@link DateTimeFormatter} for ISO Year-MonthOfYear (4 digits year)
038   */
039  public static final DateTimeFormatter ISO_YEAR_MONTH =
040          new DateTimeFormatterBuilder()
041                  .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
042                  .appendLiteral('-')
043                  .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
044                  .toFormatter().withResolverStyle(ResolverStyle.STRICT);
045

053  /**
054   * Return a pre-configured {@link TemporalParser} instance.
055   */
056  public static TemporalParser defaultTemporalParser() {
057    return new TextDateParser();
058  }
059
060  /**
061   * Get a new instance of the default implementation of TemporalParser that handles
062   * numerical dates.
063   */
064  public static TemporalParser defaultNumericalDateParser() {
065    return ThreeTenNumericalDateParser.newInstance();
066  }
067
068}
