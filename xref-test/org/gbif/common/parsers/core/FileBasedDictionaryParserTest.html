10  public class FileBasedDictionaryParserTest {

12    private static final String COMMENT_MARKER = "#";
13    private static final String TEST_DICTIONARY = "/parse/simpleDictionary.txt";
14  
15    class TestParser extends FileBasedDictionaryParser<String> {
16  
17      public TestParser(InputStream... inputs) {
18        super(false);
19        init(FileBasedDictionaryParserTest.class.getResourceAsStream(TEST_DICTIONARY));
20      }
21  
22      @Override
23      protected String fromDictFile(String value) {
24        return value;
25      }
26    }
27  
28    /**
29     * test implementation that supports commented lines
30     *
31     * @author cgendreau
32     */
33    class TestParserWithCommentSupport extends FileBasedDictionaryParser<String> {
34  
35      public TestParserWithCommentSupport(InputStream... inputs) {
36        super(false);
37        init(FileBasedDictionaryParserTest.class.getResourceAsStream(TEST_DICTIONARY), COMMENT_MARKER);
38      }
39  
40      @Override
41      protected String fromDictFile(String value) {
42        return value;
43      }
44    }
45  
46    @Test
47    public void testParse() {
48      Parsable<String> dbp = new TestParser();
49      assertParseSuccess(dbp, "32", "Tim");
50      assertParseSuccess(dbp, "32", "TIM");
51      assertParseSuccess(dbp, "32", "tIm");
52      assertParseSuccess(dbp, "38", "Markus");
53      assertParseSuccess(dbp, "38", "MarKUS");
54      assertParseSuccess(dbp, "28", "Jose");
55      assertParseSuccess(dbp, "31", COMMENT_MARKER + "carey");
56      assertParseSuccess(dbp, "31", COMMENT_MARKER + "careY");
57  
58      assertParseFailure(dbp, "Lars");
59    }
60  
61    @Test
62    public void testParseFileWithComment() {
63      Parsable<String> dbp = new TestParserWithCommentSupport();
64      assertParseSuccess(dbp, "32", "Tim");
65      assertParseFailure(dbp, COMMENT_MARKER + "carey");
66    }
67  
68    protected static void assertParseSuccess(Parsable<String> dbp, String expected, String input) {
69      ParseResult<String> parsed = dbp.parse(input);
70      assertNotNull(parsed);
71      assertEquals(expected, parsed.getPayload());
72      assertEquals(ParseResult.STATUS.SUCCESS, parsed.getStatus());
73      assertEquals(ParseResult.CONFIDENCE.DEFINITE, parsed.getConfidence());
74    }
75  
76    protected static void assertParseFailure(Parsable<String> dbp, String input) {
77      ParseResult<String> parsed = dbp.parse(input);
78      assertEquals(ParseResult.STATUS.FAIL, parsed.getStatus());
79    }
