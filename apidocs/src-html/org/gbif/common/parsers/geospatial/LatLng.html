001package org.gbif.common.parsers.geospatial;
002
003import java.util.Objects;
004import java.util.StringJoiner;
005
006/**
007 * Simple container class for an interpreted coordinate.
008 */
009public class LatLng {
010
011  private final Double lat;
012  private final Double lng;
013
014  public LatLng(double lat, double lng) {
015    this.lat = lat;
016    this.lng = lng;
017  }
018
019  public LatLng() {
020    lat = null;
021    lng = null;
022  }
023
024  public Double getLat() {
025    return lat;
026  }
027
028  public Double getLng() {
029    return lng;
030  }
031
032  @Override
033  public String toString() {
034    return new StringJoiner(", ", LatLng.class.getSimpleName() + "[", "]")
035        .add("lat=" + lat)
036        .add("lng=" + lng)
037        .toString();
038  }
039
040  @Override
041  public boolean equals(Object o) {
042    if (this == o) return true;
043    if (o == null || getClass() != o.getClass()) return false;
044    LatLng latLng = (LatLng) o;
045    return Objects.equals(lat, latLng.lat) && Objects.equals(lng, latLng.lng);
046  }
047
048  @Override
049  public int hashCode() {
050    return Objects.hash(lat, lng);
051  }
052}
