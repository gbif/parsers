45      private static Pattern FULL_YEAR = Pattern.compile("\\d\\d\\d\\d");

59      /**
60       * Verifies if day and year cannot be confused with the month and returns appropriate confidence level.
61       */
62      private static ParseResult.CONFIDENCE checkMonthDay(Date d, boolean verifyYear) {
63          if (d == null) {
64              return ParseResult.CONFIDENCE.DEFINITE;
65          }
66  
67          Calendar cal = Calendar.getInstance();
68          cal.setTime(d);
69          int day = cal.get(Calendar.DAY_OF_MONTH);
70          int month = cal.get(Calendar.MONTH) + 1;
71  
72          if (day > 12 || day == month) {
73              if (verifyYear) {
74                  int year = cal.get(Calendar.YEAR);
75                  int yearInCentury = year % 100;
76                  if (yearInCentury > 31) {
77                      return ParseResult.CONFIDENCE.DEFINITE;
78                  } else if (year < 1700) {
79                      return ParseResult.CONFIDENCE.POSSIBLE;
80                  }
81                  return ParseResult.CONFIDENCE.PROBABLE;
82              }
83              return ParseResult.CONFIDENCE.DEFINITE;
84          }
85          // day and month might have been confused
86          return ParseResult.CONFIDENCE.POSSIBLE;
87      }

89      /**
90       * Parses a numerical date or datetime string trying various formats.
91       * Parsing is done based on the string length and delimiters found in the input primarily.
92       * There is no cleaning done that removes question marks.
93       * Only superfluous whitespace is removed.
94       */
95      @Override
96      public ParseResult<Date> parse(String input) {
97          if (Strings.isNullOrEmpty(input)) {
98              return ParseResult.fail();
99          }
100 
101         final int len = input.length();
102         Date d = null;
103         ParseResult.CONFIDENCE confidence = ParseResult.CONFIDENCE.DEFINITE;
104 
105         if (input.contains("年")) {
106             d = strictParse(input, ASIAN);
107 
108         }
109         if (StringUtils.isNumeric(input)) {
110             if (len <= 4) {
111                 // year only
112                 d = strictParse(input, "yy", "yyyy");
113 
114             } else if (len == 6) {
115                 // no delimiter, year as 2 digits
116                 d = strictParse(input, CHAR6_ONLY_DIGITS);
117                 confidence = checkMonthDay(d, true);
118                 confidence = confidence == ParseResult.CONFIDENCE.DEFINITE ? ParseResult.CONFIDENCE.PROBABLE : confidence;
119             } else if (len == 8) {
120                 // no delimiter, year as 2 digits
121                 d = strictParse(input, CHAR8_ONLY_DIGITS);
122                 confidence = checkMonthDay(d, true);
123             }
124 
125         } else {
126             // find what delimiter exists how often
127             Character del = null;
128             int delimCount = 0;
129             for (char c : input.toCharArray()) {
130                 if (!Character.isDigit(c) && !Character.isAlphabetic(c)) {
131                     delimCount++;
132                     if (del == null) {
133                         del = c;
134                     } else if (!del.equals(c)) {
135                         // different delimiters in use - could be datetime
136                         d = strictParse(input, TIME_FORMATS);
137                         confidence = checkMonthDay(d, true);
138                         break;
139                     }
140                 }
141             }
142 
143             if (del == null) {
144               return ParseResult.fail();
145             }
146 
147             if (Character.isLetter(del)) {
148                 confidence = ParseResult.CONFIDENCE.POSSIBLE;
149             }
150 
151             // check if full year is given within the string
152             final boolean fullYear = FULL_YEAR.matcher(input).find();
153             try {
154                 if (delimCount == 1) {
155                     // year & month only
156                     String[] formats;
157                     if (fullYear) {
158                         formats = new String[]{"yyyy" + del + "MM", "MM" + del + "yyyy"};
159 
160                     } else {
161                         formats = new String[]{"yy" + del + "MM", "MM" + del + "yy"};
162                         confidence = ParseResult.CONFIDENCE.POSSIBLE;
163                     }
164                     d = strictParse(input, formats);
165 
166                 } else if (delimCount == 2) {
167                     if (fullYear) {
168                         // first try very common formats (main US, European, Chinese, etc)
169                         d = strictParse(input, FREQUENT_FULL);
170                         if (d == null) {
171                             // maybe a datetime format ?
172                             d = strictParse(input, TIME_FORMATS);
173                             if (d == null) {
174                                 String[] formats = new String[]{"yyyy" + del + "MM" + del + "dd", "dd" + del + "MM" + del + "yyyy",
175                                     "MM" + del + "dd" + del + "yyyy"};
176                                 d = strictParse(input, formats);
177                                 confidence = checkMonthDay(d, false);
178                             }
179                         }
180 
181                     } else {
182                         String[] formats;
183                         if (del == '-') {
184                             // iso formats use dash and start with years
185                             formats = new String[]{"yy" + del + "MM" + del + "dd", "dd" + del + "MM" + del + "yy",
186                                 "MM" + del + "dd" + del + "yy"};
187                         } else {
188                             formats = new String[]{"dd" + del + "MM" + del + "yy", "MM" + del + "dd" + del + "yy",
189                                 "yy" + del + "MM" + del + "dd"};
190                         }
191                         d = strictParse(input, formats);
192                         confidence = checkMonthDay(d, true);
193                     }
194 
195                 } else {
196                     // more than 3 delimiters, with time?
197                     d = strictParse(input, TIME_FORMATS);
198                 }
199             } catch (RuntimeException e) {
200                 return ParseResult.fail();
201             }
202         }
203 
204         if (d != null) {
205             // basic year validation to avoid year = 12321
206             if (d.before(MAX_VALID_YEAR)) {
207                 return ParseResult.success(confidence, d);
208             }
209         }
210         return ParseResult.fail();
211     }

214     /**
215      * Parses strictly swallowing errors
216      *
217      * @param input    to parse
218      * @param patterns to use in parsing
219      * @return null on error or the date
220      */
221     Date strictParse(String input, String... patterns) {
222         try {
223             return DateUtils.parseDateStrictly(input, patterns);
224         } catch (ParseException e) {
225             return null;
226         }
227     }
