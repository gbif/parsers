001package org.gbif.common.parsers;
002
003import java.net.URI;
004import java.util.List;
005import java.util.regex.Pattern;
006
007import com.google.common.base.CharMatcher;
008import com.google.common.base.Splitter;
009import com.google.common.base.Strings;
010import com.google.common.collect.Iterables;
011import com.google.common.collect.Lists;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * Greedy URL parser assuming HTTP URIs in case no schema was given.
017 * Modified version of the registry-metadata GreedyUriConverter.
018 */
019public class UrlParser {
020  private static final Logger LOG = LoggerFactory.getLogger(UrlParser.class);
021  private static final String[] MULTI_VALUE_DELIMITERS = {"|#DELIMITER#|", "|", ",", ";"};
022  private static final String HTTP_SCHEME = "http://";
023
024  // Pattern for things that are probably domains followed by a slash, without a protocol.
025  // Doesn't match IDNs etc, but this is just for people who forgot the http:// anyway.
026  private static final Pattern DOMAIN_ISH = Pattern.compile("^[A-Za-z0-9.-]{1,40}\\.[A-Za-z0-9-]{2,30}/.*");
027
028  private UrlParser() {
029  }
030
031  /**
032   * Convert a String into a java.net.URI.
033   * In case its missing the protocol prefix, it is prefixed with the default protocol.
034   *
035   * @param value The input value to be converted
036   *
037   * @return The converted value, or null if not parsable or exception occurred
038   */
039  public static URI parse(String value) {
040    value = CharMatcher.WHITESPACE.trimFrom(Strings.nullToEmpty(value));
041    if (Strings.isNullOrEmpty(value)) {
042      return null;
043    }
044
045    URI uri = null;
046    try {
047      uri = URI.create(value);
048      if (!uri.isAbsolute() && DOMAIN_ISH.matcher(value).matches()) {
049        // make into an HTTP address
050        try {
051          uri = URI.create(HTTP_SCHEME + value);
052        } catch (IllegalArgumentException e) {
053          // keep the previous scheme-less result
054        }
055      }
056
057      // verify that we have a domain
058      if (Strings.isNullOrEmpty(uri.getHost())) {
059        return null;
060      }
061
062    } catch (IllegalArgumentException e) {
063    }
064
065    return uri;
066  }
067
068
069  /**
070   * Parses a single string with null, one or many URIs concatenated together as found in dwc:associatedMedia.
071   */
072  public static List<URI> parseUriList(String uris) {
073    List<URI> result = Lists.newArrayList();
074
075    if (!Strings.isNullOrEmpty(uris)) {
076      // first try to use the entire string
077      URI uri = UrlParser.parse(uris);
078      if (uri != null) {
079        result.add(uri);
080
081      } else {
082        // try common delimiters
083        int maxValidUrls = 0;
084        for (String delimiter : MULTI_VALUE_DELIMITERS) {
085          Splitter splitter = Splitter.on(delimiter).omitEmptyStrings().trimResults();
086          String[] urls = Iterables.toArray(splitter.split(uris), String.class);
087          // avoid parsing if we haven' actually split anything
088          if (urls.length > 1) {
089            List<URI> tmp = Lists.newArrayList();
090            for (String url : urls) {
091              uri = UrlParser.parse(url);
092              if (uri != null) {
093                tmp.add(uri);
094              }
095            }
096            if (tmp.size() > maxValidUrls) {
097              result = tmp;
098              maxValidUrls = tmp.size();
099            } else if (maxValidUrls > 0 && tmp.size() == maxValidUrls) {
100              LOG.info("Unclear what delimiter is being used for concatenated URIs = {}", uris);
101            }
102          }
103        }
104      }
105    }
106    return result;
107  }
108
109}
