016package org.gbif.common.parsers.date;
017
018import org.gbif.utils.PreconditionUtils;
019
020import java.time.temporal.TemporalAccessor;
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.List;
024import java.util.Objects;
025
026import javax.annotation.Nullable;
027import javax.validation.constraints.NotNull;
028
029/**
030 * Supports multiple {@link DateTimeParser} that are considered ambiguous. Two {@link DateTimeParser} are considered
031 * ambiguous when they can potentially produce 2 different {@link TemporalAccessor}.
032 * e.g. "dd/MM/yyyy" and "MM/dd/yyyy"
033 *
034 * <p>This class will try all the parsers and keep the all the successful results.
035 *
036 * <p>This class is thread-safe once an instance is created.
037 */
038public class DateTimeMultiParser {
039
040  private final DateTimeParser preferred;
041  private final List<DateTimeParser> otherParsers;
042  private final List<DateTimeParser> allParsers;
043
044  /**
045   * Create a new instance of {@link DateTimeMultiParser}.
046   * @param parsers requires more than 1 element in list
047   */
048  DateTimeMultiParser(@NotNull List<DateTimeParser> parsers){
049    this(null, parsers);
050  }
051
052  /**
053   *
054   * Create a new instance of {@link DateTimeMultiParser}.
055   * At least 2 {@link DateTimeParser} must be provided see details on parameters.
056   *
057   * @param preferred the preferred {@link DateTimeParser} or null
058   * @param otherParsers list of {@link DateTimeParser} containing more than 1 element if no
059   *                     preferred {@link DateTimeParser} is provided. Otherwise, the list must contain at least 1 element.
060   */
061  DateTimeMultiParser(@Nullable DateTimeParser preferred, @NotNull List<DateTimeParser> otherParsers) {
062    Objects.requireNonNull(otherParsers, "otherParsers list can not be null");
063    PreconditionUtils.checkArgument(otherParsers.size() > 0, "otherParsers must contain at least 1 element");
064
065    if (preferred == null) {
066      PreconditionUtils.checkArgument(otherParsers.size() > 1, "If no preferred DateTimeParser is provided, " +
067              "the otherParsers list must contain more than 1 element");
068    }
069
070    this.preferred = preferred;
071    this.otherParsers = new ArrayList<>(otherParsers);
072
073    List<DateTimeParser> resultList = new ArrayList<>();
074
075    if (preferred != null) {
076      resultList.add(preferred);
077    }
078    resultList.addAll(otherParsers);
079
080    this.allParsers = Collections.unmodifiableList(resultList);
081  }
082
083  /**
084   * Get the list of all parsers: the preferred (if specified in the constructor) + otherParsers.
085   *
086   * @return never null
087   */
088  public List<DateTimeParser> getAllParsers(){
089    return allParsers;
090  }
091
092  /**
093   * Try to parse the input using all the parsers specified in the constructor.
094   *
095   * @param input
096   * @return {@link MultipleParseResult} instance, never null.
097   */
098  public MultipleParseResult parse(String input) {
099
100    int numberParsed = 0;
101    TemporalAccessor lastParsed = null;
102    TemporalAccessor preferredResult = null;
103    List<String> usedFormats = new ArrayList<>();
104
105    // lazily initialized assuming it should not be used most of the time
106    List<TemporalAccessor> otherResults = null;
107    for (DateTimeParser currParser : otherParsers) {
108      lastParsed = currParser.parse(input);
109      if (lastParsed != null) {
110        numberParsed++;
111        if (otherResults == null) {
112          otherResults = new ArrayList<>();
113        }
114        otherResults.add(lastParsed);
115        usedFormats.add(currParser.getOrdering().name());
116      }
117    }
118
119    // try the preferred DateTimeParser
120    if (this.preferred != null) {
121      lastParsed = this.preferred.parse(input);
122      if (lastParsed != null) {
123        numberParsed++;
124        preferredResult = lastParsed;
125      }
126    }
127
128    return new MultipleParseResult(numberParsed, usedFormats, preferredResult, otherResults);
129  }
130
131  /**
132   * Nested class representing the result of a multi-parse.
133   */
134  public static class MultipleParseResult {
135    private int numberParsed;
136    private TemporalAccessor preferredResult;
137    private List<TemporalAccessor> otherResults;
138    public List<String> formats;
139
140    public MultipleParseResult(int numberParsed, List<String> formats, TemporalAccessor preferredResult, List<TemporalAccessor> otherResults) {
141      this.numberParsed = numberParsed;
142      this.formats = formats;
143      this.preferredResult = preferredResult;
144      this.otherResults = otherResults;
145    }
146
147    public int getNumberParsed() {
148      return numberParsed;
149    }
150
151    public List<String> getFormats() {
152      return formats;
153    }
154
155    public TemporalAccessor getPreferredResult() {
156      return preferredResult;
157    }
158
159    public List<TemporalAccessor> getOtherResults() {
160      return otherResults;
161    }
162
163    /**
164     * Return the preferredResult if available otherwise the first element of otherResults.
165     * If otherResults is empty, null is returned.
166     */
167    public TemporalAccessor getResult() {
168      if (preferredResult != null) {
169        return preferredResult;
170      }
171
172      if (otherResults != null && otherResults.size() > 0) {
173        return otherResults.get(0);
174      }
175      return null;
176    }
177  }
178}
