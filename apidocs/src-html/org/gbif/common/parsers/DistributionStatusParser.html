001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.DistributionStatus;
004import org.gbif.common.parsers.core.EnumParser;
005
006/**
007 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/distribution_status.tsv.
008 */
009public class DistributionStatusParser extends EnumParser<DistributionStatus> {
010
011  private static DistributionStatusParser singletonObject = null;
012
013  private DistributionStatusParser() {
014    super(DistributionStatus.class, false);
015    // make sure we have all occurrence_status enum mapped
016    for (DistributionStatus c : DistributionStatus.values()) {
017      add(c.name(), c);
018    }
019    // use dict file last
020    init(DistributionStatusParser.class.getResourceAsStream("/dictionaries/parse/distribution_status.tsv"));
021  }
022
023  public static DistributionStatusParser getInstance() {
024    synchronized (DistributionStatusParser.class) {
025      if (singletonObject == null) {
026        singletonObject = new DistributionStatusParser();
027      }
028    }
029    return singletonObject;
030  }
031
032}
