078  @Test
079  public void testParseStringStringString() {
080    assertEquivalent("21/12/1978", DateParseUtils.parse("1978", "12", "21"), ParseResult.CONFIDENCE.DEFINITE);
081    assertEquivalent("01/01/1978", DateParseUtils.parse("1978", "01", "01"), ParseResult.CONFIDENCE.DEFINITE);
082    assertEquivalent("01/11/1978", DateParseUtils.parse("1978", "11", "01"), ParseResult.CONFIDENCE.DEFINITE);
083    assertEquivalent("01/01/0001", DateParseUtils.parse("1", "1", "1"), ParseResult.CONFIDENCE.DEFINITE);
084    assertEquals(ParseResult.STATUS.FAIL, DateParseUtils.parse("-1", "1", "1").getStatus());
085    assertEquals(ParseResult.STATUS.FAIL, DateParseUtils.parse(null, null, null).getStatus());
086  }

088  @Test
089  public void testAtomizeString() {
090    String input = "21/12/2010";
091    ParseResult<YearMonthDay> result = DateParseUtils.atomize(input);
092    LOGGER.info("{} atomized to {}", input, result);
093    assertEquals(ParseResult.STATUS.SUCCESS, result.getStatus());
094    assertEquals(ParseResult.CONFIDENCE.DEFINITE, result.getConfidence());
095    assertNotNull(result.getPayload());
096    assertEquals(new YearMonthDay("2010", "12", "21"), result.getPayload());
097
098    input = "16/07/99";
099    result = DateParseUtils.atomize(input);
100    LOGGER.info("{} atomized to {}", input, result);
101    assertEquals(ParseResult.STATUS.SUCCESS, result.getStatus());
102    assertEquals(ParseResult.CONFIDENCE.DEFINITE, result.getConfidence());
103    assertNotNull(result.getPayload());
104    assertEquals(new YearMonthDay("1999", "7", "16"), result.getPayload());
105  }

157  @Test
158  public void testNormalizeStringStringString() {
159    assertEquals(new YearMonthDay("1978", "12", "21"), DateParseUtils.normalize("1978", "DEC", "21"));
160    assertEquals(new YearMonthDay("1850", null, null), DateParseUtils.normalize("1850", "\\N", "\\N"));
161    assertEquals(new YearMonthDay("1800", "06", null), DateParseUtils.normalize("1800", "June", "\\N"));
162    // test that the year inference works ok
163    assertEquals(new YearMonthDay("1974", "03", "12"), DateParseUtils.normalize("12/3/74", null, null));
164    assertEquals(new YearMonthDay("2004", "12", "11"), DateParseUtils.normalize("11/12/04", null, null));
165    // test that 2 digit years are handled sensibly
166    assertEquals(new YearMonthDay("1997", "08", "03"), DateParseUtils.normalize("97", "8", "3"));
167    // when we see an erroneous date, try and infer the year and month
168    assertEquals(new YearMonthDay("1929", "06", null), DateParseUtils.normalize("1929-06-31", null, null));
169
170    assertEquals(new YearMonthDay("1965", "12", "02"), DateParseUtils.normalize("1965", "12", "02"));
171    assertEquals(new YearMonthDay("1992", null, null), DateParseUtils.normalize("1992", "00", "00"));
172    assertEquals(new YearMonthDay("1999", "02", "01"), DateParseUtils.normalize("1999-02-01", null, null));
173
174    assertEquals(new YearMonthDay(null, "08", "26"), DateParseUtils.normalize("32767-08-26", null, null));
175
176  }

208  private void assertEquivalent(String expected, ParseResult<Date> found, ParseResult.CONFIDENCE c) {
209    if (ParseResult.STATUS.ERROR == found.getStatus()) {
210      LOGGER.warn("Unexpected error found in parsing", found.getError());
211    }
212    assertEquals(ParseResult.STATUS.SUCCESS, found.getStatus());
213    SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy");
214    LOGGER.info("Expected [{}]: {}", expected, sdf.format(found.getPayload()));
215    assertEquals(c, found.getConfidence());
216    assertNotNull(found.getPayload());
217    try {
218      assertEquals(DateUtils.parseDateStrictly(expected, new String[] {"dd/MM/yyyy"}), found.getPayload());
219    } catch (ParseException e) {
220      fail("Error in the test class.  Expected date is not in the correct format of dd/MM/yyyy:" + expected);
221    }
222  }
