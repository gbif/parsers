016package org.gbif.common.parsers;
017
018import org.gbif.api.vocabulary.OccurrenceStatus;
019
020import org.junit.jupiter.api.Test;
021
022public class OccurrenceStatusParserTest extends ParserTestBase<OccurrenceStatus> {
023
024  public OccurrenceStatusParserTest() {
025    super(OccurrenceStatusParser.getInstance());
026  }
027
028  @Test
029  public void testParseAllEnumValues() {
030    for (OccurrenceStatus c : OccurrenceStatus.values()) {
031      assertParseSuccess(c, c.name());
032      assertParseSuccess(c, c.name().toLowerCase());
033    }
034  }
035
036  @Test
037  public void testParsing() {
038    // IUCN Red List statuses seem inappropriate to accept.  They are likely to be saying the taxon is rare, rather
039    // than whether the occurrence was present or absent.
040    assertParseFailure("EX");
041    assertParseFailure("EW");
042    assertParseFailure("CR");
043    assertParseFailure("EN");
044    assertParseFailure("VU");
045    assertParseFailure("NT");
046    assertParseFailure("LC");
047    assertParseFailure("DD");
048    assertParseFailure("NE");
049    assertParseFailure("LR/cd");
050    assertParseFailure("LR");
051    assertParseFailure("cd");
052    assertParseFailure("LR/nt");
053    assertParseFailure("LR/lc");
054    assertParseFailure("PE");
055
056    // Top 100 verbatim occurrence status values seen on 2020-05-06.
057    // (Duplicates due to different case and punctuation removed.)
058    assertParseFailure("1");
059    assertParseFailure("Damaged");
060    assertParseFailure("Defoliated");
061    assertParseFailure("doubtful");
062    assertParseFailure("Dudoso");
063    assertParseFailure("E");
064    assertParseFailure("incomplet");
065    assertParseFailure("NA");
066    assertParseFailure("Ne Sait Pas");
067    assertParseFailure("Q");
068    assertParseFailure("Sanitation fellings");
069    assertParseFailure("unknown");
070    assertParseSuccess(OccurrenceStatus.ABSENT, "Absent");
071    assertParseSuccess(OccurrenceStatus.ABSENT, "Ausente");
072    assertParseSuccess(OccurrenceStatus.ABSENT, "Non observé");
073    assertParseSuccess(OccurrenceStatus.PRESENT, "Песня, голос");
074    assertParseSuccess(OccurrenceStatus.PRESENT, "Визуально");
075    assertParseSuccess(OccurrenceStatus.PRESENT, "Летящие");
076    assertParseSuccess(OccurrenceStatus.PRESENT, "Abundant.");
077    assertParseSuccess(OccurrenceStatus.PRESENT, "Abundant (10% <= p < 20%)");
078    assertParseSuccess(OccurrenceStatus.PRESENT, "Abundant 20-99");
079    assertParseSuccess(OccurrenceStatus.PRESENT, "abundante");
080    assertParseSuccess(OccurrenceStatus.PRESENT, "Average Cover: 1-5% Maximum Cover: 1-5%");
081    assertParseSuccess(OccurrenceStatus.PRESENT, "Average Cover: 26-50% Maximum Cover: 26-50%");
082    assertParseSuccess(OccurrenceStatus.PRESENT, "collected");
083    assertParseSuccess(OccurrenceStatus.PRESENT, "common");
084    assertParseSuccess(OccurrenceStatus.PRESENT, "Common 5-19");
085    assertParseSuccess(OccurrenceStatus.PRESENT, "Common (5% <= p < 10%)");
086    assertParseSuccess(OccurrenceStatus.PRESENT, "común");
087    assertParseSuccess(OccurrenceStatus.PRESENT, "complet");
088    assertParseSuccess(OccurrenceStatus.PRESENT, "confirmed breeding");
089    assertParseSuccess(OccurrenceStatus.PRESENT, "Confirmed Present");
090    assertParseSuccess(OccurrenceStatus.PRESENT, "Dominant (20% <= p)");
091    assertParseSuccess(OccurrenceStatus.PRESENT, "established");
092    assertParseSuccess(OccurrenceStatus.PRESENT, "Frequent");
093    assertParseSuccess(OccurrenceStatus.PRESENT, "Infested");
094    assertParseSuccess(OccurrenceStatus.PRESENT, "Irregular");
095    assertParseSuccess(OccurrenceStatus.PRESENT, "locally established");
096    assertParseSuccess(OccurrenceStatus.PRESENT, "muito abundante");
097    assertParseSuccess(OccurrenceStatus.PRESENT, "Muy común");
098    assertParseSuccess(OccurrenceStatus.PRESENT, "Observed in Breeding Season");
099    assertParseSuccess(OccurrenceStatus.PRESENT, "Occasional.");
100    assertParseSuccess(OccurrenceStatus.PRESENT, "Occasional");
101    assertParseSuccess(OccurrenceStatus.PRESENT, "Outbreak");
102    assertParseSuccess(OccurrenceStatus.PRESENT, "P");
103    assertParseSuccess(OccurrenceStatus.PRESENT, "Poco común");
104    assertParseSuccess(OccurrenceStatus.PRESENT, "possible breeding");
105    assertParseSuccess(OccurrenceStatus.PRESENT, "possibly breeding");
106    assertParseSuccess(OccurrenceStatus.PRESENT, "presence");

109    assertParseSuccess(OccurrenceStatus.PRESENT, "Presente");

112    assertParseSuccess(OccurrenceStatus.PRESENT, "Probably Breeding");
