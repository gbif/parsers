28    /**
29     * Pattern for removing measurement denominations
30     */
31    private static final Pattern MEASURE_MARKER_PATTERN = Pattern.compile(".*[a-zA-Z].*");
32  
33    /**
34     * Pattern to remove measurement markers (like "m")
35     */
36    private static final Pattern REMOVE_MEASURE_MARKER_PATTERN = Pattern.compile("[a-zA-Z\" \"\"]");
37  
38    /**
39     * Pattern for recognising measurements in feet
40     */
41    private static final Pattern FEET_MARKER_PATTERN = Pattern.compile(".*ft.*|.*'.*", Pattern.CASE_INSENSITIVE);
42  
43    /**
44     * Pattern for recognising measurements in inches
45     */
46    private static final Pattern INCHES_MARKER_PATTERN = Pattern.compile(".*in.*|.*\".*", Pattern.CASE_INSENSITIVE);
47  
48    /**
49     * Pattern for recognising a range value
50     */
51    private static final Pattern SEP_MARKER_PATTERN = Pattern.compile("\\d-.*");
52  
53    /**
54     * Constant factor to convert from feet to metres.
55     */
56    private static final float FEET_TO_METRES = 0.3048f;
57  
58    /**
59     * Constant factor to convert from inches to metres.
60     */
61    private static final float INCHES_TO_METRES = 0.0254f;

64    /**
65     * The lowest elevation value recognised as valid:
66     * 10,971 m (35,994 ft) Challenger Deep, Mariana Trench[32]
67     */
68    private static final int MIN_ELEVATION = -11000;
69  
70    /**
71     * The highest elevation value recognised as valid, Mount Everest 8,848 m (29,029 ft).
72     * Highest point in the Troposphere: 17km
73     * Highest point in the Stratosphere: 55km
74     * Max altitude of airline cruises: 13km
75     * Max altitude of weather ballons: 34km
76     *
77     *
78     * @see <a href="http://en.wikipedia.org/wiki/Atmosphere_of_Earth">Atmosphere in wikipedia</a>
79     */
80    private static final int MAX_ELEVATION = 17000;
81  
82    /**
83     * The lowest elevation value recognised as valid:
84     * 10,971 m (35,994 ft) Challenger Deep, Mariana Trench[32]
85     */
86    private static final int MAX_DEPTH = Math.abs(MIN_ELEVATION);
87  
88    /**
89     * Largest holes digged into the earth are ~4km.
90     */
91    private static final int MIN_DISTANCE = -5000;
92  
93    /**
94     * Same as elevation, we use the upper end of the Troposphere.
95     */
96    private static final int MAX_DISTANCE = MAX_ELEVATION;

98    static class MeasurementWrapper<T> {
99      private T measurement;
100     private boolean isInFeet;
101     private boolean isInInches;
102     private boolean containsNonNumeric;
103     private boolean minMaxSwapped;
104     private boolean tooLarge;

107       return measurement;

119       return containsNonNumeric;
120     }
121 
122     public boolean isMinMaxSwapped() {
123       return minMaxSwapped;
124     }

127       return tooLarge;

130     public void addIssues(MeasurementWrapper<?> issues) {
131       isInFeet = isInFeet || issues.isInFeet;
132       isInInches = isInInches || issues.isInInches;
133       containsNonNumeric = containsNonNumeric || issues.containsNonNumeric;
134       minMaxSwapped = minMaxSwapped || issues.minMaxSwapped;
135       tooLarge = tooLarge || issues.tooLarge;
136     }

139   /**
140    * Takes min and max values in meters and a known precision and comes up woth a single mean value and its accuracy.
141    * This method tries also to parse common measurements given in feet or inches and converts them to meters.
142    */
143   public static MeasurementWrapper<DoubleAccuracy> parseMeterRange(String minRaw, @Nullable String maxRaw, @Nullable String precisionRaw) {
144     MeasurementWrapper<DoubleAccuracy> result = new MeasurementWrapper<DoubleAccuracy>();
145 
146     MeasurementWrapper<Double> min = parseInMeter(minRaw);
147     MeasurementWrapper<Double> max = parseInMeter(maxRaw);
148     MeasurementWrapper<Double> prec = parseInMeter(precisionRaw);
149 
150     result.addIssues(min);
151     result.addIssues(max);
152     result.addIssues(prec);
153 
154     if (min.measurement == null && max.measurement == null) {
155       // both are null, return issues only
156       return result;
157     }
158 
159     // final result vars
160     Double value;
161     Double accuracy;
162 
163     // check for swapped values and apply precision if min & max exist
164     if (min.measurement != null && max.measurement != null) {
165       // flag swapped min/max
166       if (min.measurement > max.measurement) {
167         result.minMaxSwapped = true;
168         Double oldMin = min.measurement;
169         min.measurement = max.measurement;
170         max.measurement = oldMin;
171       }
172       // apply precision to min max if we have it
173       if (prec.measurement != null) {
174         min.measurement -= prec.measurement;
175         max.measurement += prec.measurement;
176       }
177       // build the arithmetic mean and set accuracy
178       value = (min.measurement + max.measurement) / 2d;
179       accuracy = (max.measurement - min.measurement) / 2d;
180 
181     } else {
182       // use the only value and precision for accuracy
183       value = min.measurement == null ? max.measurement : min.measurement;
184       accuracy = prec.measurement;
185     }
186 
187     if (value != null) {
188       result.measurement = new DoubleAccuracy(value, accuracy);
189     }
190 
191     // finally a result, bye bye!
192     return result;
193   }

195   public static OccurrenceParseResult<DoubleAccuracy> parseElevation(@Nullable String min, @Nullable String max, @Nullable String precision) {
196 
197     MeasurementWrapper<DoubleAccuracy> elevation = parseMeterRange(min, max, precision);
198 
199     Set<OccurrenceIssue> issues = Sets.newHashSet();
200     if(elevation.containsNonNumeric) {
201       issues.add(OccurrenceIssue.ELEVATION_NON_NUMERIC);
202     }
203     if(elevation.isInFeet || elevation.isInInches) {
204       issues.add(OccurrenceIssue.ELEVATION_NOT_METRIC);
205     }
206     if(elevation.minMaxSwapped) {
207       issues.add(OccurrenceIssue.ELEVATION_MIN_MAX_SWAPPED);
208     }
209     if(elevation.tooLarge) {
210       issues.add(OccurrenceIssue.ELEVATION_UNLIKELY);
211     }
212 
213     if (elevation.measurement == null || elevation.measurement.getValue() == null) {
214       return OccurrenceParseResult.fail(issues);
215     }
216 
217     DoubleAccuracy result = elevation.measurement;
218     // record the number of records with altitude out of range
219     if (result.getValue() > MAX_ELEVATION || result.getValue() < MIN_ELEVATION) {
220       issues.add(OccurrenceIssue.ELEVATION_UNLIKELY);
221       return OccurrenceParseResult.fail(issues);
222     }
223 
224     return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result, issues);
225   }

227   public static OccurrenceParseResult<DoubleAccuracy> parseDepth(@Nullable String min, @Nullable String max, @Nullable String precision) {
228     MeasurementWrapper<DoubleAccuracy> depth = parseMeterRange(min, max, precision);
229 
230     Set<OccurrenceIssue> issues = Sets.newHashSet();
231     if(depth.containsNonNumeric) {
232       issues.add(OccurrenceIssue.DEPTH_NON_NUMERIC);
233     }
234     if(depth.isInFeet || depth.isInInches) {
235       issues.add(OccurrenceIssue.DEPTH_NOT_METRIC);
236     }
237     if(depth.minMaxSwapped) {
238       issues.add(OccurrenceIssue.DEPTH_MIN_MAX_SWAPPED);
239     }
240     if(depth.tooLarge) {
241       issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
242     }
243 
244     if (depth.measurement == null || depth.measurement.getValue() == null) {
245       return OccurrenceParseResult.fail(issues);
246     }
247 
248     DoubleAccuracy result = depth.measurement;
249 
250     // negate depth if its negative
251     if (result.getValue() < 0) {
252       result = new DoubleAccuracy(-1 * result.getValue(), result.getAccuracy());
253       issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
254     }
255 
256     // record the number of records with depth out of range
257     if (result.getValue() > MAX_DEPTH) {
258       issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
259       return OccurrenceParseResult.fail(issues);
260     }
261 
262     return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result, issues);
263   }

265   /**
266    * Parses a string supposed to be a value in meters.
267    * Accepts also feet if marked with a unit and converts them
268    */
269   public static ParseResult<Double> parseMeters(String meter) {
270     MeasurementWrapper<Double> result = parseInMeter(meter);
271     if (result.getMeasurement() == null) {
272       return ParseResult.fail();
273     }
274     return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result.getMeasurement());
275   }

277   private static MeasurementWrapper<Double> parseInMeter(String meter) {
278     MeasurementWrapper<Double> iMeter = new MeasurementWrapper<Double>();
279 
280     if (Strings.isNullOrEmpty(meter)) {
281       return iMeter;
282     }
283 
284     try {
285       iMeter.containsNonNumeric = MEASURE_MARKER_PATTERN.matcher(meter).matches();
286 
287       if (!iMeter.containsNonNumeric()) {
288         iMeter.measurement = NumberParser.parseDouble(meter);
289 
290       } else {
291         iMeter.isInFeet = FEET_MARKER_PATTERN.matcher(meter).matches();
292         iMeter.isInInches = INCHES_MARKER_PATTERN.matcher(meter).matches();
293 
294         // handle 6-7m values
295         if (SEP_MARKER_PATTERN.matcher(meter).matches()) {
296           // we have been given a range
297           try {
298             String min = meter.substring(0, meter.indexOf('-')).trim();
299             min = removeMeasurementMarkers(min);
300             String max = meter.substring(meter.indexOf('-') + 1).trim();
301             max = removeMeasurementMarkers(max);
302 
303             Double minDouble = NumberParser.parseDouble(min);
304             Double maxDouble = NumberParser.parseDouble(max);
305 
306             if(minDouble == null && maxDouble != null){
307               iMeter.measurement = maxDouble;
308             }else if(maxDouble == null && minDouble != null){
309               iMeter.measurement = maxDouble;
310             }else if (minDouble != 0 && maxDouble != 0 && maxDouble - minDouble != 0) {
311               iMeter.measurement = (maxDouble + minDouble) / 2;
312             }
313           } catch (NumberFormatException ignored) {
314           }
315 
316         } else {
317           iMeter.measurement = NumberParser.parseDouble(removeMeasurementMarkers(meter));
318         }
319 
320         if (iMeter.measurement != null) {
321           // convert to metric
322           if (iMeter.isInFeet) {
323             iMeter.measurement = convertFeetToMetres(iMeter.measurement);
324           } else if (iMeter.isInInches) {
325             iMeter.measurement = convertInchesToMetres(iMeter.measurement);
326           }
327         }
328       }
329     } catch (NumberFormatException e) {
330       LOG.debug("Unparsable meter measurement: {}", meter, e.getMessage());
331     }
332 
333     // round to centimeters
334     if (iMeter.measurement != null) {
335       iMeter.measurement = Math.round(iMeter.measurement * 100.0) / 100.0;
336     }
337     return iMeter;
338   }

340   /**
341    * Remove "m" etc.
342    *
343    * @param s to remove measurement markers from.
344    *
345    * @return a new string with all measurements removed (i.e. replaced by the empty string)
346    */
347   private static String removeMeasurementMarkers(String s) {
348     if (s == null) return null;
349     return REMOVE_MEASURE_MARKER_PATTERN.matcher(s).replaceAll("");
350   }
351 
352   private static double convertInchesToMetres(double inches) {
353     return inches * INCHES_TO_METRES;
354   }
355 
356   private static double convertFeetToMetres(double feet) {
357     return feet * FEET_TO_METRES;
358   }

360   /**
361    * @return rounded int value or null if it was null or exceeds the maximum an int can hold
362    */
363   private static Integer roundedInt(Double x) {
364     if (x == null) return null;
365 
366     Long xl = Math.round(x);
367     if (xl > Integer.MAX_VALUE) {
368       throw new IllegalArgumentException("Long too big for an integer");
369     }
370     return xl.intValue();
371   }
