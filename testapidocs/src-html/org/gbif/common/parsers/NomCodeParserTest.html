001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.NomenclaturalCode;
004import org.junit.jupiter.api.Test;
005
006public class NomCodeParserTest extends ParserTestBase<NomenclaturalCode> {
007
008    public NomCodeParserTest() {
009        super(NomCodeParser.getInstance());
010    }
011
012    /**
013     * This ensures that ALL enum values are at least parsable by the name they
014     * are created with.
015     */
016    @Test
017    public void testCompleteness() {
018        for (NomenclaturalCode t : NomenclaturalCode.values()) {
019            assertParseSuccess(t, t.name());
020            assertParseSuccess(t, t.name().toLowerCase());
021            assertParseSuccess(t, t.name().replace("_", "").toLowerCase());
022            assertParseSuccess(t, t.getAcronym());
023            assertParseSuccess(t, t.getTitle());
024        }
025    }
026
027    @Test
028    public void testFailures() {
029        assertParseFailure(null);
030        assertParseFailure("");
031        assertParseFailure("Tim");
032    }
033
034
035    @Test
036    public void testParse() {
037        // run a few basic tests to check it bootstraps and appears to work
038        assertParseSuccess(NomenclaturalCode.BOTANICAL, "ICBN");
039        assertParseSuccess(NomenclaturalCode.BOTANICAL, "botany");
040        assertParseSuccess(NomenclaturalCode.ZOOLOGICAL, "zoo");
041    }
042
043}
