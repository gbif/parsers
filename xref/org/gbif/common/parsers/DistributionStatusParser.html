1   package org.gbif.common.parsers;
2   
3   import org.gbif.api.vocabulary.DistributionStatus;
4   import org.gbif.common.parsers.core.EnumParser;
5   
6   /**
7    * Singleton implementation of the dictionary that uses the file /dictionaries/parse/distribution_status.tsv.
8    */
9   public class DistributionStatusParser extends EnumParser<DistributionStatus> {
10  
11    private static DistributionStatusParser singletonObject = null;
12  
13    private DistributionStatusParser() {
14      super(DistributionStatus.class, false);
15      // make sure we have all occurrence_status enum mapped
16      for (DistributionStatus c : DistributionStatus.values()) {
17        add(c.name(), c);
18      }
19      // use dict file last
20      init(DistributionStatusParser.class.getResourceAsStream("/dictionaries/parse/distribution_status.tsv"));
21    }
22  
23    public static DistributionStatusParser getInstance() {
24      synchronized (DistributionStatusParser.class) {
25        if (singletonObject == null) {
26          singletonObject = new DistributionStatusParser();
27        }
28      }
29      return singletonObject;
30    }
31  
32  }
