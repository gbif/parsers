001package org.gbif.common.parsers;
002
003import org.gbif.common.parsers.core.Parsable;
004import org.gbif.common.parsers.core.ParseResult;
005
006import java.io.IOException;
007
008import com.google.common.base.Strings;
009import com.google.common.io.LineProcessor;
010import org.junit.Before;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.assertNotNull;
014
015public abstract class ParserTestBase<T> implements LineProcessor<ParserTestBase.BatchParseResult>  {
016  protected final Parsable<T> parser;
017  private BatchParseResult batchResult;
018
019  public static class BatchParseResult {
020    public int total;
021    public int failed;
022  }
023
024  @Before
025  public void setupBatch() {
026    batchResult = new BatchParseResult();
027  }
028
029  public ParserTestBase(Parsable<T> parser) {
030    this.parser = parser;
031  }
032
033  protected void assertParseFailure(String input) {
034    ParseResult<T> parsed = parser.parse(input);
035    assertEquals("Expected "+input+" to fail but got "+parsed.getPayload()+" instead", ParseResult.STATUS.FAIL, parsed.getStatus());
036  }
037
038  protected void assertParseSuccess(T expected, String input) {
039    assertParseSuccess(expected, ParseResult.CONFIDENCE.DEFINITE, input);
040  }
041
042  protected void assertParseSuccess(T expected, ParseResult.CONFIDENCE confidence, String input) {
043    ParseResult<T> parsed = parser.parse(input);
044    assertNotNull(parsed);
045    assertEquals(ParseResult.STATUS.SUCCESS, parsed.getStatus());
046    assertNotNull(parsed.getPayload());
047    assertEquals(expected, parsed.getPayload());
048    if (confidence != null) {
049      assertEquals(confidence, parsed.getConfidence());
050    }
051  }
052
053  @Override
054  public boolean processLine(String line) throws IOException {
055    if (Strings.isNullOrEmpty(line)) {
056      return false;
057    }
058    ParseResult<T> parsed = parser.parse(line);
059    batchResult.total++;
060    if (parsed == null || !parsed.isSuccessful() || parsed.getPayload()==null) {
061      batchResult.failed++;
062    }
063    return true;
064  }
065
066  @Override
067  public ParserTestBase.BatchParseResult getResult() {
068    return batchResult;
069  }
070}
