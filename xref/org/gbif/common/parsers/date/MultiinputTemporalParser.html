35    private MultiinputTemporalParser(List<DateComponentOrdering> orderings) {
36      if (orderings != null && !orderings.isEmpty()) {
37        DateComponentOrderingeComponentOrdering">DateComponentOrdering[] array = orderings.toArray(new DateComponentOrdering[0]);
38        temporalParser = CustomizedTextDateParser.getInstance(array);
39      } else {
40        temporalParser = DateParsers.defaultTemporalParser();
41      }
42    }

57    /**
58     * Three dates are provided:
59     *
60     * <ul>
61     *   <li>year, month and day
62     *   <li>dateString
63     *   <li>year and dayOfYear
64     * </ul>
65     *
66     * <p>Produces a single date at the best resolution possible, ignoring missing values.
67     *
68     * <p>Years are verified to be before this year and after 1500.
69     *
70     * @return interpretation result, never null
71     */
72    public OccurrenceParseResult<TemporalAccessor> parseRecordedDate(
73        String year, String month, String day, String dateString, String dayOfYear) {
74  
75      boolean ymdProvided =
76          StringUtils.isNotBlank(year)
77              || StringUtils.isNotBlank(month)
78              || StringUtils.isNotBlank(day);
79      boolean dateStringProvided = StringUtils.isNotBlank(dateString);
80      boolean yDoyProvided = StringUtils.isNotBlank(year) && StringUtils.isNotBlank(dayOfYear);
81  
82      // If we have only a year and dayOfYear, don't parse the year alone (without the month and day)
83      if (yDoyProvided && (StringUtils.isBlank(month) && StringUtils.isBlank(day))) {
84        ymdProvided = false;
85      }
86  
87      boolean twoOrMoreProvided = (ymdProvided ? 1 : 0) + (dateStringProvided ? 1 : 0) + (yDoyProvided ? 1 : 0) >= 2;
88  
89      if (!ymdProvided && !dateStringProvided && !yDoyProvided) {
90        log.trace("Date {}|{}|{}|{}|{} is all null", year, month, day, dateString, dayOfYear);
91        return OccurrenceParseResult.fail();
92      }
93  
94      Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
95  
96      TemporalAccessor parsedTemporalAccessor;
97      ParseResult.CONFIDENCE confidence;
98  
99      // Parse all three possible dates
100     ParseResult<TemporalAccessor> parsedYMDResult =
101         ymdProvided ? temporalParser.parse(year, month, day) : ParseResult.fail();
102     ParseResult<TemporalAccessor> parsedDateResult =
103         dateStringProvided ? temporalParser.parse(dateString) : ParseResult.fail();
104     ParseResult<TemporalAccessor> parsedYearDoyResult =
105         yDoyProvided ? temporalParser.parse(year, dayOfYear) : ParseResult.fail();
106     TemporalAccessor parsedYmdTa = parsedYMDResult.getPayload();
107     TemporalAccessor parsedDateTa = parsedDateResult.getPayload();
108     TemporalAccessor parsedYearDoyTa = parsedYearDoyResult.getPayload();
109 
110     int ymdResolution = -1, dateStringResolution = -1;
111     if (ymdProvided && parsedYMDResult.isSuccessful()) {
112       ymdResolution = TemporalAccessorUtils.resolution(parsedYmdTa);
113     }
114     if (dateStringProvided && parsedDateResult.isSuccessful()) {
115       dateStringResolution = TemporalAccessorUtils.resolution(parsedDateTa);
116     }
117 
118     // Add issues if we failed to parse any dates that were present
119     if (ymdProvided && !parsedYMDResult.isSuccessful()) {
120       issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
121     }
122     if (dateStringProvided && !parsedDateResult.isSuccessful()) {
123       issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
124     }
125     if (yDoyProvided && !parsedYearDoyResult.isSuccessful()) {
126       issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
127     }
128 
129     log.trace("Date {}|{}|{}|{}|{} parsed to {}—{}—{}", year, month, day, dateString, dayOfYear, parsedYMDResult, parsedDateResult, parsedYearDoyResult);
130 
131     // If a dateString is provided with something else, handle the case where it doesn't match.
132     boolean ambiguityResolved = false;
133     if (ymdProvided
134         && dateStringProvided
135         && !TemporalAccessorUtils.sameOrContained(parsedYmdTa, parsedDateTa)
136         && parsedDateResult.getAlternativePayloads() != null) {
137 
138       // eventDate could be ambiguous (5/4/2014), but disambiguated by year-month-day.
139       Optional<TemporalAccessor> resolved =
140           TemporalAccessorUtils.resolveAmbiguousDates(
141               parsedYmdTa, parsedDateResult.getAlternativePayloads());
142       if (resolved.isPresent()) {
143         parsedDateTa = resolved.get();
144         ambiguityResolved = true;
145         log.trace("Date {}|{}|{}|{}|{} ambiguous₁ y-m-d resolved {}", year, month, day, dateString, dayOfYear, parsedDateTa);
146       }
147       // still a conflict
148       if (!ambiguityResolved) {
149         if (parsedYmdTa == null || parsedDateTa == null) {
150           log.debug("Date {}|{}|{}|{}|{} ambiguous₁ invalid", year, month, day, dateString, dayOfYear);
151           issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
152         } else {
153           log.debug("Date {}|{}|{}|{}|{} ambiguous₁ mismatch", year, month, day, dateString, dayOfYear);
154           issues.add(OccurrenceIssue.RECORDED_DATE_MISMATCH);
155         }
156       }
157     } else if (ymdProvided
158         && yDoyProvided
159         && !TemporalAccessorUtils.sameOrContained(parsedYearDoyTa, parsedDateTa)
160         && parsedDateResult.getAlternativePayloads() != null) {
161 
162       // eventDate could be ambiguous (5/4/2014), but disambiguated by year-month-day.
163       Optional<TemporalAccessor> resolved =
164           TemporalAccessorUtils.resolveAmbiguousDates(
165               parsedYearDoyTa, parsedDateResult.getAlternativePayloads());
166       if (resolved.isPresent()) {
167         parsedDateTa = resolved.get();
168         ambiguityResolved = true;
169         log.trace("Date {}|{}|{}|{}|{} ambiguous₂ y-doy resolved {}", year, month, day, dateString, dayOfYear, parsedDateTa);
170       }
171       // still a conflict
172       if (!ambiguityResolved) {
173         if (parsedYmdTa == null || parsedYearDoyTa == null) {
174           log.debug("Date {}|{}|{}|{}|{} ambiguous₂ invalid", year, month, day, dateString, dayOfYear);
175           issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
176         } else {
177           log.debug("Date {}|{}|{}|{}|{} ambiguous₂ mismatch", year, month, day, dateString, dayOfYear);
178           issues.add(OccurrenceIssue.RECORDED_DATE_MISMATCH);
179         }
180       }
181     }
182 
183     // Add an issue if there is any conflict between the dates
184     if (TemporalAccessorUtils.sameOrContainedOrNull(parsedYmdTa, parsedDateTa)
185         && TemporalAccessorUtils.sameOrContainedOrNull(parsedYmdTa, parsedYearDoyTa)
186         && TemporalAccessorUtils.sameOrContainedOrNull(parsedDateTa, parsedYearDoyTa)) {
187       confidence =
188           parsedDateTa != null
189               ? parsedDateResult.getConfidence()
190               : (parsedYmdTa != null
191                   ? parsedYMDResult.getConfidence()
192                   : parsedYearDoyResult.getConfidence());
193     } else {
194       log.debug("Date {}|{}|{}|{}|{} mismatch (conflict)", year, month, day, dateString, dayOfYear);
195       issues.add(OccurrenceIssue.RECORDED_DATE_MISMATCH);
196       confidence = PROBABLE;
197     }
198 
199     // Add an issue if the resolution af ymd / date / yDoy is different
200     if (ymdResolution > 0 && dateStringResolution > 0) {
201       if (ymdResolution != dateStringResolution) {
202         log.debug("Date {}|{}|{}|{}|{} mismatch (resolution)", year, month, day, dateString, dayOfYear);
203         issues.add(OccurrenceIssue.RECORDED_DATE_MISMATCH);
204       }
205     }
206 
207     // Best we can get from the three parts.
208     // Note 2000-01-01 and 2000-01 and 2000 will return 2000-01-01.
209     Optional<TemporalAccessor> nonConflictingTa =
210         TemporalAccessorUtils.nonConflictingDateParts(parsedYmdTa, parsedDateTa, parsedYearDoyTa);
211 
212     if (nonConflictingTa.isPresent()) {
213       parsedTemporalAccessor = nonConflictingTa.get();
214       // if one of the parses failed we can not set the confidence to DEFINITE
215       confidence =
216           ((ymdProvided && parsedYmdTa == null)
217                   || (dateStringProvided && parsedDateTa == null)
218                   || (yDoyProvided && parsedYearDoyTa == null))
219               ? PROBABLE
220               : confidence;
221     } else {
222       log.debug("Date {}|{}|{}|{}|{} mismatch (conflicting)", year, month, day, dateString, dayOfYear);
223       if (twoOrMoreProvided) {
224         issues.add(OccurrenceIssue.RECORDED_DATE_MISMATCH);
225       }
226       return OccurrenceParseResult.fail(issues);
227     }
228 
229     if (!isValidDate(parsedTemporalAccessor)) {
230       if (parsedTemporalAccessor == null) {
231         log.debug("Date {}|{}|{}|{}|{} mismatch (invalid)", year, month, day, dateString, dayOfYear);
232         issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
233       } else {
234         log.debug("Date {}|{}|{}|{}|{} mismatch (unlikely)", year, month, day, dateString, dayOfYear);
235         issues.add(OccurrenceIssue.RECORDED_DATE_UNLIKELY);
236       }
237 
238       return OccurrenceParseResult.fail(issues);
239     }
240 
241     return OccurrenceParseResult.success(confidence, parsedTemporalAccessor, issues);
242   }

254   /** @return TemporalAccessor that represents a LocalDate or LocalDateTime */
255   public OccurrenceParseResult<TemporalAccessor> parseLocalDate(
256       String dateString, Range<LocalDate> likelyRange, OccurrenceIssue unlikelyIssue, OccurrenceIssue failIssue) {
257     if (!Strings.isNullOrEmpty(dateString)) {
258       OccurrenceParseResult<TemporalAccessor> result =
259           new OccurrenceParseResult<>(temporalParser.parse(dateString));
260       // check year makes sense
261       if (result.isSuccessful() && !isValidDate(result.getPayload(), likelyRange)) {
262         log.debug("Unlikely date parsed, ignore [{}].", dateString);
263         Optional.ofNullable(unlikelyIssue).ifPresent(result::addIssue);
264       } else if (!result.isSuccessful()) {
265         Optional.ofNullable(failIssue).ifPresent(result::addIssue);
266       }
267       return result;
268     }
269     return OccurrenceParseResult.fail();
270   }

272   /**
273    * Check if a date express as TemporalAccessor falls between the predefined range. Lower bound
274    * defined by {@link #MIN_LOCAL_DATE} and upper bound by current date + 1 day
275    *
276    * @return valid or not according to the predefined range.
277    */
278   protected static boolean isValidDate(TemporalAccessor temporalAccessor) {
279     LocalDate upperBound = LocalDate.now().plusDays(1);
280     return isValidDate(temporalAccessor, Range.closed(MIN_LOCAL_DATE, upperBound));
281   }
282 
283   /** Check if a date express as TemporalAccessor falls between the provided range. */
284   protected static boolean isValidDate(
285       TemporalAccessor temporalAccessor, Range<LocalDate> likelyRange) {
286 
287     if (temporalAccessor == null) {
288       return false;
289     }
290 
291     // if partial dates should be considered valid
292     int year;
293     int month = 1;
294     int day = 1;
295     if (temporalAccessor.isSupported(ChronoField.YEAR)) {
296       year = temporalAccessor.get(ChronoField.YEAR);
297     } else {
298       return false;
299     }
300 
301     if (temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
302       month = temporalAccessor.get(ChronoField.MONTH_OF_YEAR);
303     }
304 
305     if (temporalAccessor.isSupported(ChronoField.DAY_OF_MONTH)) {
306       day = temporalAccessor.get(ChronoField.DAY_OF_MONTH);
307     }
308 
309     return likelyRange.contains(LocalDate.of(year, month, day));
310   }
