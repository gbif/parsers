017  @Test
018  public void testParseLatLng() {
019    assertExpected(CoordinateParseUtils.parseLatLng("-46,33", "51,8717"), new LatLng(-46.33, 51.8717), ParseResult.CONFIDENCE.DEFINITE);
020    assertExpected(CoordinateParseUtils.parseLatLng("10.3", "99.99"), new LatLng(10.3, 99.99), ParseResult.CONFIDENCE.DEFINITE);
021    assertExpected(CoordinateParseUtils.parseLatLng("10", "10"), new LatLng(10, 10), ParseResult.CONFIDENCE.DEFINITE);
022    assertExpected(CoordinateParseUtils.parseLatLng("90", "180"), new LatLng(90, 180), ParseResult.CONFIDENCE.DEFINITE);
023    assertExpected(CoordinateParseUtils.parseLatLng("-90", "180"), new LatLng(-90, 180), ParseResult.CONFIDENCE.DEFINITE);
024    assertExpected(CoordinateParseUtils.parseLatLng("90", "-180"), new LatLng(90, -180), ParseResult.CONFIDENCE.DEFINITE);
025    assertExpected(CoordinateParseUtils.parseLatLng("-90", "-180"), new LatLng(-90, -180), ParseResult.CONFIDENCE.DEFINITE);
026    assertExpected(CoordinateParseUtils.parseLatLng("0", "0"), new LatLng(0, 0), ParseResult.CONFIDENCE.POSSIBLE, OccurrenceIssue.ZERO_COORDINATE);
027
028    // rounding
029    assertExpected(CoordinateParseUtils.parseLatLng("2.123450678", "-8.123450678"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
030    assertExpected(CoordinateParseUtils.parseLatLng("2.123451", "-8.123450678"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
031    assertExpected(CoordinateParseUtils.parseLatLng("2.123451", "-8.123451"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE);
032    assertExpected(CoordinateParseUtils.parseLatLng("2.12345100", "-8.1234510"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE);
033    assertExpected(CoordinateParseUtils.parseLatLng("2.123", "-8.1234506"), new LatLng(2.123, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
034
035    // degree minutes seconds
036    assertExpected(CoordinateParseUtils.parseLatLng("02° 49' 52\" N", "131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
037
038    // check swapped coordinates
039    assertExpected(CoordinateParseUtils.parseLatLng("100", "40"), new LatLng(40, 100), ParseResult.CONFIDENCE.PROBABLE, OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
040    assertExpected(CoordinateParseUtils.parseLatLng("-100", "90"), new LatLng(90, -100), ParseResult.CONFIDENCE.PROBABLE, OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
041
042    // check errors
043    assertFailed(CoordinateParseUtils.parseLatLng("", "30"));
044    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("tim", "tom"), OccurrenceIssue.COORDINATE_INVALID);
045    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20,432,12", "13,4"), OccurrenceIssue.COORDINATE_INVALID);
046    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "200"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
047    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("-200", "30"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
048    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "30"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
049    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20.432,12", "13,4"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
050    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20,432,12", "13,4"), OccurrenceIssue.COORDINATE_INVALID);
051  }

053  @Test
054  public void testParseDMS() {
055    assertDMS("2°49'N", "131°47'E", 2.816667d, 131.783333d);
056
057    assertDMS("02° 49' 52\" N", "131° 47' 03\" E", 2.831111d, 131.784167d);
058    assertDMS("2°49'52\"S", "131°47'03\" W", -2.831111d, -131.784167d);
059    assertDMS("2°49'52\"  n", "131°47'03\"  O", 2.831111d, 131.784167d);
060    assertDMS("002°49'52\"N", "131°47'03\"E", 2.831111d, 131.784167d);
061    assertDMS("2°49'N", "131°47'E", 2.816667d, 131.783333d);
062    assertDMS("002°49'52''N", "131°47'03''E", 2.831111d, 131.784167d);
063    assertDMS("6º39'36\"S","35º59'59\"W", -6.66d, -35.999722);
064    assertDMS("17  02.877 N", "121  05.966 E", 17.04795, 121.099433);
065    assertDMS("08º37'S", "37º10'W", -8.616667, -37.166667);
066    assertDMS("39g30mS", "56g27mW", -39.5, -56.45);
067    assertDMS("42g24m50.00sS", "64g17m20.00sW", -42.413889, -64.288889);
068    assertDMS("42º30´S", "54º14´W", -42.5, -54.233333);
069    assertDMS("61o50'N", "30o45'E", 61.833333, 30.75);
070    assertDMS("07°35N", "38°44E", 7.583333, 38.733333);
071    assertDMS("29º32.3’N", "113º34.9’W", 29.538333, -113.581667);
072    assertDMS("5°45′30″N", "100º30′30″W", 5.758333, -100.508333);
073    assertDMS("125º52´2´´S", "39º28´47´´W", -125.867222, -39.479722);
074    assertDMS("13.1939 N", "59.5432 W", 13.1939, -59.5432);
075    assertDMS("24 06.363 N", "110 11.969 E", 24.10605d, 110.199483d);
076    // even if it's out of range that's expected here - we validate elsewhere, this is an internal method only!
077    assertDMS("122°49'52\"N", "131°47'03\"E", 122.831111d, 131.784167d);
078    assertDMS("122°49'52.39\"N", "131°47'03.23\"E", 122.83121944444444d, 131.78423055555555d);
079    assertDMS("122d49m52.39sN", "131d47m03.23sE", 122.83121944444444d, 131.78423055555555d);
080    assertDMS("122 49 52.39", "131 47 03.23", 122.83121944444444d, 131.78423055555555d);
081    assertDMS("N122°49'52.39\"", "E131°47'03.23\"", 122.83121944444444d, 131.78423055555555d);
082    assertDMS("N 122° 49' 52.39\"", "E 131° 47' 03.23\"", 122.83121944444444d, 131.78423055555555d);
083    assertDMS("N 122°49'52.39\"", "E 131°47'03.23\"", 122.83121944444444d, 131.78423055555555d);
084    assertDMS("N 122°49'52\"", "E 131°47'03.23\"", 122.831111d, 131.78423055555555d);
085    assertDMS("N122°49'52.39\"", "E131°47'03.23\"", 122.83121944444444d, 131.78423055555555d);
086    assertDMS("N36.93276", "W102.96361", 36.93276, -102.96361);
087    assertDMS("90°N", "180°", 90, 180);
088
089    // truly failing
090    assertIllegalArg("12344", true);
091    assertIllegalArg("432", false);
092    assertIllegalArg(" ", true);
093    assertIllegalArg(" ", false);
094    assertIllegalArg("131°47'132\"", false);
095    assertIllegalArg("131 47 132", false);
096    assertIllegalArg("131°47'132\"", false);
097    assertIllegalArg("043300S",true);
098    assertIllegalArg("0433S",true);
099    assertIllegalArg("043300 S",true);
100    assertIllegalArg("0433 S",true);
101  }

103  private void assertDMS(String lat, String lon, double eLat, double eLon){
104    assertEquals(eLat, CoordinateParseUtils.parseDMS(lat, true), 0.000001);
105    assertEquals(eLon, CoordinateParseUtils.parseDMS(lon, false), 0.000001);
106  }

109    try {

113      // expected!
114    }

117  @Test
118  public void testParseVerbatimCoordinates() {
119    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02° 49' 52\" N 131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
120    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02° 49' 52\" N, 131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
121    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02°49'52\"N; 131°47'03\"O"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
122    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("17d 33m 5s N/99d 30m 3s W"), new LatLng(17.551389d, -99.500833d), ParseResult.CONFIDENCE.DEFINITE);
123    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("14.93333/-91.9"), new LatLng(14.93333d, -91.9d), ParseResult.CONFIDENCE.DEFINITE);
124    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("63d 41m 39s N 170d 28m 44s W"), new LatLng(63.694167d, -170.478889d), ParseResult.CONFIDENCE.DEFINITE);
125    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("37° 28' N, 122° 6' W"), new LatLng(37.466667d, -122.1d), ParseResult.CONFIDENCE.DEFINITE);
126    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("2°49'52\"N, 131°47'03\""), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
127    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10°07'06\"N 20°48'23\"W"), new LatLng(10.118333, -20.806389), ParseResult.CONFIDENCE.DEFINITE);
128    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10º07'06\"N 20º48'23\"W"), new LatLng(10.118333, -20.806389), ParseResult.CONFIDENCE.DEFINITE);
129    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10°07'N 20°48'W"), new LatLng(10.116667, -20.8), ParseResult.CONFIDENCE.DEFINITE);
130    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10°07.55'N 20°48.55'W"), new LatLng(10.125833, -20.809167), ParseResult.CONFIDENCE.DEFINITE);
131    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("100º23'05\"N 20º35'25\"W"), new LatLng(-20.590278, 100.384722), ParseResult.CONFIDENCE.PROBABLE, OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
132
133    // failed
134    assertFailed(CoordinateParseUtils.parseVerbatimCoordinates(""));
135    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates("12344"), OccurrenceIssue.COORDINATE_INVALID);
136    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates(" "), OccurrenceIssue.COORDINATE_INVALID);
137    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates(",11.12"), OccurrenceIssue.COORDINATE_INVALID);
138    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates("122°49'52\"N, 131°47'03\"E"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
139  }

141  private void assertExpected(OccurrenceParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
142    assertNotNull(pr);
143    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
144    assertEquals(c, pr.getConfidence());
145    assertNotNull(pr.getPayload());
146    assertEquals(expected, pr.getPayload());
147    //System.out.println(Lists.newArrayList(issue));
148    if (issue == null) {
149      assertTrue(pr.getIssues().isEmpty());
150    } else {
151      assertEquals("Wrong number of issues. Found " + pr.getIssues() + ", expected " + Lists.newArrayList(issue), issue.length, pr.getIssues().size());
152      for (OccurrenceIssue iss : issue) {
153        assertTrue(pr.getIssues().contains(iss));
154      }
155    }
156  }
157
158  private void assertFailed(ParseResult<?> pr) {
159    assertNotNull(pr);
160    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
161  }
162
163  private void assertFailedWithIssues(OccurrenceParseResult<LatLng> pr, OccurrenceIssue ... issue) {
164    assertFailed(pr);
165    assertEquals(issue.length, pr.getIssues().size());
166    for (OccurrenceIssue iss : issue){
167      assertTrue(pr.getIssues().contains(iss));
168    }
169  }
