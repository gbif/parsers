35    public static String canonicalName(String scientificName, Rank rank) {
36      ParsedName pn = null;
37      try {
38        pn = PARSER.parse(scientificName, rank);

40      }

54    /**
55     * Cleans up a taxon as far as possible by removing erroneous chars etc.
56     * This does not do any parsing.
57     *
58     * @param taxon to check
59     */
60    public static String clean(String taxon) {
61      if (Strings.isNullOrEmpty(taxon) || NULL_STRINGS.contains(taxon)) {
62        return null;
63      }
64  
65      String cleanedTaxon = taxon;
66  
67      // if it is a single word and ALL "UPPERCASE", turn it into a Capitalised word
68      // Note: if we lowercase names with multiple words we might accidently create valid looking names by lowercasing the
69      // author
70      // for example ABIES ALBA REMSEN will become an Abies alba remsen which will then be interpreted badly
71      // ABIES ALBA LINNEAUS 1771 will even be Abies alba linneaus 1771, a perfectly formed zoological name
72      if (!cleanedTaxon.contains(" ") && cleanedTaxon.equals(cleanedTaxon.toUpperCase())) {
73        cleanedTaxon = cleanedTaxon.substring(0, 1) + cleanedTaxon.substring(1).toLowerCase();
74      }
75  
76      // remove the " from names with it at the beginning and end
77      while (cleanedTaxon.length()> 0 && cleanedTaxon.charAt(0) == '\"' && cleanedTaxon.charAt(cleanedTaxon.length() - 1) == '\"') {
78        if (cleanedTaxon.length() == 1) {
79          return null;
80        }
81        cleanedTaxon = cleanedTaxon.substring(1, cleanedTaxon.length() - 1);
82      }
83  
84      // remove the " from names with it just at the beginning
85      while (cleanedTaxon.length()> 0 && cleanedTaxon.charAt(0) == '\"') {
86        if (cleanedTaxon.length() == 1) {
87          return null;
88        }
89        cleanedTaxon = cleanedTaxon.substring(1, cleanedTaxon.length());
90      }
91  
92      // remove the " from names with it just at the end
93      while (cleanedTaxon.length()> 0 && cleanedTaxon.charAt(cleanedTaxon.length() - 1) == '\"') {
94        if (cleanedTaxon.length() == 1) {
95          return null;
96        }
97        cleanedTaxon = cleanedTaxon.substring(0, cleanedTaxon.length() - 1);
98      }
99  
100 
101     // remove noise
102     cleanedTaxon = CLEAN_REG_EX.matcher(cleanedTaxon).replaceAll("");
103     cleanedTaxon = cleanedTaxon.trim();
104 
105     // don't let any blacklisted names through
106     if (BlacklistedNames.contains(cleanedTaxon.toUpperCase()) || (cleanedTaxon != taxon && BlacklistedNames
107       .contains(taxon.toUpperCase()))) {
108       // blacklisted name
109       return null;
110     }
111 
112     return Strings.emptyToNull(cleanedTaxon);
113   }

115   /**
116    * Clean some noise from the author. A large proportion are "\N" for example.
117    *
118    * @param author to clean
119    *
120    * @return cleaned author
121    */
122   public static String cleanAuthor(String author) {
123     if (Strings.isNullOrEmpty(author) || NULL_STRINGS.contains(author)) {
124       return null;
125     }
126 
127     String cleanedAuthor = author;
128 
129     // remove the " from names with it at the beginning and end
130     while (cleanedAuthor.length() >0 && cleanedAuthor.charAt(0) == '\"' && cleanedAuthor.charAt(cleanedAuthor.length() - 1) == '\"') {
131       if (cleanedAuthor.length() == 1) {
132         return null;
133       }
134       cleanedAuthor = cleanedAuthor.substring(1, cleanedAuthor.length() - 1);
135     }
136 
137     // remove noise
138     cleanedAuthor = CLEAN_REG_EX.matcher(cleanedAuthor).replaceAll("");
139     cleanedAuthor = cleanedAuthor.trim();
140 
141     return Strings.emptyToNull(cleanedAuthor);
142   }

159 
160     try {

162       // Handle Aus sp. and Aus bus spp.
163       if (pn.isIndetermined()) {
164           pn.setRank(null);
165       }
166       return pn.fullName();
167 
168     } catch (UnparsableException e) {
169       // TODO: logging
170     }

176     }
