001package org.gbif.common.parsers;
002
003import org.gbif.api.model.common.MediaObject;
004import org.gbif.api.vocabulary.MediaType;
005
006import java.net.URI;
007import java.util.Set;
008import javax.annotation.Nullable;
009
010import com.google.common.base.Strings;
011import com.google.common.collect.ImmutableSet;
012import org.apache.tika.Tika;
013import org.apache.tika.mime.MimeType;
014import org.apache.tika.mime.MimeTypeException;
015import org.apache.tika.mime.MimeTypes;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019public class MediaParser {
020  private static final Logger LOG = LoggerFactory.getLogger(MediaParser.class);
021  private static final Tika TIKA = new Tika();
022  private static final MimeTypes MIME_TYPES = MimeTypes.getDefaultMimeTypes();
023  private static final String HTML_TYPE = "text/html";
024  // mime types which we consider as html links instead of real media file uris
025  private static final Set<String> HTML_MIME_TYPES = ImmutableSet
026    .of("text/x-coldfusion", "text/x-php", "text/asp", "text/aspdotnet", "text/x-cgi", "text/x-jsp", "text/x-perl",
027      HTML_TYPE, MIME_TYPES.OCTET_STREAM);
028  private static MediaParser instance = null;
029
030  public static MediaParser getInstance() {
031    synchronized (MediaParser.class) {
032      if (instance == null) {
033        instance = new MediaParser();
034      }
035    }
036    return instance;
037  }
038
039  public MediaObject detectType(MediaObject mo) {
040    if (Strings.isNullOrEmpty(mo.getFormat())) {
041      // derive from URI
042      mo.setFormat(parseMimeType(mo.getIdentifier()));
043    }
044
045    // if MIME type is text/html make it a references link instead
046    if (HTML_TYPE.equalsIgnoreCase(mo.getFormat()) && mo.getIdentifier() != null) {
047      // make file URI the references link URL instead
048      mo.setReferences(mo.getIdentifier());
049      mo.setIdentifier(null);
050      mo.setFormat(null);
051    }
052
053    if (!Strings.isNullOrEmpty(mo.getFormat())) {
054      if (mo.getFormat().startsWith("image")) {
055        mo.setType(MediaType.StillImage);
056      } else if (mo.getFormat().startsWith("audio")) {
057        mo.setType(MediaType.Sound);
058      } else if (mo.getFormat().startsWith("video")) {
059        mo.setType(MediaType.MovingImage);
060      } else {
061        LOG.debug("Unsupported media format {}", mo.getFormat());
062      }
063    }
064    return mo;
065  }
066
067  /**
068   * Parses a mime type using apache tika which can handle the following:
069   * http://svn.apache.org/repos/asf/tika/trunk/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
070   */
071  public String parseMimeType(@Nullable String format) {
072    if (format != null) {
073      format = Strings.emptyToNull(format.trim().toLowerCase());
074    }
075
076    try {
077      MimeType mime = MIME_TYPES.getRegisteredMimeType(format);
078      if (mime != null) {
079        return mime.getName();
080      }
081
082    } catch (MimeTypeException e) {
083    }
084
085    // verify this is a reasonable mime type
086    return format == null || MimeType.isValid(format) ? format : null;
087  }
088
089  /**
090   * Parses a mime type using apache tika which can handle the following:
091   * http://svn.apache.org/repos/asf/tika/trunk/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
092   */
093  public String parseMimeType(@Nullable URI uri) {
094    if (uri != null) {
095      String mime = TIKA.detect(uri.toString());
096      if (mime != null && HTML_MIME_TYPES.contains(mime.toLowerCase())) {
097        // links without any suffix default to OCTET STREAM, see:
098        // http://dev.gbif.org/issues/browse/POR-2066
099        return HTML_TYPE;
100      }
101      return mime;
102    }
103    return null;
104  }
105}
