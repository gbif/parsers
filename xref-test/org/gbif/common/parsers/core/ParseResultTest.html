16  package org.gbif.common.parsers.core;
17  
18  
19  import java.util.Date;
20  
21  import org.junit.jupiter.api.Test;
22  
23  import static org.junit.jupiter.api.Assertions.assertEquals;
24  import static org.junit.jupiter.api.Assertions.assertNotNull;
25  import static org.junit.jupiter.api.Assertions.assertNull;
26  
27  public class ParseResultTest {
28  
29    @Test
30    public void testSuccess() {
31      // check with basic string payload
32      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")));
33      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getPayload());
34      assertEquals(String.class,
35        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getPayload().getClass());
36      assertEquals(ParseResult.CONFIDENCE.DEFINITE,
37        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getConfidence());
38      assertEquals(ParseResult.STATUS.SUCCESS,
39        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getStatus());
40      assertNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getError());
41  
42      // check generics with Date payload
43      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()));
44      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getPayload());
45      assertEquals(Date.class, ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getPayload().getClass());
46      assertEquals(ParseResult.CONFIDENCE.DEFINITE,
47        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getConfidence());
48      assertEquals(ParseResult.STATUS.SUCCESS,
49        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getStatus());
50      assertNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getError());
51  
52    }
53  
54    @Test
55    public void testFail() {
56      assertNotNull(ParseResult.fail());
57      assertEquals(ParseResult.STATUS.FAIL, ParseResult.fail().getStatus());
58      assertNull(ParseResult.fail().getConfidence());
59      assertNull(ParseResult.fail().getError());
60      assertNull(ParseResult.fail().getPayload());
61    }
62  
63    @Test
64    public void testUnknownError() {
65      assertNotNull(ParseResult.error());
66      assertEquals(ParseResult.STATUS.ERROR, ParseResult.error().getStatus());
67      assertNull(ParseResult.error().getConfidence());
68      assertNull(ParseResult.error().getError());
69      assertNull(ParseResult.error().getPayload());
70    }
71  
72    @Test
73    public void testError() {
74      assertNotNull(ParseResult.error(new RuntimeException("Bingo")));
75      assertEquals(ParseResult.STATUS.ERROR, ParseResult.error(new RuntimeException("Bingo")).getStatus());
76      assertNull(ParseResult.error(new RuntimeException("Bingo")).getConfidence());
77      assertNotNull(ParseResult.error(new RuntimeException("Bingo")).getError());
78      assertEquals(RuntimeException.class, ParseResult.error(new RuntimeException("Bingo")).getError().getClass());
79      assertEquals("Bingo", ParseResult.error(new RuntimeException("Bingo")).getError().getMessage());
80      assertNull(ParseResult.error(new RuntimeException("Bingo")).getPayload());
81    }
82  }
