39  public class MediaParser {
40    private static final Logger LOG = LoggerFactory.getLogger(MediaParser.class);
41    private static final Tika TIKA = new Tika();
42    private static final MimeTypes MIME_TYPES = MimeTypes.getDefaultMimeTypes();
43    private static final String HTML_TYPE = "text/html";
44    // MIME types which we consider as HTML links instead of real media file URIs
45    private static final Set<String> HTML_MIME_TYPES = Collections.unmodifiableSet(
46        new HashSet<>(
47            Arrays.asList("text/x-coldfusion", "text/x-php", "text/asp", "text/aspdotnet", "text/x-cgi",
48                "text/x-jsp", "text/x-perl", HTML_TYPE, MimeTypes.OCTET_STREAM)));

50    // List of exceptions, could be read from a file if it grows. URLs matching this return a media file.
51    private static final Map<Pattern, String> knownUrlPatterns;
52  
53    // Add missing alias types.
54    static {
55      knownUrlPatterns = new HashMap<>();
56      knownUrlPatterns.put(Pattern.compile("http://www\\.unimus\\.no/felles/bilder/web_hent_bilde\\.php\\?id=\\d+&type=jpeg"), "image/jpeg");
57      knownUrlPatterns.put(Pattern.compile("http://www\\.jacq\\.org/image\\.php\\?filename=\\d+&method=europeana"), "image/jpeg");
58      knownUrlPatterns.put(Pattern.compile("https://images\\.ala\\.org\\.au/image/proxyImageThumbnailLarge\\?imageId=[0-9a-f-]{36}"), "image/jpeg");
59      knownUrlPatterns.put(Pattern.compile("http://[a-zA-Z0-9-]+\\.wildlifemonitoring\\.ru/get_photo\\.php\\?id=\\d+"), "image/jpeg");
60      knownUrlPatterns.put(Pattern.compile("http://procyon\\.acadiau\\.ca/ecsmith/cgi-bin/image\\.cgi\\?[0-9A-Z]+,jpeg"), "image/jpeg");
61      knownUrlPatterns.put(Pattern.compile("http://www\\.biologie\\.uni-ulm\\.de/cgi-bin/perl/sound\\.pl\\?sid=T&objid=\\d+"), "audio/vnd.wave");
62      knownUrlPatterns.put(Pattern.compile("https://dofbasen\\.dk/sound_proxy\\.php\\?referer=gbif&mode=o&snd=[0-9_]+.mp3&raw=1"), "audio/mpeg");

80    public MediaObject detectType(MediaObject mo) {
81      if (StringUtils.isEmpty(mo.getFormat())) {
82        // derive from URI
83        mo.setFormat(parseMimeType(mo.getIdentifier()));
84      }
85  
86      // if MIME type is text/html make it a references link instead
87      if (HTML_TYPE.equalsIgnoreCase(mo.getFormat()) && mo.getIdentifier() != null) {
88        // make file URI the references link URL instead
89        mo.setReferences(mo.getIdentifier());
90        mo.setIdentifier(null);
91        mo.setFormat(null);
92      }
93  
94      if (StringUtils.isNotEmpty(mo.getFormat())) {
95        if (mo.getFormat().startsWith("image")) {
96          mo.setType(MediaType.StillImage);
97        } else if (mo.getFormat().startsWith("audio")) {
98          mo.setType(MediaType.Sound);
99        } else if (mo.getFormat().startsWith("video")) {
100         mo.setType(MediaType.MovingImage);
101       } else {
102         LOG.debug("Unsupported media format {}", mo.getFormat());
103       }
104     }
105     return mo;
106   }

108   /**
109    * Parses a MIME type using Apache Tika which can handle the following:
110    * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
111    * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
112    */
113   public String parseMimeType(@Nullable String format) {
114     if (format != null) {
115       format = StringUtils.trimToNull(format.trim().toLowerCase());
116     }
117     if (format == null) {
118       return null;
119     }
120 
121     try {
122       MimeType mime = MIME_TYPES.getRegisteredMimeType(format);
123       if (mime != null) {
124         return mime.getName();
125       }
126 
127     } catch (MimeTypeException e) {
128     }
129 
130     // Failed, but return the input if it's a reasonable MIME type
131     return MimeType.isValid(format) ? format : null;
132   }
133 
134   /**
135    * Parses a MIME type using Apache Tika which can handle the following:
136    * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
137    * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
138    */
139   public String parseMimeType(@Nullable URI uri) {
140     if (uri != null) {
141       String uriString = uri.toString();
142       String mime = TIKA.detect(uriString);
143       if (mime != null && HTML_MIME_TYPES.contains(mime.toLowerCase())) {
144         // We may have something like http://example.org/imageServer?img=test.jpg, so re-run the detection on the last
145         // part of the URL query string.
146         if (uri.getQuery() != null) {
147           mime = TIKA.detect(uri.getQuery());
148           if (mime != null && !HTML_MIME_TYPES.contains(mime.toLowerCase())) {
149             return mime;
150           }
151         }
152 
153         // First check the dictionary for known exceptions.
154         for (Map.Entry<Pattern, String> p : knownUrlPatterns.entrySet()) {
155           if (p.getKey().matcher(uriString).matches()) {
156             return p.getValue();
157           }
158         }
159 
160         // links without any suffix default to OCTET STREAM, see:
161         // http://dev.gbif.org/issues/browse/POR-2066
162         return HTML_TYPE;
163       }
164       return mime;
165     }
166     return null;
167   }
