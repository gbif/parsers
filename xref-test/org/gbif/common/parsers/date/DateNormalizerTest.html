17    /**
18     * This happens when integer are exported as float/double by a data publisher.
19     */
20    @Test
21    public void testFloatNormalization(){
22      DatePartsNormalizer.NormalizedYearMonthDay result = DatePartsNormalizer.normalize("1975.0", "2.0", "1.0");
23      assertEquals(new Integer(1), result.getDay());
24      assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
25      assertEquals(new Integer(1975), result.getYear());
26    }

28    @Test
29    public void testNormalizationWithSpaces(){
30      DatePartsNormalizer.NormalizedYearMonthDay result = DatePartsNormalizer.normalize("1975 ", " 2 ", " 1");
31      assertEquals(new Integer(1), result.getDay());
32      assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
33      assertEquals(new Integer(1975), result.getYear());

37    public void testMonthNameNormalization(){
38      assertEquals(Month.JANUARY.getValue(), DatePartsNormalizer.monthNameToNumerical("January").intValue());
39      assertEquals(Month.JANUARY.getValue(), DatePartsNormalizer.monthNameToNumerical("JANUARY").intValue());
40      assertEquals(Month.JANUARY.getValue(), DatePartsNormalizer.monthNameToNumerical("Jan").intValue());

42      assertEquals(Month.DECEMBER.getValue(), DatePartsNormalizer.monthNameToNumerical("DEC").intValue());
43      assertEquals(Month.APRIL.getValue(), DatePartsNormalizer.monthNameToNumerical("April").intValue());

51    @Test
52    public void testDiscardedDateParts(){
53  
54      DatePartsNormalizer.NormalizedYearMonthDay result = DatePartsNormalizer.normalize("a", "5", "3");
55      assertTrue(result.yDiscarded());
56      assertNull(result.getYear());
57      assertNotNull(result.getMonth());
58  
59      result = DatePartsNormalizer.normalize("2000", "a", "3");
60      assertTrue(result.mDiscarded());
61      assertNull(result.getMonth());
62      assertNotNull(result.getYear());
63  
64      result = DatePartsNormalizer.normalize("2000", "5", "a");
65      assertTrue(result.dDiscarded());
66      assertNull(result.getDay());
67      assertNotNull(result.getYear());
68  
69      // empty String, null and \\N should NOT be considered as "discarded"
70      // they simply represent "not provided"
71      result = DatePartsNormalizer.normalize("2000", "5", "");
72      assertFalse(result.dDiscarded());
73      assertNull(result.getDay());
74  
75      result = DatePartsNormalizer.normalize("2000", "5", null);
76      assertFalse(result.dDiscarded());
77      assertNull(result.getDay());
78  
79      result = DatePartsNormalizer.normalize("2000", "5", "\\N");
80      assertFalse(result.dDiscarded());
81      assertNull(result.getDay());
82    }
