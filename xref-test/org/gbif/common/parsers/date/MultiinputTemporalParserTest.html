32    @Test
33    public void testIsValidDate() {
34      assertTrue(MultiinputTemporalParser.isValidDate(Year.of(2005)));
35      assertTrue(MultiinputTemporalParser.isValidDate(YearMonth.of(2005, 1)));
36      assertTrue(MultiinputTemporalParser.isValidDate(LocalDate.of(2005, 1, 1)));
37      assertTrue(MultiinputTemporalParser.isValidDate(LocalDateTime.of(2005, 1, 1, 2, 3, 4)));
38      assertTrue(MultiinputTemporalParser.isValidDate(LocalDate.now()));
39      assertTrue(MultiinputTemporalParser.isValidDate(LocalDateTime.now().plus(23, ChronoUnit.HOURS)));
40  
41      // Dates out of bounds
42      assertFalse(MultiinputTemporalParser.isValidDate(YearMonth.of(1499, 12)));
43  
44      // we tolerate an offset of 1 day
45      assertFalse(MultiinputTemporalParser.isValidDate(LocalDate.now().plusDays(2)));
46    }

48    @Test
49    public void testParseRecordedDate() {
50      MultiinputTemporalParser temporalParser = MultiinputTemporalParser.create();
51      OccurrenceParseResult<TemporalAccessor> result;
52  
53      result = temporalParser.parseRecordedDate("2005", "1", "1", "2005-01-01");
54      assertResult(2005, 1, 1, result);
55      assertEquals(0, result.getIssues().size());
56  
57      // ensure that eventDate with more precision will not record an issue and the one with most
58      // precision will be returned
59      result = temporalParser.parseRecordedDate("1996", "1", "26", "1996-01-26T01:00Z");
60      assertEquals(
61          ZonedDateTime.of(LocalDateTime.of(1996, 1, 26, 1, 0), ZoneId.of("Z")), result.getPayload());
62      assertEquals(0, result.getIssues().size());
63  
64      // if dates contradict, return the parts that agree
65      result = temporalParser.parseRecordedDate("2005", "1", "2", "2005-01-05");
66      assertResult(2005, 1, result);
67      assertEquals(1, result.getIssues().size());
68      assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
69    }

185     // providing 0 will cause a RECORDED_DATE_MISMATCH since 0 could be null but also January
186     OccurrenceParseResult<TemporalAccessor> result =
187         MultiinputTemporalParser.create().parseRecordedDate("1984", "0", "0", "1984");

366   @Test
367   public void testNow() {
368 
369     MultiinputTemporalParser temporalParser = MultiinputTemporalParser.create();

371     // Makes sure the static content is loaded
372     OccurrenceParseResult<TemporalAccessor> result =
373         temporalParser.parseRecordedDate(
374             null, null, null, DateFormatUtils.ISO_DATETIME_FORMAT.format(Calendar.getInstance()));

408   private void assertInts(Integer expected, Integer x) {

416   /**
417    * Utility method to assert a ParseResult when a LocalDate is expected. This method should not be
418    * used to test expected null results.
419    */
420   private void assertResult(Integer y, Integer m, Integer d, ParseResult<TemporalAccessor> result) {
421     // sanity checks
422     assertNotNull(result);
423 
424     LocalDate localDate = result.getPayload().query(LocalDate::from);
425     assertInts(y, localDate.getYear());
426     assertInts(m, localDate.getMonthValue());
427     assertInts(d, localDate.getDayOfMonth());
428 
429     assertEquals(LocalDate.of(y, m, d), result.getPayload());
430   }

436   /**
437    * Utility method to assert a ParseResult when a YearMonth is expected. This method should not be
438    * used to test expected null results.
439    */
440   private void assertResult(Integer y, Integer m, ParseResult<TemporalAccessor> result) {
441     // sanity checks
442     assertNotNull(result);
443 
444     YearMonth yearMonthDate = result.getPayload().query(YearMonth::from);
445     assertInts(y, yearMonthDate.getYear());
446     assertInts(m, yearMonthDate.getMonthValue());
447 
448     assertEquals(YearMonth.of(y, m), result.getPayload());
449   }

451   private void assertResult(Integer y, ParseResult<TemporalAccessor> result) {
452     // sanity checks
453     assertNotNull(result);
454 
455     Year yearDate = result.getPayload().query(Year::from);
456     assertInts(y, yearDate.getValue());
457 
458     assertEquals(Year.of(y), result.getPayload());
459   }

461   private void assertNullPayload(
462       OccurrenceParseResult<TemporalAccessor> result, OccurrenceIssue expectedIssue) {
463     assertNotNull(result);
464     assertFalse(result.isSuccessful());
465     assertNull(result.getPayload());
466 
467     if (expectedIssue != null) {
468       assertTrue(result.getIssues().contains(expectedIssue));
469     }
