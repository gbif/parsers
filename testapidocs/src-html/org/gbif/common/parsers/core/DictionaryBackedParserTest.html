001package org.gbif.common.parsers.core;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertNotNull;
010
011public class DictionaryBackedParserTest {
012
013  @Test
014  public void testParse() {
015    List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
016    source.add(new KeyValue<String, Integer>("Tim", 32));
017    source.add(new KeyValue<String, Integer>("Markus", 38));
018    source.add(new KeyValue<String, Integer>("Jose", 28));
019    DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(false);
020    dbp.init(source.iterator());
021
022    assertParsed(dbp, "Tim", 32);
023    assertParsed(dbp, "TIM", 32);
024    assertParsed(dbp, "tIm", 32);
025    assertParsed(dbp, "Markus", 38);
026    assertParsed(dbp, "MarKUS", 38);
027    assertParsed(dbp, "Jose", 28);
028    assertEquals(ParseResult.STATUS.FAIL, dbp.parse("Lars").getStatus());
029  }
030
031  private void assertParsed(DictionaryBackedParser<Integer> dbp, String input, Integer payload) {
032    assertNotNull(dbp.parse(input));
033    assertEquals(ParseResult.STATUS.SUCCESS, dbp.parse(input).getStatus());
034    assertEquals(ParseResult.CONFIDENCE.DEFINITE, dbp.parse(input).getConfidence());
035    assertEquals(payload, dbp.parse(input).getPayload());
036  }
037
038}
