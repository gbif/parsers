001package org.gbif.common.parsers.core;
002
003import org.gbif.common.parsers.core.ParseResult;
004
005import java.util.Date;
006
007import org.junit.jupiter.api.Test;
008
009import static org.junit.jupiter.api.Assertions.assertEquals;
010import static org.junit.jupiter.api.Assertions.assertNotNull;
011import static org.junit.jupiter.api.Assertions.assertNull;
012
013public class ParseResultTest {
014
015  @Test
016  public void testSuccess() {
017    // check with basic string payload
018    assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")));
019    assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getPayload());
020    assertEquals(String.class,
021      ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getPayload().getClass());
022    assertEquals(ParseResult.CONFIDENCE.DEFINITE,
023      ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getConfidence());
024    assertEquals(ParseResult.STATUS.SUCCESS,
025      ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getStatus());
026    assertNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getError());
027
028    // check generics with Date payload
029    assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()));
030    assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getPayload());
031    assertEquals(Date.class, ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getPayload().getClass());
032    assertEquals(ParseResult.CONFIDENCE.DEFINITE,
033      ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getConfidence());
034    assertEquals(ParseResult.STATUS.SUCCESS,
035      ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getStatus());
036    assertNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getError());
037
038  }
039
040  @Test
041  public void testFail() {
042    assertNotNull(ParseResult.fail());
043    assertEquals(ParseResult.STATUS.FAIL, ParseResult.fail().getStatus());
044    assertNull(ParseResult.fail().getConfidence());
045    assertNull(ParseResult.fail().getError());
046    assertNull(ParseResult.fail().getPayload());
047  }
048
049  @Test
050  public void testUnknownError() {
051    assertNotNull(ParseResult.error());
052    assertEquals(ParseResult.STATUS.ERROR, ParseResult.error().getStatus());
053    assertNull(ParseResult.error().getConfidence());
054    assertNull(ParseResult.error().getError());
055    assertNull(ParseResult.error().getPayload());
056  }
057
058  @Test
059  public void testError() {
060    assertNotNull(ParseResult.error(new RuntimeException("Bingo")));
061    assertEquals(ParseResult.STATUS.ERROR, ParseResult.error(new RuntimeException("Bingo")).getStatus());
062    assertNull(ParseResult.error(new RuntimeException("Bingo")).getConfidence());
063    assertNotNull(ParseResult.error(new RuntimeException("Bingo")).getError());
064    assertEquals(RuntimeException.class, ParseResult.error(new RuntimeException("Bingo")).getError().getClass());
065    assertEquals("Bingo", ParseResult.error(new RuntimeException("Bingo")).getError().getMessage());
066    assertNull(ParseResult.error(new RuntimeException("Bingo")).getPayload());
067  }
068}
