53    /**
54     * Parse all unique datum values found and make sure parsing doesn't get worse.
55     * If the test file is updated, values here need to be adjusted!
56     */
57    @Test
58    public void testOccurrenceValues() throws IOException {
59      final int CURRENT_TESTS_SUCCESSFUL = 14;
60  
61      int failed = 0;
62      int success = 0;
63      Set<Integer> codes = Sets.newHashSet();
64  
65      BufferedReader r = new BufferedReader(new InputStreamReader(
66            getClass().getResourceAsStream("/parse/test_datum.txt"), Charsets.UTF_8));
67      String line;
68      while ((line=r.readLine()) != null) {
69        ParseResult<Integer> parsed = parser.parse(line);
70        if (parsed.isSuccessful()) {
71          success++;
72          codes.add(parsed.getPayload());
73        } else {
74          System.out.println("Failed: " + line);
75          failed++;
76        }
77      }
78  
79      System.out.println(failed + " failed parse results");
80      System.out.println(success + " successful parse results");
81      System.out.println(codes.size() + " distinct datums parsed");
82      assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
83    }
