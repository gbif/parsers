23    @Test
24    public void singleDateRangeTest() {
25      TemporalRangeParser trp =
26          TemporalRangeParser.builder()
27              .temporalParser(MultiinputTemporalParser.create(Collections.singletonList(DMY)))
28              .create();
29  
30      OccurrenceParseResult<IsoDateInterval> result = trp.parse("1930/1929");
31      assertEquals("1929", result.getPayload().getFrom().toString());
32      assertEquals("1930", result.getPayload().getTo().toString());
33  
34      result = trp.parse("1930/1931");
35      assertEquals("1930", result.getPayload().getFrom().toString());
36      assertEquals("1931", result.getPayload().getTo().toString());
37  
38      result = trp.parse("1930/1930");
39      assertEquals("1930", result.getPayload().getFrom().toString());
40      assertEquals("1930", result.getPayload().getTo().toString());
41  
42      result = trp.parse("1930-01");
43      assertEquals("1930-01", result.getPayload().getFrom().toString());
44      assertEquals("1930-01", result.getPayload().getTo().toString());
45  
46      // Does not support
47      result = trp.parse("01/1930");
48      assertFalse(result.isSuccessful());
49      assertEquals(1, result.getIssues().size());
50      assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
51  
52      result = trp.parse("1930-01-02/1930-02-01");
53      assertEquals("1930-01-02", result.getPayload().getFrom().toString());
54      assertEquals("1930-02-01", result.getPayload().getTo().toString());
55  
56      result = trp.parse("02/01/1930");
57      assertEquals("1930-01-02", result.getPayload().getFrom().toString());
58      assertEquals("1930-01-02", result.getPayload().getTo().toString());
59  
60      result = trp.parse("1930-01-02/02-01");
61      assertEquals("1930-01-02", result.getPayload().getFrom().toString());
62      assertEquals("1930-02-01", result.getPayload().getTo().toString());
63  
64      result = trp.parse("1930-01-02/15");
65      assertEquals("1930-01-02", result.getPayload().getFrom().toString());
66      assertEquals("1930-01-15", result.getPayload().getTo().toString());
67  
68      result = trp.parse("1930-01-02/1930-01-02");
69      assertEquals("1930-01-02", result.getPayload().getFrom().toString());
70      assertEquals("1930-01-02", result.getPayload().getTo().toString());

73    @Test
74    public void ambiguousDateTest() {
75      // Use a default parser, which cannot understand like: 01/02/2020
76      TemporalRangeParser trp = TemporalRangeParser.builder().create();
77  
78      OccurrenceParseResult<IsoDateInterval> result = trp.parse("01/02/1999");
79      assertFalse(result.getIssues().isEmpty());
80      assertEquals(1, result.getIssues().size());
81      assertTrue(result.getIssues().contains(OccurrenceIssue.RECORDED_DATE_INVALID));
82  
83      // Dots are reliably European order
84      result = trp.parse("01.02.1999");
85      assertEquals("1999-02-01", result.getPayload().getFrom().toString());
86      assertEquals("1999-02-01", result.getPayload().getTo().toString());
87  
88      result = trp.parse("1999-01-02");
89      assertTrue(result.getIssues().isEmpty());
90      assertEquals("1999-01-02", result.getPayload().getFrom().toString());
91      assertEquals("1999-01-02", result.getPayload().getTo().toString());
92  
93      // Use a DMY_FORMATS parser
94      trp =
95          TemporalRangeParser.builder()
96              .temporalParser(MultiinputTemporalParser.create(Arrays.asList(DMY_FORMATS)))
97              .create();
98  
99      result = trp.parse("01/02/1999");
100     assertTrue(result.getIssues().isEmpty());
101     assertEquals("1999-02-01", result.getPayload().getFrom().toString());
102     assertEquals("1999-02-01", result.getPayload().getTo().toString());
103 
104     result = trp.parse("1999-01-02");
105     assertTrue(result.getIssues().isEmpty());
106     assertEquals("1999-01-02", result.getPayload().getFrom().toString());
107     assertEquals("1999-01-02", result.getPayload().getTo().toString());
108   }

252   @Test
253   public void testEqualResolution() {
254     TemporalRangeParser trp = TemporalRangeParser.builder().create();
255 
256     // Beginning of range will be parsed as 2000, but end of range as 2000-06.  We must have 2000/2000, not 2000/2000-06.
257     OccurrenceParseResult<IsoDateInterval> result = trp.parse("2000", "06", "06", "2000-04-01T01:02:03/2000-06-01T22:23:24", null, null);
258     assertEquals("2000", result.getPayload().getFrom().toString());
259     assertEquals("2000", result.getPayload().getTo().toString());
260     assertEquals(1, result.getIssues().size());
261     assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
262 
263     // Event date has different resolutions on each side of the range
264     result = trp.parse(null, null, null, "2009-09/2009-10-05", null, null);
265     assertEquals("2009-09", result.getPayload().getFrom().toString());
266     assertEquals("2009-10", result.getPayload().getTo().toString());
267     // TODO: A small improvement would assign INVALID rather than MISMATCH for this.
268     assertEquals(1, result.getIssues().size());
269     assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
270 
271     // Event date has different resolutions on each side of the range
272     result = trp.parse(null, null, null, "2009-09-18/2009-10-05T17:36+0200", null, null);
273     assertEquals("2009-09-18", result.getPayload().getFrom().toString());
274     assertEquals("2009-10-05", result.getPayload().getTo().toString());
275     assertEquals(1, result.getIssues().size());
276     assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
277 
278     // Event date has different resolutions on each side of the range
279     result = trp.parse(null, null, null, "2019-10-07T13:42:25Z/2019-10-07", null, null);
280     assertEquals("2019-10-07", result.getPayload().getFrom().toString());
281     assertEquals("2019-10-07", result.getPayload().getTo().toString());
282     assertEquals(1, result.getIssues().size());
283     assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
284 
285     // Another example
286     result = trp.parse(null, null, null, "1983-12-31 23:59:59/1983", null, null);
287     assertEquals("1983", result.getPayload().getFrom().toString());
288     assertEquals("1983", result.getPayload().getTo().toString());
289     assertEquals(1, result.getIssues().size());
290     assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
291 
292     // One part of the range has a time zone
293     result = trp.parse(null, null, null, "2006-03-08T2005/2006-03-08T2015+12", null, null);
294     assertEquals("2006-03-08T20:05", result.getPayload().getFrom().toString());
295     assertEquals("2006-03-08T20:15", result.getPayload().getTo().toString());
296     assertEquals(0, result.getIssues().size());
297 
298     // One part of the range has a UTC time zone
299     result = trp.parse(null, null, null, "2022-03-01T06:26:14/2022-03-01T06:47:58Z", null, null);
300     assertEquals("2022-03-01T06:26:14", result.getPayload().getFrom().toString());
301     assertEquals("2022-03-01T06:47:58", result.getPayload().getTo().toString());
302     assertEquals(0, result.getIssues().size());
303   }

355     assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
356 
357     // Inverted with a day
358     result = trp.parse("1985", "05", "19", "1985-05-20/19", null, null);
359     assertEquals("1985-05-19", result.getPayload().getFrom().toString());
360     assertEquals("1985-05-20", result.getPayload().getTo().toString());
361     assertEquals(1, result.getIssues().size());
362     assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
363 
364     // Inverted date range with time
365     result = trp.parse("2022", "07", "04", "2022-07-04T15:44/2022-07-04T14:54", null, null);
366     assertEquals("2022-07-04T14:54", result.getPayload().getFrom().toString());
367     assertEquals("2022-07-04T15:44", result.getPayload().getTo().toString());
368     assertEquals(1, result.getIssues().size());
369     assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
370 
371     // Inverted date range with zoned time
