001package org.gbif.common.parsers.core;
002
003import java.util.HashMap;
004import java.util.Iterator;
005import java.util.Map;
006
007import org.apache.commons.lang3.StringUtils;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011/**
012 * A simple parser that will initialise with source data, and will
013 * use that as a lookup to replace values. This was written with
014 * basis of record lookup and country names in mind.
015 * Future improvements to this implementation might call a
016 * dictionary web service for example, to achieve the same,
017 * but allow the abstraction of the dictionary management to a
018 * better project (separation of concerns)
019 */
020public class DictionaryBackedParser<V> implements Parsable<V> {
021  protected final Logger log = LoggerFactory.getLogger(getClass());
022  private final Map<String, V> dictionary = new HashMap<String, V>();
023  private final boolean caseSensitive;
024
025  /**
026   * @param caseSensitive If the dictionary should be case sensitive (only applicable to String keys)
027   */
028  public DictionaryBackedParser(boolean caseSensitive) {
029    this.caseSensitive = caseSensitive;
030  }
031
032  /**
033   * @param source To build the dictionary from
034   */
035  @SuppressWarnings("unchecked")
036  public void init(Iterator<KeyValue<String, V>> source) {
037    while (source.hasNext()) {
038      KeyValue<String, V> kvp = source.next();
039      add(kvp.getKey(), kvp.getValue());
040    }
041  }
042
043  final protected void add(String key, V value) {
044    if (!StringUtils.isBlank(key)) {
045      String normedKey = normalize(key);
046      if (StringUtils.isNotEmpty(normedKey)) {
047        V existingValue = dictionary.get(normedKey);
048        if (existingValue == null) {
049          dictionary.put(normedKey, value);
050        } else if (!existingValue.equals(value)) {
051          log.warn("Ignoring mapping {}â†’{} as {} is already mapped to {}", key, value, key, existingValue);
052        }
053      }
054    }
055  }
056
057  /**
058   * Normalisation of a value used both by adding to the internal dictionary and parsing values.
059   * The default does trim and uppercase the value for Strings, but leaves other types unaltered.
060   * Override this method to provide specific normalisations for parsers.
061   *
062   * @param value the value to be normalised.
063   *
064   * @return the normalised value
065   */
066  protected String normalize(String value) {
067    if (value != null) {
068      if (!caseSensitive) {
069        return StringUtils.trimToNull(value.toUpperCase());
070      } else {
071        return StringUtils.trimToNull(value);
072      }
073    }
074    return value;
075  }
076
077  /**
078   * Tries to parse the input data according to its backing dictionary.
079   * If no entry in the dictionary can be found the result will be {@link ParseResult.STATUS FAIL} otherwise the
080   * result will be a {@link org.gbif.common.parsers.core.ParseResult.CONFIDENCE DEFINITE} {@link ParseResult.STATUS
081   * SUCCESS}.
082   *
083   * @param input To lookup in the dictionary
084   *
085   * @return the replacement from the dictionary
086   */
087  @Override
088  public ParseResult<V> parse(String input) {
089    String normed = normalize(input);
090    V value = dictionary.get(normed);
091    if (value == null) {
092      return ParseResult.fail();
093    } else {
094      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, value);
095    }
096  }
097}
