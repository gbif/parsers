001package org.gbif.common.parsers;
002
003import org.gbif.api.model.common.MediaObject;
004import org.gbif.api.vocabulary.MediaType;
005
006import java.net.URI;
007
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011import static org.junit.Assert.assertNull;
012
013public class MediaParserTest {
014  private final MediaParser parser = MediaParser.getInstance();
015
016  @Test
017  public void testDetectType() throws Exception {
018    assertEquals(MediaType.StillImage, parser.detectType(buildMO("image/jp2", "abies_alba.jp2")).getType());
019    assertEquals(MediaType.StillImage, parser.detectType(buildMO("image/jpeg", null)).getType());
020    assertEquals(MediaType.StillImage, parser.detectType(buildMO("image/jpg", null)).getType());
021    assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.jp2")).getType());
022    assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.jpg")).getType());
023    assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.JPG")).getType());
024    assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.JPeg")).getType());
025    assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.TIFF")).getType());
026    assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "http://imagens3.jbrj.gov.br/fsi/server?type=image&source=rb/1/37/81/25/01378125.jpg")).getType());
027    assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "http://rrrcn.wildlifemonitoring.ru/get_photo.php?id=119797")).getType());
028
029    assertEquals(MediaType.Sound, parser.detectType(buildMO(null, "abies_alba.mp3")).getType());
030    assertEquals(MediaType.Sound, parser.detectType(buildMO(null, "abies_alba.flac")).getType());
031    assertEquals(MediaType.Sound, parser.detectType(buildMO(null, "abies_alba.ogg")).getType());
032  }
033
034  @Test
035  public void testParseMimeType() throws Exception {
036    assertNull(parser.parseMimeType((String) null));
037    assertEquals("image/jp2", parser.parseMimeType(" image/jp2\t"));
038    assertEquals("image/jpg", parser.parseMimeType("image/jpg"));
039    assertEquals("image/jpeg", parser.parseMimeType("image/jpeg"));
040    assertEquals("image/jpg", parser.parseMimeType("image/JPG"));
041    assertNull(parser.parseMimeType("JPG"));
042    assertNull(parser.parseMimeType("gif"));
043    assertNull(parser.parseMimeType("tiff"));
044
045    assertNull(parser.parseMimeType((URI) null));
046    assertEquals("image/jp2", parser.parseMimeType(URI.create("abies_alba.jp2")));
047    assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.jpg")));
048    assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.jpeg")));
049    assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.JPG")));
050    assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.JPG")));
051    assertEquals("image/gif", parser.parseMimeType(URI.create("abies_alba.gif")));
052    assertEquals("image/tiff", parser.parseMimeType(URI.create("abies_alba.tiff")));
053    assertEquals("audio/mpeg", parser.parseMimeType(URI.create("abies_alba.mp3")));
054    assertEquals("video/mpeg", parser.parseMimeType(URI.create("abies_alba.mpg")));
055
056    // we default to an html link
057    assertEquals("text/html", parser.parseMimeType(URI.create("http://www.gbif.org/image?id=12")));
058    assertEquals("text/html", parser.parseMimeType(URI.create("http://arctos.database.museum/MediaSearch.cfm?action=search")));
059    assertEquals("text/html", parser.parseMimeType(URI.create("http://arctos.database.museum/MediaSearch.php?action=search")));
060    assertEquals("text/html", parser.parseMimeType(URI.create("http://arctos.database.museum/MediaSearch.pl?action=search")));
061
062    // Audio.
063    assertNull(parser.parseMimeType("mp3"));
064    assertNull(parser.parseMimeType("mpg"));
065
066    assertEquals("audio/ogg", parser.parseMimeType("audio/ogg"));
067    assertEquals("application/ogg", parser.parseMimeType("application/ogg"));
068    assertEquals("audio/webm", parser.parseMimeType("audio/webm"));
069    assertEquals("audio/mp4", parser.parseMimeType("audio/mp4"));
070    assertEquals("audio/vnd.wave", parser.parseMimeType("audio/wav"));
071
072    assertEquals("audio/mpeg", parser.parseMimeType("audio/mp3"));
073    assertEquals("audio/mpeg", parser.parseMimeType("audio/mpeg3"));
074
075    // Known URL patterns
076    assertEquals("image/jpeg", parser.parseMimeType(URI.create("http://www.unimus.no/felles/bilder/web_hent_bilde.php?id=13080366&type=jpeg")));
077    assertEquals("image/jpeg", parser.parseMimeType(URI.create("http://www.jacq.org/image.php?filename=99618&method=europeana")));
078    assertEquals("image/jpeg", parser.parseMimeType(URI.create("https://images.ala.org.au/image/proxyImageThumbnailLarge?imageId=ffff9990-c4a7-4431-803f-3c5f14bd94ba")));
079    assertEquals("image/jpeg", parser.parseMimeType(URI.create("http://russiabirds.wildlifemonitoring.ru/get_photo.php?id=99981")));
080    assertEquals("image/jpeg", parser.parseMimeType(URI.create("http://rrrcn.wildlifemonitoring.ru/get_photo.php?id=99999")));
081    assertEquals("image/jpeg", parser.parseMimeType(URI.create("http://procyon.acadiau.ca/ecsmith/cgi-bin/image.cgi?NSPM044956P,jpeg")));
082    assertEquals("audio/vnd.wave", parser.parseMimeType(URI.create("http://www.biologie.uni-ulm.de/cgi-bin/perl/sound.pl?sid=T&objid=99999")));
083    assertEquals("audio/mpeg", parser.parseMimeType(URI.create("https://dofbasen.dk/sound_proxy.php?referer=gbif&mode=o&snd=99881321.mp3&raw=1")));
084  }
085
086  @Test
087  public void testParseHtmlMediaLink() throws Exception {
088    MediaObject mo = buildMO(null, "http://www.gbif.org/image?id=12");
089    parser.detectType(mo);
090    assertNull(mo.getFormat());
091    assertNull(mo.getIdentifier());
092    assertEquals("http://www.gbif.org/image?id=12", mo.getReferences().toString());
093    assertNull(mo.getType());
094  }
095
096  @Test
097  public void testParseKnownImageMediaLink() throws Exception {
098    MediaObject mo = buildMO(null, "http://rrrcn.wildlifemonitoring.ru/get_photo.php?id=119797");
099    parser.detectType(mo);
100    assertEquals(MediaType.StillImage, mo.getType());
101    assertEquals("http://rrrcn.wildlifemonitoring.ru/get_photo.php?id=119797", mo.getIdentifier().toString());
102    assertNull(mo.getReferences());
103    assertEquals("image/jpeg", mo.getFormat());
104  }
105
106  private MediaObject buildMO(String format, String uri) {
107    MediaObject mo = new MediaObject();
108
109    mo.setType(null);
110    mo.setFormat(format);
111    mo.setIdentifier(uri == null ? null : URI.create(uri));
112
113    return mo;
114  }
115}
