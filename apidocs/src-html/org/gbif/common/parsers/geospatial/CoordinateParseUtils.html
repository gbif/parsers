019/**
020 * Utilities for assisting in the parsing of latitude and longitude strings into Decimals.
021 */
022public class CoordinateParseUtils {
023  private final static String DMS = "\\s*(\\d{1,3})\\s*(?:°|d|º| |g|o)"  // The degrees
024                                  + "\\s*([0-6]?\\d)\\s*(?:'|m| |´|’|′)" // The minutes
025                                  + "\\s*(?:"                            // Non-capturing group
026                                  + "([0-6]?\\d(?:[,.]\\d+)?)"           // Seconds and optional decimal
027                                  + "\\s*(?:\"|''|s|´´|″)?"
028                                  + ")?\\s*";
029  private final static String DM = "\\s*(\\d{1,3})\\s*(?:°|d|º| |g|o)" // The degrees
030                                 + "\\s*(?:"                           // Non-capturing group
031                                 + "([0-6]?\\d(?:[,.]\\d+)?)"          // Minutes and optional decimal
032                                 + "\\s*(?:'|m| |´|’|′)?"
033                                 + ")?\\s*";
034  private final static String D = "\\s*(\\d{1,3}(?:[,.]\\d+)?)\\s*(?:°|d|º| |g|o|)\\s*"; // The degrees and optional decimal
035  private final static Pattern DMS_SINGLE = Pattern.compile("^" + DMS + "$", Pattern.CASE_INSENSITIVE);
036  private final static Pattern DM_SINGLE = Pattern.compile("^" + DM + "$", Pattern.CASE_INSENSITIVE);
037  private final static Pattern D_SINGLE = Pattern.compile("^" + D + "$", Pattern.CASE_INSENSITIVE);
038  private final static Pattern DMS_COORD = Pattern.compile("^" + DMS + "([NSEOW])" + "[ ,;/]?" + DMS + "([NSEOW])$", Pattern.CASE_INSENSITIVE);
039  //private final static Pattern DM_COORD = Pattern.compile("^" + DM + "([NSEOW])" + "[ ,;/]?" + DM + "([NSEOW])$", Pattern.CASE_INSENSITIVE);
040  //private final static Pattern D_COORD = Pattern.compile("^" + D + "([NSEOW])" + "[ ,;/]?" + D + "([NSEOW])$", Pattern.CASE_INSENSITIVE);
041  private final static String POSITIVE = "NEO";
042  private CoordinateParseUtils() {
043    throw new UnsupportedOperationException("Can't initialize class");
044  }

048  /**
049   * This parses string representations of latitude and longitude values. It tries its best to interpret the values and
050   * indicates any problems in its result as {@link org.gbif.api.vocabulary.OccurrenceIssue}.
051   * When the {@link ParseResult.STATUS} is FAIL the payload will be null and one or more issues should be set
052   * in {@link org.gbif.common.parsers.core.OccurrenceParseResult#getIssues()}.
053   *
054   * Coordinate precision will be 6 decimals at most, any more precise values will be rounded.
055   *
056   * Supported standard formats are the following, with dots or optionally a comma as the decimal marker, and variations
057   * on the units also accepted e.g. °, d, º, g, o.
058   * <ul>
059   *   <li>43.63871944444445</li>
060   *   <li>N43°38'19.39"</li>
061   *   <li>43°38'19.39"N</li>
062   *   <li>43°38.3232'N</li>
063   *   <li>43d 38m 19.39s N</li>
064   *   <li>43 38 19.39</li>
065   *   <li>433819N</li>
066   * </ul>
067   *
068   * @param latitude  The decimal latitude
069   * @param longitude The decimal longitude
070   *
071   * @return The parse result
072   */
073  public static OccurrenceParseResult<LatLng> parseLatLng(final String latitude, final String longitude) {
074    if (Strings.isNullOrEmpty(latitude) || Strings.isNullOrEmpty(longitude)) {
075      return OccurrenceParseResult.fail();
076    }
077    Double lat = NumberParser.parseDouble(latitude);
078    Double lng = NumberParser.parseDouble(longitude);
079    if (lat == null || lng == null) {
080      // try degree minute seconds
081      try {
082        lat = parseDMS(latitude, true);
083        lng = parseDMS(longitude, false);
084      } catch (IllegalArgumentException e) {
085        return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
086      }
087    }
088
089    return validateAndRound(lat, lng);
090  }
091
092  private static boolean inRange(double lat, double lon) {
093    if (Double.compare(lat, 90) <= 0 && Double.compare(lat, -90) >= 0 && Double.compare(lon, 180) <= 0 && Double.compare(lon, -180) >= 0) {
094      return true;
095    }
096    return false;
097  }
098
099  private static boolean isLat(String direction) {
100    if ("NS".contains(direction.toUpperCase())) {
101      return true;
102    }
103    return false;
104  }
105
106  private static int coordSign(String direction) {
107    return POSITIVE.contains(direction.toUpperCase()) ? 1 : -1;
108  }
109
110  // 02° 49' 52" N      131° 47' 03" E
111  public static OccurrenceParseResult<LatLng> parseVerbatimCoordinates(final String coordinates) {
112    if (Strings.isNullOrEmpty(coordinates)) {
113      return OccurrenceParseResult.fail();
114    }
115    Matcher m = DMS_COORD.matcher(coordinates);
116    if (m.find()) {
117      final String dir1 = m.group(4);
118      final String dir2 = m.group(8);
119      // first parse coords regardless whether they are lat or lon
120      double c1 = coordFromMatcher(m, 1,2,3, dir1);
121      double c2 = coordFromMatcher(m, 5,6,7, dir2);
122      // now see what order the coords are in:
123      if (isLat(dir1) && !isLat(dir2)) {
124        return validateAndRound(c1, c2);
125
126      } else if (!isLat(dir1) && isLat(dir2)) {
127        return validateAndRound(c2, c1);
128
129      } else {
130        return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
131      }
132
133    } else if(coordinates.length() > 4) {
134      // try to split and then use lat/lon parsing
135      for (final char delim : ",;/ ".toCharArray()) {
136        int cnt = StringUtils.countMatches(coordinates, String.valueOf(delim));
137        if (cnt == 1) {
138          String[] latlon = StringUtils.split(coordinates, delim);
139          if (latlon.length == 2) {
140            return parseLatLng(latlon[0], latlon[1]);
141          }
142        }
143      }
144    }
145    return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
146  }
147
148  private static OccurrenceParseResult<LatLng> validateAndRound(double lat, double lon) {
149    // collecting issues for result
150    Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
151
152    // round to 6 decimals
153    final double latOrig = lat;
154    final double lngOrig = lon;
155    lat = roundTo6decimals(lat);
156    lon = roundTo6decimals(lon);
157    if (Double.compare(lat, latOrig) != 0 || Double.compare(lon, lngOrig) != 0) {
158      issues.add(OccurrenceIssue.COORDINATE_ROUNDED);
159    }
160
161    // 0,0 is too suspicious
162    if (Double.compare(lat, 0) == 0 && Double.compare(lon, 0) == 0) {
163      issues.add(OccurrenceIssue.ZERO_COORDINATE);
164      return OccurrenceParseResult.success(ParseResult.CONFIDENCE.POSSIBLE, new LatLng(0, 0), issues);
165    }
166
167    // if everything falls in range
168    if (inRange(lat, lon)) {
169      return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new LatLng(lat, lon), issues);
170    }
171
172    // if lat is out of range, but in range of the lng, assume swapped coordinates.
173    // note that should we desire to trust the following records, we would need to clear the flag for the records to
174    // appear in search results and maps etc. however, this is logic decision, that goes above the capabilities of this method
175    if (Double.compare(lat, 90) > 0 || Double.compare(lat, -90) < 0) {
176      // try and swap
177      if (inRange(lon, lat)) {
178        issues.add(OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
179        return OccurrenceParseResult.success(ParseResult.CONFIDENCE.PROBABLE, new LatLng(lon, lat), issues);
180      }
181    }
182
183    // then something is out of range
184    issues.add(OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
185    return OccurrenceParseResult.fail(issues);
186
187  }

189  /**
190   * Parses a single DMS coordinate
191   * @param coord
192   * @param lat
193   * @return the converted decimal up to 6 decimals accuracy
194   */
195  @VisibleForTesting
196  protected static double parseDMS(String coord, boolean lat) {
197    final String DIRS = lat ? "NS" : "EOW";
198    coord = coord.trim().toUpperCase();
199
200    if (coord.length() > 3) {
201      // preparse the direction and remove it from the string to avoid a very complex regex
202      char dir = 'n';
203      if (DIRS.contains(String.valueOf(coord.charAt(0)))) {
204        dir = coord.charAt(0);
205        coord = coord.substring(1);
206      } else if (DIRS.contains(String.valueOf(coord.charAt(coord.length()-1)))) {
207        dir = coord.charAt(coord.length()-1);
208        coord = coord.substring(0, coord.length()-1);
209      }
210      // without the direction chuck it at the regex
211      Matcher m = DMS_SINGLE.matcher(coord);
212      if (m.find()) {
213        return coordFromMatcher(m, 1,2,3, String.valueOf(dir));
214      } else {
215        m = DM_SINGLE.matcher(coord);
216        if (m.find()) {
217          return coordFromMatcher(m, 1, 2, String.valueOf(dir));
218        } else {
219          m = D_SINGLE.matcher(coord);
220          if (m.find()) {
221            return coordFromMatcher(m, 1, String.valueOf(dir));
222          }
223        }
224      }
225    }
226    throw new IllegalArgumentException();
227  }
228
229  private static double coordFromMatcher(Matcher m, int idx1, int idx2, int idx3, String sign) {
230    return roundTo6decimals(coordSign(sign) *
231      dmsToDecimal( NumberParser.parseDouble(m.group(idx1)), NumberParser.parseDouble(m.group(idx2)), NumberParser.parseDouble(m.group(idx3)) ));
232  }
233
234  private static double coordFromMatcher(Matcher m, int idx1, int idx2, String sign) {
235    return roundTo6decimals(coordSign(sign) *
236      dmsToDecimal( NumberParser.parseDouble(m.group(idx1)), NumberParser.parseDouble(m.group(idx2)), 0.0));
237  }
238
239  private static double coordFromMatcher(Matcher m, int idx1, String sign) {
240    return roundTo6decimals(coordSign(sign) *
241      dmsToDecimal( NumberParser.parseDouble(m.group(idx1)), 0.0, 0.0));
242  }
243
244  private static double dmsToDecimal(double degree, Double minutes, Double seconds) {
245    minutes = minutes == null ? 0 : minutes;
246    seconds = seconds == null ? 0 : seconds;
247    return degree + (minutes / 60) + (seconds / 3600);
248  }
249
250  // round to 6 decimals (~1m precision) since no way we're getting anything legitimately more precise
251  private static Double roundTo6decimals(Double x) {
252    return x == null ? null : Math.round(x * Math.pow(10, 6)) / Math.pow(10, 6);
253  }
