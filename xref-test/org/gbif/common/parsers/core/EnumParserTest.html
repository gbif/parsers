15    @Test
16    public void testNormalizeDigits() throws Exception {
17      EnumParser parser = new EnumParser(Rank.class, true);
18      assertEquals("HALLO", parser.normalize("hàlló"));
19  
20      assertEquals("HALLO", parser.normalize("Hallo"));
21      assertEquals("HALLO", parser.normalize(" Hallo  "));
22      assertEquals("HALLO", parser.normalize("HallO"));
23      assertEquals("HALLO", parser.normalize("Hallo!"));
24      assertEquals("HALLO", parser.normalize("(Hallo)"));
25      assertEquals("HALLO", parser.normalize("(Hällö"));
26  
27      assertEquals("H6", parser.normalize("(h6"));
28      assertEquals("HELLOMR6", parser.normalize("hello_mr6"));
29      assertEquals("HELLOBERTOCKO", parser.normalize("Hello Bértöçkø"));
30    }

41      assertEquals("HALLO", parser.normalize("(Hallo)"));

44      assertEquals("H", parser.normalize("(h6"));
