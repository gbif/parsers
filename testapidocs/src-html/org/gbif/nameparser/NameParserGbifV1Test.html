029  @Test
030  public void convertNameType() throws Exception {
031    org.gbif.nameparser.api.ParsedName pn = new org.gbif.nameparser.api.ParsedName();
032    for (org.gbif.nameparser.api.NameType t : org.gbif.nameparser.api.NameType.values()) {
033      pn.setType(t);
034      assertNotNull(NameParserGbifV1.gbifNameType(pn));
035    }
036    pn.setCandidatus(true);
037    assertEquals(NameType.CANDIDATUS, NameParserGbifV1.gbifNameType(pn));
038
039    pn.setCandidatus(false);
040    pn.setCultivarEpithet("Bella");
041    assertEquals(NameType.CULTIVAR, NameParserGbifV1.gbifNameType(pn));
042
043    pn.setDoubtful(true);
044    assertEquals(NameType.DOUBTFUL, NameParserGbifV1.gbifNameType(pn));
045  
046    pn.addWarning(Warnings.BLACKLISTED_EPITHET);
047    assertEquals(NameType.BLACKLISTED, NameParserGbifV1.gbifNameType(pn));
048  }

072  @Test
073  public void parseQuietly() throws Exception {
074    ParsedName pn = parser.parseQuietly("Navicula var. fasciata", Rank.VARIETY);
075    assertNull(pn.getSpecificEpithet());
076    assertEquals(NameType.INFORMAL, pn.getType());
077
078    assertEquals("Abies alba", parser.parseQuietly("Abies alba Mill.").canonicalName());
079    pn = parser.parseQuietly("Abies alba x Pinus graecus L.");
080    assertEquals("Abies alba x Pinus graecus L.", pn.getScientificName());
081    assertEquals(NameType.HYBRID, pn.getType());
082    assertNull(pn.getGenusOrAbove());
083
084    assertFalse(parser.parseQuietly("Protoscenium simplex  (Cleve, 1899), JÃ¸rgensen, 1905 ", Rank.SPECIES).isParsedPartially());
085    assertFalse(parser.parseQuietly("Plagiacanthidae", Rank.SPECIES).isParsedPartially());
086
087    pn = parser.parseQuietly("Plagiacanthidae WhatTheHeck @%&/ (please dont parse me[ he?5789])", Rank.SPECIES);
088    assertTrue(pn.isParsed());
089    assertTrue(pn.isParsedPartially());
090    assertEquals("Plagiacanthidae", pn.getGenusOrAbove());
091  }

093  @Test
094  public void assertScientificName() throws Exception {
095    ParsedName pn = parser.parseQuietly("Abies sp.");
096    assertEquals("Abies sp.", pn.getScientificName());
097    assertEquals("Abies spec.", pn.canonicalName());
098    assertEquals("Abies", pn.getGenusOrAbove());
099    assertEquals(Rank.SPECIES, pn.getRank());
100    assertNull(pn.getSpecificEpithet());
101
102    pn = parser.parseQuietly("ÃAbies Mill.");
103    assertEquals("ÃAbies Mill.", pn.getScientificName());
104    assertEquals("Abies", pn.canonicalName());
105    assertEquals("Abies", pn.getGenusOrAbove());
106    assertNull(pn.getRank());
107    assertNull(pn.getSpecificEpithet());
108    assertEquals(NamePart.GENERIC, pn.getNotho());
109
110    pn = parser.parseQuietly("? hostilis Gravenhorst, 1829");
111    assertEquals("? hostilis Gravenhorst, 1829", pn.getScientificName());
112    assertEquals("? hostilis", pn.canonicalName());
113    assertEquals("?", pn.getGenusOrAbove());
114    assertEquals(Rank.SPECIES, pn.getRank());
115    assertEquals("hostilis", pn.getSpecificEpithet());
116
117    pn = parser.parseQuietly("unassigned Asteraceae");
118    assertEquals("unassigned Asteraceae", pn.getScientificName());
119    assertNull(pn.canonicalName());
120    assertNull(pn.getGenusOrAbove());
121    assertNull(pn.getRank());
122    assertNull(pn.getSpecificEpithet());
123    
124    // https://github.com/gbif/name-parser/issues/45
125    pn = parser.parseQuietly("OdontellidaeGEN");
126    assertEquals(NameType.PLACEHOLDER, pn.getType());
127    assertEquals("OdontellidaeGEN", pn.getScientificName());
128    assertEquals("Odontellidae", pn.canonicalName());
129    assertEquals("Odontellidae", pn.getGenusOrAbove());
130    assertEquals("GEN", pn.getStrain());
131    assertNull(pn.getRank());
132    assertNull(pn.getSpecificEpithet());
133    assertNull(pn.getAuthorship());
134  }
