001package org.gbif.common.parsers.date;
002
003import org.junit.Test;
004import org.threeten.bp.Month;
005
006import static org.junit.Assert.assertEquals;
007import static org.junit.Assert.assertFalse;
008import static org.junit.Assert.assertNotNull;
009import static org.junit.Assert.assertNull;
010import static org.junit.Assert.assertTrue;
011
012/**
013 * Tests related to {@link DatePartsNormalizer}
014 */
015public class DatePartsNormalizerTest {
016
017  private static DatePartsNormalizer NORMALIZER = DatePartsNormalizer.newInstance();
018
019  /**
020   * This happens when integer are exported as float/double by a data publisher.
021   */
022  @Test
023  public void testFloatNormalization(){
024    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975.0", "2.0", "1.0");
025    assertEquals(new Integer(1), result.getDay());
026    assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
027    assertEquals(new Integer(1975), result.getYear());
028  }
029
030  @Test
031  public void testNormalizationWithSpaces(){
032    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975 ", " 2 ", " 1");
033    assertEquals(new Integer(1), result.getDay());
034    assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
035    assertEquals(new Integer(1975), result.getYear());
036  }
037
038  @Test
039  public void testMonthNameNormalization(){
040    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("January").intValue());
041    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JANUARY").intValue());
042    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("Jan").intValue());
043    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JAN.").intValue());
044    assertEquals(Month.DECEMBER.getValue(), NORMALIZER.monthNameToNumerical("DEC").intValue());
045    assertEquals(Month.APRIL.getValue(), NORMALIZER.monthNameToNumerical("April").intValue());
046    assertEquals(Month.SEPTEMBER.getValue(), NORMALIZER.monthNameToNumerical("Sept").intValue());
047    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("June").intValue());
048    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("Jun").intValue());
049    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("JUNE").intValue());
050    assertEquals(Month.NOVEMBER.getValue(), NORMALIZER.monthNameToNumerical("November").intValue());
051  }
052
053  @Test
054  public void testDiscardedDateParts(){
055
056    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("a", "5", "3");
057    assertTrue(result.yDiscarded());
058    assertNull(result.getYear());
059    assertNotNull(result.getMonth());
060
061    result = NORMALIZER.normalize("2000", "a", "3");
062    assertTrue(result.mDiscarded());
063    assertNull(result.getMonth());
064    assertNotNull(result.getYear());
065
066    result = NORMALIZER.normalize("2000", "5", "a");
067    assertTrue(result.dDiscarded());
068    assertNull(result.getDay());
069    assertNotNull(result.getYear());
070
071    // empty String, null and \\N should NOT be considered as "discarded"
072    // they simply represent "not provided"
073    result = NORMALIZER.normalize("2000", "5", "");
074    assertFalse(result.dDiscarded());
075    assertNull(result.getDay());
076
077    result = NORMALIZER.normalize("2000", "5", null);
078    assertFalse(result.dDiscarded());
079    assertNull(result.getDay());
080
081    result = NORMALIZER.normalize("2000", "5", "\\N");
082    assertFalse(result.dDiscarded());
083    assertNull(result.getDay());
084  }
085}
