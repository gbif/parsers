023/**
024 * A name parser wrapped to return classic ParsedName objects from the GBIF API v1.
025 */
026public class NameParserGbifV1 implements NameParser {
027
028  private static Logger LOG = LoggerFactory.getLogger(NameParserGbifV1.class);
029  private static final Map<org.gbif.nameparser.api.NameType, NameType> NAME_TYPE_MAP = ImmutableMap.<org.gbif.nameparser.api.NameType, NameType>builder()
030      .put(org.gbif.nameparser.api.NameType.SCIENTIFIC, NameType.SCIENTIFIC)
031      .put(org.gbif.nameparser.api.NameType.VIRUS, NameType.VIRUS)
032      .put(org.gbif.nameparser.api.NameType.HYBRID_FORMULA, NameType.HYBRID)
033      .put(org.gbif.nameparser.api.NameType.INFORMAL, NameType.INFORMAL)
034      .put(org.gbif.nameparser.api.NameType.OTU, NameType.OTU)
035      .put(org.gbif.nameparser.api.NameType.PLACEHOLDER, NameType.PLACEHOLDER)
036      .put(org.gbif.nameparser.api.NameType.NO_NAME, NameType.NO_NAME)
037      .build();
038
039  private final org.gbif.nameparser.api.NameParser parser;
040
041  /**
042   * Using the default GBIF RegEx Name Parser.
043   */
044  public NameParserGbifV1() {
045    this.parser = new NameParserGBIF();
046  }
047
048  /**
049   * Using the default GBIF RegEx Name Parser with a given timeout for parsing a single name.
050   * @param timeout in milliseconds before returning an Unparsable name
051   */
052  public NameParserGbifV1(long timeout) {
053    this.parser = new NameParserGBIF(timeout);
054  }
055
056  public NameParserGbifV1(org.gbif.nameparser.api.NameParser parser) {
057    this.parser = parser;
058  }
059
060  @Override
061  public ParsedName parse(String s, @Nullable Rank rank) throws UnparsableException {
062    try {
063      return convert(s, rank, parser.parse(s, fromGbif(rank)));
064
065    } catch (UnparsableNameException e) {
066      throw new UnparsableException(NAME_TYPE_MAP.getOrDefault(e.getType(), NameType.DOUBTFUL), e.getName());
067    }
068  }
069
070  @Override
071  public ParsedName parse(String scientificName) throws UnparsableException {
072    return parse(scientificName, null);
073  }
074
075  @Override
076  public ParsedName parseQuietly(String scientificName, @Nullable Rank rank) {
077    ParsedName p;
078    try {
079      p = parse(scientificName, rank);
080
081    } catch (UnparsableException e) {
082      p = new ParsedName();
083      p.setScientificName(scientificName);
084      p.setRank(rank);
085      p.setType(e.type);
086      p.setParsed(false);
087      p.setAuthorsParsed(false);
088    }
089
090    return p;
091  }
092
093  @Override
094  public ParsedName parseQuietly(String scientificName) {
095    return parseQuietly(scientificName, null);
096  }
097
098  @Override
099  // parses the name without authorship and returns the ParsedName.canonicalName() string
100  public String parseToCanonical(String scientificName, @Nullable Rank rank) {
101    if (Strings.isNullOrEmpty(scientificName)) {
102      return null;
103    }
104    try {
105      ParsedName pn = parse(scientificName, rank);
106      if (pn != null) {
107        return pn.canonicalName();
108      }
109    } catch (UnparsableException e) {
110      LOG.warn("Unparsable name " + scientificName + " >>> " + e.getMessage());
111    }
112    return null;
113  }
114
115  @Override
116  public String parseToCanonical(String scientificName) {
117    return parseToCanonical(scientificName, null);
118  }
119
120  public String parseToCanonicalOrScientificName(String scientificName) {
121    return parseToCanonicalOrScientificName(scientificName, null);
122  }
123
124  /**
125   * Tries to parses the name without authorship and returns the ParsedName.canonicalName() string
126   * For unparsable types and other UnparsableExceptions the original scientific name is returned.
127   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
128   */
129  public String parseToCanonicalOrScientificName(String scientificName, @Nullable Rank rank) {
130    if (Strings.isNullOrEmpty(scientificName)) {
131      return null;
132    }
133    try {
134      ParsedName pn = parse(scientificName, rank);
135      if (pn != null) {
136        return pn.canonicalName();
137      }
138    } catch (UnparsableException e) {
139      LOG.warn("Unparsable name " + scientificName + " >>> " + e.getMessage());
140    }
141    return StringUtils.normalizeSpace(scientificName.trim());
142  }

146  private ParsedName convert(String scientificName, Rank rank, org.gbif.nameparser.api.ParsedName pn) throws UnparsableException {
147    // throw unparsable for all unparsable types but placeholder and for all names that have a not parsed state
148    if ((!pn.getType().isParsable() && pn.getType() != org.gbif.nameparser.api.NameType.PLACEHOLDER)
149        || pn.getState() == org.gbif.nameparser.api.ParsedName.State.NONE) {
150      throw new UnparsableException(gbifNameType(pn), scientificName);
151    }
152
153    ParsedName gbif = new ParsedName();
154
155    gbif.setType(gbifNameType(pn));
156    gbif.setScientificName(scientificName);
157
158    gbif.setGenusOrAbove(MoreObjects.firstNonNull(pn.getGenus(), pn.getUninomial()));
159    gbif.setInfraGeneric(pn.getInfragenericEpithet());
160    gbif.setSpecificEpithet(pn.getSpecificEpithet());
161    gbif.setInfraSpecificEpithet(pn.getInfraspecificEpithet());
162    gbif.setCultivarEpithet(pn.getCultivarEpithet());
163    gbif.setNotho(toGbif(pn.getNotho()));
164    gbif.setRank(toGbif(pn.getRank()));
165    // in the old API we used null instead of unranked
166    if (gbif.getRank() == rank.UNRANKED && rank.UNRANKED != rank) {
167      gbif.setRank(null);
168    }
169    gbif.setStrain(pn.getStrain());
170    gbif.setSensu(pn.getSensu());
171
172    gbif.setAuthorship(NameFormatter.authorString(pn.getCombinationAuthorship(), false));
173    gbif.setYear(pn.getCombinationAuthorship().getYear());
174    gbif.setBracketAuthorship(NameFormatter.authorString(pn.getBasionymAuthorship(), false));
175    gbif.setBracketYear(pn.getBasionymAuthorship().getYear());
176
177    gbif.setNomStatus(pn.getNomenclaturalNotes());
178    gbif.setRemarks(pn.getRemarks());
179
180    gbif.setParsed(pn.getState().isNameParsed());
181    gbif.setAuthorsParsed(pn.getState().isAuthorshipParsed());
182
183    return gbif;
184  }

186  @VisibleForTesting
187  static NameType gbifNameType(org.gbif.nameparser.api.ParsedName pn) {
188    // detect name types that only exist in the GBIF API v1
189    if (pn.isCandidatus()) {
190      return NameType.CANDIDATUS;
191    } else if (pn.isDoubtful()) {
192      return NameType.DOUBTFUL;
193    } else if (pn.getCode() == NomCode.CULTIVARS || pn.getCultivarEpithet() != null) {
194      return NameType.CULTIVAR;
195    }
196    // convert all others
197    return NAME_TYPE_MAP.get(pn.getType());
198  }
199
200  @VisibleForTesting
201  static org.gbif.api.vocabulary.NamePart toGbif(NamePart notho) {
202    return convertEnum(org.gbif.api.vocabulary.NamePart.class, notho);
203  }
204
205  @VisibleForTesting
206  static Rank toGbif(org.gbif.nameparser.api.Rank rank) {
207    if (rank == null) return null;
208    switch (rank) {
209      case SUPERSECTION: return Rank.INFRAGENERIC_NAME;
210      case SUPERSERIES: return Rank.INFRAGENERIC_NAME;
211    }
212    return convertEnum(Rank.class, rank);
213  }
214
215  @VisibleForTesting
216  static org.gbif.nameparser.api.Rank fromGbif(Rank rank) {
217    if (rank == null) return null;
218    switch (rank) {
219      case RACE: return org.gbif.nameparser.api.Rank.PROLES;
220    }
221    return convertEnum(org.gbif.nameparser.api.Rank.class, rank);
222  }
223
224  /**
225   * Converts an enumeration value into a constant with the exact same name from a different enumeration class.
226   * In case the enumeration constant name does not exist an error is thrown.
227   *
228   * @param targetClass class of the target enumeration
229   * @param value
230   * @throws IllegalArgumentException in case the enumeration name does not exist in the target class
231   */
232  private static <G extends Enum<G>> G convertEnum(Class<G> targetClass, Enum<?> value) {
233    try {
234      return value == null ? null : Enum.valueOf(targetClass, value.name());
235    } catch (IllegalArgumentException e) {
236      LOG.warn("Unable to convert {} into {}", value, targetClass);
237      return null;
238    }
239  }
