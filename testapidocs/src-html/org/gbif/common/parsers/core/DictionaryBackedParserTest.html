001package org.gbif.common.parsers.core;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.junit.jupiter.api.Test;
007
008import static org.junit.jupiter.api.Assertions.assertEquals;
009import static org.junit.jupiter.api.Assertions.assertNotNull;
010
011public class DictionaryBackedParserTest {
012
013  @Test
014  public void testParse() {
015    List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
016    source.add(new KeyValue<String, Integer>("Tim", 32));
017    source.add(new KeyValue<String, Integer>("Markus", 38));
018    source.add(new KeyValue<String, Integer>("Jose", 28));
019    DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(false);
020    dbp.init(source.iterator());
021
022    assertParsed(dbp, "Tim", 32);
023    assertParsed(dbp, "TIM", 32);
024    assertParsed(dbp, "tIm", 32);
025    assertParsed(dbp, " \t Markus \t ", 38);
026    assertParsed(dbp, "MarKUS", 38);
027    assertParsed(dbp, "Jose", 28);
028    assertEquals(ParseResult.STATUS.FAIL, dbp.parse("Lars").getStatus());
029  }
030
031  @Test
032  public void testSensitiveParse() {
033    List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
034    source.add(new KeyValue<String, Integer>("Matt", 29));
035    DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(true);
036    dbp.init(source.iterator());
037
038    assertParsed(dbp, "Matt", 29);
039    assertParsed(dbp, " \t Matt \n ", 29);
040    assertEquals(ParseResult.STATUS.FAIL, dbp.parse("MATT").getStatus());
041  }
042
043  private void assertParsed(DictionaryBackedParser<Integer> dbp, String input, Integer payload) {
044    assertNotNull(dbp.parse(input));
045    assertEquals(ParseResult.STATUS.SUCCESS, dbp.parse(input).getStatus());
046    assertEquals(ParseResult.CONFIDENCE.DEFINITE, dbp.parse(input).getConfidence());
047    assertEquals(payload, dbp.parse(input).getPayload());
048  }
049
050}
