033  private static final String LOCALDATE_TEST_FILE = "parse/date/threeten_localdate_tests.txt";
034  private static final String LOCALDATETIME_TEST_FILE = "parse/date/threeten_localdatetime_tests.txt";

036  private static final String COLUMN_SEPARATOR = ";";
037  private static final String COMMENT_MARKER = "#";

039  private static final int RAW_VAL_IDX = 0;
040  private static final int YEAR_VAL_IDX = 1;
041  private static final int MONTH_VAL_IDX = 2;
042  private static final int DAY_VAL_IDX = 3;
043  private static final int HOUR_VAL_IDX = 4;
044  private static final int MIN_VAL_IDX = 5;
045  private static final int SEC_VAL_IDX = 6;

047  private static final TemporalParser PARSER = ThreeTenNumericalDateParser.newInstance();

049  @Test
050  public void testLocalDateFromFile() {
051    assertTestFile(LOCALDATE_TEST_FILE,
052            new Function<String[], Void>() {
053              @Nullable
054              @Override
055              public Void apply(@Nullable String[] row) {
056                String raw = row[RAW_VAL_IDX];
057                try {
058                  int year = Integer.parseInt(row[YEAR_VAL_IDX]);
059                  int month = Integer.parseInt(row[MONTH_VAL_IDX]);
060                  int day = Integer.parseInt(row[DAY_VAL_IDX]);
061                  ParseResult<TemporalAccessor> result = PARSER.parse(raw);
062                  assertNotNull(raw + " generated null payload", result.getPayload());
063                  assertEquals("Test file rawValue: " + raw, LocalDate.of(year, month, day),
064                          LocalDate.from(result.getPayload()));
065                }
066                catch (NumberFormatException nfEx){
067                  fail("Error while parsing the test input file content." + nfEx.getMessage());
068                }
069                return null;
070              }
071            });
072  }

116  /**
117   * Utility function to run assertions received as Function on each rows of an input file.
118   *
119   * @param filepath
120   * @param assertRow
121   */
122  private void assertTestFile(String filepath, Function<String[], Void> assertRow) {
123    File testInputFile = FileUtils.getClasspathFile(filepath);
124    try{
125      CSVReader csv = CSVReaderFactory.build(testInputFile, COLUMN_SEPARATOR, true);
126      for (String[] row : csv) {
127        if (row == null || row[0].startsWith(COMMENT_MARKER)) {
128          continue;
129        }
130        assertRow.apply(row);
131      }
132    } catch (IOException e) {
133      fail("Problem reading testFile " + filepath + " " + e.getMessage());
134    }
135  }

137  @Test
138  public void testParseAsLocalDateTime(){
139    ThreeTenNumericalDateParser parser = ThreeTenNumericalDateParser.newInstance(Year.of(1900));
140
141    //month first with 2 digits years >_<
142    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("122178").getPayload()));
143    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12/21/78").getPayload()));
144    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12\\21\\78").getPayload()));
145    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12.21.78").getPayload()));
146    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12-21-78").getPayload()));
147    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12_21_78").getPayload()));
148
149    //month/year alone
150    assertEquals(YearMonth.of(1978,12), YearMonth.from(parser.parse("1978-12").getPayload()));
151
152    //year alone
153    assertEquals(Year.of(1978), Year.from(parser.parse("1978").getPayload()));
154   // assertEquals(Year.of(1978), Year.from(parser.parse("78").getPayload()));
155  }

157  @Test
158  public void testParseAsLocalDateByDateParts(){
159    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(PARSER.parse("1978", "12", "21").getPayload()));
160    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(PARSER.parse(1978, 12, 21).getPayload()));
161
162    assertEquals(LocalDate.of(1978, Month.DECEMBER, 1), LocalDate.from(PARSER.parse("1978", "12", "1").getPayload()));
163    assertEquals(YearMonth.of(1978,12), YearMonth.from(PARSER.parse("1978", "12", null).getPayload()));
164    assertEquals(YearMonth.of(1978,12), YearMonth.from(PARSER.parse(1978, 12, null).getPayload()));
165    assertEquals(Year.of(1978), Year.from(PARSER.parse("1978","",null).getPayload()));
166
167    // providing the day without the month should result in an error
168    assertEquals(ParseResult.STATUS.FAIL, PARSER.parse("1978", "", "2").getStatus());
169    assertEquals(ParseResult.STATUS.FAIL, PARSER.parse(1978, null, 2).getStatus());
170  }

172  @Test
173  public void testParsePreserveZoneOffset(){
174    ZonedDateTime offsetDateTime = ZonedDateTime.of(1978, 12, 21, 0, 0, 0, 0, ZoneOffset.of("+02:00"));
175    assertEquals(offsetDateTime, PARSER.parse("1978-12-21T00:00:00+02:00").getPayload());
176  }

178  @Test
179  public void testBlankDates(){
180    assertEquals(ParseResult.STATUS.FAIL, PARSER.parse(" ").getStatus());
181    assertEquals(ParseResult.STATUS.FAIL, PARSER.parse("").getStatus());
182    assertEquals(ParseResult.STATUS.FAIL, PARSER.parse(null).getStatus());
183  }
