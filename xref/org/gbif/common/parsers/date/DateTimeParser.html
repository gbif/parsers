1   package org.gbif.common.parsers.date;

3   import java.time.format.DateTimeFormatter;
4   import java.time.format.DateTimeParseException;
5   import java.time.temporal.TemporalAccessor;
6   import java.time.temporal.TemporalQuery;
7   import javax.annotation.Nullable;
8   import javax.validation.constraints.NotNull;
9   
10  import com.google.common.base.Preconditions;
11  
12  /**
13   * Adds some flexibility around {@link DateTimeFormatter} with the {@link DateTimeSeparatorNormalizer} and
14   * simple optimization the support of DateComponentOrdering.
15   * <p>
16   * This class is thread-safe once an instance is created.
17   */
18  public class DateTimeParser {
19  
20    private final DateTimeFormatter formatter;
21    private final DateTimeSeparatorNormalizer normalizer;
22    private final DateComponentOrdering ordering;
23  
24    private final TemporalQuery<?>[] types;
25    private final int minLength;
26  
27    /**
28     * Package protected constructor.
29     * Use {@link DateTimeParserBuilder}
30     */
31    DateTimeParser(@NotNull DateTimeFormatter formatter, @Nullable DateTimeSeparatorNormalizer normalizer,
32                   @NotNull DateComponentOrdering ordering, TemporalQuery<?>[] type, int minLength) {
33  
34      Preconditions.checkNotNull(formatter, "DateTimeFormatter can not be null");
35      Preconditions.checkNotNull(ordering, "DateComponentOrdering can not be null");
36      Preconditions.checkNotNull(type, "TemporalQuery can not be null");
37      Preconditions.checkArgument(minLength > 0, "minLength must be greater than 0");
38  
39      this.formatter = formatter;
40      this.ordering = ordering;
41      this.normalizer = normalizer;
42      this.minLength = minLength;
43      this.types = type;
44    }
45  
46    public DateComponentOrdering getOrdering() {
47      return ordering;
48    }
49  
50    /**
51     * Parses the provided String as a TemporalAccessor if possible, otherwise returns null.
52     * <p>
53     * This function fully support partial dates and will return the best possible date resolution based
54     * on the {@link DateComponentOrdering} provided.
55     * <p>
56     * This function will not throw DateTimeParseException but returns null in case the input
57     * can not be parsed.
58     *
59     * @return TemporalAccessor or null in case the input can not be parsed.
60     */
61    public TemporalAccessor parse(String input) {
62  
63      // return fast if minimum length is not meet
64      if (input.length() < minLength) {
65        return null;
66      }
67  
68      if (normalizer != null) {
69        input = normalizer.normalize(input);
70      }
71  
72      try {
73        if (types.length > 1) {
74          return formatter.parseBest(input, types);
75        }
76        return (TemporalAccessor) formatter.parse(input, types[0]);
77      } catch (DateTimeParseException dpe) {
78      }
79      return null;
80    }
81  
82  }
