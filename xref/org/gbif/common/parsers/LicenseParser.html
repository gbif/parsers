7   import java.net.URI;
8   import java.util.regex.Pattern;
9   import javax.annotation.Nullable;
10  
11  import com.google.common.base.Strings;

13  /**
14   * Singleton implementation of the dictionary that uses the file /dictionaries/parse/license.txt to lookup a
15   * License by its URI or its acronym/title, e.g. a lookup by "CC-BY 4.0" returns License.CC_BY_4_0.
16   * </br>
17   * The dictionary file must enforce the <a href="http://www.gbif.org/terms/licences">GBIF Licensing Policy</a>.
18   * Non-CC licenses that GBIF considers equal to one of its three supported CC licenses (CC0 1.0, CC-BY 4.0 and CC-BY-NC
19   * 4.0) can be added to this file.
20   * </br>
21   * Note a lookup by license acronym/title without a version number defaults to the latest version of that license,
22   * e.g. a lookup by "CC-BY" returns License.CC_BY_4_0.
23   */
24  public class LicenseParser extends EnumParser<License> {
25  
26    private static final String COMMENT_MARKER = "#";
27    private static final String LICENSE_FILEPATH = "/dictionaries/parse/license.tsv";
28    //allows us to remove the protocol part for http:// and https://
29    private static final Pattern REMOVE_HTTP_PATTERN = Pattern.compile("^https?:\\/\\/", Pattern.CASE_INSENSITIVE);
30    private static LicenseParser singletonObject = null;
31  
32    private LicenseParser() {
33      super(License.class, true);
34      // also make sure we have all enum values and their parameters title and url mapped
35      for (License l : License.values()) {
36        add(l.name(), l);
37        add(l.getLicenseTitle(), l);
38        add(l.getLicenseUrl(), l);
39      }
40      // use dict file last
41      init(LicenseParser.class.getResourceAsStream(LICENSE_FILEPATH), COMMENT_MARKER);
42    }
43  
44    @Override
45    protected String normalize(String value) {
46      if(value == null){
47        return null;
48      }
49      return super.normalize(REMOVE_HTTP_PATTERN.matcher(value).replaceAll(""));
50    }
51  
52    public static LicenseParser getInstance() {
53      synchronized (ContinentParser.class) {
54        if (singletonObject == null) {
55          singletonObject = new LicenseParser();
56        }
57      }
58      return singletonObject;
59    }
60  
61    /**
62     * Parse license supplied in two parts: URI and title. First parse URI. Only if no URI supplied, parse title.
63     * supplied.
64     *
65     * @param uri   optional license URI
66     * @param title optional license title
67     *
68     * @return License corresponding to license URI or title, License.UNSPECIFIED if both URI and title not supplied,
69     * otherwise defaults to License.UNSUPPORTED
70     */
71    public License parseUriThenTitle(@Nullable URI uri, @Nullable String title) {
72      if (uri == null && Strings.isNullOrEmpty(title)) {
73        return License.UNSPECIFIED;
74      }
75  
76      if (uri != null) {
77        ParseResult<License> result = singletonObject.parse(uri.toString());
78        if (result.isSuccessful()) {
79          return result.getPayload();
80        }
81      }
82  
83      if (!Strings.isNullOrEmpty(title)) {
84        ParseResult<License> result = singletonObject.parse(title);
85        if (result.isSuccessful()) {
86          return result.getPayload();
87        }
88      }
89  
90      return License.UNSUPPORTED;
91    }
92  }
