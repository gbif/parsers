025public class LicenseParserTest extends ParserTestBase<License> {

027  public LicenseParserTest() {
028    super(LicenseParser.getInstance());

030
031  /**
032   * Makes sure all License enum values and their parameters are parsed ok.
033   */
034  @Test
035  public void testParseAllEnumValues() {
036    for (License l : License.values()) {
037      assertParseSuccess(l, l.name());
038      if (l.getLicenseUrl() != null) {
039        assertParseSuccess(l, l.getLicenseUrl());
040      }
041      if (l.getLicenseTitle() != null) {
042        assertParseSuccess(l, l.getLicenseTitle());
043      }
044    }
045  }
046
047  @Test
048  public void testParseFail() {
049    assertParseFailure("Not licensed");
050    assertParseFailure("CC");
051  }

100  @Test
101  public void testParseUriThenTitle() throws URISyntaxException {
102    LicenseParser parser = LicenseParser.getInstance();
103    assertEquals(License.UNSPECIFIED, parser.parseUriThenTitle(null, null));
104
105    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(null, "CC0"));
106    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(null, "CCZero"));
107    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/publicdomain/zero/1.0"), null));
108    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/publicdomain/zero/1.0/"), null));
109    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/publicdomain/zero/1.0/legalcode"), null));
110
111    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(null, "CC-BY"));
112    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/4.0"), null));
113    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/4.0/"), null));
114    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/4.0/legalcode"), null));
115
116    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(null, "CC-BY-NC"));
117    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by-nc/4.0"), null));
118    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by-nc/4.0/"), null));
119    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by-nc/4.0/legalcode"), null));
120
121    assertEquals(License.UNSUPPORTED, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/3.0/legalcode"), null));
122    assertEquals(License.UNSUPPORTED, parser.parseUriThenTitle(new URI("http://opendatacommons.org/licenses/odbl/1.0"), null));
123  }
