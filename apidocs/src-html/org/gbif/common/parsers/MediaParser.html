004import com.google.common.collect.ImmutableMap;

016import javax.annotation.Nullable;
017import java.net.URI;

022public class MediaParser {
023  private static final Logger LOG = LoggerFactory.getLogger(MediaParser.class);
024  private static final Tika TIKA = new Tika();
025  private static final MimeTypes MIME_TYPES = MimeTypes.getDefaultMimeTypes();
026  private static final String HTML_TYPE = "text/html";
027  // MIME types which we consider as HTML links instead of real media file URIs
028  private static final Set<String> HTML_MIME_TYPES = ImmutableSet
029    .of("text/x-coldfusion", "text/x-php", "text/asp", "text/aspdotnet", "text/x-cgi", "text/x-jsp", "text/x-perl",
030      HTML_TYPE, MIME_TYPES.OCTET_STREAM);
031
032  // List of exceptions, could be read from a file if it grows. URLs matching this return a media file.
033  private static final Map<Pattern, String> knownUrlPatterns = new ImmutableMap.Builder<Pattern, String>()
034    .put(Pattern.compile("http://www\\.unimus\\.no/felles/bilder/web_hent_bilde\\.php\\?id=\\d+&type=jpeg"), "image/jpeg")
035    .put(Pattern.compile("http://www\\.jacq\\.org/image\\.php\\?filename=\\d+&method=europeana"), "image/jpeg")
036    .put(Pattern.compile("https://images\\.ala\\.org\\.au/image/proxyImageThumbnailLarge\\?imageId=[0-9a-f-]{36}"), "image/jpeg")
037    .put(Pattern.compile("http://[a-zA-Z0-9-]+\\.wildlifemonitoring\\.ru/get_photo\\.php\\?id=\\d+"), "image/jpeg")
038    .put(Pattern.compile("http://procyon\\.acadiau\\.ca/ecsmith/cgi-bin/image\\.cgi\\?[0-9A-Z]+,jpeg"), "image/jpeg")
039
040    .put(Pattern.compile("http://www\\.biologie\\.uni-ulm\\.de/cgi-bin/perl/sound\\.pl\\?sid=T&objid=\\d+"), "audio/vnd.wave")
041    .put(Pattern.compile("https://dofbasen\\.dk/sound_proxy\\.php\\?referer=gbif&mode=o&snd=[0-9_]+.mp3&raw=1"), "audio/mpeg")
042    .build();
043
044  // Add missing alias types.
045  static {
046    MediaTypeRegistry mediaTypeRegistry = MIME_TYPES.getMediaTypeRegistry();
047    mediaTypeRegistry.addAlias(org.apache.tika.mime.MediaType.audio("mpeg"), org.apache.tika.mime.MediaType.audio("mp3"));
048    mediaTypeRegistry.addAlias(org.apache.tika.mime.MediaType.audio("mpeg"), org.apache.tika.mime.MediaType.audio("mpeg3"));
049  }
050
051  private static MediaParser instance = null;
052
053  public static MediaParser getInstance() {
054    synchronized (MediaParser.class) {
055      if (instance == null) {
056        instance = new MediaParser();
057      }
058    }
059    return instance;
060  }
061
062  public MediaObject detectType(MediaObject mo) {
063    if (Strings.isNullOrEmpty(mo.getFormat())) {
064      // derive from URI
065      mo.setFormat(parseMimeType(mo.getIdentifier()));
066    }
067
068    // if MIME type is text/html make it a references link instead
069    if (HTML_TYPE.equalsIgnoreCase(mo.getFormat()) && mo.getIdentifier() != null) {
070      // make file URI the references link URL instead
071      mo.setReferences(mo.getIdentifier());
072      mo.setIdentifier(null);
073      mo.setFormat(null);
074    }
075
076    if (!Strings.isNullOrEmpty(mo.getFormat())) {
077      if (mo.getFormat().startsWith("image")) {
078        mo.setType(MediaType.StillImage);
079      } else if (mo.getFormat().startsWith("audio")) {
080        mo.setType(MediaType.Sound);
081      } else if (mo.getFormat().startsWith("video")) {
082        mo.setType(MediaType.MovingImage);
083      } else {
084        LOG.debug("Unsupported media format {}", mo.getFormat());
085      }
086    }
087    return mo;
088  }
089
090  /**
091   * Parses a MIME type using Apache Tika which can handle the following:
092   * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
093   * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
094   */
095  public String parseMimeType(@Nullable String format) {
096    if (format != null) {
097      format = Strings.emptyToNull(format.trim().toLowerCase());
098    }
099    if (format == null) {
100      return null;
101    }
102
103    try {
104      MimeType mime = MIME_TYPES.getRegisteredMimeType(format);
105      if (mime != null) {
106        return mime.getName();
107      }
108
109    } catch (MimeTypeException e) {
110    }
111
112    // Failed, but return the input if it's a reasonable MIME type
113    return MimeType.isValid(format) ? format : null;
114  }
115
116  /**
117   * Parses a MIME type using Apache Tika which can handle the following:
118   * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
119   * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
120   */
121  public String parseMimeType(@Nullable URI uri) {
122    if (uri != null) {
123      String uriString = uri.toString();
124      String mime = TIKA.detect(uriString);
125      if (mime != null && HTML_MIME_TYPES.contains(mime.toLowerCase())) {
126        // We may have something like http://example.org/imageServer?img=test.jpg, so re-run the detection on the last
127        // part of the URL query string.
128        if (uri.getQuery() != null) {
129          mime = TIKA.detect(uri.getQuery());
130          if (mime != null && !HTML_MIME_TYPES.contains(mime.toLowerCase())) {
131            return mime;
132          }
133        }
134
135        // First check the dictionary for known exceptions.
136        for (Map.Entry<Pattern, String> p : knownUrlPatterns.entrySet()) {
137          if (p.getKey().matcher(uriString).matches()) {
138            return p.getValue();
139          }
140        }
141
142        // links without any suffix default to OCTET STREAM, see:
143        // http://dev.gbif.org/issues/browse/POR-2066
144        return HTML_TYPE;
145      }
146      return mime;
147    }
148    return null;
149  }
