001package org.gbif.common.parsers.utils;
002
003import java.io.BufferedReader;
004import java.io.IOException;
005import java.io.InputStreamReader;
006import java.io.Reader;
007import java.util.HashSet;
008import java.util.Set;
009
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013/**
014 * Set of bad names in UPPERCASE.
015 */
016public final class BlacklistedNames {
017
018  private static final Logger LOGGER = LoggerFactory.getLogger(BlacklistedNames.class);
019
020  private static final Set<String> NAMES = new HashSet<String>();
021
022  private static final String BLACKLIST_FILE = "utils/blacklistedNames.tsv";
023
024  static {
025    init(BLACKLIST_FILE);
026  }
027
028  private BlacklistedNames() {
029    throw new UnsupportedOperationException("Can't initialize class");
030  }
031
032  public static boolean contains(String s) {
033    return NAMES.contains(s.toUpperCase());
034  }
035
036  public static synchronized void init(Reader reader) {
037    NAMES.clear();
038    BufferedReader br = new BufferedReader(reader);
039    try {
040      String line = br.readLine();
041      while (line != null) {
042        LOGGER.debug("Blacklisting: {}", line);
043        NAMES.add(line.toUpperCase());
044        line = br.readLine();
045      }
046    } catch (IOException e) {
047      LOGGER.debug("Exception thrown", e);
048    } finally {
049      try {
050        br.close();
051      } catch (IOException ignored) {
052      }
053    }
054  }
055
056  public static synchronized void init(String filepath) {
057    init(new InputStreamReader(BlacklistedNames.class.getClassLoader().getResourceAsStream(filepath)));
058  }
059}
