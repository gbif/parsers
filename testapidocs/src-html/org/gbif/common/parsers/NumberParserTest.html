001package org.gbif.common.parsers;
002
003import org.gbif.common.parsers.NumberParser;
004
005import org.junit.Test;
006
007import static org.junit.Assert.assertEquals;
008import static org.junit.Assert.assertNull;
009
010/**
011 *
012 */
013public class NumberParserTest {
014
015  @Test
016  public void testParseDouble() throws Exception {
017    assertNull(NumberParser.parseDouble("312,dsfds"));
018
019    assertEquals(2d, NumberParser.parseDouble("2"), 0.0);
020    assertEquals(2.123d, NumberParser.parseDouble("2.123"), 0.0);
021    assertEquals(-122.12345d, NumberParser.parseDouble("-122.12345"), 0.0);
022    assertEquals(22788130.9993d, NumberParser.parseDouble("22.788.130,9993"), 0.0);
023    assertEquals(12300d, NumberParser.parseDouble("1.23E4"), 0.0);
024
025    // These should be parsing failures due to ambiguity, see issue 23.
026    assertEquals(2.123d, NumberParser.parseDouble("2,123"), 0.0);
027    assertEquals(-2.123d, NumberParser.parseDouble("-2,123"), 0.0);
028
029    // These are unambiguous, and could be accepted
030    // assertEquals(2.123d, NumberParser.parseDouble("2,123.0"), 0.0);
031    // assertEquals(2.123d, NumberParser.parseDouble("2.123,0"), 0.0);
032
033    assertNull(NumberParser.parseDouble(null));
034    assertNull(NumberParser.parseDouble(""));
035    assertNull(NumberParser.parseDouble(" "));
036    assertNull(NumberParser.parseDouble("ds"));
037    assertNull(NumberParser.parseDouble("312,dsfds"));
038    assertNull(NumberParser.parseDouble("43-1"));
039    assertNull(NumberParser.parseDouble("43,112,321"));
040
041  }
042}
