001package org.gbif.common.parsers.geospatial;
002
003import com.google.common.base.MoreObjects;
004import com.google.common.base.Objects;
005
006/**
007 * Simple container class for double based values with an accuracy (e.g. depth in meters).
008 */
009public class DoubleAccuracy {
010
011  private final Double value;
012  private final Double accuracy;
013
014  public DoubleAccuracy(Double value, Double accuracy) {
015    this.value = value;
016    // never negative
017    this.accuracy = accuracy == null ? null : Math.abs(accuracy);
018  }
019
020  public Double getValue() {
021    return value;
022  }
023
024  public Double getAccuracy() {
025    return accuracy;
026  }
027
028  @Override
029  public String toString() {
030    return MoreObjects.toStringHelper(this)
031      .add("value", value)
032      .add("accuracy", accuracy)
033      .toString();
034  }
035
036  @Override
037  public int hashCode() {
038    return Objects.hashCode(value, accuracy);
039  }
040
041  @Override
042  public boolean equals(Object object) {
043    if (this == object) {
044      return true;
045    }
046    if (!(object instanceof DoubleAccuracy)) {
047      return false;
048    }
049
050    DoubleAccuracy that = (DoubleAccuracy) object;
051    return Objects.equal(this.value, that.value)
052           && Objects.equal(this.accuracy, that.accuracy);
053  }
054
055}
