3   import lombok.extern.slf4j.Slf4j;
4   import org.junit.jupiter.api.Test;
5   
6   import static org.junit.jupiter.api.Assertions.assertArrayEquals;
7   import static org.junit.jupiter.api.Assertions.assertNull;
8   
9   @Slf4j
10  public class DelimiterUtilsTest {
11  
12    @Test
13    public void testISORange() {
14      assertArrayEquals(
15          new String[] {"1999-01-20", "1999-01-31"},
16          DelimiterUtils.splitISODateRange("1999-01-20/31"));
17      assertArrayEquals(
18          new String[] {"1999-01", "1999-12"}, DelimiterUtils.splitISODateRange("1999-01/12"));
19      assertArrayEquals(
20          new String[] {"1999-1", "1999-2"}, DelimiterUtils.splitISODateRange("1999-1/2"));
21      assertArrayEquals(
22          new String[] {"1999-1-21", "1999-1-6"}, DelimiterUtils.splitISODateRange("1999-1-21/6"));
23      assertArrayEquals(
24          new String[] {"1999-1-02", "1999-1-6"}, DelimiterUtils.splitISODateRange("1999-1-02/6"));
25      assertNull(DelimiterUtils.splitISODateRange("1999-01/31")); // Wrong month
26      assertNull(DelimiterUtils.splitISODateRange("1999-01-20"));
27  
28      assertArrayEquals(
29          new String[] {"1999-01-20", "1999-01-31"}, DelimiterUtils.splitPeriod("1999-01-20/31"));
30      assertArrayEquals(
31          new String[] {"1999-01-20", "1999-01-20"}, DelimiterUtils.splitPeriod("1999-01-20"));
32      assertArrayEquals(
33          new String[] {"19990120", "19990230"}, DelimiterUtils.splitPeriod("19990120/19990230"));
34      assertArrayEquals(
35          new String[] {"1999-01", "1999-02"}, DelimiterUtils.splitPeriod("1999-01/02"));
36      assertArrayEquals(
37          new String[] {"1998-9-30", "1998-10-7"}, DelimiterUtils.splitPeriod("1998-9-30/10-7"));
38      assertArrayEquals(new String[] {"", ""}, DelimiterUtils.splitPeriod(null));
39    }
40  }
