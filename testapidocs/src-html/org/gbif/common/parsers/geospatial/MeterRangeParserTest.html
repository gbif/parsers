017  @Test
018  public void testParseElevation() throws Exception {
019    assertResult(MeterRangeParser.parseElevation("10", "20", null), true, 15d, 5d);
020    assertResult(MeterRangeParser.parseElevation("10", "20", "1"), true, 15d, 6d);
021    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
022    assertResult(MeterRangeParser.parseElevation("10", null, null), true, 10d, null);
023    assertResult(MeterRangeParser.parseElevation(null, "10", null), true, 10d, null);
024    assertResult(MeterRangeParser.parseElevation("-800", null, null), true, -800d, null);
025    assertResult(MeterRangeParser.parseElevation("-100", "-50", null), true, -75d, 25d);
026    assertResult(MeterRangeParser.parseElevation("110", "115", null), true, 112.5d, 2.5d);
027    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
028    assertResult(MeterRangeParser.parseElevation("10.0", "12.5", "1"), true, 11.25d, 2.25d);
029    // European commas are ambiguous, see issue 23.
030    assertResult(MeterRangeParser.parseElevation("10,0", "12,5", "1"), true, 11.25d, 2.25d);
031    assertResult(MeterRangeParser.parseElevation(null, "10000", "1"), true, 10000d, 1d);
032    assertResult(
033        MeterRangeParser.parseElevation("4061987", "4061987", null),
034        false,
035        null,
036        null,
037        OccurrenceIssue.ELEVATION_UNLIKELY);
038
039    // check units are removed
040    assertResult(
041        MeterRangeParser.parseElevation("1000m", null, "1"),
042        true,
043        1000d,
044        1d,
045        OccurrenceIssue.ELEVATION_NON_NUMERIC);
046    assertResult(
047        MeterRangeParser.parseElevation("3280ft", null, "1"),
048        true,
049        999.74,
050        1d,
051        OccurrenceIssue.ELEVATION_NOT_METRIC,
052        OccurrenceIssue.ELEVATION_NON_NUMERIC);
053
054    // check out of range
055    assertResult(
056        MeterRangeParser.parseElevation("100000000000", null, "1"),
057        false,
058        null,
059        1d,
060        OccurrenceIssue.ELEVATION_UNLIKELY);
061
062    // nonsense
063    assertResult(
064        MeterRangeParser.parseElevation("booya", "boom", "1"),
065        false,
066        null,
067        1d,
068        OccurrenceIssue.ELEVATION_NON_NUMERIC);
069  }

189  private void assertResult(
190      OccurrenceParseResult<?> pr,
191      boolean success,
192      Double elevation,
193      Double precision,
194      OccurrenceIssue... issue) {
195    if (success) {
196      assertExpected(
197          pr, new DoubleAccuracy(elevation, precision), ParseResult.CONFIDENCE.DEFINITE, issue);
198    } else {
199      assertFailed(pr);
200    }
201  }
202
203  private void assertExpected(
204      OccurrenceParseResult<?> pr,
205      Object expected,
206      ParseResult.CONFIDENCE c,
207      OccurrenceIssue... issue) {
208    assertNotNull(pr);
209    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
210    assertEquals(c, pr.getConfidence());
211    assertNotNull(pr.getPayload());
212    assertEquals(expected, pr.getPayload());
213    if (issue == null) {
214      assertTrue(pr.getIssues().isEmpty());
215    } else {
216      assertEquals(
217          "Issue mismatch, found issues: " + pr.getIssues(), issue.length, pr.getIssues().size());
218      for (OccurrenceIssue iss : issue) {
219        assertTrue(pr.getIssues().contains(iss));
220      }
221    }
222  }
223
224  private void assertFailed(ParseResult<?> pr) {
225    assertNotNull(pr);
226    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
227  }
