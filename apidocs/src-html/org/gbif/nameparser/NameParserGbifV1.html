045  private static Logger LOG = LoggerFactory.getLogger(NameParserGbifV1.class);
046  private static final Map<org.gbif.nameparser.api.NameType, NameType> NAME_TYPE_MAP = ImmutableMap.<org.gbif.nameparser.api.NameType, NameType>builder()
047      .put(org.gbif.nameparser.api.NameType.SCIENTIFIC, NameType.SCIENTIFIC)
048      .put(org.gbif.nameparser.api.NameType.VIRUS, NameType.VIRUS)
049      .put(org.gbif.nameparser.api.NameType.HYBRID_FORMULA, NameType.HYBRID)
050      .put(org.gbif.nameparser.api.NameType.INFORMAL, NameType.INFORMAL)
051      .put(org.gbif.nameparser.api.NameType.OTU, NameType.OTU)
052      .put(org.gbif.nameparser.api.NameType.PLACEHOLDER, NameType.PLACEHOLDER)
053      .put(org.gbif.nameparser.api.NameType.NO_NAME, NameType.NO_NAME)

077  @Override
078  public ParsedName parse(String s, @Nullable Rank rank) throws UnparsableException {
079    try {
080      return convert(s, rank, parser.parse(s, fromGbif(rank)));
081
082    } catch (UnparsableNameException e) {
083      throw new UnparsableException(NAME_TYPE_MAP.getOrDefault(e.getType(), NameType.DOUBTFUL), e.getName());
084    }
085  }
086
087  @Override
088  public ParsedName parse(String scientificName) throws UnparsableException {
089    return parse(scientificName, null);
090  }
091
092  @Override
093  public ParsedName parseQuietly(String scientificName, @Nullable Rank rank) {
094    ParsedName p;
095    try {
096      p = parse(scientificName, rank);
097
098    } catch (UnparsableException e) {
099      p = new ParsedName();
100      p.setScientificName(scientificName);
101      p.setRank(rank);
102      p.setType(e.type);
103      p.setParsed(false);
104      p.setParsedPartially(false);
105    }
106
107    return p;
108  }
109
110  @Override
111  public ParsedName parseQuietly(String scientificName) {
112    return parseQuietly(scientificName, null);
113  }

115  @Override
116  // parses the name without authorship and returns the ParsedName.canonicalName() string
117  public String parseToCanonical(String scientificName, @Nullable Rank rank) {
118    if (Strings.isNullOrEmpty(scientificName)) {
119      return null;
120    }
121    try {
122      ParsedName pn = parse(scientificName, rank);
123      if (pn != null) {
124        return pn.canonicalName();
125      }
126    } catch (UnparsableException e) {
127      logUnparsable(e);
128    }
129    return null;
130  }
131
132  private static void logUnparsable(UnparsableException e) {
133    if (e.type.isParsable()) {
134      LOG.debug("Unparsable {} {} >>> {}", e.type, e.name, e.getMessage());
135    } else {
136      LOG.warn("Unparsable {} {} >>> {}", e.type, e.name, e.getMessage());
137    }
138  }
139
140  @Override
141  public String parseToCanonical(String scientificName) {
142    return parseToCanonical(scientificName, null);
143  }

145  public String parseToCanonicalOrScientificName(String scientificName) {
146    return parseToCanonicalOrScientificName(scientificName, null);
147  }

149  /**
150   * Tries to parses the name without authorship and returns the ParsedName.canonicalName() string
151   * For unparsable types and other UnparsableExceptions the original scientific name is returned.
152   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
153   */
154  public String parseToCanonicalOrScientificName(String scientificName, @Nullable Rank rank) {
155    if (Strings.isNullOrEmpty(scientificName)) {
156      return null;
157    }
158    try {
159      ParsedName pn = parse(scientificName, rank);
160      if (pn != null) {
161        return pn.canonicalName();
162      }
163    } catch (UnparsableException e) {
164      logUnparsable(e);
165    }
166    return StringUtils.normalizeSpace(scientificName.trim());
167  }

171  private ParsedName convert(String scientificName, Rank rank, org.gbif.nameparser.api.ParsedName pn) throws UnparsableException {
172    // throw unparsable for all unparsable types but placeholder and for all names that have a not parsed state
173    if ((!pn.getType().isParsable() && pn.getType() != org.gbif.nameparser.api.NameType.PLACEHOLDER)
174        || pn.getState() == org.gbif.nameparser.api.ParsedName.State.NONE) {
175      throw new UnparsableException(gbifNameType(pn), scientificName);
176    }
177
178    ParsedName gbif = new ParsedName();
179
180    gbif.setType(gbifNameType(pn));
181    gbif.setScientificName(scientificName);
182
183    gbif.setGenusOrAbove(pn.getGenus() != null ? pn.getGenus(): pn.getUninomial());
184    gbif.setInfraGeneric(pn.getInfragenericEpithet());
185    gbif.setSpecificEpithet(pn.getSpecificEpithet());
186    gbif.setInfraSpecificEpithet(pn.getInfraspecificEpithet());
187    gbif.setCultivarEpithet(pn.getCultivarEpithet());
188    gbif.setNotho(toGbif(pn.getNotho()));
189    gbif.setRank(toGbif(pn.getRank()));
190    // in the old API we used null instead of unranked
191    if (gbif.getRank() == Rank.UNRANKED && Rank.UNRANKED != rank) {
192      gbif.setRank(null);
193    }
194    gbif.setStrain(pn.getPhrase());
195    gbif.setSensu(pn.getTaxonomicNote());
196
197    gbif.setAuthorship(NameFormatter.authorString(pn.getCombinationAuthorship(), false));
198    gbif.setYear(pn.getCombinationAuthorship().getYear());
199    gbif.setBracketAuthorship(NameFormatter.authorString(pn.getBasionymAuthorship(), false));
200    gbif.setBracketYear(pn.getBasionymAuthorship().getYear());
201
202    gbif.setNomStatus(pn.getNomenclaturalNote());
203    if (pn.getEpithetQualifier() != null && !pn.getEpithetQualifier().isEmpty()) {
204      StringBuilder sb = new StringBuilder();
205      for (Map.Entry<NamePart, String> pq : pn.getEpithetQualifier().entrySet()) {
206        if (sb.length() < 1) {
207          sb.append(" ");
208        }
209        sb.append(pq.getValue())
210          .append(" ")
211          .append(pn.getEpithet(pq.getKey()));
212      }
213      gbif.setRemarks(sb.toString());
214    }
215
216    // we throw UnparsableException above already for State.NONE
217    gbif.setParsed(true);
218    gbif.setParsedPartially(pn.getState() == State.PARTIAL);
219
220    return gbif;
221  }

224  public static NameType gbifNameType(org.gbif.nameparser.api.ParsedName pn) {
225    NameType t;
226    // detect name types that only exist in the GBIF API v1
227    if (pn.isDoubtful() && pn.getWarnings().contains(Warnings.BLACKLISTED_EPITHET)) {
228      t = NameType.BLACKLISTED;
229    } else if (pn.isCandidatus()) {
230      t = NameType.CANDIDATUS;
231    } else if (pn.getCode() == NomCode.CULTIVARS || pn.getCultivarEpithet() != null) {
232      t = NameType.CULTIVAR;
233    } else {
234      // convert all others
235      t = NAME_TYPE_MAP.get(pn.getType());
236    }
237    // use doubtful in too good cases
238    if (pn.isDoubtful() && (t == NameType.SCIENTIFIC || t == NameType.CULTIVAR)) {
239      return NameType.DOUBTFUL;
240    }
241    return t;
242  }

244  public static org.gbif.api.vocabulary.NamePart toGbif(NamePart notho) {
245    return convertEnum(org.gbif.api.vocabulary.NamePart.class, notho);
246  }

248  public static Rank toGbif(org.gbif.nameparser.api.Rank rank) {
249    if (rank == null) {
250      return null;
251    }
252    switch (rank) {
253      case SUPERDIVISION: return Rank.SUPERPHYLUM;
254      case DIVISION: return Rank.PHYLUM;
255      case SUBDIVISION: return Rank.SUBPHYLUM;
256      case INFRADIVISION: return Rank.INFRAPHYLUM;
257
258      case SUPERSECTION: return Rank.INFRAGENERIC_NAME;
259      case SUPERSERIES: return Rank.INFRAGENERIC_NAME;
260      
261      case MEGAFAMILY: return Rank.SUPRAGENERIC_NAME;
262      case GRANDFAMILY: return Rank.SUPRAGENERIC_NAME;
263      case EPIFAMILY: return Rank.SUPRAGENERIC_NAME;
264
265      case GIGAORDER: return Rank.SUPRAGENERIC_NAME;
266      case MIRORDER: return Rank.SUPRAGENERIC_NAME;
267      case NANORDER: return Rank.SUPRAGENERIC_NAME;
268      case HYPOORDER: return Rank.SUPRAGENERIC_NAME;
269      case MINORDER: return Rank.SUPRAGENERIC_NAME;
270
271      case SUBTERCLASS: return Rank.SUPRAGENERIC_NAME;
272
273      case REALM: return Rank.SUPRAGENERIC_NAME;
274      case SUBREALM: return Rank.SUPRAGENERIC_NAME;
275
276      default: return convertEnum(Rank.class, rank);
277    }
278  }

281  public static org.gbif.nameparser.api.Rank fromGbif(Rank rank) {
282    if (rank == null) {
283      return null;
284    }
285    if (Rank.RACE == rank) {
286      return org.gbif.nameparser.api.Rank.PROLES;
287    }
288    return convertEnum(org.gbif.nameparser.api.Rank.class, rank);
289  }

291  /**
292   * Converts an enumeration value into a constant with the exact same name from a different enumeration class.
293   * In case the enumeration constant name does not exist an error is thrown.
294   *
295   * @param targetClass class of the target enumeration
296   * @param value
297   * @throws IllegalArgumentException in case the enumeration name does not exist in the target class
298   */
299  private static <G extends Enum<G>> G convertEnum(Class<G> targetClass, Enum<?> value) {
300    try {
301      return value == null ? null : Enum.valueOf(targetClass, value.name());
302    } catch (IllegalArgumentException e) {
303      LOG.warn("Unable to convert {} into {}", value, targetClass);
304      return null;
305    }
306  }
