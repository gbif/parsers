031  @Test
032  public void testParseLatLng() {
033    assertExpected(CoordinateParseUtils.parseLatLng("-46,33", "51,8717"), new LatLng(-46.33, 51.8717), ParseResult.CONFIDENCE.DEFINITE);
034    assertExpected(CoordinateParseUtils.parseLatLng("10.3", "99.99"), new LatLng(10.3, 99.99), ParseResult.CONFIDENCE.DEFINITE);
035    assertExpected(CoordinateParseUtils.parseLatLng("10", "10"), new LatLng(10, 10), ParseResult.CONFIDENCE.DEFINITE);
036    assertExpected(CoordinateParseUtils.parseLatLng("90", "180"), new LatLng(90, 180), ParseResult.CONFIDENCE.DEFINITE);
037    assertExpected(CoordinateParseUtils.parseLatLng("-90", "180"), new LatLng(-90, 180), ParseResult.CONFIDENCE.DEFINITE);
038    assertExpected(CoordinateParseUtils.parseLatLng("90", "-180"), new LatLng(90, -180), ParseResult.CONFIDENCE.DEFINITE);
039    assertExpected(CoordinateParseUtils.parseLatLng("-90", "-180"), new LatLng(-90, -180), ParseResult.CONFIDENCE.DEFINITE);
040    assertExpected(CoordinateParseUtils.parseLatLng("0", "0"), new LatLng(0, 0), ParseResult.CONFIDENCE.POSSIBLE, OccurrenceIssue.ZERO_COORDINATE);
041
042    // rounding
043    assertExpected(CoordinateParseUtils.parseLatLng("2.123450678", "-8.123450678"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
044    assertExpected(CoordinateParseUtils.parseLatLng("2.123451", "-8.123450678"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
045    assertExpected(CoordinateParseUtils.parseLatLng("2.123451", "-8.123451"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE);
046    assertExpected(CoordinateParseUtils.parseLatLng("2.12345100", "-8.1234510"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE);
047    assertExpected(CoordinateParseUtils.parseLatLng("2.123", "-8.1234506"), new LatLng(2.123, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
048
049    // degree minutes seconds
050    assertExpected(CoordinateParseUtils.parseLatLng("02° 49' 52\" N", "131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
051
052    // check swapped coordinates
053    assertExpected(CoordinateParseUtils.parseLatLng("100", "40"), new LatLng(40, 100), ParseResult.CONFIDENCE.PROBABLE, OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
054    assertExpected(CoordinateParseUtils.parseLatLng("-100", "90"), new LatLng(90, -100), ParseResult.CONFIDENCE.PROBABLE, OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
055
056    // check errors
057    assertFailed(CoordinateParseUtils.parseLatLng("", "30"));
058    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("tim", "tom"), OccurrenceIssue.COORDINATE_INVALID);
059    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20,432,12", "13,4"), OccurrenceIssue.COORDINATE_INVALID);
060    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "200"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
061    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("-200", "30"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
062    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "30"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
063    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20.432,12", "13,4"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
064    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20,432,12", "13,4"), OccurrenceIssue.COORDINATE_INVALID);
065  }

067  @Test
068  public void testParseDMS() {
069    assertDMS("2°49'N", "131°47'E", 2.816667d, 131.783333d);
070
071    assertDMS("02° 49' 52\" N", "131° 47' 03\" E", 2.831111d, 131.784167d);
072    assertDMS("2°49'52\"S", "131°47'03\" W", -2.831111d, -131.784167d);
073    assertDMS("2°49'52\"  n", "131°47'03\"  O", 2.831111d, 131.784167d);
074    assertDMS("002°49'52\"N", "131°47'03\"E", 2.831111d, 131.784167d);
075    assertDMS("2°49'N", "131°47'E", 2.816667d, 131.783333d);
076    assertDMS("002°49'52''N", "131°47'03''E", 2.831111d, 131.784167d);
077    assertDMS("6º39'36\"S","35º59'59\"W", -6.66d, -35.999722);
078    assertDMS("17  02.877 N", "121  05.966 E", 17.04795, 121.099433);
079    assertDMS("08º37'S", "37º10'W", -8.616667, -37.166667);
080    assertDMS("39g30mS", "56g27mW", -39.5, -56.45);
081    assertDMS("42g24m50.00sS", "64g17m20.00sW", -42.413889, -64.288889);
082    assertDMS("42º30´S", "54º14´W", -42.5, -54.233333);
083    assertDMS("61o50'N", "30o45'E", 61.833333, 30.75);
084    assertDMS("07°35N", "38°44E", 7.583333, 38.733333);
085    assertDMS("29º32.3’N", "113º34.9’W", 29.538333, -113.581667);
086    assertDMS("5°45′30″N", "100º30′30″W", 5.758333, -100.508333);
087    assertDMS("125º52´2´´S", "39º28´47´´W", -125.867222, -39.479722);
088    assertDMS("13.1939 N", "59.5432 W", 13.1939, -59.5432);
089    assertDMS("24 06.363 N", "110 11.969 E", 24.10605d, 110.199483d);
090    // even if it's out of range that's expected here - we validate elsewhere, this is an internal method only!
091    assertDMS("122°49'52\"N", "131°47'03\"E", 122.831111d, 131.784167d);
092    assertDMS("122°49'52.39\"N", "131°47'03.23\"E", 122.83121944444444d, 131.78423055555555d);
093    assertDMS("122d49m52.39sN", "131d47m03.23sE", 122.83121944444444d, 131.78423055555555d);
094    assertDMS("122 49 52.39", "131 47 03.23", 122.83121944444444d, 131.78423055555555d);
095    assertDMS("N122°49'52.39\"", "E131°47'03.23\"", 122.83121944444444d, 131.78423055555555d);
096    assertDMS("N 122° 49' 52.39\"", "E 131° 47' 03.23\"", 122.83121944444444d, 131.78423055555555d);
097    assertDMS("N 122°49'52.39\"", "E 131°47'03.23\"", 122.83121944444444d, 131.78423055555555d);
098    assertDMS("N 122°49'52\"", "E 131°47'03.23\"", 122.831111d, 131.78423055555555d);
099    assertDMS("N122°49'52.39\"", "E131°47'03.23\"", 122.83121944444444d, 131.78423055555555d);
100    assertDMS("N36.93276", "W102.96361", 36.93276, -102.96361);
101    assertDMS("90°N", "180°", 90, 180);
102
103    // truly failing
104    assertIllegalArg("12344", true);
105    assertIllegalArg("432", false);
106    assertIllegalArg(" ", true);
107    assertIllegalArg(" ", false);
108    assertIllegalArg("131°47'132\"", false);
109    assertIllegalArg("131 47 132", false);
110    assertIllegalArg("131°47'132\"", false);
111    assertIllegalArg("043300S",true);
112    assertIllegalArg("0433S",true);
113    assertIllegalArg("043300 S",true);
114    assertIllegalArg("0433 S",true);
115  }

117  private void assertDMS(String lat, String lon, double eLat, double eLon){
118    assertEquals(eLat, CoordinateParseUtils.parseDMS(lat, true), 0.000001);
119    assertEquals(eLon, CoordinateParseUtils.parseDMS(lon, false), 0.000001);
120  }

131  @Test
132  public void testParseVerbatimCoordinates() {
133    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02° 49' 52\" N 131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
134    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02° 49' 52\" N, 131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
135    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02°49'52\"N; 131°47'03\"O"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
136    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("17d 33m 5s N/99d 30m 3s W"), new LatLng(17.551389d, -99.500833d), ParseResult.CONFIDENCE.DEFINITE);
137    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("14.93333/-91.9"), new LatLng(14.93333d, -91.9d), ParseResult.CONFIDENCE.DEFINITE);
138    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("63d 41m 39s N 170d 28m 44s W"), new LatLng(63.694167d, -170.478889d), ParseResult.CONFIDENCE.DEFINITE);
139    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("37° 28' N, 122° 6' W"), new LatLng(37.466667d, -122.1d), ParseResult.CONFIDENCE.DEFINITE);
140    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("2°49'52\"N, 131°47'03\""), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
141    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10°07'06\"N 20°48'23\"W"), new LatLng(10.118333, -20.806389), ParseResult.CONFIDENCE.DEFINITE);
142    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10º07'06\"N 20º48'23\"W"), new LatLng(10.118333, -20.806389), ParseResult.CONFIDENCE.DEFINITE);
143    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10°07'N 20°48'W"), new LatLng(10.116667, -20.8), ParseResult.CONFIDENCE.DEFINITE);
144    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("10°07.55'N 20°48.55'W"), new LatLng(10.125833, -20.809167), ParseResult.CONFIDENCE.DEFINITE);
145    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("100º23'05\"N 20º35'25\"W"), new LatLng(-20.590278, 100.384722), ParseResult.CONFIDENCE.PROBABLE, OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
146
147    // failed
148    assertFailed(CoordinateParseUtils.parseVerbatimCoordinates(""));
149    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates("12344"), OccurrenceIssue.COORDINATE_INVALID);
150    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates(" "), OccurrenceIssue.COORDINATE_INVALID);
151    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates(",11.12"), OccurrenceIssue.COORDINATE_INVALID);
152    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates("122°49'52\"N, 131°47'03\"E"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
153  }

155  private void assertExpected(OccurrenceParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
156    assertNotNull(pr);
157    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
158    assertEquals(c, pr.getConfidence());
159    assertNotNull(pr.getPayload());
160    assertEquals(expected, pr.getPayload());
161    if (issue == null) {
162      assertTrue(pr.getIssues().isEmpty());
163    } else {
164      assertEquals(issue.length, pr.getIssues().size(), "Wrong number of issues. Found " + pr.getIssues() + ", expected " + Arrays.asList(issue));
165      for (OccurrenceIssue iss : issue) {
166        assertTrue(pr.getIssues().contains(iss));
167      }
168    }
169  }
170
171  private void assertFailed(ParseResult<?> pr) {
172    assertNotNull(pr);
173    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
174  }
175
176  private void assertFailedWithIssues(OccurrenceParseResult<LatLng> pr, OccurrenceIssue ... issue) {
177    assertFailed(pr);
178    assertEquals(issue.length, pr.getIssues().size());
179    for (OccurrenceIssue iss : issue){
180      assertTrue(pr.getIssues().contains(iss));
181    }
182  }
