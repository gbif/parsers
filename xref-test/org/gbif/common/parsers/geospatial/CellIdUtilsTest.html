40    @Test
41    public void testCreateCellId() {
42      try {
43        assertEquals(0, CellIdUtils.toCellId((double) -90, (double) -180));
44        assertEquals(0, CellIdUtils.toCellId((double) -90, -179.9));
45        assertEquals(359, CellIdUtils.toCellId(-89.1, 179.2));
46        assertEquals(360, CellIdUtils.toCellId(-88.5, (double) -180));
47        assertEquals(6647, CellIdUtils.toCellId(-71.31, -12.4233));
48      } catch (UnableToGenerateCellIdException e) {
49        fail(e.toString());
50      }
51      try {
52        CellIdUtils.toCellId((double) -100, null);
53        fail("-100, null should throw exception");
54      } catch (UnableToGenerateCellIdException ignored) {
55      }
56    }

314   @Test
315   public void testLL2Id2LL() {
316     try {
317       List<Integer> cells = new LinkedList<>(CellIdUtils.getCellsEnclosedBy(-30, -20, 110, 130));
318       LatLngBoundingBox bb = CellIdUtils.getBoundingBoxForCells(cells);
319       assertEquals(110, (int) bb.getMinLong());
320       assertEquals(-30, (int) bb.getMinLat());
