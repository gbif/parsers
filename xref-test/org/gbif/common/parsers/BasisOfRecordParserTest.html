36    private static BasisOfRecord getValue(BasisOfRecord basisOfRecord) {
37      return
38        // Literature is replaced with Material Citation
39        (BasisOfRecord.LITERATURE == basisOfRecord ? BasisOfRecord.MATERIAL_CITATION :
40          // Unknown is replaced with Occurrence
41          (BasisOfRecord.UNKNOWN == basisOfRecord? BasisOfRecord.OCCURRENCE  : basisOfRecord));
42    }
43  
44    /**
45     * This ensures that ALL enum values are at least parsable by the name they
46     * are created with.
47     */
48    @Test
49    public void testCompleteness() {
50      for (BasisOfRecord t : BasisOfRecord.values()) {
51        System.out.println("Testing [" + t.name()
52                           + "].  Failures below might indicate new BasisOfRecord added to the BasisOfRecord enum but not to the parse file");
53  
54        assertParseSuccess(getValue(t),
55                           getValue(t).name());
56      }
57    }
58  
59    @Test
60    public void testFailures() {
61      assertParseFailure(null);
62      assertParseFailure("");
63      assertParseFailure("Tim");
64    }
65  
66    @Test
67    public void testParse() {
68      // run a few basic tests to check it bootstraps and appears to work
69      assertParseSuccess(BasisOfRecord.LIVING_SPECIMEN, "living organism");
70      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "SpeCImen");
71      assertParseSuccess(BasisOfRecord.OBSERVATION, "sound recording");
72      assertParseSuccess(BasisOfRecord.OBSERVATION, "Observation");
73      assertParseSuccess(BasisOfRecord.OBSERVATION, "O");
74      assertParseSuccess(BasisOfRecord.HUMAN_OBSERVATION, "HumanObservation");
75      assertParseSuccess(BasisOfRecord.OCCURRENCE, "unknown");
76      assertParseSuccess(BasisOfRecord.OBSERVATION, "observation");
77      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "specimen");
78      assertParseSuccess(BasisOfRecord.LIVING_SPECIMEN, "living");
79      assertParseSuccess(BasisOfRecord.LIVING_SPECIMEN, "germplasm");
80      assertParseSuccess(BasisOfRecord.FOSSIL_SPECIMEN, "fossil");
81  
82      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "50 specimens");
83      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "1250 specimens");
84      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "72 specimens");
85      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "Espèce");
86      assertParseSuccess(BasisOfRecord.LIVING_SPECIMEN, "living, growing plant");
87      assertParseSuccess(BasisOfRecord.LIVING_SPECIMEN, "germplasm");
88      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "Alcohol / Microscope");
89      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "preservé 179600");
90      assertParseSuccess(BasisOfRecord.PRESERVED_SPECIMEN, "Pinned specimen");
91  
92      // Deprecated bases
93      assertParseSuccess(BasisOfRecord.MATERIAL_CITATION, "Literature");
94      assertParseSuccess(BasisOfRecord.OCCURRENCE, "Unknown");
95    }

97    /**
98     * Parse all unique basis of record values found in our index and make sure parsing doesn't get worse.
99     * If the test file is updated, values here need to be adjusted!
100    */
101   @Test
102   public void testOccurrenceValues() throws IOException {
103     final int CURRENT_TESTS_SUCCESSFUL = 248;
104     final int CURRENT_DISTINCT = 8;
105 
106     int failed = 0;
107     int success = 0;
108     Set<BasisOfRecord> values = new HashSet<>();
109 
110     BufferedReader r = new BufferedReader(new InputStreamReader(
111       getClass().getResourceAsStream("/parse/basisofrecord/test_bor.txt"), StandardCharsets.UTF_8));
112     String line;
113     while ((line=r.readLine()) != null) {
114       ParseResult<BasisOfRecord> parsed = parser.parse(line);
115       if (parsed.isSuccessful()) {
116         success++;
117         values.add(parsed.getPayload());
118       } else {
119         System.out.println("Failed: " + line);
120         failed++;
121       }
122     }
123 
124     System.out.println(failed + " failed parse results");
125     System.out.println(success + " successful parse results");
126     System.out.println(values.size() + " distinct basis of record values parsed");
127     assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
128     assertTrue(CURRENT_DISTINCT <= values.size());
129   }
