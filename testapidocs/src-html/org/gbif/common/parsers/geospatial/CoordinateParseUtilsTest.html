017  @Test
018  public void testParseLatLng() {
019    assertExpected(CoordinateParseUtils.parseLatLng("-46,33", "51,8717"), new LatLng(-46.33, 51.8717), ParseResult.CONFIDENCE.DEFINITE);
020    assertExpected(CoordinateParseUtils.parseLatLng("10.3", "99.99"), new LatLng(10.3, 99.99), ParseResult.CONFIDENCE.DEFINITE);
021    assertExpected(CoordinateParseUtils.parseLatLng("10", "10"), new LatLng(10, 10), ParseResult.CONFIDENCE.DEFINITE);
022    assertExpected(CoordinateParseUtils.parseLatLng("90", "180"), new LatLng(90, 180), ParseResult.CONFIDENCE.DEFINITE);
023    assertExpected(CoordinateParseUtils.parseLatLng("-90", "180"), new LatLng(-90, 180), ParseResult.CONFIDENCE.DEFINITE);
024    assertExpected(CoordinateParseUtils.parseLatLng("90", "-180"), new LatLng(90, -180), ParseResult.CONFIDENCE.DEFINITE);
025    assertExpected(CoordinateParseUtils.parseLatLng("-90", "-180"), new LatLng(-90, -180), ParseResult.CONFIDENCE.DEFINITE);
026    assertExpected(CoordinateParseUtils.parseLatLng("0", "0"), new LatLng(0, 0), ParseResult.CONFIDENCE.POSSIBLE, OccurrenceIssue.ZERO_COORDINATE);
027
028    // rounding
029    assertExpected(CoordinateParseUtils.parseLatLng("2.123450678", "-8.123450678"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
030    assertExpected(CoordinateParseUtils.parseLatLng("2.123451", "-8.123450678"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
031    assertExpected(CoordinateParseUtils.parseLatLng("2.123451", "-8.123451"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE);
032    assertExpected(CoordinateParseUtils.parseLatLng("2.12345100", "-8.1234510"), new LatLng(2.123451, -8.123451), ParseResult.CONFIDENCE.DEFINITE);
033    assertExpected(CoordinateParseUtils.parseLatLng("2.123", "-8.1234506"), new LatLng(2.123, -8.123451), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.COORDINATE_ROUNDED);
034
035    // degree minutes seconds
036    assertExpected(CoordinateParseUtils.parseLatLng("02° 49' 52\" N", "131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
037
038    // check swapped coords
039    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("100", "40"), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
040    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("-100", "90"), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
041
042    // check errors
043    assertFailed(CoordinateParseUtils.parseLatLng("", "30"));
044    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("tim", "tom"), OccurrenceIssue.COORDINATE_INVALID);
045    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20,432,12", "13,4"), OccurrenceIssue.COORDINATE_INVALID);
046    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "200"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
047    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("-200", "30"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
048    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "30"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
049    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20.432,12", "13,4"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
050    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("20,432,12", "13,4"), OccurrenceIssue.COORDINATE_INVALID);
051  }

053  @Test
054  public void testParseDMS() {
055    assertDMS("2°49'N", "131°47'E", 2.816667d, 131.783333d);
056
057    assertDMS("02° 49' 52\" N", "131° 47' 03\" E", 2.831111d, 131.784167d);
058    assertDMS("2°49'52\"S", "131°47'03\" W", -2.831111d, -131.784167d);
059    assertDMS("2°49'52\"  n", "131°47'03\"  O", 2.831111d, 131.784167d);
060    assertDMS("002°49'52\"N", "131°47'03\"E", 2.831111d, 131.784167d);
061    assertDMS("2°49'N", "131°47'E", 2.816667d, 131.783333d);
062    assertDMS("002°49'52''N", "131°47'03''E", 2.831111d, 131.784167d);
063    // even if its out of range thats expected here - we validate elsewhere, this is an internal method only!
064    assertDMS("122°49'52\"N", "131°47'03\"E", 122.831111d, 131.784167d);
065    assertDMS("122°49'52.39\"N", "131°47'03.23\"E", 122.83121944444444d, 131.78423055555555d);
066    assertDMS("122d49m52.39sN", "131d47m03.23sE", 122.83121944444444d, 131.78423055555555d);
067    assertDMS("122 49 52.39", "131 47 03.23", 122.83121944444444d, 131.78423055555555d);
068    assertDMS("N122°49'52.39\"", "E131°47'03.23\"", 122.83121944444444d, 131.78423055555555d);
069    assertDMS("N 122° 49' 52.39\"", "E 131° 47' 03.23\"", 122.83121944444444d, 131.78423055555555d);
070
071    // truly failing
072    assertIllegalArg("12344", true);
073    assertIllegalArg("432", false);
074    assertIllegalArg(" ", true);
075    assertIllegalArg(" ", false);
076    assertIllegalArg("131°47'132\"", false);
077    assertIllegalArg("131 47 132", false);
078    assertIllegalArg("131°47'132\"", false);
079  }

081  private void assertDMS(String lat, String lon, double eLat, double eLon){
082    assertEquals(eLat, CoordinateParseUtils.parseDMS(lat, true), 0.000001);
083    assertEquals(eLon, CoordinateParseUtils.parseDMS(lon, false), 0.000001);
084  }

095  @Test
096  public void testParseVerbatimCoordinates() {
097    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02° 49' 52\" N 131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
098    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02° 49' 52\" N, 131° 47' 03\" E"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
099    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("02°49'52\"N; 131°47'03\"O"), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
100    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("17d 33m 5s N/99d 30m 3s W"), new LatLng(17.551389d, -99.500833d), ParseResult.CONFIDENCE.DEFINITE);
101    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("14.93333/-91.9"), new LatLng(14.93333d, -91.9d), ParseResult.CONFIDENCE.DEFINITE);
102    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("63d 41m 39s N 170d 28m 44s W"), new LatLng(63.694167d, -170.478889d), ParseResult.CONFIDENCE.DEFINITE);
103    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("37° 28' N, 122° 6' W"), new LatLng(37.466667d, -122.1d), ParseResult.CONFIDENCE.DEFINITE);
104    assertExpected( CoordinateParseUtils.parseVerbatimCoordinates("2°49'52\"N, 131°47'03\""), new LatLng(2.831111d, 131.784167d), ParseResult.CONFIDENCE.DEFINITE);
105    // failed
106    assertFailed(CoordinateParseUtils.parseVerbatimCoordinates(""));
107    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates("12344"), OccurrenceIssue.COORDINATE_INVALID);
108    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates(" "), OccurrenceIssue.COORDINATE_INVALID);
109    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates(",11.12"), OccurrenceIssue.COORDINATE_INVALID);
110    assertFailedWithIssues(CoordinateParseUtils.parseVerbatimCoordinates("122°49'52\"N, 131°47'03\"E"), OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
111  }

113  private void assertExpected(LatLng result, Double lat, Double lon) {
114    assertNotNull(result);
115    assertEquals("Latitudedifferent", lat, result.getLat());
116    assertEquals("Longitude different", lon, result.getLng());
117  }
118
119  private void assertExpected(OccurrenceParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
120    assertNotNull(pr);
121    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
122    assertEquals(c, pr.getConfidence());
123    assertNotNull(pr.getPayload());
124    assertEquals(expected, pr.getPayload());
125    //System.out.println(Lists.newArrayList(issue));
126    if (issue == null) {
127      assertTrue(pr.getIssues().isEmpty());
128    } else {
129      assertEquals("Wrong number of issues. Found " + pr.getIssues() + ", expected " + Lists.newArrayList(issue), issue.length, pr.getIssues().size());
130      for (OccurrenceIssue iss : issue) {
131        assertTrue(pr.getIssues().contains(iss));
132      }
133    }
134  }
135
136  private void assertFailed(ParseResult<?> pr) {
137    assertNotNull(pr);
138    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
139  }
140
141  private void assertFailedWithIssues(OccurrenceParseResult<LatLng> pr, OccurrenceIssue ... issue) {
142    assertFailed(pr);
143    assertEquals(issue.length, pr.getIssues().size());
144    for (OccurrenceIssue iss : issue){
145      assertTrue(pr.getIssues().contains(iss));
146    }
147  }
