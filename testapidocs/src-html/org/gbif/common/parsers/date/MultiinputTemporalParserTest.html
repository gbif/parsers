031  @Test
032  public void testIsValidDate() {
033    assertTrue(MultiinputTemporalParser.isValidDate(Year.of(2005)));
034    assertTrue(MultiinputTemporalParser.isValidDate(YearMonth.of(2005, 1)));
035    assertTrue(MultiinputTemporalParser.isValidDate(LocalDate.of(2005, 1, 1)));
036    assertTrue(MultiinputTemporalParser.isValidDate(LocalDateTime.of(2005, 1, 1, 2, 3, 4)));
037    assertTrue(MultiinputTemporalParser.isValidDate(LocalDate.now()));
038    assertTrue(MultiinputTemporalParser.isValidDate(LocalDateTime.now().plus(23, ChronoUnit.HOURS)));
039
040    // Dates out of bounds
041    assertFalse(MultiinputTemporalParser.isValidDate(YearMonth.of(1499, 12)));
042
043    // we tolerate an offset of 1 day
044    assertFalse(MultiinputTemporalParser.isValidDate(LocalDate.now().plusDays(2)));
045  }

047  @Test
048  public void testParseRecordedDate() {
049    MultiinputTemporalParser temporalParser = MultiinputTemporalParser.create();
050    OccurrenceParseResult<TemporalAccessor> result;
051
052    result = temporalParser.parseRecordedDate("2005", "1", "1", "2005-01-01");
053    assertResult(2005, 1, 1, result);
054    assertEquals(0, result.getIssues().size());
055
056    // ensure that eventDate with more precision will not record an issue and the one with most
057    // precision will be returned
058    result = temporalParser.parseRecordedDate("1996", "1", "26", "1996-01-26T01:00Z");
059    assertEquals(
060        ZonedDateTime.of(LocalDateTime.of(1996, 1, 26, 1, 0), ZoneId.of("Z")), result.getPayload());
061    assertEquals(0, result.getIssues().size());
062
063    // if dates contradict, return the parts that agree
064    result = temporalParser.parseRecordedDate("2005", "1", "2", "2005-01-05");
065    assertResult(2005, 1, result);
066    assertEquals(1, result.getIssues().size());
067    assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
068  }

182  @Test
183  public void testYearWithZeros() {
184    // providing 0 will cause a RECORDED_DATE_MISMATCH since 0 could be null but also January
185    OccurrenceParseResult<TemporalAccessor> result =
186        MultiinputTemporalParser.create().parseRecordedDate("1984", "0", "0", "1984");
187    assertResult(1984, result);
188    assertEquals(ParseResult.CONFIDENCE.PROBABLE, result.getConfidence());
189    assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
190
191    result = MultiinputTemporalParser.create().parseRecordedDate("1984", "0", "0", null);
192    assertResult(1984, result);
193    assertEquals(ParseResult.CONFIDENCE.PROBABLE, result.getConfidence());
194    assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
195
196    result = MultiinputTemporalParser.create().parseRecordedDate(null, null, null, "0-0-1984");
197    assertEquals(ParseResult.STATUS.FAIL, result.getStatus());
198    assertNullPayload(result, OccurrenceIssue.RECORDED_DATE_INVALID);
199
200    result = MultiinputTemporalParser.create().parseRecordedDate(null, null, null, "1984");
201    assertEquals(ParseResult.CONFIDENCE.DEFINITE, result.getConfidence());
202    assertTrue(result.getIssues().isEmpty());

363   * Tests that a date representing 'now' is interpreted with CONFIDENCE.DEFINITE even after
364   * v1TemporalInterpreter was instantiated. See POR-2860.
365   */
366  @Test
367  public void testNow() {
368
369    MultiinputTemporalParser temporalParser = MultiinputTemporalParser.create();
370
371    // Makes sure the static content is loaded
372    OccurrenceParseResult<TemporalAccessor> result =
373        temporalParser.parseRecordedDate(
374            null, null, null, DateFormatUtils.ISO_DATETIME_FORMAT.format(Calendar.getInstance()));
375    assertEquals(ParseResult.CONFIDENCE.DEFINITE, result.getConfidence());
376
377    // Sorry for this Thread.sleep, we need to run the v1TemporalInterpreter at least 1 second later
378    // until
379    // we refactor to inject a Calendar or we move to new Java 8 Date/Time API
380    try {
381      TimeUnit.MILLISECONDS.sleep(1_000);
382    } catch (InterruptedException e) {
383      fail(e.getMessage());
384    }
385
386    Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
387    result =
388        temporalParser.parseRecordedDate(
389            null, null, null, DateFormatUtils.ISO_DATETIME_FORMAT.format(cal.getTime()));
390    assertEquals(ParseResult.CONFIDENCE.DEFINITE, result.getConfidence());
391  }

408  private void assertInts(Integer expected, Integer x) {
409    if (expected == null) {
410      assertNull(x);
411    } else {
412      assertEquals(expected, x);
413    }
414  }

416  /**
417   * Utility method to assert a ParseResult when a LocalDate is expected. This method should not be
418   * used to test expected null results.
419   */
420  private void assertResult(Integer y, Integer m, Integer d, ParseResult<TemporalAccessor> result) {
421    // sanity checks
422    assertNotNull(result);
423
424    LocalDate localDate = result.getPayload().query(LocalDate::from);
425    assertInts(y, localDate.getYear());
426    assertInts(m, localDate.getMonthValue());
427    assertInts(d, localDate.getDayOfMonth());
428
429    assertEquals(LocalDate.of(y, m, d), result.getPayload());
430  }

432  private void assertResult(TemporalAccessor expectedTA, ParseResult<TemporalAccessor> result) {
433    assertEquals(expectedTA, result.getPayload());
434  }

436  /**
437   * Utility method to assert a ParseResult when a YearMonth is expected. This method should not be
438   * used to test expected null results.
439   */
440  private void assertResult(Integer y, Integer m, ParseResult<TemporalAccessor> result) {
441    // sanity checks
442    assertNotNull(result);
443
444    YearMonth yearMonthDate = result.getPayload().query(YearMonth::from);
445    assertInts(y, yearMonthDate.getYear());
446    assertInts(m, yearMonthDate.getMonthValue());
447
448    assertEquals(YearMonth.of(y, m), result.getPayload());
449  }
450
451  private void assertResult(Integer y, ParseResult<TemporalAccessor> result) {
452    // sanity checks
453    assertNotNull(result);
454
455    Year yearDate = result.getPayload().query(Year::from);
456    assertInts(y, yearDate.getValue());
457
458    assertEquals(Year.of(y), result.getPayload());
459  }

461  private void assertNullPayload(
462      OccurrenceParseResult<TemporalAccessor> result, OccurrenceIssue expectedIssue) {
463    assertNotNull(result);
464    assertFalse(result.isSuccessful());
465    assertNull(result.getPayload());
466
467    if (expectedIssue != null) {
468      assertTrue(result.getIssues().contains(expectedIssue));
469    }
470  }
