001package org.gbif.common.parsers;
002
003import org.apache.commons.lang3.StringUtils;
004
005import java.text.DecimalFormat;
006import java.text.NumberFormat;
007import java.text.ParsePosition;
008import java.util.Locale;
009
010/**
011 * Utils class to parse numbers trying various locales so that dots and comma based formats are both supported.
012 * All methods swallow exceptions and return null instead.
013 */
014public class NumberParser {
015
016  private NumberParser() {
017
018  }
019
020  public static Double parseDouble(String x) {
021    final String trimmed = x == null ? null : x.trim();
022    if (StringUtils.isEmpty(trimmed)) return null;
023
024    try {
025      return Double.parseDouble(x);
026    } catch (NumberFormatException e) {
027      NumberFormat format = DecimalFormat.getInstance(Locale.GERMANY);
028      ParsePosition pos = new ParsePosition(0);
029      // verify that the parsed position is at the end of the string!
030      Number num = format.parse(trimmed, pos);
031      if (num != null && trimmed.length() == pos.getIndex()) {
032        return num.doubleValue();
033      }
034    }
035    return null;
036  }
037
038  public static Integer parseInteger(String x) {
039    final String trimmed = x == null ? null : x.trim();
040    if (StringUtils.isEmpty(trimmed)) return null;
041
042    try {
043      return Integer.parseInt(x);
044    } catch (NumberFormatException e) {
045      NumberFormat format = DecimalFormat.getInstance(Locale.GERMANY);
046      ParsePosition pos = new ParsePosition(0);
047      // verify that the parsed position is at the end of the string!
048      Number num = format.parse(trimmed, pos);
049      if (num != null && trimmed.length() == pos.getIndex()) {
050        return num.intValue();
051      }
052    }
053    return null;
054  }
055}
