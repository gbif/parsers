1   package org.gbif.common.parsers.utils;
2   
3   import java.util.regex.Pattern;
4   import lombok.AccessLevel;
5   import lombok.NoArgsConstructor;
6   
7   /**
8    * Contains methods which can delimit raw string into raw array, example: "10-10-2010" to {"10",
9    * "10", "2010"} and etc.
10   */
11  @NoArgsConstructor(access = AccessLevel.PRIVATE)
12  public class DelimiterUtils {
13    // Date values
14    private static final String CHAR_PERIOD = "/";
15    private static final Pattern RGX_PERIOD = Pattern.compile(CHAR_PERIOD);
16  
17    //  1990-01-02/15, and none-ISO: 1990-1-2, but does not support 19000101/12
18    private static final String ISO_YMD_RANGE =
19        "^([0-9]{4})-(1[0-2]|0[1-9]|[1-9])-(3[01]|0[1-9]|[12][0-9]|[1-9])/(3[01]|0[1-9]|[12][0-9]|[1-9])";
20    // 1990-01/12, and none-ISO: 1990-1/2
21    private static final String ISO_YM_RANGE =
22        "^([0-9]{4})-(1[0-2]|0[1-9]|[1-9])/(1[0-2]|0[1-9]|[1-9])";
23    //  1990-01-02/09-15, and none-ISO: 1990-1-2/9-15
24    private static final String ISO_YMD_MD_RANGE =
25        "^([0-9]{4})-(1[0-2]|0[1-9]|[1-9])-(3[01]|0[1-9]|[12][0-9]|[1-9])/(1[0-2]|0[1-9]|[1-9])-(3[01]|0[1-9]|[12][0-9]|[1-9])";
26  
27    /**
28     * Attempt to split the rawPeriod into raw periods by delimiter '/' symbol, the symbol must be the
29     * only one in the rawPeriod
30     *
31     * @param rawPeriod raw string period dates
32     * @return always two elements array
33     */
34    public static String[] splitPeriod(String rawPeriod) {
35      if (rawPeriod == null || rawPeriod.isEmpty()) {
36        return new String[] {"", ""};
37      }
38      // todo can be improved by adding more date patterns
39      String[] isoRange = splitISODateRange(rawPeriod);
40      if (isoRange != null) {
41        return isoRange;
42      }
43      // If the delimiter for the first position and last position are the same, symbol only one in
44      // string,
45      // check length, the length must be greater than seven to avoid case as "1999/2", where it looks
46      // as year and month
47      boolean canSplit =
48          rawPeriod.lastIndexOf(CHAR_PERIOD) == rawPeriod.indexOf(CHAR_PERIOD)
49              && rawPeriod.length() > 7;
50      String[] splitted =
51          canSplit ? RGX_PERIOD.split(rawPeriod) : new String[] {rawPeriod, rawPeriod};
52      // Returns an array of the same length each time
53      return splitted.length < 2 ? new String[] {splitted[0], splitted[0]} : splitted;
54    }
55  
56    /** Handle day/month ranges Examples: 1991-1/3 : 1991-1, 1991-3 1991-1-2/5 : 1991-1-1, 1991-1-5 */
57    public static String[] splitISODateRange(String src) {
58      if (src.matches(ISO_YMD_RANGE) || src.matches(ISO_YM_RANGE)) {
59        String[] periods = src.split(CHAR_PERIOD);
60        String from = periods[0];
61        String to = periods[1];
62        String ym = from.substring(0, from.lastIndexOf('-')); // Get YM
63        to = ym + "-" + to;
64        return new String[] {from, to};
65      } else if (src.matches(ISO_YMD_MD_RANGE)) {
66        // 1990-01-02/09-15
67        String[] periods = src.split(CHAR_PERIOD);
68        String from = periods[0];
69        String to = periods[1];
70        String y = from.substring(0, from.indexOf('-')); // Get Y
71        to = y + "-" + to;
72        return new String[] {from, to};
73      }
74      return null;
75    }
76  }
