024public class TemporalAccessorUtilsTest {
025
026  private ZoneId EUROPE_CENTRAL_TIME = ZoneId.of("Europe/Paris");
027
028  // January 1st 2000, 00:00:00 (midnight)
029  private long YEAR2000_1JAN_EPOCH_UTC = 946684800000l;
030
031  // January 1st 2000, 02:03:04
032  private long YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC = 946692184000l;
033
034  // February 1st 2000, midnight
035  private long YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC = 949363200000l;
036
037  // January 1st 2000, midnight
038  private long YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC = 946684800000l;
039
040  @Test
041  public void testToDate(){
042    TemporalAccessor ta = LocalDate.of(2000,01,01);
043    Date date = TemporalAccessorUtils.toDate(ta, false);
044    assertEquals(YEAR2000_1JAN_EPOCH_UTC, date.getTime());
045
046    ta = LocalDateTime.of(2000, 01, 01, 2, 3, 4);
047    date = TemporalAccessorUtils.toDate(ta, false);
048    assertEquals(YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC, date.getTime());
049
050    //test partial date transformed to Date
051    //YearMonth should use the first of the month
052    ta = YearMonth.of(2000, 2);
053    date = TemporalAccessorUtils.toDate(ta, false);
054    assertEquals(YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC, date.getTime());
055
056    //Year should use the first of January
057    ta = Year.of(2000);
058    date = TemporalAccessorUtils.toDate(ta, false);
059    assertEquals(YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC, date.getTime());
060  }
061
062  @Test
063  public void testToDateWithTimeZone(){
064
065    // Test with ZonedDateTime
066    ZonedDateTime testZonedDateTime = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, EUROPE_CENTRAL_TIME);
067    Date date = TemporalAccessorUtils.toDate(testZonedDateTime, false);
068    assertEquals(testZonedDateTime.toEpochSecond(), date.getTime()/1000);
069
070    // Test the same ZonedDateTime but ignore Offset information (timezone)
071    ZonedDateTime testTimeUTC = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, TemporalAccessorUtils.UTC_ZONE_ID);
072    date = TemporalAccessorUtils.toDate(testZonedDateTime, true);
073    assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
074
075    // Ensure it is also working with OffsetDateTime
076    OffsetDateTime testOffsetDateTimeTime = testZonedDateTime.toOffsetDateTime();
077    date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, false);
078    assertEquals(testOffsetDateTimeTime.toEpochSecond(), date.getTime()/1000);
079
080    date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, true);
081    assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
082  }
083
084  @Test
085  public void testBestResolution(){
086    TemporalAccessor ta1 = Year.of(2005);
087    TemporalAccessor ta2 = YearMonth.of(2005, 1);
088    TemporalAccessor result = TemporalAccessorUtils.bestResolution(ta1, ta2).get();
089    assertEquals(YearMonth.of(2005, 1), YearMonth.from(result));
090
091    ta1 = LocalDate.of(2005, 1, 1);
092    ta2 = Year.of(2005);
093    result = TemporalAccessorUtils.bestResolution(ta1, ta2).get();
094    assertEquals(LocalDate.of(2005, 1, 1), LocalDate.from(result));
095
096    //this should not work
097    ta1 = Year.of(2005);
098    ta2 = YearMonth.of(2006, 1);
099    assertFalse(TemporalAccessorUtils.bestResolution(ta1, ta2).isPresent());
100  }
101
102  @Test
103  public void testSameOrContained() {
104    TemporalAccessor ymd = LocalDate.of(1996, 04, 26);
105
106    // Test nulls and wrong resolutions
107    assertFalse(TemporalAccessorUtils.sameOrContained(null, null));
108    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, null));
109    assertFalse(TemporalAccessorUtils.sameOrContained(null, ymd));
110
111    // Test against same year
112    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, Year.of(1996)));
113    // Test against same month
114    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, YearMonth.of(1996, 4)));
115    // Test against same date
116    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, LocalDate.of(1996, 04, 26)));
117    // Test against time on that day
118    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, LocalDateTime.of(1996, 04, 26, 1, 2, 3)));
119
120    // Then different year/month/date
121    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, Year.of(1998)));
122    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, YearMonth.of(1996, 5)));
123    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, LocalDate.of(1996, 4, 27)));
124  }

126  @Test
127  public void testLastDay(){
128    TemporalAccessor year = Year.of(1996);
129    assertEquals("1996-12-31T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(year,true).toString());
130
131    TemporalAccessor ym = YearMonth.of(1996,1);
132    assertEquals("1996-01-31T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(ym,true).toString());
133
134    ym = YearMonth.of(1996,2);
135    assertEquals("1996-02-29T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(ym,true).toString());
136
137    TemporalAccessor ymd = LocalDate.of(1996,2,3);
138    assertEquals("1996-02-03T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(ymd,true).toString());
139
140    TemporalAccessor ymdt = LocalDateTime.of(1996,2,3,1,20);
141    assertEquals("1996-02-03T01:20", TemporalAccessorUtils.toLatestLocalDateTime(ymdt,true).toString());
142  }
