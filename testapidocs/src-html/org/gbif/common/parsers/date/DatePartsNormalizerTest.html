001package org.gbif.common.parsers.date;
002
003import java.time.Month;
004
005import org.junit.jupiter.api.Test;
006
007import static org.junit.jupiter.api.Assertions.assertEquals;
008import static org.junit.jupiter.api.Assertions.assertFalse;
009import static org.junit.jupiter.api.Assertions.assertNotNull;
010import static org.junit.jupiter.api.Assertions.assertNull;
011import static org.junit.jupiter.api.Assertions.assertTrue;
012
013/**
014 * Tests related to {@link DatePartsNormalizer}
015 */
016public class DatePartsNormalizerTest {
017
018  private static DatePartsNormalizer NORMALIZER = DatePartsNormalizer.newInstance();
019
020  /**
021   * This happens when integer are exported as float/double by a data publisher.
022   */
023  @Test
024  public void testFloatNormalization(){
025    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975.0", "2.0", "1.0");
026    assertEquals(new Integer(1), result.getDay());
027    assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
028    assertEquals(new Integer(1975), result.getYear());
029  }
030
031  @Test
032  public void testNormalizationWithSpaces(){
033    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975 ", " 2 ", " 1");
034    assertEquals(new Integer(1), result.getDay());
035    assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
036    assertEquals(new Integer(1975), result.getYear());
037  }
038
039  @Test
040  public void testMonthNameNormalization(){
041    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("January").intValue());
042    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JANUARY").intValue());
043    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("Jan").intValue());
044    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JAN.").intValue());
045    assertEquals(Month.DECEMBER.getValue(), NORMALIZER.monthNameToNumerical("DEC").intValue());
046    assertEquals(Month.APRIL.getValue(), NORMALIZER.monthNameToNumerical("April").intValue());
047    assertEquals(Month.SEPTEMBER.getValue(), NORMALIZER.monthNameToNumerical("Sept").intValue());
048    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("June").intValue());
049    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("Jun").intValue());
050    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("JUNE").intValue());
051    assertEquals(Month.NOVEMBER.getValue(), NORMALIZER.monthNameToNumerical("November").intValue());
052  }
053
054  @Test
055  public void testDiscardedDateParts(){
056
057    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("a", "5", "3");
058    assertTrue(result.yDiscarded());
059    assertNull(result.getYear());
060    assertNotNull(result.getMonth());
061
062    result = NORMALIZER.normalize("2000", "a", "3");
063    assertTrue(result.mDiscarded());
064    assertNull(result.getMonth());
065    assertNotNull(result.getYear());
066
067    result = NORMALIZER.normalize("2000", "5", "a");
068    assertTrue(result.dDiscarded());
069    assertNull(result.getDay());
070    assertNotNull(result.getYear());
071
072    // empty String, null and \\N should NOT be considered as "discarded"
073    // they simply represent "not provided"
074    result = NORMALIZER.normalize("2000", "5", "");
075    assertFalse(result.dDiscarded());
076    assertNull(result.getDay());
077
078    result = NORMALIZER.normalize("2000", "5", null);
079    assertFalse(result.dDiscarded());
080    assertNull(result.getDay());
081
082    result = NORMALIZER.normalize("2000", "5", "\\N");
083    assertFalse(result.dDiscarded());
084    assertNull(result.getDay());
085  }
086}
