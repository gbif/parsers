016package org.gbif.common.parsers.core;
017
018import java.util.ArrayList;
019import java.util.List;
020
021import org.junit.jupiter.api.Test;
022
023import static org.junit.jupiter.api.Assertions.assertEquals;
024import static org.junit.jupiter.api.Assertions.assertNotNull;
025
026public class DictionaryBackedParserTest {
027
028  @Test
029  public void testParse() {
030    List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
031    source.add(new KeyValue<String, Integer>("Tim", 32));
032    source.add(new KeyValue<String, Integer>("Markus", 38));
033    source.add(new KeyValue<String, Integer>("Jose", 28));
034    DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(false);
035    dbp.init(source.iterator());
036
037    assertParsed(dbp, "Tim", 32);
038    assertParsed(dbp, "TIM", 32);
039    assertParsed(dbp, "tIm", 32);
040    assertParsed(dbp, " \t Markus \t ", 38);
041    assertParsed(dbp, "MarKUS", 38);
042    assertParsed(dbp, "Jose", 28);
043    assertEquals(ParseResult.STATUS.FAIL, dbp.parse("Lars").getStatus());
044  }
045
046  @Test
047  public void testSensitiveParse() {
048    List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
049    source.add(new KeyValue<String, Integer>("Matt", 29));
050    DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(true);
051    dbp.init(source.iterator());
052
053    assertParsed(dbp, "Matt", 29);
054    assertParsed(dbp, " \t Matt \n ", 29);
055    assertEquals(ParseResult.STATUS.FAIL, dbp.parse("MATT").getStatus());
056  }
057
058  private void assertParsed(DictionaryBackedParser<Integer> dbp, String input, Integer payload) {
059    assertNotNull(dbp.parse(input));
060    assertEquals(ParseResult.STATUS.SUCCESS, dbp.parse(input).getStatus());
061    assertEquals(ParseResult.CONFIDENCE.DEFINITE, dbp.parse(input).getConfidence());
062    assertEquals(payload, dbp.parse(input).getPayload());
063  }
064
065}
