001package org.gbif.common.parsers.date;
002
003import java.time.Duration;
004import java.time.temporal.ChronoField;
005import java.time.temporal.TemporalAccessor;
006
007import com.google.common.base.Objects;
008import org.apache.commons.lang3.builder.HashCodeBuilder;
009import org.apache.commons.lang3.builder.ToStringBuilder;
010import org.apache.commons.lang3.builder.ToStringStyle;
011
012/**
013 * Still experimental, the class should probably require to have a LocalDate otherwise the output
014 * of getResolution can not be comparable with {@link AtomizedLocalDate}
015 *
016 * AtomizedLocalDateTime is a simple immutable class to hold local date and time data from a
017 * {@link TemporalAccessor}.
018 *
019 *
020 * Thread-Safe, immutable class.
021 */
022public class AtomizedLocalDateTime {
023
024  private final AtomizedLocalDate localDate;
025  private final Integer hour;
026  private final Integer minute;
027  private final Integer second;
028  private final Integer millisecond;
029
030  private final int resolution;
031
032
033  private AtomizedLocalDateTime(AtomizedLocalDate localDate, Integer hour, Integer minute,
034                                Integer second, Integer millisecond) {
035    int res = 0;
036    this.localDate = localDate;
037    this.hour = hour;
038    this.minute = minute;
039    this.second = second;
040    this.millisecond = millisecond;
041
042    // compute resolution
043    if(localDate != null){
044      res += localDate.getResolution();
045    }
046
047    if(hour != null){
048      res ++;
049    }
050    if(minute != null){
051      res++;
052    }
053    if(second != null){
054      res++;
055    }
056    if(millisecond != null){
057      res++;
058    }
059    this.resolution = res;
060  }
061
062  public Integer getYear() {
063    if(localDate == null){
064      return null;
065    }
066    return localDate.getYear();
067  }
068
069  public Integer getMonth() {
070    if(localDate == null){
071      return null;
072    }
073    return localDate.getMonth();
074  }
075
076  public Integer getDay() {
077    if(localDate == null){
078      return null;
079    }
080    return localDate.getDay();
081  }
082
083  public Integer getHour() {
084    return hour;
085  }
086
087  public Integer getMinute() {
088    return minute;
089  }
090
091  public Integer getSecond() {
092    return second;
093  }
094
095  public Integer getMillisecond() {
096    return millisecond;
097  }
098
099  /**
100   * Get the resolution of the {@link AtomizedLocalDateTime}.
101   * Resolution represents the number of parts this date/time contains.
102   * Be aware that it is not possible to express partial time which means that the lowest
103   * resolution possible if 4 (Time with no date)
104   * @return
105   */
106  public int getResolution(){
107    return resolution;
108  }
109
110  /**
111   * Build a new instance of {@link AtomizedLocalDateTime} based on a {@link TemporalAccessor}.
112   * This is done by extracting the {@link ChronoField}.
113   * Please note this expected behavior: if a least 1 time component is available on the TemporalAccessor, the other
114   * time components will return 0 instead of null if they are not provided.
115   *
116   * AtomizedLocalDateTime.fromTemporalAccessor(LocalTime.of(HOUR, MINUTE) will return 0 as second and millisecond and
117   * the resolution will be 4.
118   */
119  public static AtomizedLocalDateTime fromTemporalAccessor(TemporalAccessor temporalAccessor) {
120    if (temporalAccessor == null) {
121      return null;
122    }
123
124    //TODO if localDate == null, we should probably return null. If we want to handle time only
125    //we should create AtomizedLocalTime
126
127    AtomizedLocalDate localDate = AtomizedLocalDate.fromTemporalAccessor(temporalAccessor);
128
129
130    Integer h = null, m = null, s = null, ms = null;
131    if (temporalAccessor.isSupported(ChronoField.HOUR_OF_DAY)) {
132      h = temporalAccessor.get(ChronoField.HOUR_OF_DAY);
133    }
134
135    if (temporalAccessor.isSupported(ChronoField.MINUTE_OF_HOUR)) {
136      m = temporalAccessor.get(ChronoField.MINUTE_OF_HOUR);
137    }
138
139    if (temporalAccessor.isSupported(ChronoField.SECOND_OF_MINUTE)) {
140      s = temporalAccessor.get(ChronoField.SECOND_OF_MINUTE);
141    }
142
143    if (temporalAccessor.isSupported(ChronoField.MILLI_OF_SECOND)) {
144      ms = temporalAccessor.get(ChronoField.MILLI_OF_SECOND);
145    } else if (temporalAccessor.isSupported(ChronoField.NANO_OF_SECOND)) {
146      ms = Long.valueOf(Duration.ofNanos(temporalAccessor.get(ChronoField.NANO_OF_SECOND)).toMillis()).intValue();
147    }
148    return new AtomizedLocalDateTime(localDate, h, m, s, ms);
149  }
150
151  @Override
152  public int hashCode() {
153    return new HashCodeBuilder().append(localDate).append(hour).append(minute).
154            append(second).append(millisecond).append(resolution).toHashCode();
155  }
156
157  @Override
158  public boolean equals(Object obj) {
159
160    if (this == obj) {
161      return true;
162    }
163    if (!(obj instanceof AtomizedLocalDateTime)) {
164      return false;
165    }
166
167    AtomizedLocalDateTime that = (AtomizedLocalDateTime) obj;
168    return Objects.equal(this.localDate, that.localDate)
169            && Objects.equal(this.hour, that.hour)
170            && Objects.equal(this.minute, that.minute)
171            && Objects.equal(this.second, that.second)
172            && Objects.equal(this.millisecond, that.millisecond)
173            && Objects.equal(this.resolution, that.resolution);
174  }
175
176  @Override
177  public String toString() {
178    return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE)
179            .append("localDate", localDate)
180            .append("hour", hour)
181            .append("minute", minute)
182            .append("second", second)
183            .append("millisecond", millisecond)
184            .append("resolution", resolution)
185            .toString();
186  }
187
188}
