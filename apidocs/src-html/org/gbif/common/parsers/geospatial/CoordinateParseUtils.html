032/**
033 * Utilities for assisting in the parsing of latitude and longitude strings into Decimals.
034 */
035public class CoordinateParseUtils {
036  private final static String DMS = "\\s*(\\d{1,3})\\s*(?:°|d|º| |g|o)"  // The degrees
037                                  + "\\s*([0-6]?\\d)\\s*(?:'|m| |´|’|′)" // The minutes
038                                  + "\\s*(?:"                            // Non-capturing group
039                                  + "([0-6]?\\d(?:[,.]\\d+)?)"           // Seconds and optional decimal
040                                  + "\\s*(?:\"|''|s|´´|″)?"
041                                  + ")?\\s*";
042  private final static String DM = "\\s*(\\d{1,3})\\s*(?:°|d|º| |g|o)" // The degrees
043                                 + "\\s*(?:"                           // Non-capturing group
044                                 + "([0-6]?\\d(?:[,.]\\d+)?)"          // Minutes and optional decimal
045                                 + "\\s*(?:'|m| |´|’|′)?"
046                                 + ")?\\s*";
047  private final static String D = "\\s*(\\d{1,3}(?:[,.]\\d+)?)\\s*(?:°|d|º| |g|o|)\\s*"; // The degrees and optional decimal
048  private final static Pattern DMS_SINGLE = Pattern.compile("^" + DMS + "$", Pattern.CASE_INSENSITIVE);
049  private final static Pattern DM_SINGLE = Pattern.compile("^" + DM + "$", Pattern.CASE_INSENSITIVE);
050  private final static Pattern D_SINGLE = Pattern.compile("^" + D + "$", Pattern.CASE_INSENSITIVE);
051  private final static Pattern DMS_COORD = Pattern.compile("^" + DMS + "([NSEOW])" + "[ ,;/]?" + DMS + "([NSEOW])$", Pattern.CASE_INSENSITIVE);
052  //private final static Pattern DM_COORD = Pattern.compile("^" + DM + "([NSEOW])" + "[ ,;/]?" + DM + "([NSEOW])$", Pattern.CASE_INSENSITIVE);
053  //private final static Pattern D_COORD = Pattern.compile("^" + D + "([NSEOW])" + "[ ,;/]?" + D + "([NSEOW])$", Pattern.CASE_INSENSITIVE);
054  private final static String POSITIVE = "NEO";
055  private CoordinateParseUtils() {
056    throw new UnsupportedOperationException("Can't initialize class");
057  }

059  private static final Logger LOG = LoggerFactory.getLogger(CoordinateParseUtils.class);

061  /**
062   * This parses string representations of latitude and longitude values. It tries its best to interpret the values and
063   * indicates any problems in its result as {@link org.gbif.api.vocabulary.OccurrenceIssue}.
064   * When the {@link ParseResult.STATUS} is FAIL the payload will be null and one or more issues should be set
065   * in {@link org.gbif.common.parsers.core.OccurrenceParseResult#getIssues()}.
066   *
067   * Coordinate precision will be 6 decimals at most, any more precise values will be rounded.
068   *
069   * Supported standard formats are the following, with dots or optionally a comma as the decimal marker, and variations
070   * on the units also accepted e.g. °, d, º, g, o.
071   * <ul>
072   *   <li>43.63871944444445</li>
073   *   <li>N43°38'19.39"</li>
074   *   <li>43°38'19.39"N</li>
075   *   <li>43°38.3232'N</li>
076   *   <li>43d 38m 19.39s N</li>
077   *   <li>43 38 19.39</li>
078   *   <li>433819N</li>
079   * </ul>
080   *
081   * @param latitude  The decimal latitude
082   * @param longitude The decimal longitude
083   *
084   * @return The parse result
085   */
086  public static OccurrenceParseResult<LatLng> parseLatLng(final String latitude, final String longitude) {
087    if (StringUtils.isEmpty(latitude) || StringUtils.isEmpty(longitude)) {
088      return OccurrenceParseResult.fail();
089    }
090    Double lat = NumberParser.parseDouble(latitude);
091    Double lng = NumberParser.parseDouble(longitude);
092    if (lat == null || lng == null) {
093      // try degree minute seconds
094      try {
095        lat = parseDMS(latitude, true);
096        lng = parseDMS(longitude, false);
097      } catch (IllegalArgumentException e) {
098        return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
099      }
100    }
101
102    return validateAndRound(lat, lng);
103  }
104
105  private static boolean inRange(double lat, double lon) {
106    if (Double.compare(lat, 90) <= 0 && Double.compare(lat, -90) >= 0 && Double.compare(lon, 180) <= 0 && Double.compare(lon, -180) >= 0) {
107      return true;
108    }
109    return false;
110  }
111
112  private static boolean isLat(String direction) {
113    if ("NS".contains(direction.toUpperCase())) {
114      return true;
115    }
116    return false;
117  }
118
119  private static int coordSign(String direction) {
120    return POSITIVE.contains(direction.toUpperCase()) ? 1 : -1;
121  }
122
123  // 02° 49' 52" N      131° 47' 03" E
124  public static OccurrenceParseResult<LatLng> parseVerbatimCoordinates(final String coordinates) {
125    if (StringUtils.isEmpty(coordinates)) {
126      return OccurrenceParseResult.fail();
127    }
128    Matcher m = DMS_COORD.matcher(coordinates);
129    if (m.find()) {
130      final String dir1 = m.group(4);
131      final String dir2 = m.group(8);
132      // first parse coords regardless whether they are lat or lon
133      double c1 = coordFromMatcher(m, 1,2,3, dir1);
134      double c2 = coordFromMatcher(m, 5,6,7, dir2);
135      // now see what order the coords are in:
136      if (isLat(dir1) && !isLat(dir2)) {
137        return validateAndRound(c1, c2);
138
139      } else if (!isLat(dir1) && isLat(dir2)) {
140        return validateAndRound(c2, c1);
141
142      } else {
143        return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
144      }
145
146    } else if(coordinates.length() > 4) {
147      // try to split and then use lat/lon parsing
148      for (final char delim : ",;/ ".toCharArray()) {
149        int cnt = StringUtils.countMatches(coordinates, String.valueOf(delim));
150        if (cnt == 1) {
151          String[] latlon = StringUtils.split(coordinates, delim);
152          if (latlon.length == 2) {
153            return parseLatLng(latlon[0], latlon[1]);
154          }
155        }
156      }
157    }
158    return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
159  }
160
161  private static OccurrenceParseResult<LatLng> validateAndRound(double lat, double lon) {
162    // collecting issues for result
163    Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
164
165    // round to 6 decimals
166    final double latOrig = lat;
167    final double lngOrig = lon;
168    lat = roundTo6decimals(lat);
169    lon = roundTo6decimals(lon);
170    if (Double.compare(lat, latOrig) != 0 || Double.compare(lon, lngOrig) != 0) {
171      issues.add(OccurrenceIssue.COORDINATE_ROUNDED);
172    }
173
174    // 0,0 is too suspicious
175    if (Double.compare(lat, 0) == 0 && Double.compare(lon, 0) == 0) {
176      issues.add(OccurrenceIssue.ZERO_COORDINATE);
177      return OccurrenceParseResult.success(ParseResult.CONFIDENCE.POSSIBLE, new LatLng(0, 0), issues);
178    }
179
180    // if everything falls in range
181    if (inRange(lat, lon)) {
182      return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new LatLng(lat, lon), issues);
183    }
184
185    // if lat is out of range, but in range of the lng, assume swapped coordinates.
186    // note that should we desire to trust the following records, we would need to clear the flag for the records to
187    // appear in search results and maps etc. however, this is logic decision, that goes above the capabilities of this method
188    if (Double.compare(lat, 90) > 0 || Double.compare(lat, -90) < 0) {
189      // try and swap
190      if (inRange(lon, lat)) {
191        issues.add(OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
192        return OccurrenceParseResult.success(ParseResult.CONFIDENCE.PROBABLE, new LatLng(lon, lat), issues);
193      }
194    }
195
196    // then something is out of range
197    issues.add(OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
198    return OccurrenceParseResult.fail(issues);
199
200  }

202  /**
203   * Parses a single DMS coordinate
204   * @param coord
205   * @param lat
206   * @return the converted decimal up to 6 decimals accuracy
207   */
208  protected static double parseDMS(String coord, boolean lat) {
209    final String DIRS = lat ? "NS" : "EOW";
210    coord = coord.trim().toUpperCase();
211
212    if (coord.length() > 3) {
213      // preparse the direction and remove it from the string to avoid a very complex regex
214      char dir = 'n';
215      if (DIRS.contains(String.valueOf(coord.charAt(0)))) {
216        dir = coord.charAt(0);
217        coord = coord.substring(1);
218      } else if (DIRS.contains(String.valueOf(coord.charAt(coord.length()-1)))) {
219        dir = coord.charAt(coord.length()-1);
220        coord = coord.substring(0, coord.length()-1);
221      }
222      // without the direction chuck it at the regex
223      Matcher m = DMS_SINGLE.matcher(coord);
224      if (m.find()) {
225        return coordFromMatcher(m, 1,2,3, String.valueOf(dir));
226      } else {
227        m = DM_SINGLE.matcher(coord);
228        if (m.find()) {
229          return coordFromMatcher(m, 1, 2, String.valueOf(dir));
230        } else {
231          m = D_SINGLE.matcher(coord);
232          if (m.find()) {
233            return coordFromMatcher(m, 1, String.valueOf(dir));
234          }
235        }
236      }
237    }
238    throw new IllegalArgumentException();
239  }
240
241  private static double coordFromMatcher(Matcher m, int idx1, int idx2, int idx3, String sign) {
242    return roundTo6decimals(coordSign(sign) *
243      dmsToDecimal( NumberParser.parseDouble(m.group(idx1)), NumberParser.parseDouble(m.group(idx2)), NumberParser.parseDouble(m.group(idx3)) ));
244  }
245
246  private static double coordFromMatcher(Matcher m, int idx1, int idx2, String sign) {
247    return roundTo6decimals(coordSign(sign) *
248      dmsToDecimal( NumberParser.parseDouble(m.group(idx1)), NumberParser.parseDouble(m.group(idx2)), 0.0));
249  }
250
251  private static double coordFromMatcher(Matcher m, int idx1, String sign) {
252    return roundTo6decimals(coordSign(sign) *
253      dmsToDecimal( NumberParser.parseDouble(m.group(idx1)), 0.0, 0.0));
254  }
255
256  private static double dmsToDecimal(double degree, Double minutes, Double seconds) {
257    minutes = minutes == null ? 0 : minutes;
258    seconds = seconds == null ? 0 : seconds;
259    return degree + (minutes / 60) + (seconds / 3600);
260  }
261
262  // round to 6 decimals (~1m precision) since no way we're getting anything legitimately more precise
263  private static Double roundTo6decimals(Double x) {
264    return x == null ? null : Math.round(x * Math.pow(10, 6)) / Math.pow(10, 6);
265  }
