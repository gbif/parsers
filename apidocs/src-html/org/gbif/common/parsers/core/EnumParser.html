001package org.gbif.common.parsers.core;
002
003import org.gbif.api.util.VocabularyUtils;
004
005import java.io.InputStream;
006import java.util.regex.Pattern;
007
008import com.google.common.base.Strings;
009
010/**
011 * Generic parser for enumerations making use of our vocabulary util to lookup an enum value from a string.
012 * Also adds stronger normalization removing any non alphanumeric characters. It is still ok to use hyphens or
013 * underscores in the enumeration values.
014 */
015public class EnumParser<T extends Enum<T>> extends FileBasedDictionaryParser<T> {
016
017  private final Class<T> clazz;
018  private final Pattern NORMALIZER;
019
020  protected EnumParser(Class<T> clazz, boolean allowDigits, final InputStream... inputs) {
021    super(false);
022
023    if (allowDigits) {
024      NORMALIZER = Pattern.compile("\\W+");
025    } else {
026      NORMALIZER = Pattern.compile("[^a-zA-Z]+");
027    }
028    this.clazz = clazz;
029    // init dicts
030    addEnumValues();
031
032    if (inputs != null) {
033      for (InputStream input : inputs) {
034        init(input);
035      }
036    }
037  }
038
039  private void addEnumValues() {
040    T[] values = clazz.getEnumConstants();
041    if (values != null) {
042      for (T val : values) {
043        add(val.name(), val);
044      }
045    }
046  }
047
048  @Override
049  protected String normalize(String value) {
050    if (Strings.isNullOrEmpty(value)) return null;
051    return NORMALIZER.matcher(value).replaceAll("").toUpperCase();
052  }
053
054  @Override
055  protected T fromDictFile(String value) {
056    try {
057      return (T) VocabularyUtils.lookupEnum(value, clazz);
058
059    } catch (RuntimeException e) {
060
061    }
062    return null;
063  }
064}
