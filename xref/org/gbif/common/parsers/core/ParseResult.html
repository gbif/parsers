1   package org.gbif.common.parsers.core;
2   
3   import org.apache.commons.lang3.builder.ToStringBuilder;
4   import org.apache.commons.lang3.builder.ToStringStyle;
5   
6   /**
7    * This represents the response of a generic parse operation.
8    */
9   public class ParseResult<T> {
10  
11    // the result code of the parse operation
12    public enum STATUS {
13      SUCCESS, FAIL, ERROR
14    }
15  
16    // the confidence of the result, applicable if status=SUCCESS
17    public enum CONFIDENCE {
18      DEFINITE, PROBABLE, POSSIBLE
19    }
20  
21    // the details of the response
22    protected STATUS status;
23    protected CONFIDENCE confidence;
24    protected T payload;
25    protected Throwable error;
26  
27    /**
28     * @param <T1>       The generic type of the payload
29     * @param confidence The confidence in the result
30     * @param payload    The payload of the parse result
31     *
32     * @return The new ParseResult which has no error and status of SUCCESS
33     */
34    public static <T1> ParseResult<T1> success(CONFIDENCE confidence, T1 payload) {
35      return new ParseResult<T1>(STATUS.SUCCESS, confidence, payload, null);
36    }
37  
38    /**
39     * @return A new parse response with only the status set to FAIL
40     */
41    public static <T1> ParseResult<T1> fail() {
42      return new ParseResult<T1>(STATUS.FAIL, null, null, null);
43    }
44  
45    /**
46     * @return A new parse response configured to indicate an error
47     */
48    public static <T1> ParseResult<T1> error() {
49      return new ParseResult<T1>(STATUS.ERROR, null, null, null);
50    }
51  
52    /**
53     * @param cause The cause of the error
54     *
55     * @return A new parse response configured with error and the cause
56     */
57    public static <T1> ParseResult<T1> error(Throwable cause) {
58      return new ParseResult<T1>(STATUS.ERROR, null, null, cause);
59    }
60  
61    /**
62     * Forces all fields to be provided
63     *
64     * @param status     The status of the response
65     * @param confidence The confidence in the result
66     * @param payload    The payload of the response
67     * @param error      The error in the response
68     */
69    public ParseResult(STATUS status, CONFIDENCE confidence, T payload, Throwable error) {
70      this.status = status;
71      this.confidence = confidence;
72      this.payload = payload;
73      this.error = error;
74    }
75  
76    public STATUS getStatus() {
77      return status;
78    }
79  
80    public CONFIDENCE getConfidence() {
81      return confidence;
82    }
83  
84    public T getPayload() {
85      return payload;
86    }
87  
88    public Throwable getError() {
89      return error;
90    }
91  
92  
93    /**
94     * Returns true if {@link #getStatus()} returns SUCCESS.
95     *
96     * @return true if the parse operation was successful.
97     */
98    public boolean isSuccessful() {
99      return status == STATUS.SUCCESS;
100   }
101 
102   @Override
103   public String toString() {
104     return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE)
105       .append("status", getStatus())
106       .append("confidence", getConfidence())
107       .append("payload", getPayload())
108       .append("error", getError()).toString();
109   }
110 }
