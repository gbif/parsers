001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.EstablishmentMeans;
004import org.gbif.common.parsers.core.EnumParser;
005
006public class EstablishmentMeansParser extends EnumParser<EstablishmentMeans> {
007
008  private static EstablishmentMeansParser singletonObject = null;
009
010  private EstablishmentMeansParser() {
011    super(EstablishmentMeans.class, false);
012    // also make sure we have all official iso countries mapped
013    for (EstablishmentMeans c : EstablishmentMeans.values()) {
014      add(c.name(), c);
015    }
016    // use dict file last
017    init(EstablishmentMeansParser.class.getResourceAsStream("/dictionaries/parse/establishment_means.tsv"));
018  }
019
020  public static EstablishmentMeansParser getInstance() {
021    synchronized (EstablishmentMeansParser.class) {
022      if (singletonObject == null) {
023        singletonObject = new EstablishmentMeansParser();
024      }
025    }
026    return singletonObject;
027  }
028
029}
