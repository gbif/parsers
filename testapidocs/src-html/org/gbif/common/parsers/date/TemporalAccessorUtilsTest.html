032/**
033 * Unit test for {@link TemporalAccessorUtils}.

035 */
036public class TemporalAccessorUtilsTest {
037
038  private ZoneId EUROPE_CENTRAL_TIME = ZoneId.of("Europe/Paris");
039
040  // January 1st 2000, 00:00:00 (midnight)
041  private long YEAR2000_1JAN_EPOCH_UTC = 946684800000L;
042
043  // January 1st 2000, 02:03:04
044  private long YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC = 946692184000L;
045
046  // February 1st 2000, midnight
047  private long YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC = 949363200000l;
048
049  // January 1st 2000, midnight
050  private long YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC = 946684800000l;
051
052  @Test
053  public void testToDate() {
054    TemporalAccessor ta = LocalDate.of(2000, 1,1);
055    Date date = TemporalAccessorUtils.toDate(ta, false);
056    assertEquals(YEAR2000_1JAN_EPOCH_UTC, date.getTime());
057
058    ta = LocalDateTime.of(2000, 1, 1, 2, 3, 4);
059    date = TemporalAccessorUtils.toDate(ta, false);
060    assertEquals(YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC, date.getTime());
061
062    //test partial date transformed to Date
063    //YearMonth should use the first of the month
064    ta = YearMonth.of(2000, 2);
065    date = TemporalAccessorUtils.toDate(ta, false);
066    assertEquals(YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC, date.getTime());
067
068    //Year should use the first of January
069    ta = Year.of(2000);
070    date = TemporalAccessorUtils.toDate(ta, false);
071    assertEquals(YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC, date.getTime());
072  }
073
074  @Test
075  public void testToDateWithTimeZone() {
076
077    // Test with ZonedDateTime
078    ZonedDateTime testZonedDateTime = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, EUROPE_CENTRAL_TIME);
079    Date date = TemporalAccessorUtils.toDate(testZonedDateTime, false);
080    assertEquals(testZonedDateTime.toEpochSecond(), date.getTime()/1000);
081
082    // Test the same ZonedDateTime but ignore Offset information (timezone)
083    ZonedDateTime testTimeUTC = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, TemporalAccessorUtils.UTC_ZONE_ID);
084    date = TemporalAccessorUtils.toDate(testZonedDateTime, true);
085    assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
086
087    // Ensure it is also working with OffsetDateTime
088    OffsetDateTime testOffsetDateTimeTime = testZonedDateTime.toOffsetDateTime();
089    date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, false);
090    assertEquals(testOffsetDateTimeTime.toEpochSecond(), date.getTime()/1000);
091
092    date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, true);
093    assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
094  }
095
096  @Test
097  public void testBestResolution() {
098    TemporalAccessor ta1 = Year.of(2005);
099    TemporalAccessor ta2 = YearMonth.of(2005, 1);
100    TemporalAccessor result = TemporalAccessorUtils.bestResolution(ta1, ta2).get();
101    assertEquals(YearMonth.of(2005, 1), YearMonth.from(result));
102
103    ta1 = LocalDate.of(2005, 1, 1);
104    ta2 = Year.of(2005);
105    result = TemporalAccessorUtils.bestResolution(ta1, ta2).get();
106    assertEquals(LocalDate.of(2005, 1, 1), LocalDate.from(result));
107
108    //this should not work
109    ta1 = Year.of(2005);
110    ta2 = YearMonth.of(2006, 1);
111    assertFalse(TemporalAccessorUtils.bestResolution(ta1, ta2).isPresent());
112  }

114  @Test
115  public void testSameOrContained() {
116    TemporalAccessor ymd = LocalDate.of(1996, 4, 26);
117
118    // Test nulls and wrong resolutions
119    assertFalse(TemporalAccessorUtils.sameOrContained(null, null));
120    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, null));
121    assertFalse(TemporalAccessorUtils.sameOrContained(null, ymd));
122
123    // Test against same year
124    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, Year.of(1996)));
125    // Test against same month
126    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, YearMonth.of(1996, 4)));
127    // Test against same date
128    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, LocalDate.of(1996, 4, 26)));
129    // Test against time on that day
130    assertTrue(TemporalAccessorUtils.sameOrContained(ymd, LocalDateTime.of(1996, 4, 26, 1, 2, 3)));
131
132    // Then different year/month/date
133    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, Year.of(1998)));
134    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, YearMonth.of(1996, 5)));
135    assertFalse(TemporalAccessorUtils.sameOrContained(ymd, LocalDate.of(1996, 4, 27)));
136  }

138  @Test
139  public void testLastDay() {
140    TemporalAccessor year = Year.of(1996);
141    assertEquals("1996-12-31T23:59:59.999", TemporalAccessorUtils.toLatestLocalDateTime(year,true).toString());
142
143    TemporalAccessor ym = YearMonth.of(1996,1);
144    assertEquals("1996-01-31T23:59:59.999", TemporalAccessorUtils.toLatestLocalDateTime(ym,true).toString());
145
146    ym = YearMonth.of(1996,2);
147    assertEquals("1996-02-29T23:59:59.999", TemporalAccessorUtils.toLatestLocalDateTime(ym,true).toString());
148
149    TemporalAccessor ymd = LocalDate.of(1996,2,3);
150    assertEquals("1996-02-03T23:59:59.999", TemporalAccessorUtils.toLatestLocalDateTime(ymd,true).toString());
151
152    TemporalAccessor ymdt = LocalDateTime.of(1996,2,3,1,20);
153    assertEquals("1996-02-03T01:20", TemporalAccessorUtils.toLatestLocalDateTime(ymdt,true).toString());
154  }

156  @Test
157  public void testWithinRange() {
158    TemporalAccessor ymd = LocalDate.of(1996,2,3);
159    TemporalAccessor begin = LocalDateTime.of(1996,2,3,1,20);
160    TemporalAccessor end = LocalDateTime.of(1996,2,3,1,20);
161    assertTrue(TemporalAccessorUtils.withinRange(begin, end, ymd));
162
163    begin = LocalDateTime.of(1996,2,3,0,0);
164    assertTrue(TemporalAccessorUtils.withinRange(begin, end, ymd));
165
166    begin = LocalDateTime.of(1996,2,2,1,20);
167    assertTrue(TemporalAccessorUtils.withinRange(begin, end, ymd));
168
169    end = LocalDateTime.of(1999,12,4,1,20);
170    assertTrue(TemporalAccessorUtils.withinRange(begin, end, ymd));
171
172    ymd = LocalDate.of(1990,2,3);
173    assertFalse(TemporalAccessorUtils.withinRange(begin, end, ymd));
174  }
