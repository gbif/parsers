001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.License;
004import org.gbif.common.parsers.core.EnumParser;
005import org.gbif.common.parsers.core.ParseResult;
006
007import java.net.URI;
008import java.util.regex.Pattern;
009import javax.annotation.Nullable;
010
011import com.google.common.base.Strings;
012
013/**
014 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/license.txt to lookup a
015 * License by its URI or its acronym/title, e.g. a lookup by "CC-BY 4.0" returns License.CC_BY_4_0.
016 * </br>
017 * The dictionary file must enforce the <a href="http://www.gbif.org/terms/licences">GBIF Licensing Policy</a>.
018 * Non-CC licenses that GBIF considers equal to one of its three supported CC licenses (CC0 1.0, CC-BY 4.0 and CC-BY-NC
019 * 4.0) can be added to this file.
020 * </br>
021 * Note a lookup by license acronym/title without a version number defaults to the latest version of that license,
022 * e.g. a lookup by "CC-BY" returns License.CC_BY_4_0.
023 */
024public class LicenseParser extends EnumParser<License> {
025
026  private static final String COMMENT_MARKER = "#";
027  private static final String LICENSE_FILEPATH = "/dictionaries/parse/license.tsv";
028  //allows us to remove the protocol part for http:// and https://
029  private static final Pattern REMOVE_HTTP_PATTERN = Pattern.compile("^https?:\\/\\/", Pattern.CASE_INSENSITIVE);
030  private static LicenseParser singletonObject = null;
031
032  private LicenseParser() {
033    super(License.class, true);
034    // also make sure we have all enum values and their parameters title and url mapped
035    for (License l : License.values()) {
036      add(l.name(), l);
037      add(l.getLicenseTitle(), l);
038      add(l.getLicenseUrl(), l);
039    }
040    // use dict file last
041    init(LicenseParser.class.getResourceAsStream(LICENSE_FILEPATH), COMMENT_MARKER);
042  }
043
044  @Override
045  protected String normalize(String value) {
046    if(value == null){
047      return null;
048    }
049    return super.normalize(REMOVE_HTTP_PATTERN.matcher(value).replaceAll(""));
050  }
051
052  public static LicenseParser getInstance() {
053    synchronized (ContinentParser.class) {
054      if (singletonObject == null) {
055        singletonObject = new LicenseParser();
056      }
057    }
058    return singletonObject;
059  }
060
061  /**
062   * Parse license supplied in two parts: URI and title. First parse URI. Only if no URI supplied, parse title.
063   * supplied.
064   *
065   * @param uri   optional license URI
066   * @param title optional license title
067   *
068   * @return License corresponding to license URI or title, License.UNSPECIFIED if both URI and title not supplied,
069   * otherwise defaults to License.UNSUPPORTED
070   */
071  public License parseUriThenTitle(@Nullable URI uri, @Nullable String title) {
072    if (uri == null && Strings.isNullOrEmpty(title)) {
073      return License.UNSPECIFIED;
074    }
075
076    if (uri != null) {
077      ParseResult<License> result = singletonObject.parse(uri.toString());
078      if (result.isSuccessful()) {
079        return result.getPayload();
080      }
081    }
082
083    if (!Strings.isNullOrEmpty(title)) {
084      ParseResult<License> result = singletonObject.parse(title);
085      if (result.isSuccessful()) {
086        return result.getPayload();
087      }
088    }
089
090    return License.UNSUPPORTED;
091  }
092}
