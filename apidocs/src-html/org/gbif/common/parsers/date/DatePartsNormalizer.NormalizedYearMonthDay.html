001package org.gbif.common.parsers.date;
002
003import org.gbif.utils.file.csv.CSVReader;
004import org.gbif.utils.file.csv.CSVReaderFactory;
005
006import java.io.IOException;
007import java.io.InputStream;
008import java.time.Month;
009import java.util.HashSet;
010import java.util.Map;
011import java.util.Set;
012
013import com.google.common.base.Charsets;
014import com.google.common.base.Splitter;
015import com.google.common.collect.Maps;
016import org.apache.commons.lang3.StringUtils;
017import org.apache.commons.lang3.builder.ToStringBuilder;
018import org.apache.commons.lang3.builder.ToStringStyle;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 * DatePartsNormalizer contract is to take String representing Year, Month and Day and return the corresponding
024 * Integer as {@link NormalizedYearMonthDay} if possible.
025 *
026 * No validation will be applied to the normalized values.
027 *
028 * Thread-Safe after creation.
029 *
030 */
031public class DatePartsNormalizer {
032
033  private static final Logger LOG = LoggerFactory.getLogger(DatePartsNormalizer.class);
034
035  private static final String STRING_NULL = "\\N";
036  private static final String COLUMN_SEPARATOR = ";";
037  private static final Splitter ROW_ELEMENT_SPLITTER = Splitter.on(',').trimResults().omitEmptyStrings();
038
039  private static final String COMMENT_MARKER = "#";
040  private static final String MONTH_FILEPATH = "/dictionaries/parse/month.csv";
041
042  private static final String[][] MONTHS = new String[Month.values().length][];
043
044  // Load all the month names and alternative month names from a file
045  static {
046    Map<String, Set<String>> monthMap = Maps.newHashMapWithExpectedSize(Month.values().length);
047    String keyName;
048    for (Month m : Month.values()) {
049      keyName = m.name().toLowerCase();
050      monthMap.put(keyName, new HashSet<String>());
051      //add the key itself
052      monthMap.get(keyName).add(keyName);
053    }
054
055    InputStream monthFileStream = DatePartsNormalizer.class.getResourceAsStream(MONTH_FILEPATH);
056
057    if (monthFileStream == null) {
058      LOG.error("Month file can not be loaded. File not found: {}", MONTH_FILEPATH);
059    } else {
060      try (CSVReader csv = CSVReaderFactory.build(monthFileStream, Charsets.UTF_8.name(), COLUMN_SEPARATOR, null, 0)) {
061        while (csv.hasNext()) {
062          String[] row = csv.next();
063          if (row == null || row[0].startsWith(COMMENT_MARKER)) {
064            continue;
065          }
066          String monthKey = row[0].toLowerCase();
067          if (monthMap.containsKey(monthKey)) {
068            for (String monthAltName : ROW_ELEMENT_SPLITTER.split(row[1])) {
069              monthMap.get(monthKey).add(monthAltName.toLowerCase());
070            }
071          } else {
072            LOG.error("Unknown month found in: {}", MONTH_FILEPATH);
073          }
074        }
075      } catch (IOException e) {
076        LOG.error("Error loading month alternative names", e);
077      }
078
079      // keep it in an array
080      int index = 0;
081      for (Month m : Month.values()) {
082        MONTHS[index] = monthMap.get(m.name().toLowerCase()).toArray(new String[0]);
083        index++;
084      }
085    }
086  }
087
088  /**
089   * Private constructor use static method {@link #newInstance()}
090   */
091  private DatePartsNormalizer(){}
092
093  public static DatePartsNormalizer newInstance(){
094    return new DatePartsNormalizer();
095  }
096
097  /**
098   * Normalize date parts value.
099   *
100   * @param year
101   * @param month
102   * @param day
103   * @return result of normalization as NormalizedYearMonthDay
104   */
105  public NormalizedYearMonthDay normalize(String year, String month, String day){
106    year = normalizeFloat(year);
107    month = normalizeFloat(month);
108    day = normalizeFloat(day);
109
110    Integer monthAsInt = parseOrNull(month);
111    if(monthAsInt == null){
112      monthAsInt = monthNameToNumerical(month);
113    }
114
115    Integer iYear = parseOrNull(year);
116    Integer iMonth = monthAsInt;
117    Integer iDay = parseOrNull(day);
118
119    boolean yearDiscarded = wasDiscarded(year, iYear);
120    boolean monthDiscarded = wasDiscarded(month, iMonth);
121    boolean dayDiscarded = wasDiscarded(day, iDay);
122
123    return new NormalizedYearMonthDay(iYear, iMonth, iDay, yearDiscarded, monthDiscarded, dayDiscarded);
124  }
125
126  /**
127   * Often months come in the form Sept. September etc. This will convert many variations into the numerical version
128   *
129   * @param month name to normalize
130   *
131   * @return the numerical value of the month (January == 1 )
132   */
133  public Integer monthNameToNumerical(String month) {
134    if (StringUtils.isNotBlank(month)) {
135      int m = 1;
136      for (String[] monthValues : MONTHS) {
137        for (String monthVal : monthValues) {
138          if (monthVal.equals(month.toLowerCase()) || (monthVal+ ".").equals(month.toLowerCase())) {
139            return m;
140          }
141        }
142        m++;
143      }
144    }
145    return null;
146  }
147
148
149  /**
150   * Often values are seen as Float rather than int, due to a database export The year "1978" is actually seen as
151   * "1978.0".  Where this is detected, the string is normalized to the INT representation
152   *
153   * @param value To check
154   *
155   * @return the integer value (as String)
156   */
157  public String normalizeFloat(String value) {
158    if (value != null && value.contains(".0")) {
159      try {
160        Double d = new Double(value);
161        if ((double) d.intValue() == d.doubleValue()) {
162          return String.valueOf(d.intValue());
163        }
164      } catch (NumberFormatException e) {
165      }
166    }
167    return value;
168  }
169
170  /**
171   * Try to parse the provided String as Integer. Returns null if not possible.
172   * This function will trim the provided String.
173   * @param integer
174   * @return
175   */
176  private Integer parseOrNull(String integer){
177    if(integer != null){
178      integer = integer.trim();
179    }
180
181    try{
182      return Integer.valueOf(integer);
183    }
184    catch(NumberFormatException nfEx){}
185    return null;
186  }
187
188  /**
189   * Assert if a String value was discarded in the normalization process.
190   *
191   * @param strValue
192   * @param intValue
193   * @return the value should be considered discarded or not
194   */
195  private boolean wasDiscarded(String strValue, Integer intValue){
196    if(StringUtils.isBlank(strValue) || STRING_NULL.equals(strValue)){
197      return false;
198    }
199    return intValue == null;
200  }
201
202  /**
203   * Hold result of the normalization process.
204   */
205  public static class NormalizedYearMonthDay {
206
207    private Integer year;
208    private Integer month;
209    private Integer day;
210
211    private boolean yDiscarded;
212    private boolean mDiscarded;
213    private boolean dDiscarded;
214
215    NormalizedYearMonthDay(Integer year, Integer month, Integer day, boolean yDiscarded, boolean mDiscarded,
216                           boolean dDiscarded){
217      this.year = year;
218      this.month = month;
219      this.day = day;
220
221      this.yDiscarded = yDiscarded;
222      this.mDiscarded = mDiscarded;
223      this.dDiscarded = dDiscarded;
224    }
225
226    public Integer getYear() {
227      return year;
228    }
229
230    public Integer getMonth() {
231      return month;
232    }
233
234    public Integer getDay() {
235      return day;
236    }
237
238    public boolean yDiscarded() {
239      return yDiscarded;
240    }
241
242    public boolean mDiscarded() {
243      return mDiscarded;
244    }
245
246    public boolean dDiscarded() {
247      return dDiscarded;
248    }
249
250    /**
251     * The NormalizedYearMonthDay contains at least one discarded part.
252     * @return
253     */
254    public boolean containsDiscardedPart(){
255      return yDiscarded || mDiscarded || dDiscarded;
256    }
257
258    @Override
259    public String toString() {
260      return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE)
261              .append("year", year)
262              .append("month", month)
263              .append("day", day)
264              .append("yDiscarded", yDiscarded)
265              .append("mDiscarded", mDiscarded)
266              .append("dDiscarded", dDiscarded)
267              .toString();
268    }
269  }
