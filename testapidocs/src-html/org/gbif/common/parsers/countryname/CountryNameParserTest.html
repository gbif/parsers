001package org.gbif.common.parsers.countryname;
002
003import org.gbif.common.parsers.FileBasedDictionaryParserTest;
004
005import org.junit.Test;
006
007public class CountryNameParserTest extends FileBasedDictionaryParserTest {
008
009  @Test
010  public void testParse() {
011    // run a few basic tests to check it bootstraps and appears to work
012    CountryNameParser cnp = CountryNameParser.getInstance();
013    assertParseFailure(cnp, "[West Indian Ocean]");
014    assertParseSuccess(cnp, "GB", "Great Britain");
015    assertParseSuccess(cnp, "GB", "Great Britain!!!");
016    assertParseFailure(cnp, "Really great britain");
017    assertParseSuccess(cnp, "GB", "Great  Britain"); // Test collapsing of multiple whitespaces
018    assertParseSuccess(cnp, "GB", " Great\tBritain "); // Test collapsing of other kinds of whitespace and trimming
019    assertParseSuccess(cnp, "DK", "Denmark");
020    assertParseSuccess(cnp, "DE", " Germany "); // Test trimming
021    assertParseSuccess(cnp, "AU", "off Australia");
022    assertParseSuccess(cnp, "AU", " off (australia)\t! \u0312 "); // all together now...
023    assertParseSuccess(cnp, "DK", "DK");
024    assertParseSuccess(cnp, "DK", "OFF DENMARK");
025    assertParseSuccess(cnp, "ME", "me");
026    assertParseSuccess(cnp, "SS", "ss");
027    assertParseSuccess(cnp, "SS", "SOUTH_SUDAN");
028    assertParseSuccess(cnp, "SS", "south sudan");
029    assertParseSuccess(cnp, "MX", "MÃ©xico");
030
031    assertParseFailure(cnp, "Borneo");
032    assertParseSuccess(cnp, "BA", "BOSNIA");
033
034  }
