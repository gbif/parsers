001package org.gbif.common.parsers;
002
003import com.google.common.base.Strings;
004import org.gbif.api.vocabulary.License;
005import org.gbif.common.parsers.core.ASCIIParser;
006import org.gbif.common.parsers.core.FileBasedDictionaryParser;
007import org.gbif.common.parsers.core.KeyValue;
008import org.gbif.common.parsers.core.ParseResult;
009
010import java.net.URI;
011import java.util.Iterator;
012import java.util.regex.Pattern;
013
014/**
015 * A license parser giving URIs, compared to {@link License} which is for our GBIF enumeration.
016 *
017 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/license_uri.txt to lookup a
018 * License by its URI or its acronym/title, e.g. a lookup by "CC-BY 4.0" returns http://creativecommons.org/licenses/by/4.0/.
019 * <br/>
020 * Note a lookup by license acronym/title without a version number defaults to the latest version of that license,
021 * e.g. a lookup by "CC-BY" returns http://creativecommons.org/licenses/by/4.0/.
022 */
023public class LicenseUriParser extends FileBasedDictionaryParser<URI> {
024
025  private static final String COMMENT_MARKER = "#";
026  private static final String LICENSE_FILEPATH = "/dictionaries/parse/license_uri.tsv";
027  // allows us to remove the protocol part for http:// and https://
028  private static final Pattern REMOVE_HTTP_PATTERN = Pattern.compile("^https?:\\/\\/", Pattern.CASE_INSENSITIVE);
029  private static final Pattern NORMALIZER = Pattern.compile("[^\\p{IsAlphabetic}\\p{N}Â©]+");
030  protected final ASCIIParser asciiParser = ASCIIParser.getInstance();
031  private static LicenseUriParser singletonObject = null;
032
033  private LicenseUriParser() {
034    super(false);
035    init(LicenseUriParser.class.getResourceAsStream(LICENSE_FILEPATH), COMMENT_MARKER);
036  }
037
038  /**
039   * @param source To build the dictionary from
040   */
041  @Override
042  public void init(Iterator<KeyValue<String, URI>> source) {
043    while (source.hasNext()) {
044      KeyValue<String, URI> kvp = source.next();
045      add(kvp.getKey(), kvp.getValue());
046      // Also adds the value, to save defining all of them.
047      add(kvp.getValue().toString(), kvp.getValue());
048    }
049  }
050
051  @Override
052  protected String normalize(String value) {
053    if (Strings.isNullOrEmpty(value)) {
054      return null;
055    }
056    ParseResult<String> ascii = asciiParser.parse(value);
057    String noHttp = REMOVE_HTTP_PATTERN.matcher(ascii.getPayload()).replaceAll("").toLowerCase();
058    return super.normalize(NORMALIZER.matcher(noHttp).replaceAll(""));
059  }
060
061  public static LicenseUriParser getInstance() {
062    synchronized (LicenseUriParser.class) {
063      if (singletonObject == null) {
064        singletonObject = new LicenseUriParser();
065      }
066    }
067    return singletonObject;
068  }
069
070  @Override
071  protected URI fromDictFile(String value) {
072    try {
073      return URI.create(value);
074    } catch (RuntimeException e) {
075      return null;
076    }
077  }
078}
