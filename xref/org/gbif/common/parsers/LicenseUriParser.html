14  package org.gbif.common.parsers;
15  
16  import org.gbif.api.vocabulary.License;
17  import org.gbif.common.parsers.core.ASCIIParser;
18  import org.gbif.common.parsers.core.FileBasedDictionaryParser;
19  import org.gbif.common.parsers.core.KeyValue;
20  import org.gbif.common.parsers.core.ParseResult;
21  
22  import java.net.URI;
23  import java.util.Iterator;
24  import java.util.regex.Pattern;
25  
26  import org.apache.commons.lang3.StringUtils;
27  
28  /**
29   * A license parser giving URIs, compared to {@link License} which is for our GBIF enumeration.
30   *
31   * Singleton implementation of the dictionary that uses the file /dictionaries/parse/license_uri.txt to lookup a
32   * License by its URI or its acronym/title, e.g. a lookup by "CC-BY 4.0" returns http://creativecommons.org/licenses/by/4.0/.
33   * <br/>
34   * Note a lookup by license acronym/title without a version number defaults to the latest version of that license,
35   * e.g. a lookup by "CC-BY" returns http://creativecommons.org/licenses/by/4.0/.
36   */
37  public class LicenseUriParser extends FileBasedDictionaryParser<URI> {
38  
39    private static final String COMMENT_MARKER = "#";
40    private static final String LICENSE_FILEPATH = "/dictionaries/parse/license_uri.tsv";
41    // allows us to remove the protocol part for http:// and https://
42    private static final Pattern REMOVE_HTTP_PATTERN = Pattern.compile("^https?:\\/\\/", Pattern.CASE_INSENSITIVE);
43    private static final Pattern NORMALIZER = Pattern.compile("[^\\p{IsAlphabetic}\\p{N}Â©]+");
44    protected final ASCIIParser asciiParser = ASCIIParser.getInstance();
45    private static LicenseUriParser singletonObject = null;
46  
47    private LicenseUriParser() {
48      super(false);
49      init(LicenseUriParser.class.getResourceAsStream(LICENSE_FILEPATH), COMMENT_MARKER);
50    }
51  
52    /**
53     * @param source To build the dictionary from
54     */
55    @Override
56    public void init(Iterator<KeyValue<String, URI>> source) {
57      while (source.hasNext()) {
58        KeyValue<String, URI> kvp = source.next();
59        add(kvp.getKey(), kvp.getValue());
60        // Also adds the value, to save defining all of them.
61        add(kvp.getValue().toString(), kvp.getValue());
62      }
63    }
64  
65    @Override
66    protected String normalize(String value) {
67      if (StringUtils.isEmpty(value)) {
68        return null;
69      }
70      ParseResult<String> ascii = asciiParser.parse(value);
71      String noHttp = REMOVE_HTTP_PATTERN.matcher(ascii.getPayload()).replaceAll("").toLowerCase();
72      return super.normalize(NORMALIZER.matcher(noHttp).replaceAll(""));
73    }
74  
75    public static LicenseUriParser getInstance() {
76      synchronized (LicenseUriParser.class) {
77        if (singletonObject == null) {
78          singletonObject = new LicenseUriParser();
79        }
80      }
81      return singletonObject;
82    }
83  
84    @Override
85    protected URI fromDictFile(String value) {
86      try {
87        return URI.create(value);
88      } catch (RuntimeException e) {
89        return null;
90      }
91    }
92  }
