001package org.gbif.common.parsers.core;
002
003import org.gbif.api.vocabulary.OccurrenceIssue;
004
005import java.util.Collection;
006import java.util.EnumSet;
007import java.util.List;
008import java.util.Set;
009
010import com.google.common.base.Preconditions;
011import com.google.common.base.Predicates;
012import com.google.common.collect.Iterables;
013import com.google.common.collect.Lists;
014
015/**
016 *
017 */
018public class OccurrenceParseResult<T> extends ParseResult<T> {
019  private final Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
020
021  public OccurrenceParseResult(STATUS status, CONFIDENCE confidence, T payload, List<T> alternativePayloads, Throwable error, Collection<OccurrenceIssue> issues) {
022    super(status, confidence, payload, alternativePayloads, error);
023    if (issues != null) {
024      // add non nulls only
025      Iterables.addAll(this.issues, Iterables.filter(issues, Predicates.notNull()));
026    }
027  }
028
029  public OccurrenceParseResult(STATUS status, CONFIDENCE confidence, T payload, List<T> alternativePayloads, Throwable error) {
030    super(status, confidence, payload, alternativePayloads, error);
031  }
032
033  public OccurrenceParseResult(ParseResult<T> result) {
034    super(result.getStatus(), result.getConfidence(), result.getPayload(), result.getAlternativePayloads(), result.getError());
035  }
036
037  /**
038   * @param <T>       The generic type of the payload
039   * @param confidence The confidence in the result
040   * @param payload    The payload of the parse result
041   *
042   * @return The new ParseResult which has no error and status of SUCCESS
043   */
044  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload) {
045    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null, null);
046  }
047
048  /**
049   * @return A new parse response with only the status set to FAIL
050   */
051  public static <T> OccurrenceParseResult<T> fail() {
052    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null, null);
053  }
054
055  /**
056   * @return A new parse response configured to indicate an error
057   */
058  public static <T> OccurrenceParseResult<T> error() {
059    return new OccurrenceParseResult<T>(STATUS.ERROR, null, null, null, null);
060  }
061
062  /**
063   * @param cause The cause of the error
064   *
065   * @return A new parse response configured with error and the cause
066   */
067  public static <T> OccurrenceParseResult<T> error(Throwable cause) {
068    return new OccurrenceParseResult<T>(STATUS.ERROR, null, null, null, cause);
069  }
070
071  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload, OccurrenceIssue issue) {
072    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null, null, Lists.newArrayList(issue));
073  }
074
075  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload, OccurrenceIssue ... issues) {
076    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null, null, Lists.newArrayList(issues));
077  }
078
079  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload, Collection<OccurrenceIssue> issues) {
080    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null, null, issues);
081  }
082
083  public static <T> OccurrenceParseResult<T> fail(OccurrenceIssue issue) {
084    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null, null, Lists.newArrayList(issue));
085  }
086
087  public static <T> OccurrenceParseResult<T> fail(OccurrenceIssue ... issues) {
088    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null, null, Lists.newArrayList(issues));
089  }
090
091  public static <T> OccurrenceParseResult<T> fail(Collection<OccurrenceIssue> issues) {
092    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null, null, issues);
093  }
094
095  /**
096   * This creates a ParseResult indicating a parse failure but it also has a payload. Depending on the context this
097   * may provide additional information about the failure.
098   *
099   * @param payload the payload of the parse result
100   * @param <T>    the generic type of the payload
101   *
102   * @return the new parse response which has a status of FAIL and an additional payload.
103   */
104  public static <T> OccurrenceParseResult<T> fail(T payload, Collection<OccurrenceIssue> issues) {
105    return new OccurrenceParseResult<T>(STATUS.FAIL, null, payload, null, null, issues);
106  }
107
108  public static <T> OccurrenceParseResult<T> fail(T payload, OccurrenceIssue ... issues) {
109    return new OccurrenceParseResult<T>(STATUS.FAIL, null, payload, null, null, Lists.newArrayList(issues));
110  }
111
112  /**
113   * This creates a ParseResult indicating a parse failure but it also has a payload. Depending on the context this
114   * may provide additional information about the failure.
115   *
116   * @param payload the payload of the parse result
117   * @param <T>    the generic type of the payload
118   *
119   * @return the new parse response which has a status of FAIL and an additional payload.
120   */
121  public static <T> OccurrenceParseResult<T> fail(T payload, OccurrenceIssue issue) {
122    return new OccurrenceParseResult<T>(STATUS.FAIL, null, payload, null, null, Lists.newArrayList(issue));
123  }
124
125
126  public Set<OccurrenceIssue> getIssues() {
127    return issues;
128  }
129
130  public void addIssue(OccurrenceIssue issue) {
131    Preconditions.checkNotNull(issue);
132    issues.add(issue);
133  }
134
135}
