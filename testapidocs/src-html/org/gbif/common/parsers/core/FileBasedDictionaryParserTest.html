001package org.gbif.common.parsers.core;
002
003import static org.junit.jupiter.api.Assertions.assertEquals;
004import static org.junit.jupiter.api.Assertions.assertNotNull;
005
006import java.io.InputStream;
007
008import org.junit.jupiter.api.Test;
009
010public class FileBasedDictionaryParserTest {
011
012  private static final String COMMENT_MARKER = "#";
013  private static final String TEST_DICTIONARY = "/parse/simpleDictionary.txt";
014
015  class TestParser extends FileBasedDictionaryParser<String> {
016
017    public TestParser(InputStream... inputs) {
018      super(false);
019      init(FileBasedDictionaryParserTest.class.getResourceAsStream(TEST_DICTIONARY));
020    }
021
022    @Override
023    protected String fromDictFile(String value) {
024      return value;
025    }
026  }
027
028  /**
029   * test implementation that supports commented lines
030   *
031   * @author cgendreau
032   */
033  class TestParserWithCommentSupport extends FileBasedDictionaryParser<String> {
034
035    public TestParserWithCommentSupport(InputStream... inputs) {
036      super(false);
037      init(FileBasedDictionaryParserTest.class.getResourceAsStream(TEST_DICTIONARY), COMMENT_MARKER);
038    }
039
040    @Override
041    protected String fromDictFile(String value) {
042      return value;
043    }
044  }
045
046  @Test
047  public void testParse() {
048    Parsable<String> dbp = new TestParser();
049    assertParseSuccess(dbp, "32", "Tim");
050    assertParseSuccess(dbp, "32", "TIM");
051    assertParseSuccess(dbp, "32", "tIm");
052    assertParseSuccess(dbp, "38", "Markus");
053    assertParseSuccess(dbp, "38", "MarKUS");
054    assertParseSuccess(dbp, "28", "Jose");
055    assertParseSuccess(dbp, "31", COMMENT_MARKER + "carey");
056    assertParseSuccess(dbp, "31", COMMENT_MARKER + "careY");
057
058    assertParseFailure(dbp, "Lars");
059  }
060
061  @Test
062  public void testParseFileWithComment() {
063    Parsable<String> dbp = new TestParserWithCommentSupport();
064    assertParseSuccess(dbp, "32", "Tim");
065    assertParseFailure(dbp, COMMENT_MARKER + "carey");
066  }
067
068  protected static void assertParseSuccess(Parsable<String> dbp, String expected, String input) {
069    ParseResult<String> parsed = dbp.parse(input);
070    assertNotNull(parsed);
071    assertEquals(expected, parsed.getPayload());
072    assertEquals(ParseResult.STATUS.SUCCESS, parsed.getStatus());
073    assertEquals(ParseResult.CONFIDENCE.DEFINITE, parsed.getConfidence());
074  }
075
076  protected static void assertParseFailure(Parsable<String> dbp, String input) {
077    ParseResult<String> parsed = dbp.parse(input);
078    assertEquals(ParseResult.STATUS.FAIL, parsed.getStatus());
079  }
080}
