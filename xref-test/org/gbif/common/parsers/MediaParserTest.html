13  public class MediaParserTest {
14    private final MediaParser parser = MediaParser.getInstance();
15  
16    @Test
17    public void testDetectType() throws Exception {
18      assertEquals(MediaType.StillImage, parser.detectType(buildMO("image/jp2", "abies_alba.jp2"))
19        .getType());
20      assertEquals(MediaType.StillImage, parser.detectType(buildMO("image/jpeg", null)).getType());
21      assertEquals(MediaType.StillImage, parser.detectType(buildMO("image/jpg", null)).getType());
22      assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.jp2")).getType());
23      assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.jpg")).getType());
24      assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.JPG")).getType());
25      assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.JPeg")).getType());
26      assertEquals(MediaType.StillImage, parser.detectType(buildMO(null, "abies_alba.TIFF")).getType());
27  
28      assertEquals(MediaType.Sound, parser.detectType(buildMO(null, "abies_alba.mp3")).getType());
29      assertEquals(MediaType.Sound, parser.detectType(buildMO(null, "abies_alba.flac")).getType());
30      assertEquals(MediaType.Sound, parser.detectType(buildMO(null, "abies_alba.ogg")).getType());
31    }
32  
33    @Test
34    public void testParseMimeType() throws Exception {
35      assertNull(parser.parseMimeType((String) null));
36      assertEquals("image/jp2", parser.parseMimeType("image/jp2"));
37      assertEquals("image/jpg", parser.parseMimeType("image/jpg"));
38      assertEquals("image/jpeg", parser.parseMimeType("image/jpeg"));
39      assertEquals("image/jpg", parser.parseMimeType("image/JPG"));
40      assertNull(parser.parseMimeType("JPG"));
41      assertNull(parser.parseMimeType("gif"));
42      assertNull(parser.parseMimeType("tiff"));
43      assertEquals("audio/mp3", parser.parseMimeType("audio/mp3"));
44      assertNull(parser.parseMimeType("mp3"));
45      assertEquals("audio/mp3", parser.parseMimeType(" audio/mp3"));
46      assertNull(parser.parseMimeType("mpg"));
47  
48  
49      assertNull(parser.parseMimeType((URI) null));
50      assertEquals("image/jp2", parser.parseMimeType(URI.create("abies_alba.jp2")));
51      assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.jpg")));
52      assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.jpeg")));
53      assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.JPG")));
54      assertEquals("image/jpeg", parser.parseMimeType(URI.create("abies_alba.JPG")));
55      assertEquals("image/gif", parser.parseMimeType(URI.create("abies_alba.gif")));
56      assertEquals("image/tiff", parser.parseMimeType(URI.create("abies_alba.tiff")));
57      assertEquals("audio/mpeg", parser.parseMimeType(URI.create("abies_alba.mp3")));
58      assertEquals("video/mpeg", parser.parseMimeType(URI.create("abies_alba.mpg")));
59  
60      // we default to an html link
61      assertEquals("text/html", parser.parseMimeType(URI.create("http://www.gbif.org/image?id=12")));
62      assertEquals("text/html", parser.parseMimeType(URI.create("http://arctos.database.museum/MediaSearch.cfm?action=search")));
63      assertEquals("text/html", parser.parseMimeType(URI.create("http://arctos.database.museum/MediaSearch.php?action=search")));
64      assertEquals("text/html", parser.parseMimeType(URI.create("http://arctos.database.museum/MediaSearch.pl?action=search")));
65    }
66  
67    @Test
68    public void testParseHtmlMediaLink() throws Exception {
69      MediaObject mo = buildMO(null, "http://www.gbif.org/image?id=12");
70      parser.detectType(mo);
71      assertNull(mo.getFormat());
72      assertNull(mo.getIdentifier());
73      assertEquals("http://www.gbif.org/image?id=12", mo.getReferences().toString());
74      assertNull(mo.getType());
75    }
76  
77    private MediaObject buildMO(String format, String uri) {
78      MediaObject mo = new MediaObject();
79  
80      mo.setType(null);
81      mo.setFormat(format);
82      mo.setIdentifier(uri == null ? null : URI.create(uri));
83  
84      return mo;
85    }
