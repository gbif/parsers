001package org.gbif.common.parsers.utils;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import org.junit.jupiter.api.Test;
007
008import static org.gbif.common.parsers.utils.ClassificationUtils.clean;
009import static org.gbif.common.parsers.utils.ClassificationUtils.cleanAuthor;
010
011import static org.junit.jupiter.api.Assertions.assertEquals;
012import static org.junit.jupiter.api.Assertions.assertNull;
013
014public class ClassificationUtilsTest {
015
016  @Test
017  public void testBlacklist() {
018    assertNull(clean("phylum"));
019    assertNull(clean("PHYLUM"));
020    assertNull(clean("Phylum"));
021    assertNull(clean("UnPLaCED_CLaSS"));
022    // assertNull(clean("{UNPLACED_CLASS")); Won't work because of the way clean is implemented at the moment.
023  }
024
025  @Test
026  public void testClean() {
027    Map<String, String> taxons = new HashMap<String, String>() {
028
029      {
030        put("", null);
031        put("Aus", "Aus");
032        put("AUS", "Aus");
033        put("AUS BUS", "AUS BUS");
034        put("\"Aus", "Aus");
035        put("\"Aus\"", "Aus");
036        put("{Aus", "Aus");
037        put("{ ", null);
038        put("\\", null);
039        put("\"", null);
040        put("\"\"", null);
041        put("\"?", null);
042        put("  Aus Bus   ", "Aus Bus");
043        put("\"{Aus\"", "Aus");
044        put("\"\"\"{Aus\"\"\"", "Aus");
045        put("\"? gryphoids", "gryphoids");
046      }
047    };
048
049    for (Map.Entry<String, String> taxon : taxons.entrySet()) {
050      assertEquals(taxon.getValue(), clean(taxon.getKey()));
051    }
052
053    assertNull(clean(null));
054    assertNull(clean(""), "Empty strings should return null");
055  }
056
057  @Test
058  public void testCleanAuthor() {
059    Map<String, String> authors = new HashMap<String, String>() {
060
061      {
062        put("Ant. Müller", "Ant. Müller");
063        put("Lineé", "Lineé");
064        put("Mull", "Mull");
065        put("(A. Müller) B. Meyer", "(A. Müller) B. Meyer");
066        put("B. Meyer", "B. Meyer");
067        put("L.", "L.");
068        put("A. Müller & B. Meier", "A. Müller & B. Meier");
069        put("A. Müller, B. Mei. C. Sone", "A. Müller, B. Mei. C. Sone");
070        put("L", "L");
071        put("\"A. Meyer\"", "A. Meyer");
072        // put("a. meier", "A. Meier");
073        put(" L.", "L.");
074        put("L. ", "L.");
075        put("L. {", "L.");
076        put("\\", null);
077        put("\\\\", null);
078        put("\"", null);
079        put("{} ", null);
080      }
081    };
082
083    for (Map.Entry<String, String> author : authors.entrySet()) {
084      assertEquals(author.getValue(), cleanAuthor(author.getKey()));
085    }
086
087    assertNull(cleanAuthor(null));
088    assertNull(cleanAuthor(""), "Empty strings should return null");
089  }
