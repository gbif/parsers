3   import java.time.LocalDate;
4   import java.time.Month;
5   import java.time.Year;
6   import java.time.YearMonth;

8   import org.junit.jupiter.api.Test;
9   
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  
12  /**
13   * Unit tests for {@link AtomizedLocalDate}.
14   *
15   */
16  public class AtomizedLocalDateTest {
17  
18    private static final int YEAR = 1978;
19    private static final Month MONTH_ENUM = Month.DECEMBER;
20    private static final int MONTH = 12;
21    private static final int DAY = 21;
22  
23    @Test
24    public void testAtomizedLocalDate(){
25      AtomizedLocalDate date = AtomizedLocalDate.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
26      assertAtomizedLocalDate(date, YEAR, MONTH, DAY);
27  
28      date = AtomizedLocalDate.fromTemporalAccessor(YearMonth.of(YEAR, MONTH_ENUM));
29      assertAtomizedLocalDate(date, YEAR, MONTH, null);
30  
31      date = AtomizedLocalDate.fromTemporalAccessor(Year.of(YEAR));
32      assertAtomizedLocalDate(date, YEAR, null, null);
33    }
34  
35    @Test
36    public void testAtomizedLocalDateEquals(){
37      AtomizedLocalDate date1 = AtomizedLocalDate.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
38      AtomizedLocalDate date2 = AtomizedLocalDate.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
39      assertEquals(date1, date2);
40    }
41  
42    private static void assertAtomizedLocalDate(AtomizedLocalDate date, Integer y, Integer m, Integer d){
43      assertEquals(y, date.getYear());
44      assertEquals(m, date.getMonth());
45      assertEquals(d, date.getDay());
46    }
47  }
