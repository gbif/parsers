036  /**
037   * Parse all unique occurrence values found for countries and make sure parsing doesnt get worse.
038   * If the test file is updated, values here need to be adjusted!
039   */
040  @Test
041  public void testOccurrenceValues() throws IOException {
042    final int CURRENT_COUNTRIES_PARSED = 251;
043    final int CURRENT_TESTS_SUCCESSFUL = 5273;
044
045    // remember number of successful parsed countries:
046    Map<Country, AtomicInteger> stats = Maps.newHashMap();
047    int values = 0;
048    int failed = 0;
049    int success = 0;
050    BufferedReader r = new BufferedReader(
051      new InputStreamReader(this.getClass().getResourceAsStream("/parse/countryname/occurrence_countries.txt"),
052        Charsets.UTF_8));
053    String line;
054    do {
055      line = r.readLine();
056      values++;
057      ParseResult<Country> parsed = parser.parse(line);
058      if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
059        Country c = parsed.getPayload();
060        if (!stats.containsKey(c)) {
061          stats.put(c, new AtomicInteger(0));
062        }
063        stats.get(c).incrementAndGet();
064        success++;
065      } else {
066        failed++;
067      }
068    } while (line != null);
069
070    System.out.println(values + " tested values");
071    System.out.println(failed + " failed parse results");
072    System.out.println(success + " successful parse results");
073    System.out.println(stats.size() + " distinct countries parsed");
074    assertTrue(failed < values);
075    assertTrue(CURRENT_COUNTRIES_PARSED <= stats.size());
076    assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
077  }
