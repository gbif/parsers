001package org.gbif.common.parsers.date;
002
003import javax.annotation.Nullable;
004import java.time.LocalDate;
005import java.time.LocalDateTime;
006import java.time.OffsetDateTime;
007import java.time.Year;
008import java.time.YearMonth;
009import java.time.ZoneId;
010import java.time.ZoneOffset;
011import java.time.temporal.ChronoField;
012import java.time.temporal.TemporalAccessor;
013import java.time.temporal.TemporalQueries;
014import java.util.Date;
015import java.util.Optional;
016
017/**
018 * Utility methods to work with {@link TemporalAccessor}
019 */
020public class TemporalAccessorUtils {
021
022  public static ZoneId UTC_ZONE_ID = ZoneOffset.UTC;
023
024  /**
025   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}.
026   * If the provided {@link TemporalAccessor} contains offset(timezone) information it will be used.
027   * See {@link #toDate(TemporalAccessor, boolean)} for more details.
028   *
029   * @param temporalAccessor
030   * @return  the Date object or null if a Date object can not be created
031   */
032  public static Date toDate(TemporalAccessor temporalAccessor) {
033    return toDate(temporalAccessor, false);
034  }
035
036  /**
037   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}, rounding a partial date/time to the start
038   * of the period.
039   *
040   * For {@link YearMonth}, the {@link java.util.Date} will represent the first day of the month.
041   * For {@link Year}, the {@link java.util.Date} will represent the first day of January.
042   *
043   * Remember that a {@link Date} object will always display the date in the current timezone.
044   *
045   * @param temporalAccessor
046   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
047   *                     be used ?
048   * @return the Date object or null if a Date object can not be created
049   */
050  public static Date toDate(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
051    return Date.from(toEarliestLocalDateTime(temporalAccessor, ignoreOffset).toInstant(ZoneOffset.UTC));
052  }
053
054  /**
055   * Transform a {@link TemporalAccessor} to a {@link java.time.LocalDateTime}, rounding a partial date/time to the
056   * start of the period.
057   *
058   * @param temporalAccessor
059   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
060   *                     be used ?
061   * @return the LocalDateTime object or null if a LocalDateTime object can not be created
062   */
063  public static LocalDateTime toEarliestLocalDateTime(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
064    if(temporalAccessor == null){
065      return null;
066    }
067
068    // Use offset if present
069    if(!ignoreOffset && temporalAccessor.isSupported(ChronoField.OFFSET_SECONDS)){
070      return temporalAccessor.query(OffsetDateTime::from).atZoneSameInstant(UTC_ZONE_ID).toLocalDateTime();
071    }
072
073    if(temporalAccessor.isSupported(ChronoField.SECOND_OF_DAY)){
074      return temporalAccessor.query(LocalDateTime::from);
075    }
076
077    // this may return null in case of partial dates
078    LocalDate localDate = temporalAccessor.query(TemporalQueries.localDate());
079
080    // try YearMonth
081    if(localDate == null && temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
082      YearMonth yearMonth = YearMonth.from(temporalAccessor);
083      localDate = yearMonth.atDay(1);
084    }
085
086    // try Year
087    if(localDate == null && temporalAccessor.isSupported(ChronoField.YEAR)) {
088      Year year = Year.from(temporalAccessor);
089      localDate = year.atDay(1);
090    }
091
092    if (localDate != null) {
093      return LocalDateTime.from(localDate.atStartOfDay(UTC_ZONE_ID));
094    }
095
096    return null;
097  }
098
099  /**
100   * Transform a {@link TemporalAccessor} to a {@link java.time.LocalDateTime}, rounding a partial date/time to the
101   * the end of the period.
102   *
103   * 1990 will be 1990-12-31,
104   * 1996-02 will be 1996-02-29
105   *
106   * @param temporalAccessor
107   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
108   *                     be used ?
109   * @return the LocalDateTime object or null if a LocalDateTime object can not be created
110   */
111  public static LocalDateTime toLatestLocalDateTime(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
112    if (temporalAccessor == null) {
113      return null;
114    }
115
116    // Use offset if present
117    if (!ignoreOffset && temporalAccessor.isSupported(ChronoField.OFFSET_SECONDS)) {
118      return temporalAccessor.query(OffsetDateTime::from).atZoneSameInstant(UTC_ZONE_ID).toLocalDateTime();
119    }
120
121    if (temporalAccessor.isSupported(ChronoField.SECOND_OF_DAY)) {
122      return temporalAccessor.query(LocalDateTime::from);
123    }
124
125    // this may return null in case of partial dates
126    LocalDate localDate = temporalAccessor.query(TemporalQueries.localDate());
127
128    // try YearMonth
129    if (localDate == null && temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
130      YearMonth yearMonth = YearMonth.from(temporalAccessor);
131      localDate = yearMonth.atEndOfMonth();
132    }
133
134    // try Year
135    if (localDate == null && temporalAccessor.isSupported(ChronoField.YEAR)) {
136      Year year = Year.from(temporalAccessor);
137      localDate = LocalDate.of(year.getValue(), 12, 31);
138    }
139
140    if (localDate != null) {
141      return LocalDateTime.from(localDate.atTime(23, 59, 59));
142    }
143
144    return null;
145  }
146
147  /**
148   * The idea of "best resolution" TemporalAccessor is to get the TemporalAccessor that offers more resolution than
149   * the other but they must NOT contradict.
150   * e.g. 2005-01 and 2005-01-01 will return 2005-01-01.
151   *
152   * Note that if one of the 2 parameters is null the other one will be considered having the best resolution
153   *
154   * @param ta1
155   * @param ta2
156   * @return TemporalAccessor representing the best resolution
157   */
158  public static Optional<TemporalAccessor> bestResolution(@Nullable TemporalAccessor ta1, @Nullable TemporalAccessor ta2){
159    //handle nulls combinations
160    if(ta1 == null && ta2 == null){
161      return Optional.empty();
162    }
163    if(ta1 == null){
164      return Optional.of(ta2);
165    }
166    if(ta2 == null){
167      return Optional.of(ta1);
168    }
169
170    AtomizedLocalDateTime ymd1 = AtomizedLocalDateTime.fromTemporalAccessor(ta1);
171    AtomizedLocalDateTime ymd2 = AtomizedLocalDateTime.fromTemporalAccessor(ta2);
172
173    // If they both provide the year, it must match
174    if(ymd1.getYear() != null && ymd2.getYear() != null && !ymd1.getYear().equals(ymd2.getYear())){
175      return Optional.empty();
176    }
177    // If they both provide the month, it must match
178    if(ymd1.getMonth() != null && ymd2.getMonth() != null && !ymd1.getMonth().equals(ymd2.getMonth())){
179      return Optional.empty();
180    }
181    // If they both provide the day, it must match
182    if(ymd1.getDay() != null && ymd2.getDay() != null && !ymd1.getDay().equals(ymd2.getDay())){
183      return Optional.empty();
184    }
185    // If they both provide the hour, it must match
186    if (ymd1.getHour() != null && ymd2.getHour() != null && !ymd1.getHour().equals(ymd2.getHour())) {
187      return Optional.empty();
188    }
189    // If they both provide the minute, it must match
190    if (ymd1.getMinute() != null && ymd2.getMinute() != null && !ymd1.getMinute().equals(ymd2.getMinute())) {
191      return Optional.empty();
192    }
193    // If they both provide the second, it must match
194    if (ymd1.getSecond() != null && ymd2.getSecond() != null && !ymd1.getSecond().equals(ymd2.getSecond())) {
195      return Optional.empty();
196    }
197    // If they both provide the millisecond, it must match
198    if (ymd1.getMillisecond() != null && ymd2.getMillisecond() != null && !ymd1.getMillisecond().equals(ymd2.getMillisecond())) {
199      return Optional.empty();
200    }
201
202    if(ymd1.getResolution() > ymd2.getResolution()){
203      return Optional.of(ta1);
204    }
205
206    return Optional.of(ta2);
207  }
208
209  /**
210   * Given two TemporalAccessor with possibly different resolutions, this method checks if they represent the same
211   * date, or if one is contained within the other.  The comparison does not go beyond date resolution.
212   *
213   * If a null is provided, false will be returned.
214   */
215  public static boolean sameOrContained(@Nullable TemporalAccessor ta1, @Nullable TemporalAccessor ta2) {
216    // handle nulls combinations
217    if (ta1 == null || ta2 == null) {
218      return false;
219    }
220
221    AtomizedLocalDate ymd1 = AtomizedLocalDate.fromTemporalAccessor(ta1);
222    AtomizedLocalDate ymd2 = AtomizedLocalDate.fromTemporalAccessor(ta2);
223
224    // we only deal with complete Local Date
225    if (ymd1.isComplete() && ymd2.isComplete()) {
226      return ymd1.equals(ymd2);
227    }
228
229    // check for equal years
230    if (!ymd1.getYear().equals(ymd2.getYear())) {
231      return false;
232    }
233
234    // compare months
235    if (ymd1.getMonth() == null || ymd2.getMonth() == null) {
236      return true;
237    }
238    if (!ymd1.getMonth().equals(ymd2.getMonth())) {
239      return false;
240    }
241
242    // compare days
243    if (ymd1.getDay() == null || ymd2.getDay() == null) {
244      return true;
245    }
246    return ymd1.getDay().equals(ymd2.getDay());
247  }
