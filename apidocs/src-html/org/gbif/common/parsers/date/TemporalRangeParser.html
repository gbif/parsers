028@Slf4j
029public class TemporalRangeParser implements Serializable {
030
031  private final MultiinputTemporalParser temporalParser;
032
033  @Builder(buildMethodName = "create")
034  private TemporalRangeParser(MultiinputTemporalParser temporalParser) {
035    if (temporalParser != null) {
036      this.temporalParser = temporalParser;
037    } else {
038      this.temporalParser = MultiinputTemporalParser.create();
039    }
040  }
041
042  public OccurrenceParseResult<IsoDateInterval> parse(String dateRange) {
043    return parse(null, null, null, dateRange, null, null);
044  }
045
046  public OccurrenceParseResult<IsoDateInterval> parse(String year, String month, String day, String dateRange) {
047    return parse(year, month, day, dateRange, null, null);
048  }
049
050  public OccurrenceParseResult<IsoDateInterval> parse(
051      String year,
052      String month,
053      String day,
054      String dateRange,
055      String startDayOfYear,
056      String endDayOfYear) {
057
058    Set<OccurrenceIssue> issues = new HashSet<>();
059
060    try {
061      // Even a single date will be split to two
062      String[] rawPeriod = DelimiterUtils.splitPeriod(dateRange);
063
064      Temporal from;
065      Temporal to;
066
067      // If eventDate is a multi-day range, with at least day precision, and year+month+day are set, we must test
068      // whether year+month+day falls within this range.
069      if (StringUtils.isNotBlank(year) && StringUtils.isNotBlank(month) && StringUtils.isNotBlank(day) && StringUtils.isNotBlank(dateRange)) {
070        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyStart = temporalParser.parseRecordedDate(null, null, null, rawPeriod[0], null);
071        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyEnd = temporalParser.parseRecordedDate(null, null, null, rawPeriod[1], null);
072        OccurrenceParseResult<TemporalAccessor> ymdOnly = temporalParser.parseRecordedDate(year, month, day, null, null);
073
074        if (dateRangeOnlyStart.isSuccessful() && dateRangeOnlyEnd.isSuccessful() && ymdOnly.isSuccessful()) {
075          if (dateRangeOnlyStart.getPayload().isSupported(ChronoField.DAY_OF_YEAR)
076            && dateRangeOnlyEnd.getPayload().isSupported(ChronoField.DAY_OF_YEAR)
077            && ymdOnly.getPayload().isSupported(ChronoField.DAY_OF_YEAR)) {
078            if (TemporalAccessorUtils.withinRange(dateRangeOnlyStart.getPayload(), dateRangeOnlyEnd.getPayload(), ymdOnly.getPayload())) {
079              // Then we can just check the startDayOfYear and endDayOfYear fields match.
080              from = parseAndSet(null, null, null, rawPeriod[0], startDayOfYear, issues);
081              to = parseAndSet(null, null, null, rawPeriod[1], endDayOfYear, issues);
082              // If the dateRange has different resolutions on either side, truncate it.
083              if (TemporalAccessorUtils.resolutionToSeconds(from) != TemporalAccessorUtils.resolutionToSeconds(to)) {
084                int requiredResolution = Math.min(TemporalAccessorUtils.resolutionToSeconds(from), TemporalAccessorUtils.resolutionToSeconds(to));
085                from = TemporalAccessorUtils.limitToResolution(from, requiredResolution);
086                to = TemporalAccessorUtils.limitToResolution(to, requiredResolution);
087                issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
088              }
089              log.trace("Range {}|{}|{}|{}|{}|{} succeeds with ymd within range {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
090              return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(from, to), issues);
091            }
092          }
093        }
094      }
095
096      // If eventDate is a range, and at least year is set, we must test whether year+month+day are set according to the
097      // constant parts of eventDate.
098      if (StringUtils.isNotBlank(year) && StringUtils.isNotBlank(dateRange)) {
099        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyStart = temporalParser.parseRecordedDate(null, null, null, rawPeriod[0], null);
100        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyEnd = temporalParser.parseRecordedDate(null, null, null, rawPeriod[1], null);
101        OccurrenceParseResult<TemporalAccessor> ymdOnly = temporalParser.parseRecordedDate(year, month, day, null, null);
102
103        if (dateRangeOnlyStart.isSuccessful() && dateRangeOnlyEnd.isSuccessful() && ymdOnly.isSuccessful()) {
104          Optional<TemporalAccessor> dateRangeConstant = TemporalAccessorUtils.nonConflictingDateParts(dateRangeOnlyStart.getPayload(), dateRangeOnlyEnd.getPayload(), null);
105
106          if (dateRangeConstant.isPresent() && ymdOnly.getPayload().equals(dateRangeConstant.get())) {
107            // Then we can just check the startDayOfYear and endDayOfYear fields match.
108            from = parseAndSet(null, null, null, rawPeriod[0], startDayOfYear, issues);
109            to = parseAndSet(null, null, null, rawPeriod[1], endDayOfYear, issues);
110            if (TemporalAccessorUtils.resolutionToSeconds(from) != TemporalAccessorUtils.resolutionToSeconds(to)) {
111              int requiredResolution = Math.min(TemporalAccessorUtils.resolutionToSeconds(from), TemporalAccessorUtils.resolutionToSeconds(to));
112              from = TemporalAccessorUtils.limitToResolution(from, requiredResolution);
113              to = TemporalAccessorUtils.limitToResolution(to, requiredResolution);
114              issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
115            }
116            log.trace("Range {}|{}|{}|{}|{}|{} succeeds with correct ymd parts {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
117            return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(from, to), issues);
118          }
119        }
120      }
121
122      // Otherwise, we will reduce the precision of the given dates until they all agree.
123
124      // Year+month+day, first part of eventDate, and startDay of year to the best we can get.
125      from = parseAndSet(year, month, day, rawPeriod[0], startDayOfYear, issues);
126      // Year+month+day, second part of eventDate, and endDayOfYear of year to the best we can get.
127      to = parseAndSet(year, month, day, rawPeriod[1], endDayOfYear, issues);
128      log.trace("Range {}|{}|{}|{}|{}|{} parsed to {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
129
130      // Return a failure, rather than a range with a missing start or end
131      if (from != null ^ to != null) {
132        log.debug("Range {}|{}|{}|{}|{}|{} fails due to missing start xor end {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
133        issues.add(OccurrenceIssue.RECORDED_DATE_MISMATCH);
134        return OccurrenceParseResult.fail(issues);
135      }
136
137      // If the resolutions are not equal, truncate such that they are.
138      if (from != null && to != null) {
139        if (TemporalAccessorUtils.resolutionToSeconds(from) != TemporalAccessorUtils.resolutionToSeconds(to)) {
140          log.trace("Range {}|{}|{}|{}|{}|{} has different resolutions, will be truncated {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
141          int requiredResolution = Math.min(TemporalAccessorUtils.resolutionToSeconds(from), TemporalAccessorUtils.resolutionToSeconds(to));
142          from = TemporalAccessorUtils.limitToResolution(from, requiredResolution);
143          to = TemporalAccessorUtils.limitToResolution(to, requiredResolution);
144        }
145      }
146
147      // Reverse order if needed
148      if (from != null && to != null) {
149        if (from.getClass() == to.getClass()) {
150          long rangeDiff = getRangeDiff(from, to);
151          if (rangeDiff < 0) {
152            log.trace("Range {}|{}|{}|{}|{}|{} will be reversed {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
153            issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
154            return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(to, from), issues);
155          }
156        } else {
157          issues.add(OccurrenceIssue.RECORDED_DATE_UNLIKELY);
158        }
159      }
160
161      if (from == null && to == null) {
162        log.debug("Range {}|{}|{}|{}|{}|{} could not be parsed", year, month, day, dateRange, startDayOfYear, endDayOfYear);
163        return OccurrenceParseResult.fail(issues);
164      } else if (from != null && to != null) {
165        log.trace("Range {}|{}|{}|{}|{}|{} succeeds {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
166        return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(from, to), issues);
167      } else {
168        log.error("From {} and to {} dates of range are unexpectedly not-null and null when parsing {}, {}, {}, {}, {}, {}",
169          from, to, year, month, day, dateRange, startDayOfYear, endDayOfYear);
170        issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
171        issues.add(OccurrenceIssue.INTERPRETATION_ERROR);
172        return OccurrenceParseResult.fail(issues);
173      }
174    } catch (Exception e) {
175      log.error("Exception when parsing dates: {}, {}, {}, {}, {}, {}, {}", year, month, day, dateRange, startDayOfYear, endDayOfYear);
176      log.error("Exception is "+e.getMessage(), e);
177      issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
178      issues.add(OccurrenceIssue.INTERPRETATION_ERROR);
179      return OccurrenceParseResult.fail(issues);
180    }
181  }
182
183  private Temporal parseAndSet(
184      String year,
185      String month,
186      String day,
187      String rawDate,
188      String dayOfYear,
189      Set<OccurrenceIssue> issues) {
190    OccurrenceParseResult<TemporalAccessor> result =
191        temporalParser.parseRecordedDate(year, month, day, rawDate, dayOfYear);
192    issues.addAll(result.getIssues());
193    if (result.isSuccessful()) {
194      return (Temporal) result.getPayload();
195    } else {
196      return null;
197    }
198  }
199
200  /** Compare dates and returns difference between FROM and TO dates in milliseconds */
201  private static long getRangeDiff(Temporal from, Temporal to) {
202    if (from == null || to == null) {
203      return 1L;
204    }
205    TemporalUnit unit = null;
206    if (from instanceof Year) {
207      unit = ChronoUnit.YEARS;
208    } else if (from instanceof YearMonth) {
209      unit = ChronoUnit.MONTHS;
210    } else if (from instanceof LocalDate) {
211      unit = ChronoUnit.DAYS;
212    } else if (from instanceof LocalDateTime
213        || from instanceof OffsetDateTime
214        || from instanceof ZonedDateTime) {
215      unit = ChronoUnit.SECONDS;
216    }
217    return from.until(to, unit);
218  }
