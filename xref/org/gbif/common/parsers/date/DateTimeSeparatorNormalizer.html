3   import org.apache.commons.lang3.StringUtils;
4   
5   import java.util.stream.IntStream;
6   
7   /**
8    * Internal (package private) normalizer used to support different separators for the same pattern.
9    */
10  class DateTimeSeparatorNormalizer {
11  
12    private final String searchChar;
13    private final String replacementChar;
14  
15    public DateTimeSeparatorNormalizer(String searchChar, String replacementChar) {
16      this.searchChar = searchChar;
17      this.replacementChar = replacementChar;
18    }
19  
20    /**
21     * Replaces all characters {@link DateTimeSeparatorNormalizer#searchChar} from {@code input} with
22     * {@link DateTimeSeparatorNormalizer#replacementChar}.
23     * (Replaces "alternative separators" with "separator")
24     *
25     * @param input date string
26     * @return normalized date string with one separator
27     */
28    public String normalize(String input) {
29      return input.chars()
30          .flatMap(p -> {
31            if (StringUtils.contains(searchChar, p))
32              return replacementChar.chars();
33            return IntStream.of(p);
34          })
35          .collect(StringBuilder::new, StringBuilder::appendCodePoint, StringBuilder::append)
36          .toString();
37    }
38  }
