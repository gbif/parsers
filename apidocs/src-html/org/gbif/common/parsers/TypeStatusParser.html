001package org.gbif.common.parsers;
002
003import org.apache.commons.lang3.StringUtils;
004import org.gbif.api.vocabulary.TypeStatus;
005import org.gbif.common.parsers.core.EnumParser;
006
007import java.util.regex.Matcher;
008import java.util.regex.Pattern;
009
010import org.gbif.common.parsers.core.ParseResult;
011
012/**
013 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/typeStatus.txt.
014 */
015public class TypeStatusParser extends EnumParser<TypeStatus> {
016
017  private static TypeStatusParser singletonObject = null;
018  private static final Pattern NAME_SEPARATOR = Pattern.compile("^(.+) (OF|FOR) ");
019
020  private TypeStatusParser() {
021    super(TypeStatus.class, false);
022    init(TypeStatusParser.class.getResourceAsStream("/dictionaries/parse/typeStatus.tsv"));
023  }
024
025  @Override
026  protected String normalize(String value) {
027    if (StringUtils.isEmpty(value)) {
028      return null;
029    }
030    // uppercase
031    value = value.toUpperCase().trim();
032
033    // keep only words before the typifiedName if existing, e.g. Holotype for "Holotype of Dianthus fruticosus ssp. amorginus Runemark"
034    Matcher m = NAME_SEPARATOR.matcher(value);
035    if (m.find()) {
036      value = m.group(1);
037    }
038    // remove whitespace and non letters
039    ParseResult<String> ascii = asciiParser.parse(value);
040
041    // remove all non-letters
042    return ascii.getPayload().chars()
043        .filter(p -> Character.isLetter((char) p))
044        .collect(StringBuilder::new, StringBuilder::appendCodePoint, StringBuilder::append)
045        .toString();
046  }
047
048
049  public static TypeStatusParser getInstance()
050    throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
051    synchronized (TypeStatusParser.class) {
052      if (singletonObject == null) {
053        singletonObject = new TypeStatusParser();
054      }
055    }
056    return singletonObject;
057  }
058}
