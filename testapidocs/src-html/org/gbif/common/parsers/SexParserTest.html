016package org.gbif.common.parsers;
017
018import org.gbif.api.vocabulary.Sex;
019import org.gbif.common.parsers.core.ParseResult;
020
021import java.io.BufferedReader;
022import java.io.IOException;
023import java.io.InputStreamReader;
024import java.nio.charset.StandardCharsets;
025import java.util.HashSet;
026import java.util.Set;
027
028import org.junit.jupiter.api.Test;
029
030import static org.junit.jupiter.api.Assertions.assertTrue;
031
032public class SexParserTest extends ParserTestBase<Sex> {
033
034  public SexParserTest() {
035    super(SexParser.getInstance());
036  }
037
038  /**
039   * Makes sure all Sex enum values are parsed ok.
040   */
041  @Test
042  public void testParseAllEnumValues() {
043    for (Sex c : Sex.values()) {
044      assertParseSuccess(c, c.name());
045      assertParseSuccess(c, c.name().toLowerCase());
046    }
047  }
048
049  @Test
050  public void testParseFail() {
051    assertParseFailure("Really great");
052    assertParseFailure("Padua");
053  }
054
055  @Test
056  public void testParseSuccess() {
057    assertParseSuccess(Sex.MALE, "männlich");
058    assertParseSuccess(Sex.MALE, "man");
059    assertParseSuccess(Sex.FEMALE, "women");
060  }
061
062  /**
063   * Parse all unique sex values found in our index and make sure parsing doesn't get worse.
064   * If the test file is updated, values here need to be adjusted!
065   */
066  @Test
067  public void testSexValues() throws IOException {
068    final int CURRENT_TESTS_SUCCESSFUL = 111;
069    final int CURRENT_DISTINCT = 3;
070
071    int failed = 0;
072    int success = 0;
073    Set<Sex> values = new HashSet<>();
074
075    BufferedReader r = new BufferedReader(new InputStreamReader(
076      getClass().getResourceAsStream("/parse/sexes.txt"), StandardCharsets.UTF_8));
077    String line;
078    while ((line=r.readLine()) != null) {
079      ParseResult<Sex> parsed = parser.parse(line);
080      if (parsed.isSuccessful()) {
081        //System.out.println("Succeeded: " + line + " → " + parsed.getPayload().name());
082        success++;
083        values.add(parsed.getPayload());
084      } else {
085        //System.out.println("Failed:    " + line);
086        failed++;
087      }
088    }
089
090    System.out.println(failed + " failed parse results");
091    System.out.println(success + " successful parse results");
092    System.out.println(values.size() + " distinct basis of record values parsed");
093    assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
094    assertTrue(CURRENT_DISTINCT <= values.size());
095  }
096}
