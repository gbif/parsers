001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.TypeStatus;
004
005import java.io.BufferedReader;
006import java.io.IOException;
007import java.io.InputStreamReader;
008import java.net.URL;
009
010import org.junit.jupiter.api.Test;
011
012import static org.junit.jupiter.api.Assertions.assertNotNull;
013import static org.junit.jupiter.api.Assertions.assertTrue;
014
015public class TypeStatusParserTest extends ParserTestBase<TypeStatus> {
016
017  public TypeStatusParserTest() {
018    super(TypeStatusParser.getInstance());
019  }
020
021  /**
022   * This ensures that ALL enum values are at least parsable by the name they
023   * are created with.
024   */
025  @Test
026  public void testCompleteness() {
027    for (TypeStatus t : TypeStatus.values()) {
028      assertParseSuccess(t, t.name());
029    }
030  }
031
032  @Test
033  public void testFailures() {
034    assertParseFailure(null);
035    assertParseFailure("");
036    assertParseFailure("Tim");
037  }
038
039
040  @Test
041  public void testParse() {
042    // run a few basic tests to check it bootstraps and appears to work
043    assertParseSuccess(TypeStatus.ALLOTYPE, "allotype");
044    assertParseSuccess(TypeStatus.HOLOTYPE, "Holotype of Abies alba");
045    assertParseSuccess(TypeStatus.HOLOTYPE, "Holotype for Abies alba");
046    assertParseSuccess(TypeStatus.PARATYPE, "paratype(s)");
047    assertParseSuccess(TypeStatus.PARATYPE, "Par√°tipo");
048  }
049
050  @Test
051  public void testFileCoverage() throws IOException {
052    // parses all values in our test file (generated from real occurrence data) and verifies we never get worse at parsing
053    URL fileUrl = this.getClass().getClassLoader().getResource("parse/typestatus/type_status.txt");
054    assertNotNull(fileUrl);
055
056    try (BufferedReader in = new BufferedReader(new InputStreamReader(fileUrl.openStream()))) {
057      readLines(in, this);
058    }
059
060    BatchParseResult result = getResult();
061    System.out.printf("%s out of %s lines failed to parse%n", result.failed, result.total);
062    assertTrue(result.failed > 0);
063    assertTrue(result.failed <= 26813);
064  }
065}
