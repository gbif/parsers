079  /**
080   * Build a single, possibly lenient, DateTimeParser.
081   *
082   * @param pattern
083   * @param hint
084   * @param type
085   * @return
086   */
087  private static DateTimeParser build(@NotNull String pattern, @NotNull DateFormatHint hint, @NotNull TemporalQuery<?>[] type){
088    Preconditions.checkNotNull(pattern);
089    Preconditions.checkNotNull(hint);
090
091    int minLength = getMinimumStringLengthForPattern(pattern);
092    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT);
093    return new DateTimeParser(dateTimeFormatter, null, hint, type, minLength);
094  }
095
096  private static DateTimeParser build(@NotNull String pattern, @NotNull DateFormatHint hint,
097                                      @NotNull TemporalQuery<?>[] type, ZoneId zoneId){
098    Preconditions.checkNotNull(pattern);
099    Preconditions.checkNotNull(hint);
100
101    int minLength = getMinimumStringLengthForPattern(pattern);
102    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withZone(zoneId)
103            .withResolverStyle(ResolverStyle.STRICT);
104    return new DateTimeParser(dateTimeFormatter, null, hint, type, minLength);
105  }

107  /**
108   * Build a single, strict, DateTimeParser with support for separator normalization.
109   *
110   * @param pattern
111   * @param hint
112   * @param separator
113   * @param alternativeSeparators
114   * @param type
115   * @return
116   */
117  private static DateTimeParser build(String pattern, DateFormatHint hint,  @NotNull TemporalQuery<?> type,
118                                      String separator, String alternativeSeparators){
119    return build(pattern, hint, new TemporalQuery[]{type}, separator, alternativeSeparators);
120  }
121
122  /**
123   * Build a single, possibly lenient, DateTimeParser with support for separator normalization.
124   *
125   * @param pattern
126   * @param hint
127   * @param separator
128   * @param alternativeSeparators
129   * @return
130   */
131  private static DateTimeParser build(String pattern, DateFormatHint hint, @NotNull TemporalQuery<?>[] type, String separator,
132                                             String alternativeSeparators){
133    Preconditions.checkNotNull(pattern);
134    Preconditions.checkNotNull(hint);
135    Preconditions.checkArgument(StringUtils.isNotBlank(separator), "separator must NOT be blank");
136    Preconditions.checkArgument(StringUtils.isNotBlank(alternativeSeparators), "alternativeSeparators must NOT be blank");
137
138    DateTimeSeparatorNormalizer dateTimeNormalizer = new DateTimeSeparatorNormalizer(CharMatcher.anyOf(alternativeSeparators), separator);
139    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT);
140    int minLength = getMinimumStringLengthForPattern(pattern);
141    return new DateTimeParser(dateTimeFormatter, dateTimeNormalizer, hint, type, minLength);
142  }
143
144  /**
145   *
146   * Build a single DateTimeParser from a baseYear.
147   *
148   *
149   * @param pattern pattern that includes a 2 digits year (-uu)
150   * @param hint
151   * @param baseYear
152   * @return
153   */
154  private static DateTimeParser build(String pattern, DateFormatHint hint, @NotNull TemporalQuery<?>[] type, Year baseYear){
155    int minLength = getMinimumStringLengthForPattern(pattern);
156    DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
157    return new DateTimeParser(dateTimeFormatter, null, hint, type, minLength);
158  }
159
160  /**
161   *
162   * Build a single DateTimeParser from a baseYear with support for separator normalization.
163   *
164   * @param pattern pattern that includes a 2 digits year (-uu)
165   * @param hint
166   * @param separator
167   * @param alternativeSeparators
168   * @param baseYear
169   * @return
170   */
171  private static DateTimeParser build(String pattern, DateFormatHint hint, @NotNull TemporalQuery<?>[] type, String separator,
172                                             String alternativeSeparators, Year baseYear){
173    Preconditions.checkNotNull(pattern);
174    Preconditions.checkNotNull(hint);
175    Preconditions.checkArgument(StringUtils.isNotBlank(separator), "separator must NOT be blank");
176    Preconditions.checkArgument(StringUtils.isNotBlank(alternativeSeparators), "alternativeSeparators must NOT be blank");
177
178    DateTimeSeparatorNormalizer dateTimeNormalizer = new DateTimeSeparatorNormalizer(CharMatcher.anyOf(alternativeSeparators), separator);
179    DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
180    int minLength = getMinimumStringLengthForPattern(pattern);
181    return new DateTimeParser(dateTimeFormatter, dateTimeNormalizer, hint, type, minLength);
182  }

186  /**
187   * From a {@link }DateTimeFormatter} pattern in String, get the minimum String length required for an input String to apply
188   * the pattern. This is used to quickly discard DateTimeFormatter simply based on String length of the input.
189   * Minimum length is the length of the pattern String minus the optional section(s) and quotes.
190   *
191   * @param pattern
192   * @return
193   */
194  private static int getMinimumStringLengthForPattern(String pattern){
195    pattern = ThreeTenNumericalDateParser.OPTIONAL_PATTERN_PART.matcher(pattern).replaceAll("").replaceAll("'", "");
196    return pattern.length();
197  }
198
199  private static DateTimeFormatter build2DigitsYearDateTimeFormatter(String pattern, Year baseYear){
200    Preconditions.checkState(pattern.matches(IS_YEAR_2_DIGITS_PATTERN) || pattern.equals(YEAR_2_DIGITS_PATTERN_SUFFIX),
201            "build2DigitsYearDateTimeFormatter can only be used for patterns with 2 digit year");
202    pattern = StringUtils.removeEnd(pattern, YEAR_2_DIGITS_PATTERN_SUFFIX);
203    return new DateTimeFormatterBuilder().append(DateTimeFormatter.ofPattern(pattern))
204            .appendValueReduced(ChronoField.YEAR, 2, 2, baseYear.getValue()).parseStrict().toFormatter();
205  }

222      return this;
223    }
224
225    /**
226     *
227     * @param pattern date pattern to use
228     * @param hint
229     * @param type expected {@link TemporalQuery} from the provided pattern
230     * @param zoneId
231     * @return the current builder
232     */
233    public ThreeTenDateParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type, ZoneId zoneId){
234      dateTimeParsers.add(DateTimeParserBuilder.build(pattern, hint, type, zoneId));
235      return this;
236    }
237
238    /**
239     *
240     * @param pattern date pattern to use
241     * @param hint
242     * @param type possible {@link TemporalQuery} from the provided pattern (ordered)
243     * @return the current builder
244     */
245    public ThreeTenDateParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?>[] type){
246      dateTimeParsers.add(DateTimeParserBuilder.build(pattern, hint, type));
247      return this;
248    }
249    /**
250     *
251     * @param pattern date pattern to use
252     * @param hint
253     * @param separator
254     * @param alternativeSeparators separator used in the pattern that should be used as replacement for alternativeSeparators

257    public ThreeTenDateParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?>[] type,
258                                                                   String separator, String alternativeSeparators
259                                                              ){
260      dateTimeParsers.add(DateTimeParserBuilder.build(pattern, hint, type, separator, alternativeSeparators));
261      return this;
262    }
263
264    public ThreeTenDateParserListBuilder append2DigitsYearDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,
265                                                                              Year baseYear){
266      dateTimeParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, baseYear));
267      return this;
268    }
269
270    public ThreeTenDateParserListBuilder append2DigitsYearDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,
271                                                                              String separator, String alternativeSeparators,
272                                                                              Year baseYear){
273      dateTimeParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, separator, alternativeSeparators, baseYear));
274      return this;

278      return ImmutableList.copyOf(dateTimeParsers);

282  /**
283   * More specific builder Builder used to build a {@link DateTimeMultiParser}.
284   */
285  public static class ThreeTenDateMultiParserListBuilder {
286    private DateTimeParser preferred;
287    private List<DateTimeParser> otherParsers = Lists.newArrayList();
288
289    public ThreeTenDateMultiParserListBuilder preferredDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type){
290      preferred = DateTimeParserBuilder.build(pattern, hint, type);
291      return this;
292    }
293
294    public ThreeTenDateMultiParserListBuilder preferredDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,  Year year){
295      preferred = DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, year);
296      return this;
297    }
298
299    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type){
300      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, type));
301      return this;
302    }
303
304    public ThreeTenDateMultiParserListBuilder appendDateTimeFormatter(DateTimeFormatter dateTimeFormatter, DateFormatHint hint, TemporalQuery<?> type, int minLength){
305      otherParsers.add(new DateTimeParser(dateTimeFormatter, null, hint, new TemporalQuery[]{type},  minLength));
306      return this;
307    }
308
309    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type, Year year){
310      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, year));
311      return this;
312    }
313
314    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,
315                                                               String separator, String alternativeSeparators){
316      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, separator, alternativeSeparators));
317      return this;
318    }
319
320    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,
321                                                               String separator, String alternativeSeparators, Year year){
322      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, separator, alternativeSeparators, year));
323      return this;
324    }
325
326    /**
327     * Ensure the builder is used with content we expect.
328     * Currently (this could change) we should only have one DateTimeParser per DateFormatHint.
329     *
330     * @throws IllegalStateException
331     */
332    private void validate() throws IllegalStateException {
333      Set<DateFormatHint> hints = Sets.newHashSet();
334      if(preferred != null){
335        hints.add(preferred.getHint());
336      }
337
338      for(DateTimeParser parser : otherParsers){
339        if(!hints.add(parser.getHint())){
340          throw new IllegalStateException("DateFormatHint can only be used once in a DateTimeMultiParser." +
341                  "[" + parser.getHint() + "]");
342        }
343      }
344    }
345
346    public DateTimeMultiParser build() throws IllegalStateException {
347      validate();
348      return new DateTimeMultiParser(preferred, otherParsers);
349    }
350  }
