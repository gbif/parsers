001package org.gbif.common.parsers;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertNull;
007
008public class UrlParserTest {
009
010  @Test
011  public void testParse() throws Exception {
012    assertNull(UrlParser.parse(null));
013    assertNull(UrlParser.parse(""));
014    assertNull(UrlParser.parse(" "));
015    assertNull(UrlParser.parse("-"));
016    assertNull(UrlParser.parse("tim.png"));
017    assertNull(UrlParser.parse("images/logo.gif"));
018
019    assertEquals("http://www.gbif.org/logo.png", UrlParser.parse(" http://www.gbif.org/logo.png").toString());
020    assertEquals("http://www.gbif.org/logo.png", UrlParser.parse("www.gbif.org/logo.png").toString());
021    assertEquals("https://www.gbif.org/logo.png", UrlParser.parse(" https://www.gbif.org/logo.png").toString());
022    assertEquals("ftp://www.gbif.org/logo.png", UrlParser.parse(" ftp://www.gbif.org/logo.png").toString());
023    assertEquals("http://www.gbif.org/image?id=12&format=gif,jpg",
024      UrlParser.parse("http://www.gbif.org/image?id=12&format=gif,jpg").toString());
025  }
026
027  @Test
028  public void testParseUriList() throws Exception {
029    assertEquals(0, UrlParser.parseUriList(null).size());
030    assertEquals(0, UrlParser.parseUriList("").size());
031    assertEquals(0, UrlParser.parseUriList(" ").size());
032    assertEquals(0, UrlParser.parseUriList("-").size());
033
034    assertEquals(1, UrlParser.parseUriList("http://gbif.org/logo.png").size());
035    assertEquals(1, UrlParser.parseUriList(" http://gbif.org/logo.png").size());
036    assertEquals(1, UrlParser.parseUriList("www.gbif.org/logo.png").size());
037    assertEquals(1, UrlParser.parseUriList("www.gbif.org/image?id=12").size());
038    assertEquals(1, UrlParser.parseUriList("http://www.gbif.org/image?id=12").size());
039    assertEquals(1, UrlParser.parseUriList("http://www.gbif.org/image?id=12&format=gif,jpg").size());
040
041    assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png, http://gbif.org/logo2.png").size());
042    assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png; http://gbif.org/logo2.png").size());
043    assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png | http://gbif.org/logo2.png").size());
044    assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png |#DELIMITER#| http://gbif.org/logo2.png").size());
045
046    assertEquals(3, UrlParser.parseUriList("http://gbif.org/logo.png, http://gbif.org/logo2.png, http://gbif.org/logo3.png").size());
047
048    assertEquals(2, UrlParser.parseUriList("imagens3.jbrj.gov.br/fsi/server?type=image&source=rb/0/13/11/84/00131184.jpg|imagens3.jbrj.gov.br/fsi/server?type=image&source=rb/0/13/11/84/00131184-1.jpg").size());
049  }
050}
