36    @Test
37    public void testTextDateParsing(){
38      ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2nd jan. 2018");
39      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
40  
41      parseResult = TEXTDATE_PARSER.parse("2018 january 2");
42      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
43  
44      parseResult = TEXTDATE_PARSER.parse("2018-01-02");
45      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
46  
47      parseResult = TEXTDATE_PARSER.parse("2018/01/02");
48      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
49  
50      parseResult = TEXTDATE_PARSER.parse("2018年1月2日");
51      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
52  
53      parseResult = TEXTDATE_PARSER.parse("23-March-1969");
54      assertEquals(LocalDate.of(1969, Month.MARCH, 23), LocalDate.from(parseResult.getPayload()));
55  
56      parseResult = TEXTDATE_PARSER.parse("19-Feb-85");
57      assertTrue(!parseResult.isSuccessful());
58    }

63      assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
64  
65      parseResult = TEXTDATE_PARSER.parse("1978-01-01T02:12:43Z");

68      parseResult = TEXTDATE_PARSER.parse("2018-10-15 16:21:48-03");
69      assertEquals(LocalDate.of(2018, Month.OCTOBER, 15), LocalDate.from(parseResult.getPayload()));
70  
71      parseResult = TEXTDATE_PARSER.parse("2015-09-17 10:10:03");
72      assertEquals(LocalDate.of(2015, Month.SEPTEMBER, 17), LocalDate.from(parseResult.getPayload()));
73    }

78      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));

80      parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "02.0");

83      parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "");

191   @Test
192   public void testPreferredFormatDateParsing() {
193     ParseResult<TemporalAccessor> parseResult;
194 
195     // Without any orderings, all possibilities are returned.
196     parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30.128");
197     assertEquals(CONFIDENCE.POSSIBLE, parseResult.getConfidence());
198     assertEquals(2, parseResult.getAlternativePayloads().size());
199 
200     // If a single DateComponentOrdering is provided, the date *must* be in that format to be parsed.
201     parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", YMDTZ);
202     assertEquals(STATUS.SUCCESS, parseResult.getStatus());
203 
204     parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", YMDT);
205     assertEquals(STATUS.FAIL, parseResult.getStatus());
206 
207     parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", YMD);
208     assertEquals(STATUS.FAIL, parseResult.getStatus());
209 
210     parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", DMYT);
211     assertEquals(STATUS.FAIL, parseResult.getStatus());
212 
213     // Another example
214     parseResult = TEXTDATE_PARSER.parse("2018-02-01", YMDTZ);
215     assertEquals(STATUS.FAIL, parseResult.getStatus());
216 
217     parseResult = TEXTDATE_PARSER.parse("2018-02-01", YMDT);
218     assertEquals(STATUS.FAIL, parseResult.getStatus());
219 
220     parseResult = TEXTDATE_PARSER.parse("2018-02-01", YMD);
221     assertEquals(STATUS.SUCCESS, parseResult.getStatus());
222 
223     parseResult = TEXTDATE_PARSER.parse("2018-02-01", DMY);
224     assertEquals(STATUS.FAIL, parseResult.getStatus());
225 
226     // And another
227     parseResult = TEXTDATE_PARSER.parse("1/2/2018", YMDTZ);
228     assertEquals(STATUS.FAIL, parseResult.getStatus());
229     parseResult = TEXTDATE_PARSER.parse("1/2/2018", YMDT);
230     assertEquals(STATUS.FAIL, parseResult.getStatus());
231     parseResult = TEXTDATE_PARSER.parse("1/2/2018", YMDT);
232     assertEquals(STATUS.FAIL, parseResult.getStatus());
233     parseResult = TEXTDATE_PARSER.parse("1/2/2018", DMYT);
234     assertEquals(STATUS.FAIL, parseResult.getStatus());
235     parseResult = TEXTDATE_PARSER.parse("1/2/2018", DMY);
236     assertEquals(STATUS.SUCCESS, parseResult.getStatus());
237     assertLocalDateResultEquals("2018-02-01", parseResult);
238     parseResult = TEXTDATE_PARSER.parse("1/2/2018", MDYT);
239     assertEquals(STATUS.FAIL, parseResult.getStatus());
240     parseResult = TEXTDATE_PARSER.parse("1/2/2018", MDY);
241     assertEquals(STATUS.SUCCESS, parseResult.getStatus());
242     assertLocalDateResultEquals("2018-01-02", parseResult);
243 
244     // Last one
245     parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30+0100", DMYT);
246     assertLocalDateTimeResultEquals("2018-02-01T11:20:30", parseResult);
247 
248     // If an array of DateComponentOrderings is provided, then ISO dates are matched, unambiguous dates
249     // parsed, and ambiguous dates are parsed according to the ordering.
250 
251     // Typical Australian, British, etc format:
252     parseResult = TEXTDATE_PARSER.parse("2018-02-01", DMY_FORMATS);
253     assertLocalDateResultEquals("2018-02-01", parseResult);
254     parseResult = TEXTDATE_PARSER.parse("1/2/2018", DMY_FORMATS);
255     assertLocalDateResultEquals("2018-02-01", parseResult);
256     parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30", DMY_FORMATS);
257     assertLocalDateTimeResultEquals("2018-02-01T11:20:30", parseResult);
258     parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", DMY_FORMATS);
259     assertZonedDateTimeResultEquals("2018-02-01T11:20:30+01:00", parseResult);
260     parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30.128", DMY_FORMATS);
261     assertLocalDateTimeResultEquals("2018-02-01T11:20:30.128", parseResult);
262     parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30+0100", DMY_FORMATS);
263     assertLocalDateTimeResultEquals("2018-02-01T11:20:30", parseResult);
264     parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30.128", DMY_FORMATS);
265     assertLocalDateTimeResultEquals("2018-02-01T11:20:30.128", parseResult);
266 
267     // Other European/African formats:
268     parseResult = TEXTDATE_PARSER.parse("1.2.2018", DMY_FORMATS);
269     assertLocalDateResultEquals("2018-02-01", parseResult);
270     parseResult = TEXTDATE_PARSER.parse("01.02.2018", DMY_FORMATS);
271     assertLocalDateResultEquals("2018-02-01", parseResult);
272     parseResult = TEXTDATE_PARSER.parse("1/2-2018", DMY_FORMATS);
273     assertLocalDateResultEquals("2018-02-01", parseResult);
274 
275     // Typical American format:
276     parseResult = TEXTDATE_PARSER.parse("1/2/2018", MDY_FORMATS);
277     assertLocalDateResultEquals("2018-01-02", parseResult);
278     parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30.128", MDY_FORMATS);
279     assertLocalDateTimeResultEquals("2018-01-02T11:20:30.128", parseResult);
280     parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30.128", MDY_FORMATS);
281     assertLocalDateTimeResultEquals("2018-01-02T11:20:30.128", parseResult);
282     parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30+0100", MDY_FORMATS);
283     assertZonedDateTimeResultEquals("2018-01-02T11:20:30+01:00", parseResult);
284 
285     // Supporting a local format plus ISO in a single query
286     DateComponentOrdering[] dmy_and_iso = new DateComponentOrdering[] {DMYT, DMY, ISO_ETC};
287     parseResult = TEXTDATE_PARSER.parse("2018-01-02 11:20:30+0100", dmy_and_iso);
288     assertZonedDateTimeResultEquals("2018-01-02T11:20:30+01:00", parseResult);
289     parseResult = TEXTDATE_PARSER.parse("2018-01-02", dmy_and_iso);
290     assertLocalDateResultEquals("2018-01-02", parseResult);
291     parseResult = TEXTDATE_PARSER.parse("2018-01", dmy_and_iso);
292     assertYearMonthResultEquals("2018-01", parseResult);
293     parseResult = TEXTDATE_PARSER.parse("2018年1月2日", dmy_and_iso);
294     assertLocalDateResultEquals("2018-01-02", parseResult);
295     parseResult = TEXTDATE_PARSER.parse("2/1/2018 11:20:30+0100", dmy_and_iso);
296     assertZonedDateTimeResultEquals("2018-01-02T11:20:30+01:00", parseResult);
297     parseResult = TEXTDATE_PARSER.parse("2/1/2018 11:20:30", dmy_and_iso);
298     assertLocalDateTimeResultEquals("2018-01-02T11:20:30", parseResult);
299     parseResult = TEXTDATE_PARSER.parse("2/1/2018", dmy_and_iso);
300     assertLocalDateResultEquals("2018-01-02", parseResult);
301     parseResult = TEXTDATE_PARSER.parse("2-January-2018", dmy_and_iso);
302     assertLocalDateResultEquals("2018-01-02", parseResult);
303   }
