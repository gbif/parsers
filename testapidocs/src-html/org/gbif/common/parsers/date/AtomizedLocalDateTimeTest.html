001package org.gbif.common.parsers.date;
002
003import org.junit.Test;
004import org.threeten.bp.LocalDate;
005import org.threeten.bp.LocalDateTime;
006import org.threeten.bp.LocalTime;
007import org.threeten.bp.Month;
008
009import static org.junit.Assert.assertEquals;
010
011/**
012 *
013 */
014public class AtomizedLocalDateTimeTest {
015
016  private static final int YEAR = 1978;
017  private static final Month MONTH_ENUM = Month.DECEMBER;
018  private static final int MONTH = 12;
019  private static final int DAY = 21;
020
021  private static final int HOUR = 2;
022  private static final int MINUTE = 3;
023  private static final int SECOND = 4;
024  private static final int MILLISECOND = 5;
025  // not a mistake, LocalDateTime.of take nanoseconds and not milliseconds
026  private static final int NANOSECOND = 5000000;
027
028  @Test
029  public void testAtomizedLocalDateTime(){
030    AtomizedLocalDateTime dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDateTime.of(YEAR, MONTH_ENUM, DAY, HOUR,
031            MINUTE, SECOND, NANOSECOND));
032    assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, HOUR, MINUTE, SECOND, MILLISECOND);
033
034    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDateTime.of(YEAR, MONTH_ENUM, DAY, HOUR,
035            MINUTE));
036    assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, HOUR, MINUTE, 0, 0);
037    assertEquals(7, dateTime.getResolution());
038
039    // this will probably not be supported ------
040    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalTime.of(HOUR, MINUTE));
041    assertAtomizedLocalDateTime(dateTime, null, null, null, HOUR, MINUTE, 0, 0);
042    assertEquals(4, dateTime.getResolution());
043
044    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalTime.of(HOUR, MINUTE));
045    assertAtomizedLocalDateTime(dateTime, null, null, null, HOUR, MINUTE, 0, 0);
046    assertEquals(4, dateTime.getResolution());
047    // end of this will probably not be supported ------
048
049    // ensure it works with a LocalDate alone (without time component)
050    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
051    assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, null, null, null, null);
052  }
053
054  @Test
055  public void testAtomizedLocalDateEquals(){
056    AtomizedLocalDateTime date1 = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
057    AtomizedLocalDateTime date2 = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
058    assertEquals(date1, date2);
059  }
060
061  private static void assertAtomizedLocalDateTime(AtomizedLocalDateTime dateTime, Integer year, Integer month,
062                                                  Integer day, Integer hour, Integer minute, Integer second,
063                                                  Integer millisecond){
064    assertEquals(year, dateTime.getYear());
065    assertEquals(month, dateTime.getMonth());
066    assertEquals(day, dateTime.getDay());
067    assertEquals(hour, dateTime.getHour());
068    assertEquals(minute, dateTime.getMinute());
069    assertEquals(second, dateTime.getSecond());
070    assertEquals(millisecond, dateTime.getMillisecond());
071  }
072}
