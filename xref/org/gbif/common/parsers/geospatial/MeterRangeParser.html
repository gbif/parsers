17  /**
18   * Utilities for parsing min/max meter measurements in general plus specific additions and
19   * validations for elevation, depth and distance above surface.
20   *
21   * Accepts metres and converts centimetres, kilometres, nautical miles, fathoms, feet and inches to
22   * metres.

24   * TODO: consider to use the JScience Library:
25   *  http://jscience.org/api/javax/measure/unit/package-summary.html

204   /**
205    * Takes min and max values in metres and a known precision and calculates a single mean value and
206    * its accuracy. This method tries also to parse common measurements given in fathoms, feet,
207    * inches, km or cm and converts them to metres.
208    */
209   public static MeasurementWrapper<DoubleAccuracy> parseMeterRange(
210       String minRaw, @Nullable String maxRaw, @Nullable String precisionRaw) {
211     MeasurementWrapper<DoubleAccuracy> result = new MeasurementWrapper<DoubleAccuracy>();
212 
213     MeasurementWrapper<Double> min = parseInMeter(minRaw);
214     MeasurementWrapper<Double> max = parseInMeter(maxRaw);
215     MeasurementWrapper<Double> prec = parseInMeter(precisionRaw);
216 
217     result.addIssues(min);
218     result.addIssues(max);
219     result.addIssues(prec);
220 
221     if (min.measurement == null && max.measurement == null) {
222       // both are null, return issues only
223       return result;
224     }
225 
226     // final result vars
227     Double value;
228     Double accuracy;
229 
230     // check for swapped values and apply precision if min & max exist
231     if (min.measurement != null && max.measurement != null) {
232       // flag swapped min/max
233       if (min.measurement > max.measurement) {
234         result.minMaxSwapped = true;
235         Double oldMin = min.measurement;
236         min.measurement = max.measurement;
237         max.measurement = oldMin;
238       }
239       // apply precision to min max if we have it
240       if (prec.measurement != null) {
241         min.measurement -= prec.measurement;
242         max.measurement += prec.measurement;
243       }
244       // build the arithmetic mean and set accuracy
245       value = (min.measurement + max.measurement) / 2d;
246       accuracy = (max.measurement - min.measurement) / 2d;
247 
248     } else {
249       // use the only value and precision for accuracy
250       value = min.measurement == null ? max.measurement : min.measurement;
251       accuracy = prec.measurement;
252     }
253 
254     if (value != null) {
255       result.measurement = new DoubleAccuracy(value, accuracy);
256     }
257 
258     // finally a result, bye bye!
259     return result;
260   }

262   public static OccurrenceParseResult<DoubleAccuracy> parseElevation(@Nullable String min, @Nullable String max, @Nullable String precision) {
263 
264     MeasurementWrapper<DoubleAccuracy> elevation = parseMeterRange(min, max, precision);
265 
266     Set<OccurrenceIssue> issues = new HashSet<>();
267     if(elevation.containsNonNumeric) {
268       issues.add(OccurrenceIssue.ELEVATION_NON_NUMERIC);
269     }
270     if(elevation.isInNauticalMiles || elevation.isInFathoms
271       || elevation.isInFeet || elevation.isInInches) {
272       issues.add(OccurrenceIssue.ELEVATION_NOT_METRIC);
273     }
274     if(elevation.minMaxSwapped) {
275       issues.add(OccurrenceIssue.ELEVATION_MIN_MAX_SWAPPED);
276     }
277     if(elevation.tooLarge) {
278       issues.add(OccurrenceIssue.ELEVATION_UNLIKELY);
279     }
280 
281     if (elevation.measurement == null || elevation.measurement.getValue() == null) {
282       return OccurrenceParseResult.fail(issues);
283     }
284 
285     DoubleAccuracy result = elevation.measurement;
286     // record the number of records with altitude out of range
287     if (result.getValue() > MAX_ELEVATION || result.getValue() < MIN_ELEVATION) {
288       issues.add(OccurrenceIssue.ELEVATION_UNLIKELY);
289       return OccurrenceParseResult.fail(issues);
290     }
291 
292     return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result, issues);
293   }

295   public static OccurrenceParseResult<DoubleAccuracy> parseDepth(@Nullable String min, @Nullable String max, @Nullable String precision) {
296     MeasurementWrapper<DoubleAccuracy> depth = parseMeterRange(min, max, precision);
297 
298     Set<OccurrenceIssue> issues = new HashSet<>();
299     if(depth.containsNonNumeric) {
300       issues.add(OccurrenceIssue.DEPTH_NON_NUMERIC);
301     }
302     if(depth.isInNauticalMiles || depth.isInFathoms || depth.isInFeet || depth.isInInches) {
303       issues.add(OccurrenceIssue.DEPTH_NOT_METRIC);
304     }
305     if(depth.minMaxSwapped) {
306       issues.add(OccurrenceIssue.DEPTH_MIN_MAX_SWAPPED);
307     }
308     if(depth.tooLarge) {
309       issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
310     }
311 
312     if (depth.measurement == null || depth.measurement.getValue() == null) {
313       return OccurrenceParseResult.fail(issues);
314     }
315 
316     DoubleAccuracy result = depth.measurement;
317 
318     // negate depth if its negative
319     if (result.getValue() < 0) {
320       result = new DoubleAccuracy(-1 * result.getValue(), result.getAccuracy());
321       issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
322     }
323 
324     // record the number of records with depth out of range
325     if (result.getValue() > MAX_DEPTH) {
326       issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
327       return OccurrenceParseResult.fail(issues);
328     }
329 
330     return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result, issues);
331   }

333   /**
334    * Parses a string supposed to be a value in metres.
335    * Accepts nautical miles, fathoms, feet, inches if marked with a unit and converts them
336    */
337   public static ParseResult<Double> parseMeters(String meter) {
338     MeasurementWrapper<Double> result = parseInMeter(meter);
339     if (result.getMeasurement() == null) {
340       return ParseResult.fail();
341     }
342     return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result.getMeasurement());
343   }

345   private static MeasurementWrapper<Double> parseInMeter(String meter) {
346     MeasurementWrapper<Double> iMeter = new MeasurementWrapper<>();
347 
348     if (StringUtils.isEmpty(meter)) {
349       return iMeter;
350     }
351 
352     try {
353       iMeter.containsNonNumeric = MEASURE_MARKER_PATTERN.matcher(meter).find();
354 
355       if (!iMeter.containsNonNumeric()) {
356         iMeter.measurement = NumberParser.parseDouble(meter);
357 
358       } else {
359         iMeter.isInNauticalMiles = NAUTICAL_MILES_MARKER_PATTERN.matcher(meter).find();
360         iMeter.isInFathoms = FATHOMS_MARKER_PATTERN.matcher(meter).find();
361         iMeter.isInFeet = FEET_MARKER_PATTERN.matcher(meter).find();
362         iMeter.isInInches = INCHES_MARKER_PATTERN.matcher(meter).find();
363         iMeter.isInKm = KM_MARKER_PATTERN.matcher(meter).find();
364         iMeter.isInCm = CM_MARKER_PATTERN.matcher(meter).find();
365 
366         // handle 6-7m values
367         if (SEP_MARKER_PATTERN.matcher(meter).find()) {
368           // we have been given a range
369           try {
370             String min = meter.substring(0, meter.indexOf('-')).trim();
371             min = removeMeasurementMarkers(min);
372             String max = meter.substring(meter.indexOf('-') + 1).trim();
373             max = removeMeasurementMarkers(max);
374 
375             Double minDouble = NumberParser.parseDouble(min);
376             Double maxDouble = NumberParser.parseDouble(max);
377 
378             if (minDouble == null && maxDouble != null){
379               iMeter.measurement = maxDouble;
380             } else if(maxDouble == null && minDouble != null){
381               iMeter.measurement = maxDouble;
382             } else if (minDouble != null && maxDouble != null && minDouble != 0 && maxDouble != 0
383               && maxDouble - minDouble != 0) {
384               iMeter.measurement = (maxDouble + minDouble) / 2;
385             }
386           } catch (NumberFormatException ignored) {
387           }
388 
389         } else {
390           iMeter.measurement = NumberParser.parseDouble(removeMeasurementMarkers(meter));
391         }
392 
393         if (iMeter.measurement != null) {
394           // convert to metric
395           if (iMeter.isInNauticalMiles) {
396             iMeter.measurement = convertNauticalMilesToMetres(iMeter.measurement);
397           } else if (iMeter.isInFathoms) {
398             iMeter.measurement = convertFathomsToMetres(iMeter.measurement);
399           } else if (iMeter.isInFeet) {
400             iMeter.measurement = convertFeetToMetres(iMeter.measurement);
401           } else if (iMeter.isInInches) {
402             iMeter.measurement = convertInchesToMetres(iMeter.measurement);
403           } else if (iMeter.isInKm){
404             iMeter.measurement = convertKmToMetres(iMeter.measurement);
405           } else if (iMeter.isInCm){
406             iMeter.measurement = convertCmToMetres(iMeter.measurement);
407           }
408         }
409       }
410     } catch (NumberFormatException e) {
411       LOG.debug("Unparsable metre measurement: {}, {}", meter, e.getMessage());
412     }
413 
414     // round to centimetres
415     if (iMeter.measurement != null) {
416       iMeter.measurement = Math.round(iMeter.measurement * 100.0) / 100.0;
417     }
418     return iMeter;
419   }

421   /**
422    * Remove "m" etc.
423    *
424    * @param s to remove measurement markers from.
425    *
426    * @return a new string with all measurements removed (i.e. replaced by the empty string)
427    */
428   private static String removeMeasurementMarkers(String s) {
429     if (s == null) return null;
430     return REMOVE_MEASURE_MARKER_PATTERN.matcher(s).replaceAll("");
431   }

433   private static double convertNauticalMilesToMetres(double nauticalMiles) {
434     return nauticalMiles * NAUTICAL_MILES_TO_METRES;
435   }
436 
437   private static double convertFathomsToMetres(double fathoms) {
438     return fathoms * FATHOMS_TO_METRES;
439   }

441   private static double convertFeetToMetres(double feet) {

443   }

445   private static double convertInchesToMetres(double inches) {
446     return inches * INCHES_TO_METRES;

449   private static double convertKmToMetres(double km) {

451   }

453   private static double convertCmToMetres(double cm) {
454     return cm * CM_TO_METRES;

457   /**
458    * @return rounded int value or null if it was null or exceeds the maximum an int can hold
459    */
460   private static Integer roundedInt(Double x) {
461     if (x == null) return null;
462 
463     Long xl = Math.round(x);
464     if (xl > Integer.MAX_VALUE) {
465       throw new IllegalArgumentException("Long too big for an integer");
466     }
467     return xl.intValue();
468   }
