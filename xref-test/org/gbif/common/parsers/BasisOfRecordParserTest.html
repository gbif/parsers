74    /**
75     * Parse all unique basis of record values found in our index and make sure parsing doesn't get worse.
76     * If the test file is updated, values here need to be adjusted!
77     */
78    @Test
79    public void testOccurrenceValues() throws IOException {
80      final int CURRENT_TESTS_SUCCESSFUL = 248;
81      final int CURRENT_DISTINCT = 9;
82  
83      int failed = 0;
84      int success = 0;
85      Set<BasisOfRecord> values = Sets.newHashSet();
86  
87      BufferedReader r = new BufferedReader(new InputStreamReader(
88        getClass().getResourceAsStream("/parse/basisofrecord/test_bor.txt"), Charsets.UTF_8));
89      String line;
90      while ((line=r.readLine()) != null) {
91        ParseResult<BasisOfRecord> parsed = parser.parse(line);
92        if (parsed.isSuccessful()) {
93          success++;
94          values.add(parsed.getPayload());
95        } else {
96          System.out.println("Failed: " + line);
97          failed++;
98        }
99      }
100 
101     System.out.println(failed + " failed parse results");
102     System.out.println(success + " successful parse results");
103     System.out.println(values.size() + " distinct basis of record values parsed");
104     assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
105     assertTrue(CURRENT_DISTINCT <= values.size());
106   }
