24  /**
25   * AtomizedLocalDate is a simple immutable class to hold local date data from a
26   * {@link TemporalAccessor}.
27   *
28   * Thread-Safe, immutable class.
29   */
30  public class AtomizedLocalDate {
31  
32    // Maximum resolution for a complete Local Date
33    public static final int COMPLETE_LOCAL_DATE_RESOLUTION = 3;
34  
35    private final Integer year;
36    private final Integer month;
37    private final Integer day;
38  
39    private final int resolution;
40  
41    private AtomizedLocalDate(Integer year, Integer month, Integer day) {
42      this.year = year;
43      this.month = month;
44      this.day = day;
45  
46      int res = 0;
47      if (year != null) {
48        res++;
49      }
50      if (month != null) {
51        res++;
52      }
53      if (day != null) {
54        res++;
55      }
56      resolution = res;
57    }
58  
59    /**
60     * Build a new instance of {@link AtomizedLocalDate} based on a {@link TemporalAccessor}.
61     * This is done by extracting the {@link ChronoField}.
62     * @param temporalAccessor
63     * @return
64     */
65    public static AtomizedLocalDate fromTemporalAccessor(TemporalAccessor temporalAccessor) {
66      if (temporalAccessor == null) {
67        return null;
68      }
69  
70      Integer y = null, m = null, d = null;
71      if (temporalAccessor.isSupported(ChronoField.YEAR)) {
72        y = temporalAccessor.get(ChronoField.YEAR);
73      }
74  
75      if (temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
76        m = temporalAccessor.get(ChronoField.MONTH_OF_YEAR);
77      }
78  
79      if (temporalAccessor.isSupported(ChronoField.DAY_OF_MONTH)) {
80        d = temporalAccessor.get(ChronoField.DAY_OF_MONTH);
81      }
82  
83      return new AtomizedLocalDate(y, m, d);
84    }
85  
86    public Integer getYear() {
87      return year;
88    }
89  
90    public Integer getMonth() {
91      return month;
92    }
93  
94    public Integer getDay() {
95      return day;
96    }
97  
98    public int getResolution() {
99      return resolution;
100   }

102   /**
103    * Is the resolution of this local date matches the maximum.
104    * @return
105    */
106   public boolean isComplete() {
107     return COMPLETE_LOCAL_DATE_RESOLUTION == resolution;
108   }
