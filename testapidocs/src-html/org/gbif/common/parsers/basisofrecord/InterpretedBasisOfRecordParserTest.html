001package org.gbif.common.parsers.basisofrecord;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004import org.gbif.common.parsers.InterpretedParserTest;
005
006import org.junit.Test;
007
008/**
009 *
010 */
011public class InterpretedBasisOfRecordParserTest extends InterpretedParserTest<BasisOfRecord> {
012
013  public InterpretedBasisOfRecordParserTest() {
014    super(InterpretedBasisOfRecordParser.getInstance());
015  }
016
017  /**
018   * This ensures that ALL enum values are at least parsable by the name they
019   * are created with.
020   */
021  @Test
022  public void testCompleteness() {
023    for (BasisOfRecord t : BasisOfRecord.values()) {
024      System.out.println("Testing [" + t.name()
025                         + "].  Failures below might indicate new BasisOfRecord added to the BasisOfRecord enum but not to the parse file");
026      assertParseSuccess(parser, t, t.name());
027    }
028  }
029
030  @Test
031  public void testFailures() {
032    assertParseFailure(parser, null);
033    assertParseFailure(parser, "");
034    assertParseFailure(parser, "Tim");
035  }
036
037
038  @Test
039  public void testParse() {
040    // run a few basic tests to check it bootstraps and appears to work
041    assertParseSuccess(parser, BasisOfRecord.LIVING_SPECIMEN, "living organism");
042    assertParseSuccess(parser, BasisOfRecord.PRESERVED_SPECIMEN, "SpeCImen");
043    assertParseSuccess(parser, BasisOfRecord.OBSERVATION, "sound recording");
044    assertParseSuccess(parser, BasisOfRecord.OBSERVATION, "Observation");
045    assertParseSuccess(parser, BasisOfRecord.OBSERVATION, "O");
046  }
047
048}
