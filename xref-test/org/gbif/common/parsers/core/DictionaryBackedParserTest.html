11  public class DictionaryBackedParserTest {
12  
13    @Test
14    public void testParse() {
15      List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
16      source.add(new KeyValue<String, Integer>("Tim", 32));
17      source.add(new KeyValue<String, Integer>("Markus", 38));
18      source.add(new KeyValue<String, Integer>("Jose", 28));
19      DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(false);
20      dbp.init(source.iterator());
21  
22      assertParsed(dbp, "Tim", 32);
23      assertParsed(dbp, "TIM", 32);
24      assertParsed(dbp, "tIm", 32);
25      assertParsed(dbp, " \t Markus \t ", 38);
26      assertParsed(dbp, "MarKUS", 38);
27      assertParsed(dbp, "Jose", 28);
28      assertEquals(ParseResult.STATUS.FAIL, dbp.parse("Lars").getStatus());
29    }
30  
31    @Test
32    public void testSensitiveParse() {
33      List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
34      source.add(new KeyValue<String, Integer>("Matt", 29));
35      DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(true);
36      dbp.init(source.iterator());
37  
38      assertParsed(dbp, "Matt", 29);
39      assertParsed(dbp, " \t Matt \n ", 29);
40      assertEquals(ParseResult.STATUS.FAIL, dbp.parse("MATT").getStatus());
41    }
42  
43    private void assertParsed(DictionaryBackedParser<Integer> dbp, String input, Integer payload) {
44      assertNotNull(dbp.parse(input));
45      assertEquals(ParseResult.STATUS.SUCCESS, dbp.parse(input).getStatus());
46      assertEquals(ParseResult.CONFIDENCE.DEFINITE, dbp.parse(input).getConfidence());
47      assertEquals(payload, dbp.parse(input).getPayload());
48    }
