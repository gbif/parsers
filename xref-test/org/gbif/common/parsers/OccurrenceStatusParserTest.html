18  import org.junit.jupiter.api.Test;

20  public class OccurrenceStatusParserTest extends ParserTestBase<OccurrenceStatus> {
21  
22    public OccurrenceStatusParserTest() {
23      super(OccurrenceStatusParser.getInstance());
24    }
25  
26    @Test
27    public void testParseAllEnumValues() {
28      for (OccurrenceStatus c : OccurrenceStatus.values()) {
29        assertParseSuccess(c, c.name());
30        assertParseSuccess(c, c.name().toLowerCase());
31      }
32    }
33  
34    @Test
35    public void testParsing() {

37      // than whether the occurrence was present or absent.
