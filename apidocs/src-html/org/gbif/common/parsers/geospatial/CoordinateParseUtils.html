019/**
020 * Utilities for assisting in the parsing of latitude and longitude strings into Decimals.
021 */
022public class CoordinateParseUtils {
023  private final static String DMS = "\\s*(\\d{1,3})\\s*[°d ]"
024                                  + "\\s*([0-6]?\\d)\\s*['m ]"
025                                  + "\\s*(?:"
026                                    + "([0-6]?\\d(?:[,.]\\d+)?)"
027                                    + "\\s*(?:\"|''|s)?"
028                                  + ")?\\s*";
029  private final static Pattern DMS_SINGLE = Pattern.compile("^" + DMS + "$", Pattern.CASE_INSENSITIVE);
030  private final static Pattern DMS_COORD = Pattern.compile("^" + DMS + "([NSEOW])" + "[ ,;/]?" + DMS + "([NSEOW])$", Pattern.CASE_INSENSITIVE);
031  private final static String POSITIVE = "NEO";
032  private CoordinateParseUtils() {
033    throw new UnsupportedOperationException("Can't initialize class");
034  }
035
036  private static final Logger LOG = LoggerFactory.getLogger(CoordinateParseUtils.class);
037
038  /**
039   * This parses string representations of latitude and longitude values. It tries its best to interpret the values and
040   * indicates any problems in its result as {@link org.gbif.api.vocabulary.OccurrenceIssue}.
041   * When the {@link ParseResult.STATUS} is FAIL the payload will be null and one or more issues should be set
042   * in {@link org.gbif.common.parsers.core.OccurrenceParseResult#getIssues()}.
043   *
044   * Coordinate precision will be 6 decimals at most, any more precise values will be rounded.
045   *
046   * Supported standard formats are the following, with dots or optionally a comma as the decimal marker:
047   * <ul>
048   *   <li>43.63871944444445</li>
049   *   <li>N43°38'19.39"</li>
050   *   <li>43°38'19.39"N</li>
051   *   <li>43d 38m 19.39s N</li>
052   *   <li>43 38 19.39</li>
053   * </ul>
054   *
055   * @param latitude  The decimal latitude
056   * @param longitude The decimal longitude
057   *
058   * @return The parse result
059   */
060  public static OccurrenceParseResult<LatLng> parseLatLng(final String latitude, final String longitude) {
061    if (Strings.isNullOrEmpty(latitude) || Strings.isNullOrEmpty(longitude)) {
062      return OccurrenceParseResult.fail();
063    }
064    Double lat = NumberParser.parseDouble(latitude);
065    Double lng = NumberParser.parseDouble(longitude);
066    if (lat == null || lng == null) {
067      // try degree minute seconds
068      try {
069        lat = parseDMS(latitude, true);
070        lng = parseDMS(longitude, false);
071      } catch (IllegalArgumentException e) {
072        return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
073      }
074    }
075
076    return validateAndRound(lat, lng);
077  }
078
079  private static boolean inRange(double lat, double lon) {
080    if (Double.compare(lat, 90) <= 0 && Double.compare(lat, -90) >= 0 && Double.compare(lon, 180) <= 0 && Double.compare(lon, -180) >= 0) {
081      return true;
082    }
083    return false;
084  }
085
086  private static boolean isLat(String direction) {
087    if ("NS".contains(direction.toUpperCase())) {
088      return true;
089    }
090    return false;
091  }
092
093  private static int coordSign(String direction) {
094    return POSITIVE.contains(direction.toUpperCase()) ? 1 : -1;
095  }
096
097  // 02° 49' 52" N      131° 47' 03" E
098  public static OccurrenceParseResult<LatLng> parseVerbatimCoordinates(final String coordinates) {
099    if (Strings.isNullOrEmpty(coordinates)) {
100      return OccurrenceParseResult.fail();
101    }
102    Matcher m = DMS_COORD.matcher(coordinates);
103    if (m.find()) {
104      final String dir1 = m.group(4);
105      final String dir2 = m.group(8);
106      // first parse coords regardless whether they are lat or lon
107      double c1 = coordFromMatcher(m, 1,2,3, dir1);
108      double c2 = coordFromMatcher(m, 5,6,7, dir2);
109      // now see what order the coords are in:
110      if (isLat(dir1) && !isLat(dir2)) {
111        return validateAndRound(c1, c2);
112
113      } else if (!isLat(dir1) && isLat(dir2)) {
114        return validateAndRound(c2, c1);
115
116      } else {
117        return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
118      }
119
120    } else if(coordinates.length() > 4) {
121      // try to split and then use lat/lon parsing
122      for (final char delim : ",;/ ".toCharArray()) {
123        int cnt = StringUtils.countMatches(coordinates, String.valueOf(delim));
124        if (cnt == 1) {
125          String[] latlon = StringUtils.split(coordinates, delim);
126          if (latlon.length == 2) {
127            return parseLatLng(latlon[0], latlon[1]);
128          }
129        }
130      }
131    }
132    return OccurrenceParseResult.fail(OccurrenceIssue.COORDINATE_INVALID);
133  }
134
135  private static OccurrenceParseResult<LatLng> validateAndRound(double lat, double lon) {
136    // collecting issues for result
137    Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
138
139    // round to 6 decimals
140    final double latOrig = lat;
141    final double lngOrig = lon;
142    lat = roundTo6decimals(lat);
143    lon = roundTo6decimals(lon);
144    if (Double.compare(lat, latOrig) != 0 || Double.compare(lon, lngOrig) != 0) {
145      issues.add(OccurrenceIssue.COORDINATE_ROUNDED);
146    }
147
148    // 0,0 is too suspicious
149    if (Double.compare(lat, 0) == 0 && Double.compare(lon, 0) == 0) {
150      issues.add(OccurrenceIssue.ZERO_COORDINATE);
151      return OccurrenceParseResult.success(ParseResult.CONFIDENCE.POSSIBLE, new LatLng(0, 0), issues);
152    }
153
154    // if everything falls in range
155    if (inRange(lat, lon)) {
156      return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new LatLng(lat, lon), issues);
157    }
158
159    // if lat is out of range, but in range of the lng,
160    // assume swapped coordinates.
161    // note that should we desire to trust the following records, we would need to clear the flag for the records to
162    // appear in
163    // search results and maps etc. however, this is logic decision, that goes above the capabilities of this method
164    if (Double.compare(lat, 90) > 0 || Double.compare(lat, -90) < 0) {
165      // try and swap
166      if (inRange(lon, lat)) {
167        issues.add(OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
168        return OccurrenceParseResult.fail(new LatLng(lat, lon), issues);
169      }
170    }
171
172    // then something is out of range
173    issues.add(OccurrenceIssue.COORDINATE_OUT_OF_RANGE);
174    return OccurrenceParseResult.fail(issues);
175
176  }

178  /**
179   * Parses a single DMS coordinate
180   * @param coord
181   * @param lat
182   * @return the converted decimal up to 6 decimals accuracy
183   */
184  @VisibleForTesting
185  protected static double parseDMS(String coord, boolean lat) {
186    final String DIRS = lat ? "NS" : "EOW";
187    coord = coord.trim().toUpperCase();
188
189    if (coord.length() > 3) {
190      // preparse the direction and remove it from the string to avoid a very complex regex
191      char dir = 'n';
192      if (DIRS.contains(String.valueOf(coord.charAt(0)))) {
193        dir = coord.charAt(0);
194        coord = coord.substring(1);
195      } else if (DIRS.contains(String.valueOf(coord.charAt(coord.length()-1)))) {
196        dir = coord.charAt(coord.length()-1);
197        coord = coord.substring(0, coord.length()-1);
198      }
199      // without the direction chuck it at the regex
200      Matcher m = DMS_SINGLE.matcher(coord);
201      if (m.find()) {
202        return coordFromMatcher(m, 1,2,3, String.valueOf(dir));
203      }
204    }
205    throw new IllegalArgumentException();
206  }
207
208  private static double coordFromMatcher(Matcher m, int idx1, int idx2, int idx3, String sign) {
209    return roundTo6decimals(coordSign(sign) *
210        dmsToDecimal( NumberParser.parseDouble(m.group(idx1)), NumberParser.parseDouble(m.group(idx2)), NumberParser.parseDouble(m.group(idx3)) ));
211  }
212  private static double dmsToDecimal(double degree, Double minutes, Double seconds) {
213    minutes = minutes == null ? 0 : minutes;
214    seconds = seconds == null ? 0 : seconds;
215    return degree + (minutes / 60) + (seconds / 3600);
216  }
217
218  // round to 6 decimals (~1m precision) since no way we're getting anything legitimately more precise
219  private static Double roundTo6decimals(Double x) {
220    return x == null ? null : Math.round(x * Math.pow(10, 6)) / Math.pow(10, 6);
221  }
