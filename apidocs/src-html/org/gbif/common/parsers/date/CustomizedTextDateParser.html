001package org.gbif.common.parsers.date;
002
003import java.io.Serializable;
004import java.time.temporal.TemporalAccessor;
005
006import org.apache.commons.lang3.ArrayUtils;
007import org.gbif.common.parsers.core.ParseResult;
008
009import javax.annotation.Nullable;
010
011/**
012 * Wrap a set of DateTimeFormatters to <code>parse</code> method.
013 * <p>
014 * If the ISO format parsing process fails, use <code>orderings</code> formatters to parse the date.
015 */
016public class CustomizedTextDateParser implements TemporalParser, Serializable {
017  private DateComponentOrdering[] orderings;
018  private TextDateParser parser = new TextDateParser();
019
020  /**
021   * @param orderings a set of DateTimeFormatters
022   */
023  public static TemporalParser getInstance(DateComponentOrdering[] orderings) {
024    CustomizedTextDateParser ptdp = new CustomizedTextDateParser();
025    ptdp.orderings = orderings;
026    return ptdp;
027  }
028
029  @Override
030  public ParseResult<TemporalAccessor> parse(String input) {
031    if (ArrayUtils.isNotEmpty(orderings)) {
032      return parser.parse(input, orderings);
033    } else {
034      return parser.parse(input);
035    }
036  }
037
038  @Override
039  public ParseResult<TemporalAccessor> parse(String input,
040                                             @Nullable DateComponentOrdering ordering) {
041    return parser.parse(input, ordering);
042  }
043
044  @Override
045  public ParseResult<TemporalAccessor> parse(String input,
046                                             @Nullable DateComponentOrdering[] orderings) {
047    return parser.parse(input, orderings);
048  }
049
050  @Override
051  public ParseResult<TemporalAccessor> parse(
052    @Nullable String year,
053    @Nullable String month,
054    @Nullable String day) {
055    return parser.parse(year, month, day);
056  }
057
058  @Override
059  public ParseResult<TemporalAccessor> parse(
060    @Nullable Integer year,
061    @Nullable Integer month,
062    @Nullable Integer day) {
063    return parser.parse(year, month, day);
064  }
065}
