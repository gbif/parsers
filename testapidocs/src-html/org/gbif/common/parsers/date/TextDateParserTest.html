050  @Test
051  public void testTextDateParsing(){
052    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2nd jan. 2018");
053    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
054
055    parseResult = TEXTDATE_PARSER.parse("2018 january 2");
056    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
057
058    parseResult = TEXTDATE_PARSER.parse("2018-01-02");
059    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
060
061    parseResult = TEXTDATE_PARSER.parse("2018/01/02");
062    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
063
064    parseResult = TEXTDATE_PARSER.parse("2018年1月2日");
065    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
066
067    parseResult = TEXTDATE_PARSER.parse("23-March-1969");
068    assertEquals(LocalDate.of(1969, Month.MARCH, 23), LocalDate.from(parseResult.getPayload()));
069
070    parseResult = TEXTDATE_PARSER.parse("19-Feb-85");
071    assertTrue(!parseResult.isSuccessful());
072  }

074  @Test
075  public void testTextDateTimeParsing(){
076    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("1978-1-1T00:00");
077    assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
078
079    parseResult = TEXTDATE_PARSER.parse("1978-01-01T02:12:43Z");
080    assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
081
082    parseResult = TEXTDATE_PARSER.parse("2018-10-15 16:21:48-03");
083    assertEquals(LocalDate.of(2018, Month.OCTOBER, 15), LocalDate.from(parseResult.getPayload()));
084
085    parseResult = TEXTDATE_PARSER.parse("2015-09-17 10:10:03");
086    assertEquals(LocalDate.of(2015, Month.SEPTEMBER, 17), LocalDate.from(parseResult.getPayload()));
087  }

133    // Some of the many ways of specifying a whole year.
134    // So many cases aren't really needed, as we're only taking the start of the interval anyway.
135    // Just with years
136    testDateRangeStart(1999, 1, 1, "1999");
137    testDateRangeStart(1999, 1, 1, "1999/2000");

141    testDateRangeStart(1999, 1, 1, "1999-01/12");
142    testDateRangeStart(1999, 1, 1, "1999-01/P12M");
143    // With years and days
144    testDateRangeStart(1999, 1, 1, "1999-001/1999-365");
145    testDateRangeStart(1999, 1, 1, "1999-001/P365D");
146    // With years, months and days
147    testDateRangeStart(1999, 1, 1, "1999-01-01/1999-12-31");
148    testDateRangeStart(1999, 1, 1, "1999-01-01/12-31");
149    testDateRangeStart(1999, 1, 1, "1999-01-01/P365D");
150    // With years, months, days and a time

153    testDateRangeStart(1999, 1, 1, "1999-01-01T00:00:00/1999-12-31T24:00:00");
154    testDateRangeStart(1999, 1, 1, "1999-01-01T00:00:00+0000/1999-12-31T24:00:00+0000");
155

205  @Test
206  public void testPreferredFormatDateParsing() {
207    ParseResult<TemporalAccessor> parseResult;
208
209    // Without any orderings, all possibilities are returned.
210    parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30.128");
211    assertEquals(CONFIDENCE.POSSIBLE, parseResult.getConfidence());
212    assertEquals(2, parseResult.getAlternativePayloads().size());
213
214    // If a single DateComponentOrdering is provided, the date *must* be in that format to be parsed.
215    parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", YMDTZ);
216    assertEquals(STATUS.SUCCESS, parseResult.getStatus());
217
218    parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", YMDT);
219    assertEquals(STATUS.FAIL, parseResult.getStatus());
220
221    parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", YMD);
222    assertEquals(STATUS.FAIL, parseResult.getStatus());
223
224    parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", DMYT);
225    assertEquals(STATUS.FAIL, parseResult.getStatus());
226
227    // Another example
228    parseResult = TEXTDATE_PARSER.parse("2018-02-01", YMDTZ);
229    assertEquals(STATUS.FAIL, parseResult.getStatus());
230
231    parseResult = TEXTDATE_PARSER.parse("2018-02-01", YMDT);
232    assertEquals(STATUS.FAIL, parseResult.getStatus());
233
234    parseResult = TEXTDATE_PARSER.parse("2018-02-01", YMD);
235    assertEquals(STATUS.SUCCESS, parseResult.getStatus());
236
237    parseResult = TEXTDATE_PARSER.parse("2018-02-01", DMY);
238    assertEquals(STATUS.FAIL, parseResult.getStatus());
239
240    // And another
241    parseResult = TEXTDATE_PARSER.parse("1/2/2018", YMDTZ);
242    assertEquals(STATUS.FAIL, parseResult.getStatus());
243    parseResult = TEXTDATE_PARSER.parse("1/2/2018", YMDT);
244    assertEquals(STATUS.FAIL, parseResult.getStatus());
245    parseResult = TEXTDATE_PARSER.parse("1/2/2018", YMDT);
246    assertEquals(STATUS.FAIL, parseResult.getStatus());
247    parseResult = TEXTDATE_PARSER.parse("1/2/2018", DMYT);
248    assertEquals(STATUS.FAIL, parseResult.getStatus());
249    parseResult = TEXTDATE_PARSER.parse("1/2/2018", DMY);
250    assertEquals(STATUS.SUCCESS, parseResult.getStatus());
251    assertLocalDateResultEquals("2018-02-01", parseResult);
252    parseResult = TEXTDATE_PARSER.parse("1/2/2018", MDYT);
253    assertEquals(STATUS.FAIL, parseResult.getStatus());
254    parseResult = TEXTDATE_PARSER.parse("1/2/2018", MDY);
255    assertEquals(STATUS.SUCCESS, parseResult.getStatus());
256    assertLocalDateResultEquals("2018-01-02", parseResult);
257
258    // Last one
259    parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30+0100", DMYT);
260    assertLocalDateTimeResultEquals("2018-02-01T11:20:30", parseResult);
261
262    // If an array of DateComponentOrderings is provided, then ISO dates are matched, unambiguous dates
263    // parsed, and ambiguous dates are parsed according to the ordering.
264
265    // Typical Australian, British, etc format:
266    parseResult = TEXTDATE_PARSER.parse("2018-02-01", DMY_FORMATS);
267    assertLocalDateResultEquals("2018-02-01", parseResult);
268    parseResult = TEXTDATE_PARSER.parse("1/2/2018", DMY_FORMATS);
269    assertLocalDateResultEquals("2018-02-01", parseResult);
270    parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30", DMY_FORMATS);
271    assertLocalDateTimeResultEquals("2018-02-01T11:20:30", parseResult);
272    parseResult = TEXTDATE_PARSER.parse("2018-02-01T11:20:30+0100", DMY_FORMATS);
273    assertZonedDateTimeResultEquals("2018-02-01T11:20:30+01:00", parseResult);
274    parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30.128", DMY_FORMATS);
275    assertLocalDateTimeResultEquals("2018-02-01T11:20:30.128", parseResult);
276    parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30+0100", DMY_FORMATS);
277    assertLocalDateTimeResultEquals("2018-02-01T11:20:30", parseResult);
278    parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30.128", DMY_FORMATS);
279    assertLocalDateTimeResultEquals("2018-02-01T11:20:30.128", parseResult);
280
281    // Other European/African formats:
282    parseResult = TEXTDATE_PARSER.parse("1.2.2018", DMY_FORMATS);
283    assertLocalDateResultEquals("2018-02-01", parseResult);
284    parseResult = TEXTDATE_PARSER.parse("01.02.2018", DMY_FORMATS);
285    assertLocalDateResultEquals("2018-02-01", parseResult);
286    parseResult = TEXTDATE_PARSER.parse("1/2-2018", DMY_FORMATS);
287    assertLocalDateResultEquals("2018-02-01", parseResult);
288
289    // Typical American format:
290    parseResult = TEXTDATE_PARSER.parse("1/2/2018", MDY_FORMATS);
291    assertLocalDateResultEquals("2018-01-02", parseResult);
292    parseResult = TEXTDATE_PARSER.parse("1/2/2018T11:20:30.128", MDY_FORMATS);
293    assertLocalDateTimeResultEquals("2018-01-02T11:20:30.128", parseResult);
294    parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30.128", MDY_FORMATS);
295    assertLocalDateTimeResultEquals("2018-01-02T11:20:30.128", parseResult);
296    parseResult = TEXTDATE_PARSER.parse("1/2/2018 11:20:30+0100", MDY_FORMATS);
297    assertZonedDateTimeResultEquals("2018-01-02T11:20:30+01:00", parseResult);
298
299    // Supporting a local format plus ISO in a single query
300    DateComponentOrdering[] dmy_and_iso = new DateComponentOrdering[] {DMYT, DMY, ISO_ETC};
301    parseResult = TEXTDATE_PARSER.parse("2018-01-02 11:20:30+0100", dmy_and_iso);
302    assertZonedDateTimeResultEquals("2018-01-02T11:20:30+01:00", parseResult);
303    parseResult = TEXTDATE_PARSER.parse("2018-01-02", dmy_and_iso);
304    assertLocalDateResultEquals("2018-01-02", parseResult);
305    parseResult = TEXTDATE_PARSER.parse("2018-01", dmy_and_iso);
306    assertYearMonthResultEquals("2018-01", parseResult);
307    parseResult = TEXTDATE_PARSER.parse("2018年1月2日", dmy_and_iso);
308    assertLocalDateResultEquals("2018-01-02", parseResult);
309    parseResult = TEXTDATE_PARSER.parse("2/1/2018 11:20:30+0100", dmy_and_iso);
310    assertZonedDateTimeResultEquals("2018-01-02T11:20:30+01:00", parseResult);
311    parseResult = TEXTDATE_PARSER.parse("2/1/2018 11:20:30", dmy_and_iso);
312    assertLocalDateTimeResultEquals("2018-01-02T11:20:30", parseResult);
313    parseResult = TEXTDATE_PARSER.parse("2/1/2018", dmy_and_iso);
314    assertLocalDateResultEquals("2018-01-02", parseResult);
315    parseResult = TEXTDATE_PARSER.parse("2-January-2018", dmy_and_iso);
316    assertLocalDateResultEquals("2018-01-02", parseResult);
317  }

323  private void testDateRangeStart(int y, int m, int d, int hh, int mm, int ss, String inDate) {
324    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse(inDate);
325    assertTrue(parseResult.isSuccessful());
326    assertEquals(LocalDateTime.of(y, m, d, hh, mm, ss), TemporalAccessorUtils.toEarliestLocalDateTime(parseResult.getPayload(), true));
327  }
