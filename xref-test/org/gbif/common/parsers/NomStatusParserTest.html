26  import org.apache.commons.lang3.StringUtils;
27  import org.junit.jupiter.api.Test;
28  
29  import static org.junit.jupiter.api.Assertions.assertNotNull;
30  import static org.junit.jupiter.api.Assertions.assertTrue;
31  
32  /**
33   * Parses a single nomenclatural status out of a given string.
34   * TODO: Improve parser and return a status set instead, see nomStatus.txt file with comma separated values
35   */
36  public class NomStatusParserTest extends ParserTestBase<NomenclaturalStatus> {
37  
38      public NomStatusParserTest() {
39          super(NomStatusParser.getInstance());
40      }
41  
42      /**
43       * This ensures that ALL enum values are at least parsable by the name they
44       * are created with.
45       */
46      @Test
47      public void testCompleteness() {
48          for (NomenclaturalStatus t : NomenclaturalStatus.values()) {
49              assertParseSuccess(t, t.name());
50              assertParseSuccess(t, t.name().toLowerCase());
51              assertParseSuccess(t, t.name().replace("_", "").toLowerCase());
52              if (StringUtils.isNotEmpty(t.getLatinLabel())) {
53                  assertParseSuccess(t, t.getLatinLabel());
54              }
55              if (StringUtils.isNotEmpty(t.getAbbreviatedLabel())) {
56                  assertParseSuccess(t, t.getAbbreviatedLabel());
57              }
58          }
59      }
60  
61      @Test
62      public void testFailures() {
63          assertParseFailure(null);
64          assertParseFailure("");
65          assertParseFailure("Tim");
66      }
67  
68      @Test
69      public void testParse() {
70          // run a few basic tests to check it bootstraps and appears to work
71          assertParseSuccess(NomenclaturalStatus.NUDUM, "nom nud");
72          assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, "later HOmonym");
73          assertParseSuccess(NomenclaturalStatus.NEW_COMBINATION, ParseResult.CONFIDENCE.PROBABLE, "comb nov , rejected name, homotypic syn\tcomb nov , rejected name, homotypic syn");
74          assertParseSuccess(NomenclaturalStatus.CONSERVED, "nom cons , see art 14");
75          assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, ParseResult.CONFIDENCE.PROBABLE, "nom illeg ; (1775), non mill (1768)");
76          assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, "nom illegit , arts 53 1, 53 3");
77          assertParseSuccess(NomenclaturalStatus.INVALID, ParseResult.CONFIDENCE.PROBABLE, "nom inval , art 32 1(b), see art 18 3");
78          assertParseSuccess(NomenclaturalStatus.NEW_SPECIES, "sp nov , heterotypic syn");
79      }
80  
81      @Test
82      public void testFileCoverage() throws IOException {
83          // parses all values in our test file (generated from real occurrence data) and verifies we never get worse at parsing
84          URL fileUrl = this.getClass().getClassLoader().getResource("parse/nom_status.txt");
85          assertNotNull(fileUrl);
86  
87          try (BufferedReader in = new BufferedReader(new InputStreamReader(fileUrl.openStream()))) {
88              readLines(in, this);
89          }
90  
91          BatchParseResult result = getResult();
92          System.out.printf("%s out of %s lines failed to parse%n", result.failed, result.total);
93          assertTrue(result.failed > 0);
94          assertTrue(result.failed <= 166); // out of 747
95      }
96  }
