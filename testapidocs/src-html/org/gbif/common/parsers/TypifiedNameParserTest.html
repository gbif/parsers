001package org.gbif.common.parsers;
002
003import org.junit.jupiter.api.Test;
004
005public class TypifiedNameParserTest extends ParserTestBase<String> {
006
007  public TypifiedNameParserTest() {
008    super(TypifiedNameParser.getInstance());
009  }
010
011  @Test
012  public void testParse() throws Exception {
013    assertParseSuccess("Abies alba", null, " Holotype of Abies alba");
014    assertParseSuccess("Dianthus fruticosus subsp. amorginus Runemark", null, "Holotype of Dianthus fruticosus ssp. amorginus Runemark");
015    assertParseSuccess("Abies alba", null, " Holotype of: Abies alba");
016    assertParseSuccess("Abies alba", null, " Holotype of  Abies alba.");
017
018    assertParseFailure("Part of Holotype");
019    assertParseFailure("Figured Specimen");
020//    assertParseFailure("Cast of Figured Specimen");
021    assertParseFailure("Cast of holotype");
022    assertParseFailure("Cast of syntype");
023    assertParseFailure("mark");
024  }
025
026}
