001package org.gbif.common.parsers;
002
003import org.gbif.api.model.checklistbank.ParsedName;
004import org.gbif.common.parsers.core.Parsable;
005import org.gbif.common.parsers.core.ParseResult;
006import org.gbif.nameparser.NameParser;
007import org.gbif.nameparser.UnparsableException;
008
009import java.util.regex.Matcher;
010import java.util.regex.Pattern;
011
012import com.google.common.base.Strings;
013import com.google.common.collect.Range;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Singleton implementation using regex to extract a scientific name after a typestatus from a string.
019 * For example given the input "Holotype of Dianthus fruticosus ssp. amorginus Runemark"
020 * the parser will extract Dianthus fruticosus ssp. amorginus Runemark.
021 */
022public class TypifiedNameParser implements Parsable<String> {
023  protected final Logger log = LoggerFactory.getLogger(getClass());
024  private static TypifiedNameParser singletonObject = null;
025
026  private static final Range<Integer> REASONABLE_NAME_SIZE_RANGE = Range.closed(4, 40);
027  private static final NameParser NAME_PARSER = new NameParser();
028  private static final Pattern NAME_SEPARATOR = Pattern.compile("\\sOF\\W*\\s+\\W*(.+)\\W*\\s*$", Pattern.CASE_INSENSITIVE);
029  private static final Pattern CLEAN_WHITESPACE = Pattern.compile("\\s+");
030
031  private TypifiedNameParser() {
032  }
033
034  @Override
035  public ParseResult<String> parse(String input) {
036    if (!Strings.isNullOrEmpty(input)) {
037      Matcher m = NAME_SEPARATOR.matcher(input);
038      if (m.find()) {
039        String name = m.group(1);
040        // make sure the name does not end with "type", see http://dev.gbif.org/issues/browse/POR-2703
041        if (!name.endsWith("type")) {
042          try {
043            ParsedName pn = NAME_PARSER.parse(name,null);
044            return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, pn.canonicalNameComplete());
045
046          } catch (UnparsableException e) {
047            log.debug("Cannot parse typified name: [{}] from input [{}]", name, input);
048            name = CLEAN_WHITESPACE.matcher(name).replaceAll(" ").trim();
049            if (REASONABLE_NAME_SIZE_RANGE.contains(name.length())) {
050              return ParseResult.success(ParseResult.CONFIDENCE.POSSIBLE, name);
051            }
052          }
053        }
054      }
055    }
056    return ParseResult.fail();
057  }
058
059  public static TypifiedNameParser getInstance() {
060    synchronized (TypifiedNameParser.class) {
061      if (singletonObject == null) {
062        singletonObject = new TypifiedNameParser();
063      }
064    }
065    return singletonObject;
066  }
067
068
069}
