30    private static final Joiner DATE_JOINER = Joiner.on("-").skipNulls();

35    // enumeration of the fields supported in the date handling
36    public enum DATE_FIELD {
37      DAY, MONTH, YEAR

40    // Dictionary for interpreting string months
41    // Important that the FIRST is the canonical form, followed
42    // by abbreviations.  Be very careful not to add any ambiguous terms
43    // such as "J" "A" etc.  Do not add a "." at the end of abbreviations
44    // as normalization will automatically do this.
45    protected static final String[][] MONTHS = {
46        {"January", "Jan", "J", "Ene", "Ja"}, // Ene. is abbreviated Spanish
47        {"February", "Feb", "F", "Fe"},
48        {"March", "Mar"},
49        {"April", "Apr", "Ap", "Abr"}, // Abr. is abbreviated Spanish
50        {"May"},
51        {"June", "Jun"},
52        {"July", "Jul"},
53        {"August", "Aug", "Au", "Ago"}, // Ago seen a lot in data
54        {"September", "Sep", "Sept", "Set", "S"}, // Set is a common mispelling
55        {"October", "Oct", "O", "Oc"},
56        {"November", "Nov", "N", "No"},
57        {"December", "Dec", "D", "De"}
58    };

60    protected static final int CURRENT_YEAR = new GregorianCalendar().get(Calendar.YEAR);

74    /**
75     * Takes a String representation that is in the format specified and attempts to construct a date.
76     *
77     * @param input  To convert to a Date()
78     * @param format The format of the date, e.g. dd/mm/yyyy
79     *
80     * @return The result of the parsing
81     */
82    public static ParseResult<Date> parse(String input, String format) {
83      try {
84        Date d = DateUtils.parseDate(input, format);
85        LOGGER.debug("Parsed input[{}] in format[{}] to date[{}]", new Object[] {input, format, d});
86        return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, d);
87      } catch (ParseException e) {
88        LOGGER.debug("Exception caught trying to parse a date", e);
89        return ParseResult.fail();
90      }
91    }

93    /**
94     * Utility to check that the field appears valid in it's status use. Day must fall within 1-31, month within 1-12 and
95     * year within 1-<currentyear>, null will always be invalid.
96     *
97     * @param value To test
98     * @param field The context that the value is being used
99     *
100    * @return true or false depending on the validity of the value in it's context
101    */
102   public static boolean isValidUse(String value, DATE_FIELD field) {
103     if (value == null) return false;
104 
105     try {
106       int valueAsInt = Integer.parseInt(value);
107       switch (field) {
108         case DAY:
109           return valueAsInt >= 1 && valueAsInt <= 31;
110         case MONTH:
111           return valueAsInt >= 1 && valueAsInt <= 12;
112         case YEAR:
113           return valueAsInt >= 1 && valueAsInt <= new GregorianCalendar().get(Calendar.YEAR);
114       }
115     } catch (NumberFormatException e) {
116       LOGGER.debug("{} does not appear valid as a {}", value, field);
117     }
118     return false;
119   }

121   /**
122    * Checks if a {@link YearMonthDay} object <i>maybe</i> represents a valid date in the sense that the date really
123    * exists (e.g. no February 31st)
124    * As the components of a YearMonthDay may be null it only checks this validity if all components are not null.
125    *
126    * @param ymd to check
127    *
128    * @return true if it may be a valid partial date, false if it is sure that it is an invalid date.
129    */
130   public static boolean isDateCoercible(YearMonthDay ymd) {
131     if (ymd.getYear() == null || ymd.getMonth() == null || ymd.getDay() == null) {
132       return true;
133     }
134 
135     // SimpleDateFormat is not thread-safe. Do not reuse or if you reuse use Thread locals or other synchronization
136     SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
137     sdf.setLenient(false);
138     return sdf.parse(DATE_JOINER.join(ymd.getYear(), ymd.getMonth(), ymd.getDay()), new ParsePosition(0)) != null;
139   }

141   /**
142    * Tries to create a valid date (complete or partial) out of three Strings for year, month and day.
143    * If all three components are null the parse will fail.
144    *
145    * @param year  to parse. May be null.
146    * @param month to parse. May be null.
147    * @param day   to parse. May be null.
148    *
149    * @return result of the parsing
150    */
151   public static ParseResult<Date> parse(@Nullable String year, @Nullable String month, @Nullable String day) {
152     if (year == null && month == null && day == null) {
153       return ParseResult.fail();
154     }
155 
156     month = normalizeMonth(month);
157     Date d = STRING_TO_DATE_PARSER.strictParse(day + "/" + month + "/" + year, new String[]{"dd/MM/yyyy", "dd/mm/yyyy"});
158     if (d != null) {
159       return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, d);
160     }
161     return ParseResult.fail();
162   }

164   /**
165    * Note that YearMonthDay can be sparsely populated
166    *
167    * @param input To convert to a YearMonthDay()
168    *
169    * @return The result of the parsing
170    */
171   public static YearMonthDay atomize(Date input) {
172     YearMonthDay ymd = new YearMonthDay();
173     Calendar c = Calendar.getInstance();
174     c.setTime(input);
175     ymd.setDay(String.valueOf(c.get(Calendar.DAY_OF_MONTH)));
176     ymd.setMonth(String.valueOf(c.get(Calendar.MONTH) + 1)); // months are zero indexed
177     ymd.setYear(String.valueOf(c.get(Calendar.YEAR)));
178     return ymd;
179   }

182   public static YearMonthDay normalize(String year, String month, String day) {
183     YearMonthDay ymd = new YearMonthDay();
184     // normalize float values
185     year = normalizeFloat(year);
186     month = normalizeFloat(month);
187     day = normalizeFloat(day);
188 
189     // normalize string months
190     month = normalizeMonth(month);
191 
192     // remove nulls
193     day = "\\N".equalsIgnoreCase(day) ? null : day;
194     month = "\\N".equalsIgnoreCase(month) ? null : month;
195     year = "\\N".equalsIgnoreCase(year) ? null : year;
196     year = inferCentury(year);
197     if (isValidUse(year, DATE_FIELD.YEAR)) {
198       ymd.setYear(year);
199     }
200     if (isValidUse(month, DATE_FIELD.MONTH)) {
201       ymd.setMonth(month);
202     }
203     if (isValidUse(day, DATE_FIELD.DAY)) {
204       ymd.setDay(day);
205     }
206 
207     // if everything is null, it is possible that
208     // the year contains a full date.  Attempt a parse of that
209     if (ymd.representsNull() && StringUtils.isNotBlank(year)) {
210       LOGGER.debug("Attempting to infer day, month and year from year field: {}", year);
211       ParseResult<YearMonthDay> result = atomize(year);
212       if (ParseResult.STATUS.SUCCESS == result.getStatus()) {
213         LOGGER.debug("Year[{}] resulted in: {}", year, result.getPayload());
214         ymd = result.getPayload();
215       } else {
216         try {
217           // last ditch attempt to get a date
218           // bad days are converted by the gregorian calendar (31/06/2000 for example becomes 01/07/2000)
219           Date p = DateUtils.parseDate(year, new String[] {"yyyy-MM-dd", "yyyy/MM/dd"});
220           GregorianCalendar gc = new GregorianCalendar();
221           gc.setTime(p);
222           ymd.setYear(String.valueOf(gc.get(Calendar.YEAR)));
223           ymd.setMonth(String.valueOf(1+gc.get(Calendar.MONTH)));
224           ymd.setDay(String.valueOf(gc.get(Calendar.DAY_OF_MONTH)));
225 
226         } catch (ParseException e) {
227         }
228       }
229     }
230 
231     if (!isDateCoercible(ymd)) {
232       ymd.setYear(null);
233       ymd.setMonth(null);
234       ymd.setDay(null);
235     }
236 
237     // Final check before returning
238     if (!isValidUse(ymd.getYear(), DATE_FIELD.YEAR)) {
239       ymd.setYear(null);
240     }
241     if (!isValidUse(ymd.getMonth(), DATE_FIELD.MONTH)) {
242       ymd.setMonth(null);
243     }
244     if (!isValidUse(ymd.getDay(), DATE_FIELD.DAY)) {
245       ymd.setDay(null);
246     }
247 
248     return ymd;
249   }

251   /**
252    * Should the year be lower than 100, this will assume it is current century if makes sense, otherwise uses previous
253    * century. At time of writing (2010) a value of 1-99 will be treated as 2010, whereas 11 will be considered 1911.
254    * Next year, the same code would interpret 11 as 2011.  This is not foolproof, but much more likely than the year
255    * 0011.
256    */
257   public static String inferCentury(String year) {
258     try {
259       int y = Integer.parseInt(year);
260       if (y >= 0 && y <= 99) {
261         // if it is in the future, it can't be
262         if ((CURRENT_YEAR - (CURRENT_YEAR % 100)) + y > CURRENT_YEAR) {
263           y = CURRENT_YEAR - (CURRENT_YEAR % 100) - 100 + y;
264         } else {
265           y = CURRENT_YEAR - (CURRENT_YEAR % 100) + y;
266         }
267         return String.valueOf(y);
268       }
269     } catch (NumberFormatException e) {
270       // no problem, it does not look like a 2 digit year anyway
271     }
272     return year;
273   }

275   /**
276    * Often values are seen as Float rather than int, due to a database export The year "1978" is actually seen as
277    * "1978.0".  Where this is detected, the string is normalized to the INT representation
278    *
279    * @param value To check
280    *
281    * @return the integer value
282    */
283   public static String normalizeFloat(String value) {
284     if (value != null && value.contains(".0")) {
285       try {
286         Double d = new Double(value);
287         if ((double) d.intValue() == d.doubleValue()) {
288           return String.valueOf(d.intValue());
289         }
290       } catch (NumberFormatException e) {
291       }
292     }
293     return value;
294   }

296   /**
297    * Parses the date and then atomizes it to the year month and day
298    *
299    * @param input To convert to a YearMonthDay()
300    *
301    * @return The result of the parsing
302    */
303   public static ParseResult<YearMonthDay> atomize(String input) {
304     input = normalizeMonth(input);
305     ParseResult<Date> parsed = parse(input);
306     if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
307       // atomize
308       YearMonthDay atomized = atomize(parsed.getPayload());
309       // construct a combined result of the 2 stages
310       return ParseResult.success(parsed.getConfidence(), atomized);
311 
312     } else if (ParseResult.STATUS.ERROR == parsed.getStatus()) {
313       return ParseResult.error(parsed.getError());
314     }
315 
316     return ParseResult.fail();
317   }

319   /**
320    * Often months come in the form Sept. September etc. This will convert many variations into the numerical version
321    *
322    * @param input To normalize
323    *
324    * @return the normalized month
325    */
326   public static String normalizeMonth(String input) {
327     if (input != null) {
328       int m = 1;
329       for (String[] monthValues : MONTHS) {
330         for (String month : monthValues) {
331           if (StringUtils.equalsIgnoreCase(month, input) || StringUtils.equalsIgnoreCase(month + ".", input)) {
332             return padTo2Chars(m);
333           }
334         }
335         m++;
336       }
337     }
338     return input;
339   }

341   /**
342    * @param value To convert into a 2 char String.  Must be 1-12 inclusive or else strange results might occur
343    *
344    * @return 1 becomes "01", 5 becomes "05", 10 becomes "10" etc
345    */
346   protected static String padTo2Chars(int value) {
347     return value < 10 ? "0" + value : String.valueOf(value);
348   }
