031/**
032 * Utilities for parsing min/max meter measurements in general plus specific additions and
033 * validations for elevation, depth and distance above surface.
034 *
035 * Accepts metres and converts centimetres, kilometres, nautical miles, fathoms, feet and inches to
036 * metres.
037 *
038 * TODO: consider to use the JScience Library:
039 *  http://jscience.org/api/javax/measure/unit/package-summary.html
040 *  http://jscience.org/api/javax/measure/unit/Unit.html
041 */
042public class MeterRangeParser {
043  private static final Logger LOG = LoggerFactory.getLogger(MeterRangeParser.class);

045  /**
046   * Pattern for removing measurement denominations
047   */
048  private static final Pattern MEASURE_MARKER_PATTERN = Pattern.compile("[a-zA-Zµ]");
049
050  /**
051   * Pattern to remove measurement markers (like "m", "FT.")
052   */
053  private static final Pattern REMOVE_MEASURE_MARKER_PATTERN = Pattern.compile("[a-zA-Zµ\" ]\\.?");

058  private static final Pattern NAUTICAL_MILES_MARKER_PATTERN = Pattern.compile("nm|nmi", Pattern.CASE_INSENSITIVE);

061   * Pattern for recognising measurements in fathoms

064

068  private static final Pattern FEET_MARKER_PATTERN = Pattern.compile("ft|'|feet", Pattern.CASE_INSENSITIVE);
069
070  /**

073  private static final Pattern INCHES_MARKER_PATTERN = Pattern.compile("in|\"|inch|inches", Pattern.CASE_INSENSITIVE);

076   * Pattern for recognising measurements in km
077   */
078  private static final Pattern KM_MARKER_PATTERN = Pattern.compile("km|kilometres|kilometers", Pattern.CASE_INSENSITIVE);
079
080  /**
081   * Pattern for recognising measurements in cm

083  private static final Pattern CM_MARKER_PATTERN = Pattern.compile("cm|centimetres|centimeters", Pattern.CASE_INSENSITIVE);

100  /**
101   * Constant factor to convert from feet to metres.

103  private static final float FEET_TO_METRES = 0.3048f;
104
105  /**
106   * Constant factor to convert from inches to metres.

108  private static final float INCHES_TO_METRES = 0.0254f;
109
110  /**
111   * Constant factor to convert from km to metres.

113  private static final float KM_TO_METRES = 1000f;

116   * Constant factor to convert from cm to metres.

118  private static final float CM_TO_METRES = 0.1f;

120  /**
121   * The lowest elevation value recognised as valid:
122   * 10,971 m (35,994 ft) Challenger Deep, Mariana Trench[32]
123   */
124  private static final int MIN_ELEVATION = -11000;
125
126  /**
127   * The highest elevation value recognised as valid, Mount Everest 8,848 m (29,029 ft).
128   * Highest point in the Troposphere: 17km
129   * Highest point in the Stratosphere: 55km
130   * Max altitude of airline cruises: 13km
131   * Max altitude of weather ballons: 34km
132   *
133   * @see <a href="http://en.wikipedia.org/wiki/Atmosphere_of_Earth">Atmosphere in wikipedia</a>
134   */
135  private static final int MAX_ELEVATION = 17000;
136
137  /**
138   * The lowest elevation value recognised as valid:
139   * 10,971 m (35,994 ft) Challenger Deep, Mariana Trench[32]
140   */
141  private static final int MAX_DEPTH = Math.abs(MIN_ELEVATION);
142
143  /**
144   * Largest holes dug into the earth are ~4km.
145   */
146  private static final int MIN_DISTANCE = -5000;
147
148  /**
149   * Same as elevation, we use the upper end of the Troposphere.
150   */
151  private static final int MAX_DISTANCE = MAX_ELEVATION;

205    public void addIssues(MeasurementWrapper<?> issues) {
206      isInNauticalMiles = isInNauticalMiles || issues.isInNauticalMiles;
207      isInFathoms = isInFathoms || issues.isInFathoms;
208      isInFeet = isInFeet || issues.isInFeet;
209      isInInches = isInInches || issues.isInInches;
210      isInCm = isInCm || issues.isInCm;
211      isInKm = isInKm || issues.isInKm;
212      containsNonNumeric = containsNonNumeric || issues.containsNonNumeric;
213      minMaxSwapped = minMaxSwapped || issues.minMaxSwapped;
214      tooLarge = tooLarge || issues.tooLarge;
215    }

218  /**
219   * Takes min and max values in metres and a known precision and calculates a single mean value and
220   * its accuracy. This method tries also to parse common measurements given in fathoms, feet,
221   * inches, km or cm and converts them to metres.
222   */
223  public static MeasurementWrapper<DoubleAccuracy> parseMeterRange(
224      String minRaw, @Nullable String maxRaw, @Nullable String precisionRaw) {
225    MeasurementWrapper<DoubleAccuracy> result = new MeasurementWrapper<DoubleAccuracy>();
226
227    MeasurementWrapper<Double> min = parseInMeter(minRaw);
228    MeasurementWrapper<Double> max = parseInMeter(maxRaw);
229    MeasurementWrapper<Double> prec = parseInMeter(precisionRaw);
230
231    result.addIssues(min);
232    result.addIssues(max);
233    result.addIssues(prec);
234
235    if (min.measurement == null && max.measurement == null) {
236      // both are null, return issues only
237      return result;
238    }
239
240    // final result vars
241    Double value;
242    Double accuracy;
243
244    // check for swapped values and apply precision if min & max exist
245    if (min.measurement != null && max.measurement != null) {
246      // flag swapped min/max
247      if (min.measurement > max.measurement) {
248        result.minMaxSwapped = true;
249        Double oldMin = min.measurement;
250        min.measurement = max.measurement;
251        max.measurement = oldMin;
252      }
253      // apply precision to min max if we have it
254      if (prec.measurement != null) {
255        min.measurement -= prec.measurement;
256        max.measurement += prec.measurement;
257      }
258      // build the arithmetic mean and set accuracy
259      value = (min.measurement + max.measurement) / 2d;
260      accuracy = (max.measurement - min.measurement) / 2d;
261
262    } else {
263      // use the only value and precision for accuracy
264      value = min.measurement == null ? max.measurement : min.measurement;
265      accuracy = prec.measurement;
266    }
267
268    if (value != null) {
269      result.measurement = new DoubleAccuracy(value, accuracy);
270    }
271
272    // finally a result, bye bye!
273    return result;
274  }

276  public static OccurrenceParseResult<DoubleAccuracy> parseElevation(@Nullable String min, @Nullable String max, @Nullable String precision) {
277
278    MeasurementWrapper<DoubleAccuracy> elevation = parseMeterRange(min, max, precision);
279
280    Set<OccurrenceIssue> issues = new HashSet<>();
281    if(elevation.containsNonNumeric) {
282      issues.add(OccurrenceIssue.ELEVATION_NON_NUMERIC);
283    }
284    if(elevation.isInNauticalMiles || elevation.isInFathoms
285      || elevation.isInFeet || elevation.isInInches) {
286      issues.add(OccurrenceIssue.ELEVATION_NOT_METRIC);
287    }
288    if(elevation.minMaxSwapped) {
289      issues.add(OccurrenceIssue.ELEVATION_MIN_MAX_SWAPPED);
290    }
291    if(elevation.tooLarge) {
292      issues.add(OccurrenceIssue.ELEVATION_UNLIKELY);
293    }
294
295    if (elevation.measurement == null || elevation.measurement.getValue() == null) {
296      return OccurrenceParseResult.fail(issues);
297    }
298
299    DoubleAccuracy result = elevation.measurement;
300    // record the number of records with altitude out of range
301    if (result.getValue() > MAX_ELEVATION || result.getValue() < MIN_ELEVATION) {
302      issues.add(OccurrenceIssue.ELEVATION_UNLIKELY);
303      return OccurrenceParseResult.fail(issues);
304    }
305
306    return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result, issues);
307  }

309  public static OccurrenceParseResult<DoubleAccuracy> parseDepth(@Nullable String min, @Nullable String max, @Nullable String precision) {
310    MeasurementWrapper<DoubleAccuracy> depth = parseMeterRange(min, max, precision);
311
312    Set<OccurrenceIssue> issues = new HashSet<>();
313    if(depth.containsNonNumeric) {
314      issues.add(OccurrenceIssue.DEPTH_NON_NUMERIC);
315    }
316    if(depth.isInNauticalMiles || depth.isInFathoms || depth.isInFeet || depth.isInInches) {
317      issues.add(OccurrenceIssue.DEPTH_NOT_METRIC);
318    }
319    if(depth.minMaxSwapped) {
320      issues.add(OccurrenceIssue.DEPTH_MIN_MAX_SWAPPED);
321    }
322    if(depth.tooLarge) {
323      issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
324    }
325
326    if (depth.measurement == null || depth.measurement.getValue() == null) {
327      return OccurrenceParseResult.fail(issues);
328    }
329
330    DoubleAccuracy result = depth.measurement;
331
332    // negate depth if its negative
333    if (result.getValue() < 0) {
334      result = new DoubleAccuracy(-1 * result.getValue(), result.getAccuracy());
335      issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
336    }
337
338    // record the number of records with depth out of range
339    if (result.getValue() > MAX_DEPTH) {
340      issues.add(OccurrenceIssue.DEPTH_UNLIKELY);
341      return OccurrenceParseResult.fail(issues);
342    }
343
344    return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result, issues);
345  }

347  /**
348   * Parses a string supposed to be a value in metres.
349   * Accepts nautical miles, fathoms, feet, inches if marked with a unit and converts them
350   */
351  public static ParseResult<Double> parseMeters(String meter) {
352    MeasurementWrapper<Double> result = parseInMeter(meter);
353    if (result.getMeasurement() == null) {
354      return ParseResult.fail();
355    }
356    return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, result.getMeasurement());
357  }

359  private static MeasurementWrapper<Double> parseInMeter(String meter) {
360    MeasurementWrapper<Double> iMeter = new MeasurementWrapper<>();
361
362    if (StringUtils.isEmpty(meter)) {
363      return iMeter;
364    }
365
366    try {
367      iMeter.containsNonNumeric = MEASURE_MARKER_PATTERN.matcher(meter).find();
368
369      if (!iMeter.containsNonNumeric()) {
370        iMeter.measurement = NumberParser.parseDouble(meter);
371
372      } else {
373        iMeter.isInNauticalMiles = NAUTICAL_MILES_MARKER_PATTERN.matcher(meter).find();
374        iMeter.isInFathoms = FATHOMS_MARKER_PATTERN.matcher(meter).find();
375        iMeter.isInFeet = FEET_MARKER_PATTERN.matcher(meter).find();
376        iMeter.isInInches = INCHES_MARKER_PATTERN.matcher(meter).find();
377        iMeter.isInKm = KM_MARKER_PATTERN.matcher(meter).find();
378        iMeter.isInCm = CM_MARKER_PATTERN.matcher(meter).find();
379
380        // handle 6-7m values
381        if (SEP_MARKER_PATTERN.matcher(meter).find()) {
382          // we have been given a range
383          try {
384            String min = meter.substring(0, meter.indexOf('-')).trim();
385            min = removeMeasurementMarkers(min);
386            String max = meter.substring(meter.indexOf('-') + 1).trim();
387            max = removeMeasurementMarkers(max);
388
389            Double minDouble = NumberParser.parseDouble(min);
390            Double maxDouble = NumberParser.parseDouble(max);
391
392            if (minDouble == null && maxDouble != null){
393              iMeter.measurement = maxDouble;
394            } else if(maxDouble == null && minDouble != null){
395              iMeter.measurement = maxDouble;
396            } else if (minDouble != null && maxDouble != null && minDouble != 0 && maxDouble != 0
397              && maxDouble - minDouble != 0) {
398              iMeter.measurement = (maxDouble + minDouble) / 2;
399            }
400          } catch (NumberFormatException ignored) {
401          }
402
403        } else {
404          iMeter.measurement = NumberParser.parseDouble(removeMeasurementMarkers(meter));
405        }
406
407        if (iMeter.measurement != null) {
408          // convert to metric
409          if (iMeter.isInNauticalMiles) {
410            iMeter.measurement = convertNauticalMilesToMetres(iMeter.measurement);
411          } else if (iMeter.isInFathoms) {
412            iMeter.measurement = convertFathomsToMetres(iMeter.measurement);
413          } else if (iMeter.isInFeet) {
414            iMeter.measurement = convertFeetToMetres(iMeter.measurement);
415          } else if (iMeter.isInInches) {
416            iMeter.measurement = convertInchesToMetres(iMeter.measurement);
417          } else if (iMeter.isInKm){
418            iMeter.measurement = convertKmToMetres(iMeter.measurement);
419          } else if (iMeter.isInCm){
420            iMeter.measurement = convertCmToMetres(iMeter.measurement);
421          }
422        }
423      }
424    } catch (NumberFormatException e) {
425      LOG.debug("Unparsable metre measurement: {}, {}", meter, e.getMessage());
426    }
427
428    // round to centimetres
429    if (iMeter.measurement != null) {
430      iMeter.measurement = Math.round(iMeter.measurement * 100.0) / 100.0;
431    }
432    return iMeter;
433  }

435  /**
436   * Remove "m" etc.
437   *
438   * @param s to remove measurement markers from.
439   *
440   * @return a new string with all measurements removed (i.e. replaced by the empty string)
441   */
442  private static String removeMeasurementMarkers(String s) {
443    if (s == null) return null;
444    return REMOVE_MEASURE_MARKER_PATTERN.matcher(s).replaceAll("");
445  }
446
447  private static double convertNauticalMilesToMetres(double nauticalMiles) {
448    return nauticalMiles * NAUTICAL_MILES_TO_METRES;
449  }
450
451  private static double convertFathomsToMetres(double fathoms) {
452    return fathoms * FATHOMS_TO_METRES;
453  }
454
455  private static double convertFeetToMetres(double feet) {
456    return feet * FEET_TO_METRES;
457  }
458
459  private static double convertInchesToMetres(double inches) {
460    return inches * INCHES_TO_METRES;
461  }
462
463  private static double convertKmToMetres(double km) {
464    return km * KM_TO_METRES;
465  }
466
467  private static double convertCmToMetres(double cm) {
468    return cm * CM_TO_METRES;
469  }

471  /**
472   * @return rounded int value or null if it was null or exceeds the maximum an int can hold
473   */
474  private static Integer roundedInt(Double x) {
475    if (x == null) return null;
476
477    Long xl = Math.round(x);
478    if (xl > Integer.MAX_VALUE) {
479      throw new IllegalArgumentException("Long too big for an integer");
480    }
481    return xl.intValue();
482  }
