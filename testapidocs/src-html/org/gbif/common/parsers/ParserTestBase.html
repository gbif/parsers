001package org.gbif.common.parsers;
002
003import com.google.common.base.Strings;
004import com.google.common.io.LineProcessor;
005import org.gbif.common.parsers.core.Parsable;
006import org.gbif.common.parsers.core.ParseResult;
007import org.junit.Before;
008
009import java.io.IOException;
010
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertNotNull;
013
014public abstract class ParserTestBase<T> implements LineProcessor<ParserTestBase.BatchParseResult>  {
015  protected final Parsable<T> parser;
016  private BatchParseResult batchResult;
017
018  public static class BatchParseResult {
019    public int total;
020    public int failed;
021  }
022
023  @Before
024  public void setupBatch() {
025    batchResult = new BatchParseResult();
026  }
027
028  public ParserTestBase(Parsable<T> parser) {
029    this.parser = parser;
030  }
031
032  protected void assertParseFailure(String input) {
033    ParseResult<T> parsed = parser.parse(input);
034    assertEquals("Expected "+input+" to fail but got "+parsed.getPayload()+" instead", ParseResult.STATUS.FAIL, parsed.getStatus());
035  }
036
037  protected void assertParseSuccess(T expected, String input) {
038    assertParseSuccess(expected, ParseResult.CONFIDENCE.DEFINITE, input);
039  }
040
041  protected void assertParseSuccess(T expected, ParseResult.CONFIDENCE confidence, String input) {
042    ParseResult<T> parsed = parser.parse(input);
043    assertNotNull(parsed);
044//    System.out.println(parsed);
045    assertEquals("BAD PARSING OF: "+input, ParseResult.STATUS.SUCCESS, parsed.getStatus());
046    assertNotNull(parsed.getPayload());
047    assertEquals("BAD PARSING OF: "+input, expected, parsed.getPayload());
048    if (confidence != null) {
049      assertEquals(confidence, parsed.getConfidence());
050    }
051  }
052
053  @Override
054  public boolean processLine(String line) throws IOException {
055    if (Strings.isNullOrEmpty(line)) {
056      return false;
057    }
058    ParseResult<T> parsed = parser.parse(line);
059    batchResult.total++;
060    if (parsed == null || !parsed.isSuccessful() || parsed.getPayload()==null) {
061      batchResult.failed++;
062    }
063    return true;
064  }
065
066  @Override
067  public ParserTestBase.BatchParseResult getResult() {
068    return batchResult;
069  }
070}
