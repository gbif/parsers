014package org.gbif.common.parsers.date;
015
016import java.time.LocalDate;
017import java.time.LocalDateTime;
018import java.time.OffsetDateTime;
019import java.time.Year;
020import java.time.YearMonth;
021import java.time.ZoneId;
022import java.time.ZoneOffset;
023import java.time.temporal.ChronoField;
024import java.time.temporal.TemporalAccessor;
025import java.time.temporal.TemporalQueries;
026import java.util.Date;
027import java.util.Optional;
028
029import javax.annotation.Nullable;
030
031/**
032 * Utility methods to work with {@link TemporalAccessor}
033 */
034public class TemporalAccessorUtils {
035
036  public static ZoneId UTC_ZONE_ID = ZoneOffset.UTC;
037
038  /**
039   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}.
040   * If the provided {@link TemporalAccessor} contains offset(timezone) information it will be used.
041   * See {@link #toDate(TemporalAccessor, boolean)} for more details.
042   *
043   * @param temporalAccessor
044   * @return  the Date object or null if a Date object can not be created
045   */
046  public static Date toDate(TemporalAccessor temporalAccessor) {
047    return toDate(temporalAccessor, false);
048  }
049
050  /**
051   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}, rounding a partial date/time to the start
052   * of the period.
053   *
054   * For {@link YearMonth}, the {@link java.util.Date} will represent the first day of the month.
055   * For {@link Year}, the {@link java.util.Date} will represent the first day of January.
056   *
057   * Remember that a {@link Date} object will always display the date in the current timezone.
058   *
059   * @param temporalAccessor
060   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
061   *                     be used ?
062   * @return the Date object or null if a Date object can not be created
063   */
064  public static Date toDate(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
065    return Date.from(toEarliestLocalDateTime(temporalAccessor, ignoreOffset).toInstant(ZoneOffset.UTC));
066  }
067
068  /**
069   * Transform a {@link TemporalAccessor} to a {@link java.time.LocalDateTime}, rounding a partial date/time to the
070   * start of the period.
071   *
072   * @param temporalAccessor
073   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
074   *                     be used ?
075   * @return the LocalDateTime object or null if a LocalDateTime object can not be created
076   */
077  public static LocalDateTime toEarliestLocalDateTime(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
078    if(temporalAccessor == null){
079      return null;
080    }
081
082    // Use offset if present
083    if(!ignoreOffset && temporalAccessor.isSupported(ChronoField.OFFSET_SECONDS)){
084      return temporalAccessor.query(OffsetDateTime::from).atZoneSameInstant(UTC_ZONE_ID).toLocalDateTime();
085    }
086
087    if(temporalAccessor.isSupported(ChronoField.SECOND_OF_DAY)){
088      return temporalAccessor.query(LocalDateTime::from);
089    }
090
091    // this may return null in case of partial dates
092    LocalDate localDate = temporalAccessor.query(TemporalQueries.localDate());
093
094    // try YearMonth
095    if(localDate == null && temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
096      YearMonth yearMonth = YearMonth.from(temporalAccessor);
097      localDate = yearMonth.atDay(1);
098    }
099
100    // try Year
101    if(localDate == null && temporalAccessor.isSupported(ChronoField.YEAR)) {
102      Year year = Year.from(temporalAccessor);
103      localDate = year.atDay(1);
104    }
105
106    if (localDate != null) {
107      return LocalDateTime.from(localDate.atStartOfDay(UTC_ZONE_ID));
108    }
109
110    return null;
111  }
112
113  /**
114   * Transform a {@link TemporalAccessor} to a {@link java.time.LocalDateTime}, rounding a partial date/time to the
115   * the end of the period.
116   *
117   * 1990 will be 1990-12-31,
118   * 1996-02 will be 1996-02-29
119   *
120   * @param temporalAccessor
121   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
122   *                     be used ?
123   * @return the LocalDateTime object or null if a LocalDateTime object can not be created
124   */
125  public static LocalDateTime toLatestLocalDateTime(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
126    if (temporalAccessor == null) {
127      return null;
128    }
129
130    // Use offset if present
131    if (!ignoreOffset && temporalAccessor.isSupported(ChronoField.OFFSET_SECONDS)) {
132      return temporalAccessor.query(OffsetDateTime::from).atZoneSameInstant(UTC_ZONE_ID).toLocalDateTime();
133    }
134
135    if (temporalAccessor.isSupported(ChronoField.SECOND_OF_DAY)) {
136      return temporalAccessor.query(LocalDateTime::from);
137    }
138
139    // this may return null in case of partial dates
140    LocalDate localDate = temporalAccessor.query(TemporalQueries.localDate());
141
142    // try YearMonth
143    if (localDate == null && temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
144      YearMonth yearMonth = YearMonth.from(temporalAccessor);
145      localDate = yearMonth.atEndOfMonth();
146    }
147
148    // try Year
149    if (localDate == null && temporalAccessor.isSupported(ChronoField.YEAR)) {
150      Year year = Year.from(temporalAccessor);
151      localDate = LocalDate.of(year.getValue(), 12, 31);
152    }
153
154    if (localDate != null) {
155      return LocalDateTime.from(localDate.atTime(23, 59, 59));
156    }
157
158    return null;
159  }

161  /**
162   * The idea of "best resolution" TemporalAccessor is to get the TemporalAccessor that offers more resolution than
163   * the other but they must NOT contradict.
164   * e.g. 2005-01 and 2005-01-01 will return 2005-01-01.
165   *
166   * Note that if one of the 2 parameters is null the other one will be considered having the best resolution
167   *
168   * @param ta1
169   * @param ta2
170   * @return TemporalAccessor representing the best resolution
171   */
172  public static Optional<TemporalAccessor> bestResolution(@Nullable TemporalAccessor ta1, @Nullable TemporalAccessor ta2){
173    //handle nulls combinations
174    if(ta1 == null && ta2 == null){
175      return Optional.empty();
176    }
177    if(ta1 == null){
178      return Optional.of(ta2);
179    }
180    if(ta2 == null){
181      return Optional.of(ta1);
182    }
183
184    AtomizedLocalDateTime ymd1 = AtomizedLocalDateTime.fromTemporalAccessor(ta1);
185    AtomizedLocalDateTime ymd2 = AtomizedLocalDateTime.fromTemporalAccessor(ta2);
186
187    // If they both provide the year, it must match
188    if(ymd1.getYear() != null && ymd2.getYear() != null && !ymd1.getYear().equals(ymd2.getYear())){
189      return Optional.empty();
190    }
191    // If they both provide the month, it must match
192    if(ymd1.getMonth() != null && ymd2.getMonth() != null && !ymd1.getMonth().equals(ymd2.getMonth())){
193      return Optional.empty();
194    }
195    // If they both provide the day, it must match
196    if(ymd1.getDay() != null && ymd2.getDay() != null && !ymd1.getDay().equals(ymd2.getDay())){
197      return Optional.empty();
198    }
199    // If they both provide the hour, it must match
200    if (ymd1.getHour() != null && ymd2.getHour() != null && !ymd1.getHour().equals(ymd2.getHour())) {
201      return Optional.empty();
202    }
203    // If they both provide the minute, it must match
204    if (ymd1.getMinute() != null && ymd2.getMinute() != null && !ymd1.getMinute().equals(ymd2.getMinute())) {
205      return Optional.empty();
206    }
207    // If they both provide the second, it must match
208    if (ymd1.getSecond() != null && ymd2.getSecond() != null && !ymd1.getSecond().equals(ymd2.getSecond())) {
209      return Optional.empty();
210    }
211    // If they both provide the millisecond, it must match
212    if (ymd1.getMillisecond() != null && ymd2.getMillisecond() != null && !ymd1.getMillisecond().equals(ymd2.getMillisecond())) {
213      return Optional.empty();
214    }
215
216    if(ymd1.getResolution() > ymd2.getResolution()){
217      return Optional.of(ta1);
218    }
219
220    return Optional.of(ta2);
221  }
222
223  /**
224   * Given two TemporalAccessor with possibly different resolutions, this method checks if they represent the same
225   * date, or if one is contained within the other.  The comparison does not go beyond date resolution.
226   *
227   * If a null is provided, false will be returned.
228   */
229  public static boolean sameOrContained(@Nullable TemporalAccessor ta1, @Nullable TemporalAccessor ta2) {
230    // handle nulls combinations
231    if (ta1 == null || ta2 == null) {
232      return false;
233    }
234
235    AtomizedLocalDate ymd1 = AtomizedLocalDate.fromTemporalAccessor(ta1);
236    AtomizedLocalDate ymd2 = AtomizedLocalDate.fromTemporalAccessor(ta2);
237
238    // we only deal with complete Local Date
239    if (ymd1.isComplete() && ymd2.isComplete()) {
240      return ymd1.equals(ymd2);
241    }
242
243    // check for equal years
244    if (!ymd1.getYear().equals(ymd2.getYear())) {
245      return false;
246    }
247
248    // compare months
249    if (ymd1.getMonth() == null || ymd2.getMonth() == null) {
250      return true;
251    }
252    if (!ymd1.getMonth().equals(ymd2.getMonth())) {
253      return false;
254    }
255
256    // compare days
257    if (ymd1.getDay() == null || ymd2.getDay() == null) {
258      return true;
259    }
260    return ymd1.getDay().equals(ymd2.getDay());
261  }
