015/**
016 * Utilities to work on classifications.
017 */
018public final class ClassificationUtils {
019
020  // used to clean up bad characters
021  private static final Pattern CLEAN_REG_EX = Pattern.compile("[{}ยง';_|$%!?]+");
022
023  // common null strings to ignore for fast performance.
024  // Less frequent ones are kept in the blacklisted names dictionary!
025  public static final Set<String> NULL_STRINGS =
026    new HashSet<String>(Arrays.asList("/N", "\\", "\\\\", "\\N", "\\\\N", "null", "NULL", "Null"));
027
028  private ClassificationUtils() {
029    throw new UnsupportedOperationException("Can't initialize class");
030  }
031
032  /**
033   * Parses a canonical name at a specific Rank.
034   */
035  public static String canonicalName(String scientificName, Rank rank) {
036    ParsedName pn = null;
037    try {
038      pn = PARSER.parse(scientificName, rank);
039    } catch (UnparsableException e) {
040    }
041    return pn.canonicalNameWithMarker();
042  }
043
044  /**
045   * Parses a scientific name and creates the canonical name including a potential hybrid and rank marker
046   * plus the cultivar and strain names if existing.
047   * Note: This method once used to only include the hybrid marker - if that is still needed revert to buildName
048   * method.
049   */
050  public static String canonicalName(String scientificName) {
051    return canonicalName(scientificName,null);
052  }
053
054  /**
055   * Cleans up a taxon as far as possible by removing erroneous chars etc.
056   * This does not do any parsing.
057   *
058   * @param taxon to check
059   */
060  public static String clean(String taxon) {
061    if (StringUtils.isEmpty(taxon) || NULL_STRINGS.contains(taxon)) {
062      return null;
063    }
064
065    String cleanedTaxon = taxon;
066
067    // if it is a single word and ALL "UPPERCASE", turn it into a Capitalised word
068    // Note: if we lowercase names with multiple words we might accidently create valid looking names by lowercasing the
069    // author
070    // for example ABIES ALBA REMSEN will become an Abies alba remsen which will then be interpreted badly
071    // ABIES ALBA LINNEAUS 1771 will even be Abies alba linneaus 1771, a perfectly formed zoological name
072    if (!cleanedTaxon.contains(" ") && cleanedTaxon.equals(cleanedTaxon.toUpperCase())) {
073      cleanedTaxon = cleanedTaxon.substring(0, 1) + cleanedTaxon.substring(1).toLowerCase();
074    }
075
076    // remove the " from names with it at the beginning and end
077    while (cleanedTaxon.length()> 0 && cleanedTaxon.charAt(0) == '\"' && cleanedTaxon.charAt(cleanedTaxon.length() - 1) == '\"') {
078      if (cleanedTaxon.length() == 1) {
079        return null;
080      }
081      cleanedTaxon = cleanedTaxon.substring(1, cleanedTaxon.length() - 1);
082    }
083
084    // remove the " from names with it just at the beginning
085    while (cleanedTaxon.length()> 0 && cleanedTaxon.charAt(0) == '\"') {
086      if (cleanedTaxon.length() == 1) {
087        return null;
088      }
089      cleanedTaxon = cleanedTaxon.substring(1, cleanedTaxon.length());
090    }
091
092    // remove the " from names with it just at the end
093    while (cleanedTaxon.length()> 0 && cleanedTaxon.charAt(cleanedTaxon.length() - 1) == '\"') {
094      if (cleanedTaxon.length() == 1) {
095        return null;
096      }
097      cleanedTaxon = cleanedTaxon.substring(0, cleanedTaxon.length() - 1);
098    }
099
100
101    // remove noise
102    cleanedTaxon = CLEAN_REG_EX.matcher(cleanedTaxon).replaceAll("");
103    cleanedTaxon = cleanedTaxon.trim();
104
105    // don't let any blacklisted names through
106    if (BlacklistedNames.contains(cleanedTaxon.toUpperCase()) || (cleanedTaxon != taxon && BlacklistedNames
107      .contains(taxon.toUpperCase()))) {
108      // blacklisted name
109      return null;
110    }
111
112    return StringUtils.trimToNull(cleanedTaxon);
113  }
114
115  /**
116   * Clean some noise from the author. A large proportion are "\N" for example.
117   *
118   * @param author to clean
119   *
120   * @return cleaned author
121   */
122  public static String cleanAuthor(String author) {
123    if (StringUtils.isEmpty(author) || NULL_STRINGS.contains(author)) {
124      return null;
125    }
126
127    String cleanedAuthor = author;
128
129    // remove the " from names with it at the beginning and end
130    while (cleanedAuthor.length() >0 && cleanedAuthor.charAt(0) == '\"' && cleanedAuthor.charAt(cleanedAuthor.length() - 1) == '\"') {
131      if (cleanedAuthor.length() == 1) {
132        return null;
133      }
134      cleanedAuthor = cleanedAuthor.substring(1, cleanedAuthor.length() - 1);
135    }
136
137    // remove noise
138    cleanedAuthor = CLEAN_REG_EX.matcher(cleanedAuthor).replaceAll("");
139    cleanedAuthor = cleanedAuthor.trim();
140
141    return StringUtils.trimToNull(cleanedAuthor);
142  }

153  /**
154   * Parses a scientific name of a specific rank.
155   * @deprecated unsufficiently documented; should be in NameParser project if it is needed at all
156   */
157  @Deprecated
158  public static String parseName(String scientificName, Rank rank) {
159
160    try {
161      ParsedName pn = PARSER.parse(scientificName, rank);
162      // Handle Aus sp. and Aus bus spp.
163      if (pn.isIndetermined()) {
164          pn.setRank(null);
165      }
166      return pn.fullName();
167
168    } catch (UnparsableException e) {
169      // TODO: logging
170    }
171
172    // looks dirty, so try and normalize it as best we can and get a canonical at least
173    String canon = PARSER.parseToCanonical(scientificName, rank);
174    if (canon != null) {
175      return canon;
176    }
177
178    return scientificName;
179  }
