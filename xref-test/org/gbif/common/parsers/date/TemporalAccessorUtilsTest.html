26    private ZoneId EUROPE_CENTRAL_TIME = ZoneId.of("Europe/Paris");
27  
28    // January 1st 2000, 00:00:00 (midnight)
29    private long YEAR2000_1JAN_EPOCH_UTC = 946684800000l;
30  
31    // January 1st 2000, 02:03:04
32    private long YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC = 946692184000l;
33  
34    // February 1st 2000, midnight
35    private long YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC = 949363200000l;
36  
37    // January 1st 2000, midnight
38    private long YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC = 946684800000l;
39  
40    @Test
41    public void testToDate(){
42      TemporalAccessor ta = LocalDate.of(2000,01,01);
43      Date date = TemporalAccessorUtils.toDate(ta, false);
44      assertEquals(YEAR2000_1JAN_EPOCH_UTC, date.getTime());
45  
46      ta = LocalDateTime.of(2000, 01, 01, 2, 3, 4);
47      date = TemporalAccessorUtils.toDate(ta, false);
48      assertEquals(YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC, date.getTime());
49  
50      //test partial date transformed to Date
51      //YearMonth should use the first of the month
52      ta = YearMonth.of(2000, 2);
53      date = TemporalAccessorUtils.toDate(ta, false);
54      assertEquals(YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC, date.getTime());
55  
56      //Year should use the first of January
57      ta = Year.of(2000);
58      date = TemporalAccessorUtils.toDate(ta, false);
59      assertEquals(YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC, date.getTime());
60    }

62    @Test
63    public void testToDateWithTimeZone(){
64  
65      // Test with ZonedDateTime
66      ZonedDateTime testZonedDateTime = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, EUROPE_CENTRAL_TIME);
67      Date date = TemporalAccessorUtils.toDate(testZonedDateTime, false);
68      assertEquals(testZonedDateTime.toEpochSecond(), date.getTime()/1000);
69  
70      // Test the same ZonedDateTime but ignore Offset information (timezone)
71      ZonedDateTime testTimeUTC = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, TemporalAccessorUtils.UTC_ZONE_ID);
72      date = TemporalAccessorUtils.toDate(testZonedDateTime, true);
73      assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
74  
75      // Ensure it is also working with OffsetDateTime
76      OffsetDateTime testOffsetDateTimeTime = testZonedDateTime.toOffsetDateTime();
77      date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, false);
78      assertEquals(testOffsetDateTimeTime.toEpochSecond(), date.getTime()/1000);
79  
80      date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, true);
81      assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
82    }

84    @Test
85    public void testGetBestResolutionTemporalAccessor(){
86      TemporalAccessor ta1 = Year.of(2005);
87      TemporalAccessor ta2 = YearMonth.of(2005, 1);
88      TemporalAccessor result = TemporalAccessorUtils.getBestResolutionTemporalAccessor(ta1, ta2);
89      assertEquals(YearMonth.of(2005, 1), YearMonth.from(result));
90  
91      ta1 = LocalDate.of(2005, 1, 1);
92      ta2 = Year.of(2005);
93      result = TemporalAccessorUtils.getBestResolutionTemporalAccessor(ta1, ta2);
94      assertEquals(LocalDate.of(2005, 1, 1), LocalDate.from(result));
95  
96      //this should not work
97      ta1 = Year.of(2005);
98      ta2 = YearMonth.of(2006, 1);
99      result = TemporalAccessorUtils.getBestResolutionTemporalAccessor(ta1, ta2);
100     assertNull(result);
101   }

103   @Test
104   public void testRepresentsSameYMD() {
105     TemporalAccessor ta1 = LocalDate.of(1996, 01, 26);
106 
107     // Test nulls and wrong resolutions
108     assertFalse(TemporalAccessorUtils.representsSameYMD(null, null));
109     assertFalse(TemporalAccessorUtils.representsSameYMD(ta1, null));
110     assertFalse(TemporalAccessorUtils.representsSameYMD(null, ta1));
111     assertFalse(TemporalAccessorUtils.representsSameYMD(ta1, Year.of(1996)));
112 
113     // Test 2 dates representing the same day in the year, at different resolution
114     TemporalAccessor ta2 = LocalDateTime.of(1996, 01, 26, 1, 2);
115     assertTrue(TemporalAccessorUtils.representsSameYMD(ta1, ta2));
116   }
