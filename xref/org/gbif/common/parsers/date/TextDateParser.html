14  /**
15   * Main class to parse a date represented as a single String or as date parts into a {@link TemporalAccessor}.
16   * If the String contains letters, the {@link TextualMonthDateTokenizer} and the {@link DatePartsNormalizer} will be used.
17   * Otherwise, a {@link TemporalParser} configured for numerical dates will be used.
18   *
19   * This class is basically a decorator on top of default NumericalDateParser to handle months written in text.
20   */
21  class TextDateParser implements TemporalParser {
22    /*
23     * ISO format intervals, which are datetime/datetime or datetime/period.
24     * Match 2013/2014, 2013/P1Y, 2013-02/03.
25     * Fail 2013-12, 2013/02, 2018/10-23.
26     */
27    private static final Pattern ISO_YEAR_RANGE_PATTERN = Pattern.compile("^([12][0-9]{3})/([P12]\\d[^/]+)$");
28    private static final Pattern ISO_DATE_RANGE_PATTERN = Pattern.compile("^([12][0-9]{3}[^/]+)/([^/]{2,})$");
29  
30    // This regex is not complete and will NOT handle date when the time zone is provided as text GMT
31    private static final Pattern NUMERICAL_DATE_PATTERN =  Pattern.compile("[^a-zA-VX-Z]+[\\dT\\d]?[^a-zA-Z]+[Z]?$");
32    private static final TextualMonthDateTokenizer TEXT_MONTH_TOKENIZER = TextualMonthDateTokenizer.newInstance();
33  
34    private static final TemporalParser NUMERICAL_DATE_PARSER = ThreeTenNumericalDateParser.newInstance();
35    private static final DatePartsNormalizer DATE_PARTS_NORMALIZER = DatePartsNormalizer.newInstance();
36  
37    @Override
38    public ParseResult<TemporalAccessor> parse(String input) {
39  
40      if (StringUtils.isBlank(input)) {
41        return ParseResult.fail();
42      }
43  
44      Matcher matcher = ISO_YEAR_RANGE_PATTERN.matcher(input);
45      if (!matcher.matches()) {
46        matcher = ISO_DATE_RANGE_PATTERN.matcher(input);
47      }
48  
49      if (matcher.matches()) {
50        String from = matcher.group(1);
51        // String to = matcher.group(2);
52        return NUMERICAL_DATE_PARSER.parse(from, DateFormatHint.NONE);
53      }
54  
55      // Check if the input text contains only punctuations and numbers
56      // Also accept the T marker (e.g. 1978-12-21T02:12) from the ISO format
57      // and the W week marker (e.g. 2018-W43).
58      // We could also simply try to parse it but it is probably not optimal
59      if(NUMERICAL_DATE_PATTERN.matcher(input).matches()) {
60        return NUMERICAL_DATE_PARSER.parse(input, DateFormatHint.NONE);
61      }
62  
63      TextualMonthDateTokenizer.DateTokens dt = TEXT_MONTH_TOKENIZER.tokenize(input);
64      // for now we only handle cases where we can find year, month, day with confidence.
65      if(!dt.containsDiscardedTokens() && dt.size() == 3){
66        DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DATE_PARTS_NORMALIZER.normalize(
67                dt.getToken(TextualMonthDateTokenizer.TokenType.INT_4).getToken(),
68                dt.getToken(TextualMonthDateTokenizer.TokenType.TEXT).getToken(),
69                dt.getToken(TextualMonthDateTokenizer.TokenType.INT_2).getToken());
70  
71        //no handling for partial dates with textual month for now
72        if(normalizedYearMonthDay.getYear() != null &&
73                normalizedYearMonthDay.getMonth() != null &&
74                normalizedYearMonthDay.getDay() != null){
75          try {
76            return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE,
77                    LocalDate.of(normalizedYearMonthDay.getYear(),
78                            normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay()));
79          }
80          catch (DateTimeException ignore){
81            //simply ignore bad dates
82          }
83        }
84      }
85      return ParseResult.fail();
86    }
87  
88    /**
89     * For now this is directly delegated the NumericalDateParser.
90     *
91     * @param input
92     * @param hint help to speed up the parsing and possibly return a better confidence
93     * @return
94     */
95    @Override
96    public ParseResult<TemporalAccessor> parse(String input, @Nullable DateFormatHint hint) {
97      return NUMERICAL_DATE_PARSER.parse(input, hint);
98    }
99  
100   /**
101    * Parse date parts into a TemporalAccessor.
102    * The {@link DatePartsNormalizer} will be applied on raw data.
103    *
104    * @param year
105    * @param month
106    * @param day
107    * @return
108    */
109   public ParseResult<TemporalAccessor> parse(String year, String month, String day) {
110     DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DATE_PARTS_NORMALIZER.normalize(
111             year, month, day);
112 
113     // TODO when we can get the day but not the month in NormalizedYearMonthDay the parsing fails
114     // we could actually keep the year by calling ThreeTenNumericalDateParser.parse(normalizedYearMonthDay.getYear(),
115     // null, null); the CONFIDENCE would be set at PROBABLE.
116 
117     ParseResult<TemporalAccessor> parseResult = NUMERICAL_DATE_PARSER.parse(normalizedYearMonthDay.getYear(),
118             normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay());
119 
120     //If we got a successful parsing BUT a part of the date was discarded we reduce confidence.
121     if(parseResult.isSuccessful() && normalizedYearMonthDay.containsDiscardedPart()){
122       return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, parseResult.getPayload());
123     }
124 
125     return parseResult;
126   }
127 
128   @Override
129   public ParseResult<TemporalAccessor> parse(@Nullable Integer year, @Nullable Integer month, @Nullable Integer day) {
130     return NUMERICAL_DATE_PARSER.parse(year, month, day);
131   }
