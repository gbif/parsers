44    /**
45     * Parse all unique multimedia licenses values found in our index and make sure parsing doesn't get worse.
46     * If the test file is updated, values here need to be adjusted!
47     */
48    @Test
49    public void testOccurrenceValues() throws IOException {
50      final int CURRENT_TESTS_SUCCESSFUL = 75;
51      final int CURRENT_DISTINCT = 15;
52  
53      int failed = 0;
54      int success = 0;
55      Set<URI> values = Sets.newHashSet();
56  
57      BufferedReader r = new BufferedReader(new InputStreamReader(getClass().getResourceAsStream("/parse/license_uri.txt"), Charsets.UTF_8));
58      String line;
59      while ((line=r.readLine()) != null) {
60        ParseResult<URI> parsed = parser.parse(line);
61        if (parsed.isSuccessful()) {
62          // System.out.println("Worked: " + line + " â†’ " + parsed.getPayload());
63          success++;
64          values.add(parsed.getPayload());
65        } else {
66          System.out.println("Failed: " + line);
67          failed++;
68        }
69      }
70  
71      System.out.println(failed + " failed parse results");
72      System.out.println(success + " successful parse results");
73      System.out.println(values.size() + " distinct license values parsed");
74      assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
75      assertTrue(CURRENT_DISTINCT <= values.size());
76    }
