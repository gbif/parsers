024  @Test
025  public void testTextDateParsing(){
026    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2nd jan. 2018");
027    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
028
029    parseResult = TEXTDATE_PARSER.parse("2018 january 2");
030    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
031
032    parseResult = TEXTDATE_PARSER.parse("2018-01-02");
033    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
034
035    parseResult = TEXTDATE_PARSER.parse("2018/01/02");
036    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
037
038    parseResult = TEXTDATE_PARSER.parse("2018年1月2日");
039    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
040  }
041
042  @Test
043  public void testTextDateTimeParsing(){
044    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("1978-1-1T00:00");
045    assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
046
047    parseResult = TEXTDATE_PARSER.parse("1978-01-01T02:12:43Z");
048    assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
049  }
050
051  @Test
052  public void testDateParts(){
053    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2018.0", "1.0", "02.0");
054    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
055
056    parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "02.0");
057    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
058
059    parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "");
060    assertEquals(YearMonth.of(2018, Month.JANUARY), YearMonth.from(parseResult.getPayload()));
061  }

093  @Test
094  public void testDateRanges() {
095    // Some of the many ways of specifying a whole year.
096    // So many cases aren't really needed, as we're only taking the start of the interval anyway.
097    // Just with years
098    testDateRangeStart(1999, 1, 1, "1999");
099    testDateRangeStart(1999, 1, 1, "1999/2000");
100    testDateRangeStart(1999, 1, 1, "1999/P1Y");
101    // With years and months
102    testDateRangeStart(1999, 1, 1, "1999-01/1999-12");
103    testDateRangeStart(1999, 1, 1, "1999-01/12");
104    testDateRangeStart(1999, 1, 1, "1999-01/P12M");
105    // With years and days
106    testDateRangeStart(1999, 1, 1, "1999-001/1999-365");
107    testDateRangeStart(1999, 1, 1, "1999-001/P365D");
108    // With years, months and days
109    testDateRangeStart(1999, 1, 1, "1999-01-01/1999-12-31");
110    testDateRangeStart(1999, 1, 1, "1999-01-01/12-31");
111    testDateRangeStart(1999, 1, 1, "1999-01-01/P365D");
112    // With years, months, days and a time
113    testDateRangeStart(1999, 1, 1, "1999-01-01T00/1999-12-31T24");
114    testDateRangeStart(1999, 1, 1, "1999-01-01T00:00/1999-12-31T24:00");
115    testDateRangeStart(1999, 1, 1, "1999-01-01T00:00:00/1999-12-31T24:00:00");
116    testDateRangeStart(1999, 1, 1, "1999-01-01T00:00:00+0000/1999-12-31T24:00:00+0000");
117
118    // A whole month.
119    // With years and months
120    testDateRangeStart(1999, 2, 1, "1999-02");
121    testDateRangeStart(1999, 2, 1, "1999-02/1999-03");
122    testDateRangeStart(1999, 2, 1, "1999-02/03");
123    testDateRangeStart(1999, 2, 1, "1999-02/P1M");
124    // With years and days
125    testDateRangeStart(1999, 2, 1, "1999-032/1999-59");
126    testDateRangeStart(1999, 2, 1, "1999-032/P28D");
127    // With years, months and days
128    testDateRangeStart(1999, 2, 1, "1999-02-01/1999-02-28");
129    testDateRangeStart(1999, 2, 1, "1999-02-01/28");
130    testDateRangeStart(1999, 2, 1, "1999-02-01/P28D");
131    // With years, months, days and a time
132    testDateRangeStart(1999, 2, 1, "1999-02-01T00/1999-02-28T24");
133    testDateRangeStart(1999, 2, 1, "1999-02-01T00:00/1999-02-28T24:00");
134    testDateRangeStart(1999, 2, 1, "1999-02-01T00:00:00/1999-02-28T24:00:00");
135    testDateRangeStart(1999, 2, 1, "1999-02-01T00:00:00+0000/1999-02-28T24:00:00+0000");
136
137    // A period of several days, from 2004-12-30 to 2005-03-13
138    // With years and days
139    testDateRangeStart(2004, 12, 30, "2004-365/2005-072");
140    testDateRangeStart(2004, 12, 30, "2004-365/P74D");
141    // With years, months and days
142    testDateRangeStart(2004, 12, 30, "2004-12-30/2005-03-13");
143    //testDateRangeStart(2004, 12, 30, "2004-12-30/31");
144    testDateRangeStart(2004, 12, 30, "2004-12-30/P74D");
145    // With years, months, days and a time
146    testDateRangeStart(2004, 12, 30, "2004-12-30T00/2005-03-13T24");
147    testDateRangeStart(2004, 12, 30, "2004-12-30T00:00/2005-03-13T24:00");
148    testDateRangeStart(2004, 12, 30, "2004-12-30T00:00:00/2005-03-13T24:00:00");
149    testDateRangeStart(2004, 12, 30, "2004-12-30T00:00:00+0000/2005-03-13T24:00:00+0000");
150
151    // A period of hours
152    testDateRangeStart(2004, 12, 30, 8, 0, 0, "2004-12-30T08/2005-03-13T18");
153    testDateRangeStart(2004, 12, 30, 8, 0, 0, "2004-12-30T08:00/2005-03-13T18:00");
154    testDateRangeStart(2004, 12, 30, 8, 0, 0, "2004-12-30T08:00:00/2005-03-13T18:00:00");
155    testDateRangeStart(2004, 12, 30, 8, 0, 0, "2004-12-30T08:00:00+0000/2005-03-13T18:00:00+0000");
156  }

163    assertFalse(TEXTDATE_PARSER.parse("1986/03").isSuccessful());
164    assertFalse(TEXTDATE_PARSER.parse("1986/03-13").isSuccessful());
165  }

167  private void testDateRangeStart(int y, int m, int d, String inDate) {
168    testDateRangeStart(y, m, d, 0, 0, 0, inDate);
169  }
170
171  private void testDateRangeStart(int y, int m, int d, int hh, int mm, int ss, String inDate) {
172    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse(inDate);
173    assertTrue(parseResult.isSuccessful());
174    assertEquals(LocalDateTime.of(y, m, d, hh, mm, ss), TemporalAccessorUtils.toEarliestLocalDateTime(parseResult.getPayload(), true));
175  }
