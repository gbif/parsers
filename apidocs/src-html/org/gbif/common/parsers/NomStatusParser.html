001package org.gbif.common.parsers;
002
003import com.google.common.base.Strings;
004import com.google.common.collect.ImmutableMap;
005import org.gbif.api.vocabulary.NomenclaturalStatus;
006import org.gbif.common.parsers.core.EnumParser;
007import org.gbif.common.parsers.core.ParseResult;
008
009import java.io.InputStream;
010import java.util.Map;
011import java.util.regex.Pattern;
012
013/**
014 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/nomStatus.txt.
015 */
016public class NomStatusParser extends EnumParser<NomenclaturalStatus> {
017    private Pattern CLEAN_PREFIX = Pattern.compile("\\s*\\.?\\s*");
018    private Map<String, NomenclaturalStatus> PREFIXES = ImmutableMap.<String, NomenclaturalStatus>builder()
019        .put("nom illeg", NomenclaturalStatus.ILLEGITIMATE)
020        .put("nom inval", NomenclaturalStatus.INVALID)
021        .put("comb nov", NomenclaturalStatus.NEW_COMBINATION)
022        .put("nom nov", NomenclaturalStatus.REPLACEMENT)
023        .put("nom nud", NomenclaturalStatus.NUDUM)
024        .put("nom rej", NomenclaturalStatus.REJECTED)
025        .put("unavailable", NomenclaturalStatus.INVALID)
026        .build();
027  private static NomStatusParser singletonObject = null;
028
029  private NomStatusParser(InputStream... file) {
030    super(NomenclaturalStatus.class, false, file);
031    // also make sure we have all enum knowledge mapped
032    for (NomenclaturalStatus ns : NomenclaturalStatus.values()) {
033      add(ns.getLatinLabel(), ns);
034      add(ns.getAbbreviatedLabel(), ns);
035    }
036  }
037
038    @Override
039    public ParseResult<NomenclaturalStatus> parse(String input) {
040        ParseResult<NomenclaturalStatus> result = super.parse(input);
041        if (!result.isSuccessful() && !Strings.isNullOrEmpty(input)) {
042            String normed = CLEAN_PREFIX.matcher(input).replaceFirst(" ").trim().toLowerCase();
043            if (!Strings.isNullOrEmpty(normed)) {
044                // try generic parsing of status prefixes only
045                for (Map.Entry<String, NomenclaturalStatus> entry : PREFIXES.entrySet()) {
046                    if (normed.startsWith(entry.getKey())) {
047                        return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, entry.getValue());
048                    }
049                }
050            }
051        }
052        return result;
053    }
054
055    public static NomStatusParser getInstance()
056    throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
057    synchronized (NomStatusParser.class) {
058      if (singletonObject == null) {
059        singletonObject = new NomStatusParser(NomStatusParser.class.getResourceAsStream("/dictionaries/parse/nomStatus.txt"));
060      }
061    }
062    return singletonObject;
063  }
064
065
066}
