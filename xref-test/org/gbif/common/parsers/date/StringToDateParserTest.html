17  public class StringToDateParserTest {
18    private static final Logger LOG = LoggerFactory.getLogger(DateParseUtilsTest.class);
19  
20    @Test
21    public void testParse() throws Exception {
22      StringToDateParser parser = new StringToDateParser();
23  
24      assertEquivalent("21/12/1978", parser.parse("211278"), ParseResult.CONFIDENCE.PROBABLE);
25      assertEquivalent("21/12/1978", parser.parse("21/12/1978"), ParseResult.CONFIDENCE.DEFINITE);
26      assertEquivalent("21/12/1978", parser.parse("21/12/78"), ParseResult.CONFIDENCE.DEFINITE);
27  
28      assertEquals(ParseResult.STATUS.FAIL, parser.parse("fsfgr/12/78").getStatus());
29      assertEquals(ParseResult.STATUS.FAIL, parser.parse("fsfgr").getStatus());
30      assertEquals(ParseResult.STATUS.FAIL, parser.parse("//").getStatus());
31      assertEquals(ParseResult.STATUS.FAIL, parser.parse("0-0-0").getStatus());
32      assertEquals(ParseResult.STATUS.FAIL, parser.parse("%$-s2").getStatus());
33      assertEquals(ParseResult.STATUS.FAIL, parser.parse("as-fds-sdf").getStatus());
34      assertEquals(ParseResult.STATUS.FAIL, parser.parse("88/88/88").getStatus());
35      assertEquals(ParseResult.STATUS.FAIL, parser.parse(" ").getStatus());
36      assertEquals(ParseResult.STATUS.FAIL, parser.parse("").getStatus());
37      assertEquals(ParseResult.STATUS.FAIL, parser.parse(null).getStatus());
38    }
39  
40    private void assertEquivalent(String expected, ParseResult<Date> found, ParseResult.CONFIDENCE c) {
41      if (ParseResult.STATUS.ERROR == found.getStatus()) {
42        LOG.warn("Unexpected error found in parsing", found.getError());
43      }
44      assertEquals(ParseResult.STATUS.SUCCESS, found.getStatus());
45      assertEquals(c, found.getConfidence());
46      assertNotNull(found.getPayload());
47      try {
48        assertEquals(DateUtils.parseDateStrictly(expected, new String[] {"dd/MM/yyyy"}), found.getPayload());
49      } catch (ParseException e) {
50        fail("Error in the test class.  Expected date is not in the correct format of dd/MM/yyyy:" + expected);
51      }
52    }
53  }
