049  /**
050   * Parse all unique occurrence values found for countries and make sure parsing doesnt get worse.
051   * If the test file is updated, values here need to be adjusted!
052   */
053  @Test
054  public void testOccurrenceValues() throws IOException {
055    final int CURRENT_COUNTRIES_PARSED = 251;
056    final int CURRENT_TESTS_SUCCESSFUL = 5273;
057
058    // remember number of successful parsed countries:
059    Map<Country, AtomicInteger> stats = new HashMap<>();
060    int values = 0;
061    int failed = 0;
062    int success = 0;
063    BufferedReader r = new BufferedReader(
064      new InputStreamReader(this.getClass().getResourceAsStream("/parse/countryname/occurrence_countries.txt"),
065        StandardCharsets.UTF_8));
066    String line;
067    do {
068      line = r.readLine();
069      values++;
070      ParseResult<Country> parsed = parser.parse(line);
071      if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
072        Country c = parsed.getPayload();
073        if (!stats.containsKey(c)) {
074          stats.put(c, new AtomicInteger(0));
075        }
076        stats.get(c).incrementAndGet();
077        success++;
078      } else {
079        failed++;
080      }
081    } while (line != null);
082
083    System.out.println(values + " tested values");
084    System.out.println(failed + " failed parse results");
085    System.out.println(success + " successful parse results");
086    System.out.println(stats.size() + " distinct countries parsed");
087    assertTrue(failed < values);
088    assertTrue(CURRENT_COUNTRIES_PARSED <= stats.size());
089    assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
090  }
