001package org.gbif.common.parsers.date;
002
003import static org.gbif.common.parsers.date.DateComponentOrdering.DMY_FORMATS;
004import static org.junit.jupiter.api.Assertions.assertEquals;
005import static org.junit.jupiter.api.Assertions.assertFalse;
006
007import java.time.LocalDate;
008import java.time.Month;
009import java.time.YearMonth;
010import java.time.temporal.TemporalAccessor;
011import org.gbif.common.parsers.core.ParseResult;
012
013import org.junit.jupiter.api.Test;
014
015public class CustomizedTextDateParserTest {
016
017  private final TemporalParser parser =  CustomizedTextDateParser.getInstance(DMY_FORMATS);
018
019  /**
020   * Can parse: 1969-02
021   * Cannot parse: 1969/02
022   */
023  @Test
024  public void testTextDateParsing(){
025    ParseResult<TemporalAccessor> parseResult =  parser.parse("1/2/2000");
026    assertEquals(LocalDate.of(2000, Month.FEBRUARY, 1), LocalDate.from(parseResult.getPayload()));
027    parseResult =  parser.parse("2000-2-1");
028    assertEquals(LocalDate.of(2000, Month.FEBRUARY, 1), LocalDate.from(parseResult.getPayload()));
029    parseResult = parser.parse("23-March-1969");
030    assertEquals(LocalDate.of(1969, Month.MARCH, 23), LocalDate.from(parseResult.getPayload()));
031    parseResult = parser.parse("1969-02");
032    assertEquals("1969-02", YearMonth.from(parseResult.getPayload()).toString());
033  }
034
035  /**
036   * Because TextDateParse does not initiate ThreeTenNumericalDateParse with start year, like 19
037   */
038  @Test
039  public void shouldFail(){
040    ParseResult<TemporalAccessor> parseResult = parser.parse("7/10/08");
041    assertFalse(parseResult.isSuccessful());
042    parseResult = parser.parse("17/10/78");
043    assertFalse(parseResult.isSuccessful());
044    parseResult = parser.parse("1969/02");
045    assertFalse(parseResult.isSuccessful());
046  }
047}
