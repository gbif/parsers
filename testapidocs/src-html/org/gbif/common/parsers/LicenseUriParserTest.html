001package org.gbif.common.parsers;
002
003import org.gbif.common.parsers.core.ParseResult;
004import org.junit.jupiter.api.Test;
005
006import java.io.BufferedReader;
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.net.URI;
010import java.nio.charset.StandardCharsets;
011import java.util.HashSet;
012import java.util.Set;
013
014import static org.junit.jupiter.api.Assertions.assertTrue;
015
016public class LicenseUriParserTest extends ParserTestBase<URI> {
017
018  public LicenseUriParserTest() {
019    super(LicenseUriParser.getInstance());
020  }
021
022  @Test
023  public void testFailures() {
024    assertParseFailure(null);
025    assertParseFailure("");
026    assertParseFailure("Matt");
027  }
028
029  @Test
030  public void testParse() {
031    // run a few basic tests to check it bootstraps and appears to work
032    assertParseSuccess(URI.create("http://creativecommons.org/licenses/by-sa/3.0/"), "CC BY-SA 3.0");
033    assertParseSuccess(URI.create("http://creativecommons.org/licenses/by/4.0/"), "cc-by");
034    assertParseSuccess(URI.create("http://creativecommons.org/licenses/by-nc/4.0/"), "http://creativecommons.org/licenses/by-nc/4.0/");
035    assertParseSuccess(URI.create("http://creativecommons.org/licenses/by/4.0/"), "http://creativecommons.org/licenses/by/4.0/");
036    assertParseSuccess(URI.create("http://creativecommons.org/licenses/by/4.0/"), "https://creativecommons.org/licenses/by/4.0/");
037    assertParseSuccess(URI.create("http://creativecommons.org/licenses/by/4.0/"), "http://creativecommons.org/licenses/by/4.0/legalcode");
038    assertParseSuccess(URI.create("http://creativecommons.org/licenses/by/4.0/"), "https://creativecommons.org/licenses/by/4.0/legalcode#languages");
039  }
040
041  /**
042   * Parse all unique multimedia licenses values found in our index and make sure parsing doesn't get worse.
043   * If the test file is updated, values here need to be adjusted!
044   */
045  @Test
046  public void testOccurrenceValues() throws IOException {
047    final int CURRENT_TESTS_SUCCESSFUL = 75;
048    final int CURRENT_DISTINCT = 15;
049
050    int failed = 0;
051    int success = 0;
052    Set<URI> values = new HashSet<>();
053
054    BufferedReader r = new BufferedReader(
055        new InputStreamReader(getClass().getResourceAsStream("/parse/license_uri.txt"), StandardCharsets.UTF_8));
056    String line;
057    while ((line=r.readLine()) != null) {
058      ParseResult<URI> parsed = parser.parse(line);
059      if (parsed.isSuccessful()) {
060        // System.out.println("Worked: " + line + " â†’ " + parsed.getPayload());
061        success++;
062        values.add(parsed.getPayload());
063      } else {
064        System.out.println("Failed: " + line);
065        failed++;
066      }
067    }
068
069    System.out.println(failed + " failed parse results");
070    System.out.println(success + " successful parse results");
071    System.out.println(values.size() + " distinct license values parsed");
072    assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
073    assertTrue(CURRENT_DISTINCT <= values.size());
074  }
075}
