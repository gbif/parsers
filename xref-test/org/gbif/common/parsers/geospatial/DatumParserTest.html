36    @Test
37    public void testParse() {
38      // run a few basic tests to check it bootstraps and appears to work
39      assertParseSuccess(4269, "EPSG::4269");
40      assertParseSuccess(6317, " EPSG::6317");
41      assertParseSuccess(4269, "EPSG:4269");
42      assertParseSuccess(4269, " EPSG :: 4269 ");
43      assertParseSuccess(4269, "ESPG:4269");
44      // known srs names
45      assertParseSuccess(6269, "NAD83");
46      assertParseSuccess(4326, "WGS84");
47      assertParseSuccess(6269, "nad83");
48      assertParseSuccess(6301, "tokYo!");
49      assertParseSuccess(4326, "WGS_1984");
50      assertParseSuccess(6202, "AGD66");

52      assertParseSuccess(7405, "OSGB36");
53      assertParseSuccess(7405, "Ordnance_Survey");
54      assertParseSuccess(7405, "Ordnance_Survey_36");
55      assertParseSuccess(7405, "OSGB36");
56      assertParseSuccess(6209, "ARC 1950");
57      assertParseSuccess(4222, "CAPE");
58      assertParseSuccess(6131, "indian");
59      assertParseSuccess(6760, "WGS66");
60      assertParseSuccess(6760, "World Geodetic System 1966");
61    }

63    /**
64     * Parse all unique datum values found and make sure parsing doesn't get worse.
65     * If the test file is updated, values here need to be adjusted!
66     */
67    @Test
68    public void testOccurrenceValues() throws IOException {
69      final int CURRENT_TESTS_SUCCESSFUL = 14;
70  
71      int failed = 0;
72      int success = 0;
73      Set<Integer> codes = new HashSet<>();
74  
75      BufferedReader r = new BufferedReader(new InputStreamReader(
76            getClass().getResourceAsStream("/parse/test_datum.txt"), StandardCharsets.UTF_8));
77      String line;
78      while ((line=r.readLine()) != null) {
79        ParseResult<Integer> parsed = parser.parse(line);
80        if (parsed.isSuccessful()) {
81          success++;
82          codes.add(parsed.getPayload());
83        } else {
84          System.out.println("Failed: " + line);
85          failed++;
86        }
87      }
88  
89      System.out.println(failed + " failed parse results");
90      System.out.println(success + " successful parse results");
91      System.out.println(codes.size() + " distinct datums parsed");
92      assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
93    }
