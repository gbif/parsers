38    private ZoneId EUROPE_CENTRAL_TIME = ZoneId.of("Europe/Paris");
39  
40    // January 1st 2000, 00:00:00 (midnight)
41    private long YEAR2000_1JAN_EPOCH_UTC = 946684800000L;
42  
43    // January 1st 2000, 02:03:04
44    private long YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC = 946692184000L;
45  
46    // February 1st 2000, midnight
47    private long YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC = 949363200000l;
48  
49    // January 1st 2000, midnight
50    private long YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC = 946684800000l;
51  
52    @Test
53    public void testToDate(){
54      TemporalAccessor ta = LocalDate.of(2000, 1,1);
55      Date date = TemporalAccessorUtils.toDate(ta, false);
56      assertEquals(YEAR2000_1JAN_EPOCH_UTC, date.getTime());
57  
58      ta = LocalDateTime.of(2000, 1, 1, 2, 3, 4);
59      date = TemporalAccessorUtils.toDate(ta, false);
60      assertEquals(YEAR2000_1JAN_2H_3M_4S_EPOCH_UTC, date.getTime());
61  
62      //test partial date transformed to Date
63      //YearMonth should use the first of the month
64      ta = YearMonth.of(2000, 2);
65      date = TemporalAccessorUtils.toDate(ta, false);
66      assertEquals(YEAR2000_2FEB_0H_0M_0S_EPOCH_UTC, date.getTime());
67  
68      //Year should use the first of January
69      ta = Year.of(2000);
70      date = TemporalAccessorUtils.toDate(ta, false);
71      assertEquals(YEAR2000_1JAN_0H_0M_0S_EPOCH_UTC, date.getTime());
72    }
73  
74    @Test
75    public void testToDateWithTimeZone(){
76  
77      // Test with ZonedDateTime
78      ZonedDateTime testZonedDateTime = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, EUROPE_CENTRAL_TIME);
79      Date date = TemporalAccessorUtils.toDate(testZonedDateTime, false);
80      assertEquals(testZonedDateTime.toEpochSecond(), date.getTime()/1000);
81  
82      // Test the same ZonedDateTime but ignore Offset information (timezone)
83      ZonedDateTime testTimeUTC = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, TemporalAccessorUtils.UTC_ZONE_ID);
84      date = TemporalAccessorUtils.toDate(testZonedDateTime, true);
85      assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
86  
87      // Ensure it is also working with OffsetDateTime
88      OffsetDateTime testOffsetDateTimeTime = testZonedDateTime.toOffsetDateTime();
89      date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, false);
90      assertEquals(testOffsetDateTimeTime.toEpochSecond(), date.getTime()/1000);
91  
92      date = TemporalAccessorUtils.toDate(testOffsetDateTimeTime, true);
93      assertEquals(testTimeUTC.toEpochSecond(), date.getTime()/1000);
94    }

96    @Test
97    public void testBestResolution(){
98      TemporalAccessor ta1 = Year.of(2005);
99      TemporalAccessor ta2 = YearMonth.of(2005, 1);
100     TemporalAccessor result = TemporalAccessorUtils.bestResolution(ta1, ta2).get();
101     assertEquals(YearMonth.of(2005, 1), YearMonth.from(result));
102 
103     ta1 = LocalDate.of(2005, 1, 1);
104     ta2 = Year.of(2005);
105     result = TemporalAccessorUtils.bestResolution(ta1, ta2).get();
106     assertEquals(LocalDate.of(2005, 1, 1), LocalDate.from(result));
107 
108     //this should not work
109     ta1 = Year.of(2005);
110     ta2 = YearMonth.of(2006, 1);
111     assertFalse(TemporalAccessorUtils.bestResolution(ta1, ta2).isPresent());
112   }

114   @Test
115   public void testSameOrContained() {
116     TemporalAccessor ymd = LocalDate.of(1996, 4, 26);
117 
118     // Test nulls and wrong resolutions
119     assertFalse(TemporalAccessorUtils.sameOrContained(null, null));
120     assertFalse(TemporalAccessorUtils.sameOrContained(ymd, null));
121     assertFalse(TemporalAccessorUtils.sameOrContained(null, ymd));
122 
123     // Test against same year
124     assertTrue(TemporalAccessorUtils.sameOrContained(ymd, Year.of(1996)));
125     // Test against same month
126     assertTrue(TemporalAccessorUtils.sameOrContained(ymd, YearMonth.of(1996, 4)));
127     // Test against same date
128     assertTrue(TemporalAccessorUtils.sameOrContained(ymd, LocalDate.of(1996, 4, 26)));
129     // Test against time on that day
130     assertTrue(TemporalAccessorUtils.sameOrContained(ymd, LocalDateTime.of(1996, 4, 26, 1, 2, 3)));
131 
132     // Then different year/month/date
133     assertFalse(TemporalAccessorUtils.sameOrContained(ymd, Year.of(1998)));
134     assertFalse(TemporalAccessorUtils.sameOrContained(ymd, YearMonth.of(1996, 5)));
135     assertFalse(TemporalAccessorUtils.sameOrContained(ymd, LocalDate.of(1996, 4, 27)));
136   }

138   @Test
139   public void testLastDay(){
140     TemporalAccessor year = Year.of(1996);
141     assertEquals("1996-12-31T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(year,true).toString());
142 
143     TemporalAccessor ym = YearMonth.of(1996,1);
144     assertEquals("1996-01-31T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(ym,true).toString());
145 
146     ym = YearMonth.of(1996,2);
147     assertEquals("1996-02-29T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(ym,true).toString());
148 
149     TemporalAccessor ymd = LocalDate.of(1996,2,3);
150     assertEquals("1996-02-03T23:59:59", TemporalAccessorUtils.toLatestLocalDateTime(ymd,true).toString());
151 
152     TemporalAccessor ymdt = LocalDateTime.of(1996,2,3,1,20);
153     assertEquals("1996-02-03T01:20", TemporalAccessorUtils.toLatestLocalDateTime(ymdt,true).toString());
154   }
