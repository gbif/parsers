080  /**
081   * Build a single, possibly lenient, DateTimeParser.
082   *
083   * @param pattern
084   * @param hint
085   * @param type
086   * @return
087   */
088  private static DateTimeParser build(@NotNull String pattern, @NotNull DateFormatHint hint, @NotNull TemporalQuery<?>[] type){
089    Preconditions.checkNotNull(pattern);
090    Preconditions.checkNotNull(hint);
091
092    int minLength = getMinimumStringLengthForPattern(pattern);
093    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT);
094    return new DateTimeParser(dateTimeFormatter, null, hint, type, minLength);
095  }
096
097  private static DateTimeParser build(@NotNull String pattern, @NotNull DateFormatHint hint,
098                                      @NotNull TemporalQuery<?>[] type, ZoneId zoneId){
099    Preconditions.checkNotNull(pattern);
100    Preconditions.checkNotNull(hint);
101
102    int minLength = getMinimumStringLengthForPattern(pattern);
103    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withZone(zoneId)
104            .withResolverStyle(ResolverStyle.STRICT);
105    return new DateTimeParser(dateTimeFormatter, null, hint, type, minLength);
106  }

108  /**
109   * Build a single, strict, DateTimeParser with support for separator normalization.
110   *
111   * @param pattern
112   * @param hint
113   * @param separator
114   * @param alternativeSeparators
115   * @param type
116   * @return
117   */
118  private static DateTimeParser build(String pattern, DateFormatHint hint,  @NotNull TemporalQuery<?> type,
119                                      String separator, String alternativeSeparators){
120    return build(pattern, hint, new TemporalQuery[]{type}, separator, alternativeSeparators);
121  }

123  /**
124   * Build a single, possibly lenient, DateTimeParser with support for separator normalization.
125   *
126   * @param pattern
127   * @param hint
128   * @param separator
129   * @param alternativeSeparators
130   * @return
131   */
132  private static DateTimeParser build(String pattern, DateFormatHint hint, @NotNull TemporalQuery<?>[] type, String separator,
133                                             String alternativeSeparators){
134    Preconditions.checkNotNull(pattern);
135    Preconditions.checkNotNull(hint);
136    Preconditions.checkArgument(StringUtils.isNotBlank(separator), "separator must NOT be blank");
137    Preconditions.checkArgument(StringUtils.isNotBlank(alternativeSeparators), "alternativeSeparators must NOT be blank");
138
139    DateTimeSeparatorNormalizer dateTimeNormalizer = new DateTimeSeparatorNormalizer(CharMatcher.anyOf(alternativeSeparators), separator);
140    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT);
141    int minLength = getMinimumStringLengthForPattern(pattern);
142    return new DateTimeParser(dateTimeFormatter, dateTimeNormalizer, hint, type, minLength);
143  }

145  /**
146   *
147   * Build a single DateTimeParser from a baseYear.
148   *
149   *
150   * @param pattern pattern that includes a 2 digits year (-uu)
151   * @param hint
152   * @param baseYear
153   * @return
154   */
155  private static DateTimeParser build(String pattern, DateFormatHint hint, @NotNull TemporalQuery<?>[] type, Year baseYear){
156    int minLength = getMinimumStringLengthForPattern(pattern);
157    DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
158    return new DateTimeParser(dateTimeFormatter, null, hint, type, minLength);
159  }

161  /**
162   *
163   * Build a single DateTimeParser from a baseYear with support for separator normalization.
164   *
165   * @param pattern pattern that includes a 2 digits year (-uu)
166   * @param hint
167   * @param separator
168   * @param alternativeSeparators
169   * @param baseYear
170   * @return
171   */
172  private static DateTimeParser build(String pattern, DateFormatHint hint, @NotNull TemporalQuery<?>[] type, String separator,
173                                             String alternativeSeparators, Year baseYear){
174    Preconditions.checkNotNull(pattern);
175    Preconditions.checkNotNull(hint);
176    Preconditions.checkArgument(StringUtils.isNotBlank(separator), "separator must NOT be blank");
177    Preconditions.checkArgument(StringUtils.isNotBlank(alternativeSeparators), "alternativeSeparators must NOT be blank");
178
179    DateTimeSeparatorNormalizer dateTimeNormalizer = new DateTimeSeparatorNormalizer(CharMatcher.anyOf(alternativeSeparators), separator);
180    DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
181    int minLength = getMinimumStringLengthForPattern(pattern);
182    return new DateTimeParser(dateTimeFormatter, dateTimeNormalizer, hint, type, minLength);
183  }

187  /**
188   * From a {@link }DateTimeFormatter} pattern in String, get the minimum String length required for an input String to apply
189   * the pattern. This is used to quickly discard DateTimeFormatter simply based on String length of the input.
190   * Minimum length is the length of the pattern String minus the optional section(s) and quotes.
191   *
192   * @param pattern
193   * @return
194   */
195  private static int getMinimumStringLengthForPattern(String pattern){
196    pattern = ThreeTenNumericalDateParser.OPTIONAL_PATTERN_PART.matcher(pattern).replaceAll("").replaceAll("'", "");
197    return pattern.length();
198  }
199
200  private static DateTimeFormatter build2DigitsYearDateTimeFormatter(String pattern, Year baseYear){
201    Preconditions.checkState(pattern.matches(IS_YEAR_2_DIGITS_PATTERN) || pattern.equals(YEAR_2_DIGITS_PATTERN_SUFFIX),
202            "build2DigitsYearDateTimeFormatter can only be used for patterns with 2 digit year");
203    pattern = StringUtils.removeEnd(pattern, YEAR_2_DIGITS_PATTERN_SUFFIX);
204    return new DateTimeFormatterBuilder().append(DateTimeFormatter.ofPattern(pattern))
205            .appendValueReduced(ChronoField.YEAR, 2, 2, baseYear.getValue()).parseStrict().toFormatter();
206  }

298  /**
299   * More specific builder Builder used to build a {@link DateTimeMultiParser}.
300   */
301  public static class ThreeTenDateMultiParserListBuilder {
302    private DateTimeParser preferred;
303    private List<DateTimeParser> otherParsers = Lists.newArrayList();
304
305    public ThreeTenDateMultiParserListBuilder preferredDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type){
306      preferred = DateTimeParserBuilder.build(pattern, hint, type);
307      return this;
308    }
309
310    public ThreeTenDateMultiParserListBuilder preferredDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,  Year year){
311      preferred = DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, year);
312      return this;
313    }
314
315    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type){
316      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, type));
317      return this;
318    }
319
320    public ThreeTenDateMultiParserListBuilder appendDateTimeFormatter(DateTimeFormatter dateTimeFormatter, DateFormatHint hint, TemporalQuery<?> type, int minLength){
321      otherParsers.add(new DateTimeParser(dateTimeFormatter, null, hint, new TemporalQuery[]{type},  minLength));
322      return this;
323    }
324
325    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type, Year year){
326      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, year));
327      return this;
328    }
329
330    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,
331                                                               String separator, String alternativeSeparators){
332      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, separator, alternativeSeparators));
333      return this;
334    }
335
336    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, TemporalQuery<?> type,
337                                                               String separator, String alternativeSeparators, Year year){
338      otherParsers.add(DateTimeParserBuilder.build(pattern, hint, new TemporalQuery[]{type}, separator, alternativeSeparators, year));
339      return this;
340    }
341
342    /**
343     * Ensure the builder is used with content we expect.
344     * Currently (this could change) we should only have one DateTimeParser per DateFormatHint.
345     *
346     * @throws IllegalStateException
347     */
348    private void validate() throws IllegalStateException {
349      Set<DateFormatHint> hints = Sets.newHashSet();
350      if(preferred != null){
351        hints.add(preferred.getHint());
352      }
353
354      for(DateTimeParser parser : otherParsers){
355        if(!hints.add(parser.getHint())){
356          throw new IllegalStateException("DateFormatHint can only be used once in a DateTimeMultiParser." +
357                  "[" + parser.getHint() + "]");
358        }
359      }
360    }
361
362    public DateTimeMultiParser build() throws IllegalStateException {
363      validate();
364      return new DateTimeMultiParser(preferred, otherParsers);
365    }
366  }
