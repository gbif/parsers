026  /**
027   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}.
028   * If the provided {@link TemporalAccessor} contains offset(timezone) information it will be used.
029   * See {@link #toDate(TemporalAccessor, boolean)} for more details.
030   *
031   * @param temporalAccessor
032   * @return  the Date object or null if a Date object can not be created
033   */
034  public static Date toDate(TemporalAccessor temporalAccessor) {
035    return toDate(temporalAccessor, false);
036  }
037
038  /**
039   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}.
040   *
041   * For {@link YearMonth}, the {@link java.util.Date} will represent the first day of the month.
042   * For {@link Year}, the {@link java.util.Date} will represent the first day of January.
043   *
044   * Remember that a {@link Date} object will always display the date in the current timezone.
045   *
046   * @param temporalAccessor
047   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
048   *                     be used ?
049   * @return the Date object or null if a Date object can not be created
050   */
051  public static Date toDate(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
052    if(temporalAccessor == null){
053      return null;
054    }
055
056    if(!ignoreOffset && temporalAccessor.isSupported(ChronoField.OFFSET_SECONDS)){
057      return Date.from(temporalAccessor.query(ZonedDateTime::from).toInstant());
058    }
059
060    if(temporalAccessor.isSupported(ChronoField.SECOND_OF_DAY)){
061      return Date.from(temporalAccessor.query(LocalDateTime::from).atZone(UTC_ZONE_ID).toInstant());
062    }
063
064    // this may return null in case of partial dates
065    LocalDate localDate = temporalAccessor.query(TemporalQueries.localDate());
066
067    // try YearMonth
068    if(localDate == null && temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
069      YearMonth yearMonth = YearMonth.from(temporalAccessor);
070      localDate = yearMonth.atDay(1);
071    }
072
073    // try Year
074    if(localDate == null && temporalAccessor.isSupported(ChronoField.YEAR)) {
075      Year year = Year.from(temporalAccessor);
076      localDate = year.atDay(1);
077    }
078
079    if (localDate != null) {
080      return  Date.from(localDate.atStartOfDay(UTC_ZONE_ID).toInstant());
081    }
082
083    return null;
084  }

130  /**
131   * The idea of "best resolution" TemporalAccessor is to get the TemporalAccessor that offers more resolution than
132   * the other but they must NOT contradict.
133   * e.g. 2005-01 and 2005-01-01 will return 2005-01-01.
134   *
135   * Note that if one of the 2 parameters is null the other one will be considered having the best resolution
136   *
137   * @param ta1
138   * @param ta2
139   * @return TemporalAccessor representing the best resolution or null
140   */
141  public static TemporalAccessor getBestResolutionTemporalAccessor(@Nullable TemporalAccessor ta1,
142                                                                    @Nullable TemporalAccessor ta2){
143    //handle nulls combinations
144    if(ta1 == null && ta2 == null){
145      return null;
146    }
147    if(ta1 == null){
148      return ta2;
149    }
150    if(ta2 == null){
151      return ta1;
152    }
153
154    AtomizedLocalDate ymd1 = AtomizedLocalDate.fromTemporalAccessor(ta1);
155    AtomizedLocalDate ymd2 = AtomizedLocalDate.fromTemporalAccessor(ta2);
156
157    // If they both provide the year, it must match
158    if(ymd1.getYear() != null && ymd2.getYear() != null && !ymd1.getYear().equals(ymd2.getYear())){
159      return null;
160    }
161    // If they both provide the month, it must match
162    if(ymd1.getMonth() != null && ymd2.getMonth() != null && !ymd1.getMonth().equals(ymd2.getMonth())){
163      return null;
164    }
165    // If they both provide the day, it must match
166    if(ymd1.getDay() != null && ymd2.getDay() != null && !ymd1.getDay().equals(ymd2.getDay())){
167      return null;
168    }
169
170    if(ymd1.getResolution() > ymd2.getResolution()){
171      return ta1;
172    }
173    return ta2;
174  }
175
176  /**
177   * Given two TemporalAccessor with possibly different resolutions, this method checks if they represent the
178   * same YMD.
179   * If a null is provided, false will be returned. If one of the 2 TemporalAccessor provides resolution
180   * less than YMD, false will be returned.
181   *
182   * @param ta1
183   * @param ta2
184   *
185   * @return
186   */
187  public static boolean representsSameYMD(@Nullable TemporalAccessor ta1, @Nullable TemporalAccessor ta2) {
188
189    //handle nulls combinations
190    if (ta1 == null || ta2 == null) {
191      return false;
192    }
193
194    AtomizedLocalDate ymd1 = AtomizedLocalDate.fromTemporalAccessor(ta1);
195    AtomizedLocalDate ymd2 = AtomizedLocalDate.fromTemporalAccessor(ta2);
196
197    // we only deal with complete Local Date
198    if (!ymd1.isComplete() || !ymd2.isComplete()) {
199      return false;
200    }
201    return ymd1.equals(ymd2);
202  }
