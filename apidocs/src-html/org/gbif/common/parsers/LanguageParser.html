001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.Language;
004import org.gbif.common.parsers.core.EnumParser;
005import org.gbif.utils.file.csv.CSVReader;
006import org.gbif.utils.file.csv.CSVReaderFactory;
007
008import java.util.Arrays;
009import java.util.HashSet;
010import java.util.List;
011import java.util.Locale;
012import java.util.Set;
013import java.util.regex.Pattern;
014
015import org.apache.commons.lang3.StringUtils;
016
017/**
018 * Singleton implementation of the case insensitive ISO 639-1 language dictionary
019 * that uses the org.gbif.common.api.Language enumeration.
020 * Parsing results will be upper cased 2 letter codes.
021 * Understood inputs are 2 or 3 letter (both terminological and bibliographical) ISO codes
022 * and natural language names given in any of the ISO languages.
023 */
024public class LanguageParser extends EnumParser<Language> {
025
026  private static LanguageParser singletonObject = null;
027  private static final Pattern LOCALE = Pattern.compile("^[a-zA-Z]{2,3}_[a-zA-Z]");
028  private static final List<Pattern> REMOVE_FROM_NAME_PATTERNS = Arrays.asList(
029      // remove brackets
030      Pattern.compile("\\(.\\)"),
031      // remove French ", langues"
032      Pattern.compile(", ?langues"),
033      // remove English " languages"
034      Pattern.compile(" languages")
035  );
036
037
038  private LanguageParser() {
039    super(Language.class, false, LanguageParser.class.getResourceAsStream("/dictionaries/parse/language.tsv"));
040
041    // make sure we have all enum values mapped
042    for (Language r : Language.values()) {
043      add(r.name(), r);
044      add(r.getTitleEnglish(), r);
045      add(r.getTitleNative(), r);
046      add(r.getIso2LetterCode(), r);
047      add(r.getIso3LetterCode(), r);
048    }
049
050    // make sure we have all enum values mapped
051    for (Locale l : Locale.getAvailableLocales()) {
052      Language lang = Language.fromIsoCode(l.getISO3Language());
053      if (lang == null) {
054        log.warn("ISO code {} not part of our language enumeration", lang);
055        continue;
056      }
057      add(l.getISO3Language(), lang);
058      add(l.getDisplayLanguage(), lang);
059      add(l.getLanguage(), lang);
060      for (Locale l2 : Locale.getAvailableLocales()) {
061        add(l.getDisplayLanguage(l2), lang);
062      }
063    }
064
065    // OFFICIAL LIST, downloaded from
066    // http://www.loc.gov/standards/iso639-2/ascii_8bits.html
067
068    // An alpha-3 (bibliographic) code
069    // an alpha-3 (terminologic) code (when given)
070    // an alpha-2 code (when given)
071    // an English name
072    // a French name
073    try {
074      CSVReader r = CSVReaderFactory.build(LanguageParser.class.getResourceAsStream("/dictionaries/parse/ISO-639-2_utf-8.txt"), "UTF8", "|", null, 0);
075      while(r.hasNext()) {
076        String[] row = r.next();
077        if (row.length>2) {
078          // ISO 2 letter code
079          String alpha2 = row[2];
080          if (!StringUtils.isBlank(alpha2)) {
081            Language lang = Language.fromIsoCode(alpha2);
082            if (lang == null || lang == Language.UNKNOWN) {
083              log.warn("ISO code {} not part of our language enumeration", alpha2);
084              continue;
085            }
086            // alpha-3 (bibliographic)
087            add(row[0], lang);
088            // alpha-3 (terminologic)
089            add(row[1], lang);
090            // English
091            for (String l : mutateLanguageName(row[3])) {
092              add(l, lang);
093            }
094            // French
095            for (String l : mutateLanguageName(row[4])) {
096              add(l, lang);
097            }
098          }
099        }
100      }
101    } catch (Exception e) {
102      log.error("Cannot initiate language parser: {}", e.getMessage());
103      throw new IllegalStateException("Cannot initiate language parser", e);
104    }
105
106
107    // ISO 639-3 list from SIL:
108    // http://www-01.sil.org/iso639-3/download.asp
109    try {
110      // Id     Part2B  Part2T  Part1   Scope   Language_Type   Ref_Name        Comment
111      CSVReader r = CSVReaderFactory.buildTabReader(LanguageParser.class.getResourceAsStream("/dictionaries/parse/iso-639-3-sil.tab"), "UTF8", 1);
112      while(r.hasNext()) {
113        String[] row = r.next();
114        if (row.length>2) {
115          String alpha2 = row[3];
116          if (!StringUtils.isBlank(alpha2)) {
117            Language lang = Language.fromIsoCode(alpha2);
118            if (lang == null || lang == Language.UNKNOWN) {
119              log.warn("ISO code {} not part of our language enumeration", alpha2);
120              continue;
121            }
122            // 3-letter code
123            add(row[0], lang);
124            // 3-letter code part2B
125            add(row[1], lang);
126            // 3-letter code part2T
127            add(row[2], lang);
128            // name
129            add(row[6], lang);
130          }
131        }
132      }
133    } catch (Exception e) {
134      log.error("Cannot initiate language parser: {}", e.getMessage());
135      throw new IllegalStateException("Cannot initiate language parser", e);
136    }
137  }
138
139  private Set<String> mutateLanguageName(String lang) {
140    Set<String> langs = new HashSet<>();
141    for (String l : lang.split(";")) {
142      langs.add(l);
143      // also remove common patterns
144      for (Pattern p : REMOVE_FROM_NAME_PATTERNS) {
145        langs.add(p.matcher(l).replaceAll(""));
146      }
147    }
148    return langs;
149  }
150
151  @Override
152  protected String normalize(String value) {
153    if (value != null) {
154      /**
155       * A language string could come in as a locale like "en_US" or if it was constructed improperly "eng_US", so
156       * extract only the part before the underscore. Only if it contains an "_" is parsing attempted.
157       * Whether it actually represents an ISO 369 language code is left for the language parser to determine.
158       */
159      if (LOCALE.matcher(value).find()) {
160        int index = value.indexOf("_");
161        // only allow underscore
162        if (index > 1 && index < 4 ) {
163          return super.normalize(value.substring(0, index));
164        }
165      }
166      return super.normalize(value);
167    }
168    return null;
169  }
170
171  public static LanguageParser getInstance() {
172    synchronized (LanguageParser.class) {
173      if (singletonObject == null) {
174        singletonObject = new LanguageParser();
175      }
176    }
177    return singletonObject;
178  }
179}
