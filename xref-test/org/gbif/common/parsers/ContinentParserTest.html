24        assertParseSuccess(c, c.getTitle());
25      }

36    public void testParseSuccess() {

39      assertParseSuccess(Continent.NORTH_AMERICA, "Caribbean");
40      assertParseSuccess(Continent.NORTH_AMERICA, "North america");
