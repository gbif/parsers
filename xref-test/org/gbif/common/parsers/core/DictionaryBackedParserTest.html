14  package org.gbif.common.parsers.core;
15  
16  import java.util.ArrayList;
17  import java.util.List;
18  
19  import org.junit.jupiter.api.Test;
20  
21  import static org.junit.jupiter.api.Assertions.assertEquals;
22  import static org.junit.jupiter.api.Assertions.assertNotNull;
23  
24  public class DictionaryBackedParserTest {
25  
26    @Test
27    public void testParse() {
28      List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
29      source.add(new KeyValue<String, Integer>("Tim", 32));
30      source.add(new KeyValue<String, Integer>("Markus", 38));
31      source.add(new KeyValue<String, Integer>("Jose", 28));
32      DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(false);
33      dbp.init(source.iterator());
34  
35      assertParsed(dbp, "Tim", 32);
36      assertParsed(dbp, "TIM", 32);
37      assertParsed(dbp, "tIm", 32);
38      assertParsed(dbp, " \t Markus \t ", 38);
39      assertParsed(dbp, "MarKUS", 38);
40      assertParsed(dbp, "Jose", 28);
41      assertEquals(ParseResult.STATUS.FAIL, dbp.parse("Lars").getStatus());
42    }
43  
44    @Test
45    public void testSensitiveParse() {
46      List<KeyValue<String, Integer>> source = new ArrayList<KeyValue<String, Integer>>();
47      source.add(new KeyValue<String, Integer>("Matt", 29));
48      DictionaryBackedParser<Integer> dbp = new DictionaryBackedParser<Integer>(true);
49      dbp.init(source.iterator());
50  
51      assertParsed(dbp, "Matt", 29);
52      assertParsed(dbp, " \t Matt \n ", 29);
53      assertEquals(ParseResult.STATUS.FAIL, dbp.parse("MATT").getStatus());
54    }
55  
56    private void assertParsed(DictionaryBackedParser<Integer> dbp, String input, Integer payload) {
57      assertNotNull(dbp.parse(input));
58      assertEquals(ParseResult.STATUS.SUCCESS, dbp.parse(input).getStatus());
59      assertEquals(ParseResult.CONFIDENCE.DEFINITE, dbp.parse(input).getConfidence());
60      assertEquals(payload, dbp.parse(input).getPayload());
61    }
62  
63  }
