001package org.gbif.common.parsers;
002
003import com.google.common.base.Strings;
004import com.google.common.collect.ImmutableSet;
005import org.apache.tika.Tika;
006import org.apache.tika.mime.MediaTypeRegistry;
007import org.apache.tika.mime.MimeType;
008import org.apache.tika.mime.MimeTypeException;
009import org.apache.tika.mime.MimeTypes;
010import org.gbif.api.model.common.MediaObject;
011import org.gbif.api.vocabulary.MediaType;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015import javax.annotation.Nullable;
016import java.net.URI;
017import java.util.Set;
018
019public class MediaParser {
020  private static final Logger LOG = LoggerFactory.getLogger(MediaParser.class);
021  private static final Tika TIKA = new Tika();
022  private static final MimeTypes MIME_TYPES = MimeTypes.getDefaultMimeTypes();
023  private static final String HTML_TYPE = "text/html";
024  // MIME types which we consider as HTML links instead of real media file URIs
025  private static final Set<String> HTML_MIME_TYPES = ImmutableSet
026    .of("text/x-coldfusion", "text/x-php", "text/asp", "text/aspdotnet", "text/x-cgi", "text/x-jsp", "text/x-perl",
027      HTML_TYPE, MIME_TYPES.OCTET_STREAM);
028
029  // Add missing alias types.
030  static {
031    MediaTypeRegistry mediaTypeRegistry = MIME_TYPES.getMediaTypeRegistry();
032    mediaTypeRegistry.addAlias(org.apache.tika.mime.MediaType.audio("mpeg"), org.apache.tika.mime.MediaType.audio("mp3"));
033    mediaTypeRegistry.addAlias(org.apache.tika.mime.MediaType.audio("mpeg"), org.apache.tika.mime.MediaType.audio("mpeg3"));
034  }
035
036  private static MediaParser instance = null;
037
038  public static MediaParser getInstance() {
039    synchronized (MediaParser.class) {
040      if (instance == null) {
041        instance = new MediaParser();
042      }
043    }
044    return instance;
045  }
046
047  public MediaObject detectType(MediaObject mo) {
048    if (Strings.isNullOrEmpty(mo.getFormat())) {
049      // derive from URI
050      mo.setFormat(parseMimeType(mo.getIdentifier()));
051    }
052
053    // if MIME type is text/html make it a references link instead
054    if (HTML_TYPE.equalsIgnoreCase(mo.getFormat()) && mo.getIdentifier() != null) {
055      // make file URI the references link URL instead
056      mo.setReferences(mo.getIdentifier());
057      mo.setIdentifier(null);
058      mo.setFormat(null);
059    }
060
061    if (!Strings.isNullOrEmpty(mo.getFormat())) {
062      if (mo.getFormat().startsWith("image")) {
063        mo.setType(MediaType.StillImage);
064      } else if (mo.getFormat().startsWith("audio")) {
065        mo.setType(MediaType.Sound);
066      } else if (mo.getFormat().startsWith("video")) {
067        mo.setType(MediaType.MovingImage);
068      } else {
069        LOG.debug("Unsupported media format {}", mo.getFormat());
070      }
071    }
072    return mo;
073  }
074
075  /**
076   * Parses a MIME type using Apache Tika which can handle the following:
077   * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
078   * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
079   */
080  public String parseMimeType(@Nullable String format) {
081    if (format != null) {
082      format = Strings.emptyToNull(format.trim().toLowerCase());
083    }
084    if (format == null) {
085      return null;
086    }
087
088    try {
089      MimeType mime = MIME_TYPES.getRegisteredMimeType(format);
090      if (mime != null) {
091        return mime.getName();
092      }
093
094    } catch (MimeTypeException e) {
095    }
096
097    // Failed, but return the input if it's a reasonable MIME type
098    return MimeType.isValid(format) ? format : null;
099  }
100
101  /**
102   * Parses a MIME type using Apache Tika which can handle the following:
103   * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
104   * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
105   */
106  public String parseMimeType(@Nullable URI uri) {
107    if (uri != null) {
108      String mime = TIKA.detect(uri.toString());
109      if (mime != null && HTML_MIME_TYPES.contains(mime.toLowerCase())) {
110        // We may have something like http://example.org/imageServer?img=test.jpg, so re-run the detection on the last
111        // part of the URL query string.
112        if (uri.getQuery() != null) {
113          mime = TIKA.detect(uri.getQuery());
114          if (mime != null && !HTML_MIME_TYPES.contains(mime.toLowerCase())) {
115            return mime;
116          }
117        }
118
119        // links without any suffix default to OCTET STREAM, see:
120        // http://dev.gbif.org/issues/browse/POR-2066
121        return HTML_TYPE;
122      }
123      return mime;
124    }
125    return null;
126  }
127}
