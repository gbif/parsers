50    public OccurrenceParseResult<IsoDateInterval> parse(
51        String year,
52        String month,
53        String day,
54        String dateRange,
55        String startDayOfYear,
56        String endDayOfYear) {
57      // Even a single date will be split to two
58      String[] rawPeriod = DelimiterUtils.splitPeriod(dateRange);
59  
60      Temporal from;
61      Temporal to;
62      Set<OccurrenceIssue> issues = new HashSet<>();
63  
64      // If eventDate is a multi-day range, with at least day precision, and year+month+day are set, we must test
65      // whether year+month+day falls within this range.
66      if (StringUtils.isNotBlank(year) && StringUtils.isNotBlank(month) && StringUtils.isNotBlank(day) && StringUtils.isNotBlank(dateRange)) {
67        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyStart = temporalParser.parseRecordedDate(null, null, null, rawPeriod[0], null);
68        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyEnd = temporalParser.parseRecordedDate(null, null, null, rawPeriod[1], null);
69        OccurrenceParseResult<TemporalAccessor> ymdOnly = temporalParser.parseRecordedDate(year, month, day, null, null);
70  
71        if (dateRangeOnlyStart.isSuccessful() && dateRangeOnlyEnd.isSuccessful() && ymdOnly.isSuccessful()) {
72          if (dateRangeOnlyStart.getPayload().isSupported(ChronoField.DAY_OF_YEAR)
73            && dateRangeOnlyEnd.getPayload().isSupported(ChronoField.DAY_OF_YEAR)
74            && ymdOnly.getPayload().isSupported(ChronoField.DAY_OF_YEAR)) {
75              if (TemporalAccessorUtils.withinRange(dateRangeOnlyStart.getPayload(), dateRangeOnlyEnd.getPayload(), ymdOnly.getPayload())) {
76              // Then we can just check the startDayOfYear and endDayOfYear fields match.
77              from = parseAndSet(null, null, null, rawPeriod[0], startDayOfYear, issues);
78              to = parseAndSet(null, null, null, rawPeriod[1], endDayOfYear, issues);
79              // If the dateRange has different resolutions on either side, truncate it.
80              if (TemporalAccessorUtils.resolutionToSeconds(from) != TemporalAccessorUtils.resolutionToSeconds(to)) {
81                int requiredResolution = Math.min(TemporalAccessorUtils.resolutionToSeconds(from), TemporalAccessorUtils.resolutionToSeconds(to));
82                from = TemporalAccessorUtils.limitToResolution(from, requiredResolution);
83                to = TemporalAccessorUtils.limitToResolution(to, requiredResolution);
84                issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
85              }
86              log.trace("Range {}|{}|{}|{}|{}|{} succeeds with ymd within range {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
87              return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(from, to), issues);
88            }
89          }
90        }
91      }
92  
93      // If eventDate is a range, and at least year is set, we must test whether year+month+day are set according to the
94      // constant parts of eventDate.
95      if (StringUtils.isNotBlank(year) && StringUtils.isNotBlank(dateRange)) {
96        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyStart = temporalParser.parseRecordedDate(null, null, null, rawPeriod[0], null);
97        OccurrenceParseResult<TemporalAccessor> dateRangeOnlyEnd = temporalParser.parseRecordedDate(null, null, null, rawPeriod[1], null);
98        OccurrenceParseResult<TemporalAccessor> ymdOnly = temporalParser.parseRecordedDate(year, month, day, null, null);
99  
100       if (dateRangeOnlyStart.isSuccessful() && dateRangeOnlyEnd.isSuccessful() && ymdOnly.isSuccessful()) {
101         Optional<TemporalAccessor> dateRangeConstant = TemporalAccessorUtils.nonConflictingDateParts(dateRangeOnlyStart.getPayload(), dateRangeOnlyEnd.getPayload(), null);
102 
103         if (dateRangeConstant.isPresent() && ymdOnly.getPayload().equals(dateRangeConstant.get())) {
104           // Then we can just check the startDayOfYear and endDayOfYear fields match.
105           from = parseAndSet(null, null, null, rawPeriod[0], startDayOfYear, issues);
106           to = parseAndSet(null, null, null, rawPeriod[1], endDayOfYear, issues);
107           if (TemporalAccessorUtils.resolutionToSeconds(from) != TemporalAccessorUtils.resolutionToSeconds(to)) {
108             int requiredResolution = Math.min(TemporalAccessorUtils.resolutionToSeconds(from), TemporalAccessorUtils.resolutionToSeconds(to));
109             from = TemporalAccessorUtils.limitToResolution(from, requiredResolution);
110             to = TemporalAccessorUtils.limitToResolution(to, requiredResolution);
111             issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
112           }
113           log.trace("Range {}|{}|{}|{}|{}|{} succeeds with correct ymd parts {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
114           return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(from, to), issues);
115         }
116       }
117     }
118 
119     // Otherwise, we will reduce the precision of the given dates until they all agree.
120 
121     // Year+month+day, first part of eventDate, and startDay of year to the best we can get.
122     from = parseAndSet(year, month, day, rawPeriod[0], startDayOfYear, issues);
123     // Year+month+day, second part of eventDate, and endDayOfYear of year to the best we can get.
124     to = parseAndSet(year, month, day, rawPeriod[1], endDayOfYear, issues);
125     log.trace("Range {}|{}|{}|{}|{}|{} parsed to {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
126 
127     // Return a failure, rather than a range with a missing start or end
128     if (from != null ^ to != null) {
129       log.debug("Range {}|{}|{}|{}|{}|{} fails due to missing start xor end {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
130       issues.add(OccurrenceIssue.RECORDED_DATE_MISMATCH);
131       return OccurrenceParseResult.fail(issues);
132     }
133 
134     // If the resolutions are not equal, truncate such that they are.
135     if (from != null && to != null) {
136       if (TemporalAccessorUtils.resolutionToSeconds(from) != TemporalAccessorUtils.resolutionToSeconds(to)) {
137         log.trace("Range {}|{}|{}|{}|{}|{} has different resolutions, will be truncated {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
138         int requiredResolution = Math.min(TemporalAccessorUtils.resolutionToSeconds(from), TemporalAccessorUtils.resolutionToSeconds(to));
139         from = TemporalAccessorUtils.limitToResolution(from, requiredResolution);
140         to = TemporalAccessorUtils.limitToResolution(to, requiredResolution);
141       }
142     }
143 
144     // Reverse order if needed
145     if (from != null && to != null) {
146       if (from.getClass() == to.getClass()) {
147         long rangeDiff = getRangeDiff(from, to);
148         if (rangeDiff < 0) {
149           log.trace("Range {}|{}|{}|{}|{}|{} will be reversed {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
150           issues.add(OccurrenceIssue.RECORDED_DATE_INVALID);
151           return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(to, from), issues);
152         }
153       } else {
154         issues.add(OccurrenceIssue.RECORDED_DATE_UNLIKELY);
155       }
156     }
157 
158     if (from == null && to == null) {
159       log.debug("Range {}|{}|{}|{}|{}|{} could not be parsed", year, month, day, dateRange, startDayOfYear, endDayOfYear);
160       return OccurrenceParseResult.fail(issues);
161     } else if (from != null && to != null) {
162       log.trace("Range {}|{}|{}|{}|{}|{} succeeds {}→{}", year, month, day, dateRange, startDayOfYear, endDayOfYear, from, to);
163       return OccurrenceParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IsoDateInterval(from, to), issues);
164     } else {
165       log.error("From {} and to {} dates of range are unexpectedly not-null and null when parsing {}, {}, {}, {}, {}, {}",
166         from, to, year, month, day, dateRange, startDayOfYear, endDayOfYear);
167       issues.add(OccurrenceIssue.INTERPRETATION_ERROR);
168       return OccurrenceParseResult.fail(issues);
169     }
170   }

172   private Temporal parseAndSet(
173       String year,
174       String month,
175       String day,
176       String rawDate,
177       String dayOfYear,
178       Set<OccurrenceIssue> issues) {
179     OccurrenceParseResult<TemporalAccessor> result =
180         temporalParser.parseRecordedDate(year, month, day, rawDate, dayOfYear);
181     issues.addAll(result.getIssues());
182     if (result.isSuccessful()) {
183       return (Temporal) result.getPayload();
184     } else {
185       return null;
186     }
187   }
188 
189   /** Compare dates and returns difference between FROM and TO dates in milliseconds */
190   private static long getRangeDiff(Temporal from, Temporal to) {
191     if (from == null || to == null) {
192       return 1L;
193     }
194     TemporalUnit unit = null;
195     if (from instanceof Year) {
196       unit = ChronoUnit.YEARS;
197     } else if (from instanceof YearMonth) {
198       unit = ChronoUnit.MONTHS;
199     } else if (from instanceof LocalDate) {
200       unit = ChronoUnit.DAYS;
201     } else if (from instanceof LocalDateTime
202         || from instanceof OffsetDateTime
203         || from instanceof ZonedDateTime) {
204       unit = ChronoUnit.SECONDS;
205     }
206     return from.until(to, unit);
207   }
