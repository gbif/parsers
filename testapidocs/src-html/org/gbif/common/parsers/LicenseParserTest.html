003import org.gbif.api.vocabulary.License;

012public class LicenseParserTest extends ParserTestBase<License> {
013
014  public LicenseParserTest() {
015    super(LicenseParser.getInstance());

017
018  /**
019   * Makes sure all License enum values and their parameters are parsed ok.
020   */
021  @Test
022  public void testParseAllEnumValues() {
023    for (License l : License.values()) {
024      assertParseSuccess(l, l.name());
025      if (l.getLicenseUrl() != null) {
026        assertParseSuccess(l, l.getLicenseUrl());
027      }
028      if (l.getLicenseTitle() != null) {
029        assertParseSuccess(l, l.getLicenseTitle());
030      }
031    }
032  }
033
034  @Test
035  public void testParseFail() {
036    assertParseFailure("Not licensed");
037    assertParseFailure("CC");
038  }

087  @Test
088  public void testParseUriThenTitle() throws URISyntaxException {
089    LicenseParser parser = LicenseParser.getInstance();
090    assertEquals(License.UNSPECIFIED, parser.parseUriThenTitle(null, null));
091
092    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(null, "CC0"));
093    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(null, "CCZero"));
094    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/publicdomain/zero/1.0"), null));
095    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/publicdomain/zero/1.0/"), null));
096    assertEquals(License.CC0_1_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/publicdomain/zero/1.0/legalcode"), null));
097
098    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(null, "CC-BY"));
099    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/4.0"), null));
100    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/4.0/"), null));
101    assertEquals(License.CC_BY_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/4.0/legalcode"), null));
102
103    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(null, "CC-BY-NC"));
104    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by-nc/4.0"), null));
105    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by-nc/4.0/"), null));
106    assertEquals(License.CC_BY_NC_4_0, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by-nc/4.0/legalcode"), null));
107
108    assertEquals(License.UNSUPPORTED, parser.parseUriThenTitle(new URI("http://creativecommons.org/licenses/by/3.0/legalcode"), null));
109    assertEquals(License.UNSUPPORTED, parser.parseUriThenTitle(new URI("http://opendatacommons.org/licenses/odbl/1.0"), null));
110  }
