020/**
021 * Utilities for parsing dates. Methods are implemented in static to reduce object creation and shorten coding for
022 * clients. It is not anticipated that these methods would ever be mocked in unit tests.
023 * Note: The returned date will always be in the systems default TimeZone. BE CAREFUL!
024 */
025public class DateParseUtils {
026
027  private static final Logger LOGGER = LoggerFactory.getLogger(DateParseUtils.class);
028
029  private static final Joiner DATE_JOINER = Joiner.on("-").skipNulls();
030
031  // parsers are held as statics (effectively singletons) as they are all threadsafe
032  private static  final StringToDateParser STRING_TO_DATE_PARSER = new StringToDateParser();
033
034  // enumeration of the fields supported in the date handling
035  public enum DATE_FIELD {
036    DAY, MONTH, YEAR
037  }
038
039  // Dictionary for interpreting string months
040  // Important that the FIRST is the canonical form, followed
041  // by abbreviations.  Be very careful not to add any ambiguous terms
042  // such as "J" "A" etc.  Do not add a "." at the end of abbreviations
043  // as normalization will automatically do this.
044  protected static final String[][] MONTHS = {{"January", "Jan", "J", "Ene", "Ja"},  // Ene. is abbreviated Spanish
045    {"February", "Feb", "F", "Fe"}, {"March", "Mar"}, {"April", "Apr", "Ap", "Abr"},  // Abr. is abbreviated Spanish
046    {"May"}, {"June", "Jun"}, {"July", "Jul"}, {"August", "Aug", "Au", "Ago"},
047    // Ago seen a lot in data
048    {"September", "Sep", "Sept", "Set", "S"}, // Set is a common mispelling
049    {"October", "Oct", "O", "Oc"}, {"Novermber", "Nov", "N", "No"}, {"December", "Dec", "D", "De"},};
050
051  protected static final int CURRENT_YEAR = new GregorianCalendar().get(Calendar.YEAR);
052
053  /**
054   * Takes a String representation and attempts to construct a date
055   *
056   * @param input To convert to a Date()
057   *
058   * @return The result of the parsing
059   */
060  public static ParseResult<Date> parse(String input) {
061    input = normalizeMonth(input);
062    return STRING_TO_DATE_PARSER.parse(input);
063  }
064
065  /**
066   * Takes a String representation that is in the format specified and attempts to construct a date.
067   *
068   * @param input  To convert to a Date()
069   * @param format The format of the date, e.g. dd/mm/yyyy
070   *
071   * @return The result of the parsing
072   */
073  public static ParseResult<Date> parse(String input, String format) {
074    try {
075      Date d = DateUtils.parseDate(input, format);
076      LOGGER.debug("Parsed input[{}] in format[{}] to date[{}]", new Object[] {input, format, d});
077      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, d);
078    } catch (ParseException e) {
079      LOGGER.debug("Exception caught trying to parse a date", e);
080      return ParseResult.fail();
081    }
082  }
083
084  /**
085   * Utility to check that the field appears valid in it's status use. Day must fall within 1-31, month within 1-12 and
086   * year within 1-<currentyear>
087   *
088   * @param value To test
089   * @param field The context that the value is being used
090   *
091   * @return true or false depending on the validity of the value in it's context
092   */
093  public static boolean isValidUse(String value, DATE_FIELD field) {
094    try {
095      int valueAsInt = Integer.parseInt(value);
096      switch (field) {
097        case DAY:
098          return valueAsInt >= 1 && valueAsInt <= 31;
099        case MONTH:
100          return valueAsInt >= 1 && valueAsInt <= 12;
101        case YEAR:
102          return valueAsInt >= 1 && valueAsInt <= new GregorianCalendar().get(Calendar.YEAR);
103      }
104    } catch (NumberFormatException e) {
105      LOGGER.debug("{} does not appear valid as a {}", new Object[] {value, field, e});
106    }
107    return false;
108  }
109
110  /**
111   * Checks if a {@link YearMonthDay} object <i>maybe</i> represents a valid date in the sense that the date really
112   * exists (e.g. no February 31st)
113   * As the components of a YearMonthDay may be null it only checks this validity if all components are not null.
114   *
115   * @param ymd to check
116   *
117   * @return true if it may be a valid partial date, false if it is sure that it is an invalid date.
118   */
119  public static boolean isDateCoercible(YearMonthDay ymd) {
120    if (ymd.getYear() == null || ymd.getMonth() == null || ymd.getDay() == null) {
121      return true;
122    }
123
124    // SimpleDateFormat is not thread-safe. Do not reuse or if you reuse use Thread locals or other synchronization
125    SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
126    sdf.setLenient(false);
127    return sdf.parse(DATE_JOINER.join(ymd.getYear(), ymd.getMonth(), ymd.getDay()), new ParsePosition(0)) != null;
128  }
129
130  /**
131   * Tries to create a valid date (complete or partial) out of three Strings for year, month and day.
132   * If all three components are null the parse will fail.
133   *
134   * @param year  to parse. May be null.
135   * @param month to parse. May be null.
136   * @param day   to parse. May be null.
137   *
138   * @return result of the parsing
139   */
140  public static ParseResult<Date> parse(@Nullable String year, @Nullable String month, @Nullable String day) {
141    if (year == null && month == null && day == null) {
142      return ParseResult.fail();
143    }
144
145    month = normalizeMonth(month);
146    Date d = STRING_TO_DATE_PARSER.strictParse(day + "/" + month + "/" + year, new String[]{"dd/MM/yyyy", "dd/mm/yyyy"});
147    if (d != null) {
148      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, d);
149    }
150    return ParseResult.fail();
151  }
152
153  /**
154   * Note that YearMonthDay can be sparsely populated
155   *
156   * @param input To convert to a YearMonthDay()
157   *
158   * @return The result of the parsing
159   */
160  public static YearMonthDay atomize(Date input) {
161    YearMonthDay ymd = new YearMonthDay();
162    Calendar c = Calendar.getInstance();
163    c.setTime(input);
164    ymd.setDay(String.valueOf(c.get(Calendar.DAY_OF_MONTH)));
165    ymd.setMonth(String.valueOf(c.get(Calendar.MONTH) + 1)); // months are zero indexed
166    ymd.setYear(String.valueOf(c.get(Calendar.YEAR)));
167    return ymd;
168  }

171  public static YearMonthDay normalize(String year, String month, String day) {
172    YearMonthDay ymd = new YearMonthDay();
173    // normalize float values
174    year = normalizeFloat(year);
175    month = normalizeFloat(month);
176    day = normalizeFloat(day);
177
178    // normalize string months
179    month = normalizeMonth(month);
180
181    // remove nulls
182    day = "\\N".equalsIgnoreCase(day) ? null : day;
183    month = "\\N".equalsIgnoreCase(month) ? null : month;
184    year = "\\N".equalsIgnoreCase(year) ? null : year;
185    year = inferCentury(year);
186    if (isValidUse(year, DATE_FIELD.YEAR)) {
187      ymd.setYear(year);
188    }
189    if (isValidUse(month, DATE_FIELD.MONTH)) {
190      ymd.setMonth(month);
191    }
192    if (isValidUse(day, DATE_FIELD.DAY)) {
193      ymd.setDay(day);
194    }
195
196    // if everything is null, it is possible that
197    // the year contains a full date.  Attempt a parse of that
198    if (ymd.representsNull() && StringUtils.isNotBlank(year)) {
199      LOGGER.debug("Attempting to infer day, month and year from year field: {}", year);
200      ParseResult<YearMonthDay> result = atomize(year);
201      if (ParseResult.STATUS.SUCCESS == result.getStatus()) {
202        LOGGER.debug("Year[{}] resulted in: {}", year, result.getPayload());
203        ymd = result.getPayload();
204      } else {
205        try {
206          // last ditch attempt to get a date
207          Date p = DateUtils.parseDate(year, new String[] {"yyyy-MM-dd", "yyyy/MM/dd"});
208          GregorianCalendar gc = new GregorianCalendar();
209          gc.setTime(p);
210          ymd.setYear(String.valueOf(gc.get(Calendar.YEAR)));
211          ymd.setMonth(String.valueOf(gc.get(Calendar.MONTH)));
212          // we deliberately do not set the day since this is a last ditch to catch bad
213          // dates, and a common reason is the bad day (31/06/2000 for example)
214
215        } catch (ParseException e) {
216        }
217      }
218    }
219
220    if (!isDateCoercible(ymd)) {
221      ymd.setYear(null);
222      ymd.setMonth(null);
223      ymd.setDay(null);
224    }
225
226    // Final check before returning
227    if (!isValidUse(ymd.getYear(), DATE_FIELD.YEAR)) {
228      ymd.setYear(null);
229    }
230    if (!isValidUse(ymd.getMonth(), DATE_FIELD.MONTH)) {
231      ymd.setMonth(null);
232    }
233    if (!isValidUse(ymd.getDay(), DATE_FIELD.DAY)) {
234      ymd.setDay(null);
235    }
236
237    return ymd;
238  }

240  /**
241   * Should the year be lower than 100, this will assume it is current century if makes sense, otherwise uses previous
242   * century. At time of writing (2010) a value of 1-99 will be treated as 2010, whereas 11 will be considered 1911.
243   * Next year, the same code would interpret 11 as 2011.  This is not foolproof, but much more likely than the year
244   * 0011.
245   */
246  public static String inferCentury(String year) {
247    try {
248      int y = Integer.parseInt(year);
249      if (y >= 0 && y <= 99) {
250        // if it is in the future, it can't be
251        if ((CURRENT_YEAR - (CURRENT_YEAR % 100)) + y > CURRENT_YEAR) {
252          y = CURRENT_YEAR - (CURRENT_YEAR % 100) - 100 + y;
253        } else {
254          y = CURRENT_YEAR - (CURRENT_YEAR % 100) + y;
255        }
256        return String.valueOf(y);
257      }
258    } catch (NumberFormatException e) {
259      // no problem, it does not look like a 2 digit year anyway
260    }
261    return year;
262  }

264  /**
265   * Often values are seen as Float rather than int, due to a database export The year "1978" is actually seen as
266   * "1978.0".  Where this is detected, the string is normalized to the INT representation
267   *
268   * @param value To check
269   *
270   * @return the integer value
271   */
272  public static String normalizeFloat(String value) {
273    if (value != null && value.contains(".0")) {
274      try {
275        Double d = new Double(value);
276        if ((double) d.intValue() == d.doubleValue()) {
277          return String.valueOf(d.intValue());
278        }
279      } catch (NumberFormatException e) {
280      }
281    }
282    return value;
283  }

285  /**
286   * Parses the date and then atomizes it to the year month and day
287   *
288   * @param input To convert to a YearMonthDay()
289   *
290   * @return The result of the parsing
291   */
292  public static ParseResult<YearMonthDay> atomize(String input) {
293    input = normalizeMonth(input);
294    ParseResult<Date> parsed = parse(input);
295    if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
296      // atomize
297      YearMonthDay atomized = atomize(parsed.getPayload());
298      // construct a combined result of the 2 stages
299      return ParseResult.success(parsed.getConfidence(), atomized);
300
301    } else if (ParseResult.STATUS.ERROR == parsed.getStatus()) {
302      return ParseResult.error(parsed.getError());
303    }
304
305    return ParseResult.fail();
306  }

308  /**
309   * Often months come in the form Sept. September etc. This will convert many variations into the numerical version
310   *
311   * @param input To normalize
312   *
313   * @return the normalized month
314   */
315  public static String normalizeMonth(String input) {
316    if (input != null) {
317      int m = 1;
318      for (String[] monthValues : MONTHS) {
319        for (String month : monthValues) {
320          if (StringUtils.equalsIgnoreCase(month, input) || StringUtils.equalsIgnoreCase(month + ".", input)) {
321            return padTo2Chars(m);
322          }
323        }
324        m++;
325      }
326    }
327    return input;
328  }

330  /**
331   * @param value To convert into a 2 char String.  Must be 1-12 inclusive or else strange results might occur
332   *
333   * @return 1 becomes "01", 5 becomes "05", 10 becomes "10" etc
334   */
335  protected static String padTo2Chars(int value) {
336    return value < 10 ? "0" + value : String.valueOf(value);
337  }
