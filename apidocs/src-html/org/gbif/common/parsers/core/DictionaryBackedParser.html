001package org.gbif.common.parsers.core;
002
003import java.util.HashMap;
004import java.util.Iterator;
005import java.util.Map;
006
007import com.google.common.base.Preconditions;
008import com.google.common.base.Strings;
009import org.apache.commons.lang3.StringUtils;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013/**
014 * A simple parser that will initialise with source data, and will
015 * use that as a lookup to replace values. This was written with
016 * basis of record lookup and country names in mind.
017 * Future improvements to this implementation might call a
018 * dictionary web service for example, to achieve the same,
019 * but allow the abstraction of the dictionary management to a
020 * better project (separation of concerns)
021 */
022public class DictionaryBackedParser<V> implements Parsable<V> {
023  protected final Logger log = LoggerFactory.getLogger(getClass());
024  private final Map<String, V> dictionary = new HashMap<String, V>();
025  private boolean caseSensitive = false;
026
027  /**
028   * @param caseSensitive If the dictionary should be case sensitive (only applicable to String keys)
029   */
030  public DictionaryBackedParser(boolean caseSensitive) {
031    this.caseSensitive = caseSensitive;
032  }
033
034  /**
035   * @param source To build the dictionary from
036   */
037  @SuppressWarnings("unchecked")
038  public void init(Iterator<KeyValue<String, V>> source) {
039    while (source.hasNext()) {
040      KeyValue<String, V> kvp = source.next();
041      add(kvp.getKey(), kvp.getValue());
042    }
043  }
044
045  final protected void add(String key, V value) {
046    Preconditions.checkNotNull("Adding a null value to the parser dictionary is not allowed", value);
047    String normedKey = normalize(key);
048    if (!Strings.isNullOrEmpty(normedKey)) {
049      V existingValue = dictionary.get(normedKey);
050      if (existingValue != null && !existingValue.equals(value)) {
051        log.warn("Ignore value {} mapped to more than one result", key);
052      } else {
053        dictionary.put(normedKey, value);
054      }
055    }
056  }
057
058  /**
059   * Normalisation of a value used both by adding to the internal dictionary and parsing values.
060   * The default does trim and uppercase the value for Strings, but leaves other types unaltered.
061   * Override this method to provide specific normalisations for parsers.
062   *
063   * @param value the value to be normalised.
064   *
065   * @return the normalised value
066   */
067  protected String normalize(String value) {
068    if (value != null) {
069      if (!caseSensitive) {
070        return StringUtils.trimToNull(value.toUpperCase());
071      }
072    }
073    return value;
074  }
075
076  /**
077   * Tries to parse the input data according to its backing dictionary.
078   * If no entry in the dictionary can be found the result will be {@link ParseResult.STATUS FAIL} otherwise the
079   * result will be a {@link org.gbif.common.parsers.core.ParseResult.CONFIDENCE DEFINITE} {@link ParseResult.STATUS
080   * SUCCESS}.
081   *
082   * @param input To lookup in the dictionary
083   *
084   * @return the replacement from the dictionary
085   */
086  @Override
087  public ParseResult<V> parse(String input) {
088    String normed = normalize(input);
089    V value = dictionary.get(normed);
090    if (value == null) {
091      return ParseResult.fail();
092    } else {
093      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, value);
094    }
095  }
096}
