016package org.gbif.common.parsers;
017
018import org.gbif.api.vocabulary.TypeStatus;
019
020import java.io.BufferedReader;
021import java.io.IOException;
022import java.io.InputStreamReader;
023import java.net.URL;
024
025import org.junit.jupiter.api.Test;
026
027import static org.junit.jupiter.api.Assertions.assertNotNull;
028import static org.junit.jupiter.api.Assertions.assertTrue;
029
030public class TypeStatusParserTest extends ParserTestBase<TypeStatus> {
031
032  public TypeStatusParserTest() {
033    super(TypeStatusParser.getInstance());
034  }
035
036  /**
037   * This ensures that ALL enum values are at least parsable by the name they
038   * are created with.
039   */
040  @Test
041  public void testCompleteness() {
042    for (TypeStatus t : TypeStatus.values()) {
043      assertParseSuccess(t, t.name());
044    }
045  }
046
047  @Test
048  public void testFailures() {
049    assertParseFailure(null);
050    assertParseFailure("");
051    assertParseFailure("Tim");
052  }
053
054
055  @Test
056  public void testParse() {
057    // run a few basic tests to check it bootstraps and appears to work
058    assertParseSuccess(TypeStatus.ALLOTYPE, "allotype");
059    assertParseSuccess(TypeStatus.HOLOTYPE, "Holotype of Abies alba");
060    assertParseSuccess(TypeStatus.HOLOTYPE, "Holotype for Abies alba");
061    assertParseSuccess(TypeStatus.PARATYPE, "paratype(s)");
062    assertParseSuccess(TypeStatus.PARATYPE, "Par√°tipo");
063  }
064
065  @Test
066  public void testFileCoverage() throws IOException {
067    // parses all values in our test file (generated from real occurrence data) and verifies we never get worse at parsing
068    URL fileUrl = this.getClass().getClassLoader().getResource("parse/typestatus/type_status.txt");
069    assertNotNull(fileUrl);
070
071    try (BufferedReader in = new BufferedReader(new InputStreamReader(fileUrl.openStream()))) {
072      readLines(in, this);
073    }
074
075    BatchParseResult result = getResult();
076    System.out.printf("%s out of %s lines failed to parse%n", result.failed, result.total);
077    assertTrue(result.failed > 0);
078    assertTrue(result.failed <= 26813);
079  }
080}
