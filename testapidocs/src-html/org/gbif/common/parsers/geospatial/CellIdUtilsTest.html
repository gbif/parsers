038  @Test
039  public void testCreateCellId() {
040    try {
041      assertEquals(0, CellIdUtils.toCellId((double) -90, (double) -180));
042      assertEquals(0, CellIdUtils.toCellId((double) -90, -179.9));
043      assertEquals(359, CellIdUtils.toCellId(-89.1, 179.2));
044      assertEquals(360, CellIdUtils.toCellId(-88.5, (double) -180));
045      assertEquals(6647, CellIdUtils.toCellId(-71.31, -12.4233));
046    } catch (UnableToGenerateCellIdException e) {
047      fail(e.toString());
048    }
049    try {
050      CellIdUtils.toCellId((double) -100, null);
051      fail("-100, null should throw exception");
052    } catch (UnableToGenerateCellIdException ignored) {
053    }
054  }

310   * Test a conversion from and to ids
311   */
312  @Test
313  public void testLL2Id2LL() {
314    try {
315      List<Integer> cells = new ArrayList<>(CellIdUtils.getCellsEnclosedBy(-30, -20, 110, 130));
316      LatLngBoundingBox bb = CellIdUtils.getBoundingBoxForCells(cells);
317      assertEquals(110, (int) bb.getMinLong());
318      assertEquals(-30, (int) bb.getMinLat());
319      assertEquals(130, (int) bb.getMaxLong());
320      assertEquals(-20, (int) bb.getMaxLat());
321    } catch (UnableToGenerateCellIdException e) {
322      fail("Incorrectly thrown - the lat longs are valid: " + e.getMessage());
