3   import com.google.common.base.Strings;
4   import com.google.common.collect.ImmutableMap;
5   import org.gbif.api.vocabulary.NomenclaturalStatus;
6   import org.gbif.common.parsers.core.EnumParser;
7   import org.gbif.common.parsers.core.ParseResult;
8   
9   import java.io.InputStream;
10  import java.util.Map;
11  import java.util.regex.Pattern;
12  
13  /**
14   * Singleton implementation of the dictionary that uses the file /dictionaries/parse/nomStatus.txt.
15   */
16  public class NomStatusParser extends EnumParser<NomenclaturalStatus> {
17      private Pattern CLEAN_PREFIX = Pattern.compile("\\s*\\.?\\s*");
18      private Map<String, NomenclaturalStatus> PREFIXES = ImmutableMap.<String, NomenclaturalStatus>builder()
19          .put("nom illeg", NomenclaturalStatus.ILLEGITIMATE)
20          .put("nom inval", NomenclaturalStatus.INVALID)
21          .put("comb nov", NomenclaturalStatus.NEW_COMBINATION)
22          .put("nom nov", NomenclaturalStatus.REPLACEMENT)
23          .put("nom nud", NomenclaturalStatus.NUDUM)
24          .put("nom rej", NomenclaturalStatus.REJECTED)
25          .put("unavailable", NomenclaturalStatus.INVALID)
26          .build();
27    private static NomStatusParser singletonObject = null;
28  
29    private NomStatusParser(InputStream... file) {
30      super(NomenclaturalStatus.class, false, file);
31      // also make sure we have all enum knowledge mapped
32      for (NomenclaturalStatus ns : NomenclaturalStatus.values()) {
33        add(ns.getLatinLabel(), ns);
34        add(ns.getAbbreviatedLabel(), ns);
35      }
36    }
37  
38      @Override
39      public ParseResult<NomenclaturalStatus> parse(String input) {
40          ParseResult<NomenclaturalStatus> result = super.parse(input);
41          if (!result.isSuccessful() && !Strings.isNullOrEmpty(input)) {
42              String normed = CLEAN_PREFIX.matcher(input).replaceFirst(" ").trim().toLowerCase();
43              if (!Strings.isNullOrEmpty(normed)) {
44                  // try generic parsing of status prefixes only
45                  for (Map.Entry<String, NomenclaturalStatus> entry : PREFIXES.entrySet()) {
46                      if (normed.startsWith(entry.getKey())) {
47                          return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, entry.getValue());
48                      }
49                  }
50              }
51          }
52          return result;
53      }
54  
55      public static NomStatusParser getInstance()
56      throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
57      synchronized (NomStatusParser.class) {
58        if (singletonObject == null) {
59          singletonObject = new NomStatusParser(NomStatusParser.class.getResourceAsStream("/dictionaries/parse/nomStatus.tsv"));
60        }
61      }
62      return singletonObject;
63    }
