001package org.gbif.common.parsers.geospatial;
002
003import com.google.common.base.Objects;
004
005/**
006 * Simple container class for an interpreted coordinate.
007 */
008public class LatLng {
009
010  private final Double lat;
011  private final Double lng;
012
013  public LatLng(double lat, double lng) {
014    this.lat = lat;
015    this.lng = lng;
016  }
017
018  public LatLng() {
019    lat = null;
020    lng = null;
021  }
022
023  public Double getLat() {
024    return lat;
025  }
026
027  public Double getLng() {
028    return lng;
029  }
030
031  @Override
032  public String toString() {
033    return Objects.toStringHelper(this)
034      .add("lat", lat)
035      .add("lng", lng)
036      .toString();
037  }
038
039  @Override
040  public int hashCode() {
041    return Objects.hashCode(lat, lng);
042  }
043
044  @Override
045  public boolean equals(Object object) {
046    if (this == object) {
047      return true;
048    }
049    if (!(object instanceof LatLng)) {
050      return false;
051    }
052
053    LatLng that = (LatLng) object;
054    return Objects.equal(this.lat, that.lat)
055           && Objects.equal(this.lng, that.lng);
056  }
057}
