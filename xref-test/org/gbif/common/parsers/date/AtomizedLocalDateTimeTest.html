15  public class AtomizedLocalDateTimeTest {
16  
17    private static final int YEAR = 1978;
18    private static final Month MONTH_ENUM = Month.DECEMBER;
19    private static final int MONTH = 12;
20    private static final int DAY = 21;
21  
22    private static final int HOUR = 2;
23    private static final int MINUTE = 3;
24    private static final int SECOND = 4;
25    private static final int MILLISECOND = 5;
26    // not a mistake, LocalDateTime.of take nanoseconds and not milliseconds
27    private static final int NANOSECOND = 5000000;
28  
29    @Test
30    public void testAtomizedLocalDateTime(){
31      AtomizedLocalDateTime dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDateTime.of(YEAR, MONTH_ENUM, DAY, HOUR, MINUTE, SECOND, NANOSECOND));
32      assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, HOUR, MINUTE, SECOND, MILLISECOND);
33  
34      dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDateTime.of(YEAR, MONTH_ENUM, DAY, HOUR, MINUTE));
35      assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, HOUR, MINUTE, 0, 0);
36      assertEquals(7, dateTime.getResolution());
37  
38      // this will probably not be supported ------
39      dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalTime.of(HOUR, MINUTE));
40      assertAtomizedLocalDateTime(dateTime, null, null, null, HOUR, MINUTE, 0, 0);
41      assertEquals(4, dateTime.getResolution());
42  
43      dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalTime.of(HOUR, MINUTE));
44      assertAtomizedLocalDateTime(dateTime, null, null, null, HOUR, MINUTE, 0, 0);
45      assertEquals(4, dateTime.getResolution());
46      // end of this will probably not be supported ------
47  
48      // ensure it works with a LocalDate alone (without time component)
49      dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
50      assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, null, null, null, null);
51    }
52  
53    @Test
54    public void testAtomizedLocalDateEquals(){
55      AtomizedLocalDateTime date1 = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
56      AtomizedLocalDateTime date2 = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
57      assertEquals(date1, date2);
58    }
59  
60    private static void assertAtomizedLocalDateTime(AtomizedLocalDateTime dateTime, Integer year, Integer month,
61                                                    Integer day, Integer hour, Integer minute, Integer second,
62                                                    Integer millisecond){
63      assertEquals(year, dateTime.getYear());
64      assertEquals(month, dateTime.getMonth());
65      assertEquals(day, dateTime.getDay());
66      assertEquals(hour, dateTime.getHour());
67      assertEquals(minute, dateTime.getMinute());
68      assertEquals(second, dateTime.getSecond());
69      assertEquals(millisecond, dateTime.getMillisecond());
70    }
71  }
