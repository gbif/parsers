14  package org.gbif.common.parsers;
15  
16  import java.net.URI;
17  import java.util.ArrayList;
18  import java.util.Arrays;
19  import java.util.List;
20  import java.util.regex.Pattern;
21  import java.util.stream.Collectors;
22  
23  import org.apache.commons.lang3.StringUtils;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  /**
28   * Greedy URL parser assuming HTTP URIs in case no schema was given.
29   * Modified version of the registry-metadata GreedyUriConverter.
30   */
31  public class UrlParser {
32    private static final Logger LOG = LoggerFactory.getLogger(UrlParser.class);
33    private static final String[] MULTI_VALUE_DELIMITERS = {"|#DELIMITER#|", "|", ",", ";"};
34    private static final String HTTP_SCHEME = "http://";
35  
36    // Pattern for things that are probably domains followed by a slash, without a protocol.
37    // Doesn't match IDNs etc, but this is just for people who forgot the http:// anyway.
38    // The longest TLD currently in existence is 24 characters long, but can be up to 63 according to specs.
39    private static final Pattern DOMAIN_ISH = Pattern.compile("^[A-Za-z0-9.-]{1,60}\\.[A-Za-z]{2,10}(?:/.*)?");
40  
41    private UrlParser() {
42    }
43  
44    /**
45     * Convert a String into a java.net.URI.
46     * In case its missing the protocol prefix, it is prefixed with the default protocol.
47     *
48     * @param value The input value to be converted
49     *
50     * @return The converted value, or null if not parsable or exception occurred
51     */
52    public static URI parse(String value) {
53      value = StringUtils.trimToEmpty(value);
54      if (StringUtils.isEmpty(value)) {
55        return null;
56      }
57  
58      URI uri = null;
59      try {
60        uri = URI.create(value);
61        if (!uri.isAbsolute() && DOMAIN_ISH.matcher(value).matches()) {
62          // make into an HTTP address
63          try {
64            uri = URI.create(HTTP_SCHEME + value);
65          } catch (IllegalArgumentException e) {
66            // keep the previous scheme-less result
67          }
68        }
69  
70        // verify that we have a domain
71        if (StringUtils.isEmpty(uri.getHost())) {
72          return null;
73        }
74  
75      } catch (IllegalArgumentException ignored) {
76      }
77  
78      return uri;
79    }
80  
81  
82    /**
83     * Parses a single string with null, one or many URIs concatenated together as found in dwc:associatedMedia.
84     */
85    public static List<URI> parseUriList(String uris) {
86      List<URI> result = new ArrayList<>();
87  
88      if (StringUtils.isNotEmpty(uris)) {
89        // first try to use the entire string
90        URI uri = UrlParser.parse(uris);
91        if (uri != null) {
92          result.add(uri);
93  
94        } else {
95          // try common delimiters
96          int maxValidUrls = 0;
97          for (String delimiter : MULTI_VALUE_DELIMITERS) {
98            List<String> urls = Arrays.stream(StringUtils.splitByWholeSeparator(uris, delimiter))
99                .map(org.gbif.utils.text.StringUtils::trim)
100               .filter(StringUtils::isNotEmpty)
101               .collect(Collectors.toList());
102 
103           // avoid parsing if we haven't actually split anything
104           if (urls.size() > 1) {
105             List<URI> tmp = new ArrayList<>();
106             for (String url : urls) {
107               uri = UrlParser.parse(url);
108               if (uri != null) {
109                 tmp.add(uri);
110               }
111             }
112             if (tmp.size() > maxValidUrls) {
113               result = tmp;
114               maxValidUrls = tmp.size();
115             } else if (maxValidUrls > 0 && tmp.size() == maxValidUrls) {
116               LOG.info("Unclear what delimiter is being used for concatenated URIs = {}", uris);
117             }
118           }
119         }
120       }
121     }
122     return result;
123   }
