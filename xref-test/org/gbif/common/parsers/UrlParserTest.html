10    @Test
11    public void testParse() throws Exception {
12      assertNull(UrlParser.parse(null));
13      assertNull(UrlParser.parse(""));
14      assertNull(UrlParser.parse(" "));
15      assertNull(UrlParser.parse("-"));
16      assertNull(UrlParser.parse("tim.png"));
17      assertNull(UrlParser.parse("images/logo.gif"));
18  
19      assertEquals("http://www.gbif.org/logo.png", UrlParser.parse(" http://www.gbif.org/logo.png").toString());
20      assertEquals("http://www.gbif.org/logo.png", UrlParser.parse("www.gbif.org/logo.png").toString());
21      assertEquals("https://www.gbif.org/logo.png", UrlParser.parse(" https://www.gbif.org/logo.png").toString());
22      assertEquals("ftp://www.gbif.org/logo.png", UrlParser.parse(" ftp://www.gbif.org/logo.png").toString());
23      assertEquals("http://www.gbif.org/image?id=12&format=gif,jpg",
24        UrlParser.parse("http://www.gbif.org/image?id=12&format=gif,jpg").toString());
25    }

27    @Test
28    public void testParseUriList() throws Exception {
29      assertEquals(0, UrlParser.parseUriList(null).size());
30      assertEquals(0, UrlParser.parseUriList("").size());
31      assertEquals(0, UrlParser.parseUriList(" ").size());
32      assertEquals(0, UrlParser.parseUriList("-").size());
33  
34      assertEquals(1, UrlParser.parseUriList("http://gbif.org/logo.png").size());
35      assertEquals(1, UrlParser.parseUriList(" http://gbif.org/logo.png").size());
36      assertEquals(1, UrlParser.parseUriList("www.gbif.org/logo.png").size());
37      assertEquals(1, UrlParser.parseUriList("www.gbif.org/image?id=12").size());
38      assertEquals(1, UrlParser.parseUriList("http://www.gbif.org/image?id=12").size());
39      assertEquals(1, UrlParser.parseUriList("http://www.gbif.org/image?id=12&format=gif,jpg").size());
40  
41      assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png, http://gbif.org/logo2.png").size());
42      assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png; http://gbif.org/logo2.png").size());
43      assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png | http://gbif.org/logo2.png").size());
44      assertEquals(2, UrlParser.parseUriList("http://gbif.org/logo.png |#DELIMITER#| http://gbif.org/logo2.png").size());
45  
46      assertEquals(3, UrlParser.parseUriList("http://gbif.org/logo.png, http://gbif.org/logo2.png, http://gbif.org/logo3.png").size());
47    }
