001package org.gbif.common.parsers.core;
002
003import org.gbif.api.vocabulary.OccurrenceIssue;
004
005import java.util.Collection;
006import java.util.EnumSet;
007import java.util.Set;
008
009import com.google.common.base.Preconditions;
010import com.google.common.base.Predicates;
011import com.google.common.collect.Iterables;
012import com.google.common.collect.Lists;
013
014/**
015 *
016 */
017public class OccurrenceParseResult<T> extends ParseResult<T> {
018  private final Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
019
020  public OccurrenceParseResult(STATUS status, CONFIDENCE confidence, T payload, Throwable error, Collection<OccurrenceIssue> issues) {
021    super(status, confidence, payload, error);
022    if (issues != null) {
023      // add non nulls only
024      Iterables.addAll(this.issues, Iterables.filter(issues, Predicates.notNull()));
025    }
026  }
027
028  public OccurrenceParseResult(STATUS status, CONFIDENCE confidence, T payload, Throwable error) {
029    super(status, confidence, payload, error);
030  }
031
032  public OccurrenceParseResult(ParseResult<T> result) {
033    super(result.getStatus(), result.getConfidence(), result.getPayload(), result.getError());
034  }
035
036  /**
037   * @param <T>       The generic type of the payload
038   * @param confidence The confidence in the result
039   * @param payload    The payload of the parse result
040   *
041   * @return The new ParseResult which has no error and status of SUCCESS
042   */
043  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload) {
044    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null);
045  }
046
047  /**
048   * @return A new parse response with only the status set to FAIL
049   */
050  public static <T> OccurrenceParseResult<T> fail() {
051    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null);
052  }
053
054  /**
055   * @return A new parse response configured to indicate an error
056   */
057  public static <T> OccurrenceParseResult<T> error() {
058    return new OccurrenceParseResult<T>(STATUS.ERROR, null, null, null);
059  }
060
061  /**
062   * @param cause The cause of the error
063   *
064   * @return A new parse response configured with error and the cause
065   */
066  public static <T> OccurrenceParseResult<T> error(Throwable cause) {
067    return new OccurrenceParseResult<T>(STATUS.ERROR, null, null, cause);
068  }
069
070  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload, OccurrenceIssue issue) {
071    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null, Lists.newArrayList(issue));
072  }
073
074  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload, OccurrenceIssue ... issues) {
075    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null, Lists.newArrayList(issues));
076  }
077
078  public static <T> OccurrenceParseResult<T> success(CONFIDENCE confidence, T payload, Collection<OccurrenceIssue> issues) {
079    return new OccurrenceParseResult<T>(STATUS.SUCCESS, confidence, payload, null, issues);
080  }
081
082  public static <T> OccurrenceParseResult<T> fail(OccurrenceIssue issue) {
083    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null, Lists.newArrayList(issue));
084  }
085
086  public static <T> OccurrenceParseResult<T> fail(OccurrenceIssue ... issues) {
087    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null, Lists.newArrayList(issues));
088  }
089
090  public static <T> OccurrenceParseResult<T> fail(Collection<OccurrenceIssue> issues) {
091    return new OccurrenceParseResult<T>(STATUS.FAIL, null, null, null, issues);
092  }
093
094  /**
095   * This creates a ParseResult indicating a parse failure but it also has a payload. Depending on the context this
096   * may provide additional information about the failure.
097   *
098   * @param payload the payload of the parse result
099   * @param <T>    the generic type of the payload
100   *
101   * @return the new parse response which has a status of FAIL and an additional payload.
102   */
103  public static <T> OccurrenceParseResult<T> fail(T payload, Collection<OccurrenceIssue> issues) {
104    return new OccurrenceParseResult<T>(STATUS.FAIL, null, payload, null, issues);
105  }
106
107  public static <T> OccurrenceParseResult<T> fail(T payload, OccurrenceIssue ... issues) {
108    return new OccurrenceParseResult<T>(STATUS.FAIL, null, payload, null, Lists.newArrayList(issues));
109  }
110
111  /**
112   * This creates a ParseResult indicating a parse failure but it also has a payload. Depending on the context this
113   * may provide additional information about the failure.
114   *
115   * @param payload the payload of the parse result
116   * @param <T>    the generic type of the payload
117   *
118   * @return the new parse response which has a status of FAIL and an additional payload.
119   */
120  public static <T> OccurrenceParseResult<T> fail(T payload, OccurrenceIssue issue) {
121    return new OccurrenceParseResult<T>(STATUS.FAIL, null, payload, null, Lists.newArrayList(issue));
122  }
123
124
125  public Set<OccurrenceIssue> getIssues() {
126    return issues;
127  }
128
129  public void addIssue(OccurrenceIssue issue) {
130    Preconditions.checkNotNull(issue);
131    issues.add(issue);
132  }
133
134}
