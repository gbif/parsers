001package org.gbif.common.parsers.date;
002
003import java.time.temporal.TemporalAccessor;
004import java.util.ArrayList;
005import java.util.List;
006import javax.annotation.Nullable;
007import javax.validation.constraints.NotNull;
008
009import com.google.common.base.Preconditions;
010import com.google.common.collect.ImmutableList;
011import com.google.common.collect.Lists;
012
013/**
014 * Supports multiple {@link DateTimeParser} that are considered ambiguous. Two {@link DateTimeParser} are considered
015 * ambiguous when they can potentially produce 2 different {@link TemporalAccessor}.
016 * e.g. "dd/MM/yyyy" and "MM/dd/yyyy"
017 *
018 * <p>This class will try all the parsers and keep the all the successful results.
019 *
020 * <p>This class is thread-safe once an instance is created.
021 */
022public class DateTimeMultiParser {
023
024  private final DateTimeParser preferred;
025  private final List<DateTimeParser> otherParsers;
026  private final List<DateTimeParser> allParsers;
027
028  /**
029   * Create a new instance of {@link DateTimeMultiParser}.
030   * @param parsers requires more than 1 element in list
031   */
032  DateTimeMultiParser(@NotNull List<DateTimeParser> parsers){
033    this(null, parsers);
034  }
035
036  /**
037   *
038   * Create a new instance of {@link DateTimeMultiParser}.
039   * At least 2 {@link DateTimeParser} must be provided see details on parameters.
040   *
041   * @param preferred the preferred {@link DateTimeParser} or null
042   * @param otherParsers list of {@link DateTimeParser} containing more than 1 element if no
043   *                     preferred {@link DateTimeParser} is provided. Otherwise, the list must contain at least 1 element.
044   */
045  DateTimeMultiParser(@Nullable DateTimeParser preferred, @NotNull List<DateTimeParser> otherParsers) {
046
047    Preconditions.checkNotNull(otherParsers, "otherParsers list can not be null");
048    Preconditions.checkArgument(otherParsers.size() > 0, "otherParsers must contain at least 1 element");
049
050    if (preferred == null) {
051      Preconditions.checkArgument(otherParsers.size() > 1, "If no preferred DateTimeParser is provided, " +
052              "the otherParsers list must contain more than 1 element");
053    }
054
055    this.preferred = preferred;
056    this.otherParsers = Lists.newArrayList(otherParsers);
057
058    ImmutableList.Builder immutableListBuilder = new ImmutableList.Builder<DateTimeParser>();
059    if (preferred != null) {
060      immutableListBuilder.add(preferred);
061    }
062    immutableListBuilder.addAll(otherParsers);
063
064    this.allParsers = immutableListBuilder.build();
065  }
066
067  /**
068   * Get the list of all parsers: the preferred (if specified in the constructor) + otherParsers.
069   *
070   * @return never null
071   */
072  public List<DateTimeParser> getAllParsers(){
073    return allParsers;
074  }
075
076  /**
077   * Try to parse the input using all the parsers specified in the constructor.
078   *
079   * @param input
080   * @return {@link MultipleParseResult} instance, never null.
081   */
082  public MultipleParseResult parse(String input) {
083
084    int numberParsed = 0;
085    TemporalAccessor lastParsed = null;
086    TemporalAccessor preferredResult = null;
087    List<String> usedFormats = new ArrayList<>();
088
089    // lazily initialized assuming it should not be used most of the time
090    List<TemporalAccessor> otherResults = null;
091    for (DateTimeParser currParser : otherParsers) {
092      lastParsed = currParser.parse(input);
093      if (lastParsed != null) {
094        numberParsed++;
095        if (otherResults == null) {
096          otherResults = Lists.newArrayList();
097        }
098        otherResults.add(lastParsed);
099        usedFormats.add(currParser.getOrdering().name());
100      }
101    }
102
103    // try the preferred DateTimeParser
104    if (this.preferred != null) {
105      lastParsed = this.preferred.parse(input);
106      if (lastParsed != null) {
107        numberParsed++;
108        preferredResult = lastParsed;
109      }
110    }
111
112    return new MultipleParseResult(numberParsed, usedFormats, preferredResult, otherResults);
113  }
114
115  /**
116   * Nested class representing the result of a multi-parse.
117   */
118  public static class MultipleParseResult {
119    private int numberParsed;
120    private TemporalAccessor preferredResult;
121    private List<TemporalAccessor> otherResults;
122    public List<String> formats;
123
124    public MultipleParseResult(int numberParsed, List<String> formats, TemporalAccessor preferredResult, List<TemporalAccessor> otherResults) {
125      this.numberParsed = numberParsed;
126      this.formats = formats;
127      this.preferredResult = preferredResult;
128      this.otherResults = otherResults;
129    }
130
131    public int getNumberParsed() {
132      return numberParsed;
133    }
134
135    public List<String> getFormats() {
136      return formats;
137    }
138
139    public TemporalAccessor getPreferredResult() {
140      return preferredResult;
141    }
142
143    public List<TemporalAccessor> getOtherResults() {
144      return otherResults;
145    }
146
147    /**
148     * Return the preferredResult if available otherwise the first element of otherResults.
149     * If otherResults is empty, null is returned.
150     */
151    public TemporalAccessor getResult() {
152      if (preferredResult != null) {
153        return preferredResult;
154      }
155
156      if (otherResults != null && otherResults.size() > 0) {
157        return otherResults.get(0);
158      }
159      return null;
160    }
161  }
162}
