16  public class DatePartsNormalizerTest {

18    private static DatePartsNormalizer NORMALIZER = DatePartsNormalizer.newInstance();

20    /**
21     * This happens when integer are exported as float/double by a data publisher.
22     */
23    @Test
24    public void testFloatNormalization(){
25      DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975.0", "2.0", "1.0");
26      assertEquals(new Integer(1), result.getDay());
27      assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
28      assertEquals(new Integer(1975), result.getYear());
29    }
30  
31    @Test
32    public void testNormalizationWithSpaces(){
33      DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975 ", " 2 ", " 1");
34      assertEquals(new Integer(1), result.getDay());
35      assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
36      assertEquals(new Integer(1975), result.getYear());
37    }

39    @Test
40    public void testMonthNameNormalization(){
41      assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("January").intValue());
42      assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JANUARY").intValue());
43      assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("Jan").intValue());
44      assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JAN.").intValue());
45      assertEquals(Month.DECEMBER.getValue(), NORMALIZER.monthNameToNumerical("DEC").intValue());
46      assertEquals(Month.APRIL.getValue(), NORMALIZER.monthNameToNumerical("April").intValue());
47      assertEquals(Month.SEPTEMBER.getValue(), NORMALIZER.monthNameToNumerical("Sept").intValue());
48      assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("June").intValue());
49      assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("Jun").intValue());
50      assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("JUNE").intValue());
51      assertEquals(Month.NOVEMBER.getValue(), NORMALIZER.monthNameToNumerical("November").intValue());
52    }

54    @Test
55    public void testDiscardedDateParts(){
56  
57      DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("a", "5", "3");
58      assertTrue(result.yDiscarded());
59      assertNull(result.getYear());
60      assertNotNull(result.getMonth());
61  
62      result = NORMALIZER.normalize("2000", "a", "3");
63      assertTrue(result.mDiscarded());
64      assertNull(result.getMonth());
65      assertNotNull(result.getYear());
66  
67      result = NORMALIZER.normalize("2000", "5", "a");
68      assertTrue(result.dDiscarded());
69      assertNull(result.getDay());
70      assertNotNull(result.getYear());
71  
72      // empty String, null and \\N should NOT be considered as "discarded"
73      // they simply represent "not provided"
74      result = NORMALIZER.normalize("2000", "5", "");
75      assertFalse(result.dDiscarded());
76      assertNull(result.getDay());
77  
78      result = NORMALIZER.normalize("2000", "5", null);
79      assertFalse(result.dDiscarded());
80      assertNull(result.getDay());
81  
82      result = NORMALIZER.normalize("2000", "5", "\\N");
83      assertFalse(result.dDiscarded());
84      assertNull(result.getDay());
85    }
