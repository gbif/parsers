18  import java.io.Serializable;
19  import java.time.DateTimeException;
20  import java.time.LocalDate;
21  import java.time.temporal.TemporalAccessor;
22  import java.util.regex.Matcher;
23  import java.util.regex.Pattern;
24  
25  import javax.annotation.Nullable;
26  
27  import org.apache.commons.lang3.StringUtils;
28  
29  /**
30   * Main class to parse a date represented as a single String or as date parts into a {@link TemporalAccessor}.
31   * If the String contains letters, the {@link TextualMonthDateTokenizer} and the {@link DatePartsNormalizer} will be used.
32   * Otherwise, a {@link TemporalParser} configured for numerical dates will be used.
33   *
34   * This class is basically a decorator on top of default NumericalDateParser to handle months written in text.
35   */
36  class TextDateParser implements TemporalParser, Serializable {
37    /*
38     * ISO format intervals, which are datetime/datetime or datetime/period.
39     * Match 2013/2014, 2013/P1Y, 2013-02/03.
40     * Fail 2013-12, 2013/02, 2018/10-23.
41     */
42    private static final Pattern ISO_YEAR_RANGE_PATTERN = Pattern.compile("^([12][0-9]{3})/([P12]\\d[^/]+)$");
43    private static final Pattern ISO_DATE_RANGE_PATTERN = Pattern.compile("^([12][0-9]{3}[^/]+)/([^/]{2,})$");
44  
45    // This regex is not complete and will NOT handle date when the time zone is provided as text GMT
46    private static final Pattern NUMERICAL_DATE_PATTERN =  Pattern.compile("[^a-zA-VX-Z]+[\\dT\\d]?[^a-zA-Z]+[Z]?$");
47    private static final TextualMonthDateTokenizer TEXT_MONTH_TOKENIZER = TextualMonthDateTokenizer.newInstance();
48  
49    private static final TemporalParser NUMERICAL_DATE_PARSER = ThreeTenNumericalDateParser.newInstance();
50    private static final DatePartsNormalizer DATE_PARTS_NORMALIZER = DatePartsNormalizer.newInstance();
51  
52    /**
53     * Parse a date using the default (ISO) formats, or other unambiguous parsing.
54     */
55    @Override
56    public ParseResult<TemporalAccessor> parse(String input) {
57      if (StringUtils.isBlank(input)) {
58        return ParseResult.fail();
59      }
60  
61      Matcher matcher = ISO_YEAR_RANGE_PATTERN.matcher(input);
62      if (!matcher.matches()) {
63        matcher = ISO_DATE_RANGE_PATTERN.matcher(input);
64      }
65  
66      if (matcher.matches()) {
67        String from = matcher.group(1);
68        // String to = matcher.group(2);
69        return NUMERICAL_DATE_PARSER.parse(from, DateComponentOrdering.ISO_ETC);
70      }
71  
72      // Check if the input text contains only punctuations and numbers
73      // Also accept the T marker (e.g. 1978-12-21T02:12) from the ISO format
74      // and the W week marker (e.g. 2018-W43).
75      // We could also simply try to parse it but it is probably not optimal
76      if (NUMERICAL_DATE_PATTERN.matcher(input).matches()) {
77        return NUMERICAL_DATE_PARSER.parse(input, DateComponentOrdering.ISO_ETC);
78      }
79  
80      TextualMonthDateTokenizer.DateTokens dt = TEXT_MONTH_TOKENIZER.tokenize(input);
81      // for now we only handle cases where we can find year, month, day with confidence.
82      if (!dt.containsDiscardedTokens() && dt.size() == 3) {
83        DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DATE_PARTS_NORMALIZER.normalize(
84          dt.getToken(TextualMonthDateTokenizer.TokenType.INT_4).getToken(),
85          dt.getToken(TextualMonthDateTokenizer.TokenType.TEXT).getToken(),
86          dt.getToken(TextualMonthDateTokenizer.TokenType.INT_2).getToken());
87  
88        // no handling for partial dates with textual month for now
89        if (normalizedYearMonthDay.getYear() != null &&
90          normalizedYearMonthDay.getMonth() != null &&
91          normalizedYearMonthDay.getDay() != null) {
92          try {
93            return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE,
94              LocalDate.of(normalizedYearMonthDay.getYear(),
95                normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay()));
96          } catch (DateTimeException ignore) {
97            // simply ignore bad dates
98          }
99        }
100     }
101     return ParseResult.fail();
102   }
103 
104   /**
105    * Parse a date restricted to the provided date component ordering.
106    *
107    * For now this is directly delegated the NumericalDateParser.
108    */
109   @Override
110   public ParseResult<TemporalAccessor> parse(String input, @Nullable DateComponentOrdering ordering) {
111     return NUMERICAL_DATE_PARSER.parse(input, ordering);
112   }
113 
114   /**
115    * Parse a date, and if given an ambiguous date, like 2/3/2000, use the orderings in turn to try
116    * and parse the date.
117    * <p>
118    * NOTE, this behaviour <strong>differs</strong> from <code>parse(String input, DateComponentOrdering ordering)</code>
119    */
120   @Override
121   public ParseResult<TemporalAccessor> parse(String input, DateComponentOrdering[] orderings) {
122     ParseResult <TemporalAccessor> result = this.parse(input);
123     if (result.isSuccessful()) {
124       return result;
125     } else {
126       return NUMERICAL_DATE_PARSER.parse(input, orderings);
127     }
128   }
129 
130   /**
131    * Parse date parts into a TemporalAccessor.
132    * The {@link DatePartsNormalizer} will be applied on raw data.
133    */
134   @Override
135   public ParseResult<TemporalAccessor> parse(String year, String month, String day) {
136     DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DATE_PARTS_NORMALIZER.normalize(
137             year, month, day);
138 
139     // TODO when we can get the day but not the month in NormalizedYearMonthDay the parsing fails
140     // we could actually keep the year by calling ThreeTenNumericalDateParser.parse(normalizedYearMonthDay.getYear(),
141     // null, null); the CONFIDENCE would be set at PROBABLE.
142 
143     ParseResult<TemporalAccessor> parseResult = NUMERICAL_DATE_PARSER.parse(normalizedYearMonthDay.getYear(),
144             normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay());
145 
146     //If we got a successful parsing BUT a part of the date was discarded we reduce confidence.
147     if(parseResult.isSuccessful() && normalizedYearMonthDay.containsDiscardedPart()){
148       return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, parseResult.getPayload());
149     }
150 
151     return parseResult;
152   }
153 
154   @Override
155   public ParseResult<TemporalAccessor> parse(@Nullable Integer year, @Nullable Integer month, @Nullable Integer day) {
156     return NUMERICAL_DATE_PARSER.parse(year, month, day);
157   }
158 
159   /**
160    * Parse date parts into a TemporalAccessor.
161    * The {@link DatePartsNormalizer} will be applied on raw data.
162    */
163   @Override
164   public ParseResult<TemporalAccessor> parse(String year, String dayOfYear) {
165     DatePartsNormalizer.NormalizedYearDayOfYear normalizedYearDayOfYear = DATE_PARTS_NORMALIZER.normalize(
166       year, dayOfYear);
167 
168     ParseResult<TemporalAccessor> parseResult = NUMERICAL_DATE_PARSER.parse(normalizedYearDayOfYear.getYear(),
169       normalizedYearDayOfYear.getDayOfYear());
170 
171     // If we got a successful parsing BUT a part of the date was discarded we reduce confidence.
172     if (parseResult.isSuccessful() && normalizedYearDayOfYear.containsDiscardedPart()) {
173       return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, parseResult.getPayload());
174     }
175 
176     return parseResult;
177   }
178 
179   @Override
180   public ParseResult<TemporalAccessor> parse(@Nullable Integer year, @Nullable Integer dayOfYear) {
181     return NUMERICAL_DATE_PARSER.parse(year, dayOfYear);
182   }
