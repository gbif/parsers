6   public class CitesAppendixParser extends EnumParser<CitesAppendix> {
7   
8     private static CitesAppendixParser singletonObject = null;
9   
10    private CitesAppendixParser() {
11      super(CitesAppendix.class, true);
12      // also make sure we have all official iso countries mapped
13      for (CitesAppendix c : CitesAppendix.values()) {
14        add(c.name(), c);
15      }
16      // use dict file last
17      init(CitesAppendixParser.class.getResourceAsStream("/dictionaries/parse/cites.tsv"));
18    }
19  
20    public static CitesAppendixParser getInstance() {
21      synchronized (CitesAppendixParser.class) {
22        if (singletonObject == null) {
23          singletonObject = new CitesAppendixParser();
24        }
25      }
26      return singletonObject;
27    }

29  }
