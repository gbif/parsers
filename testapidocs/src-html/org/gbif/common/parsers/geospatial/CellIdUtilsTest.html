297   * Test a conversion from and to ids
298   */
299  @Test
300  public void testLL2Id2LL() {
301    try {
302      List<Integer> cells = new LinkedList<>(CellIdUtils.getCellsEnclosedBy(-30, -20, 110, 130));
303      LatLngBoundingBox bb = CellIdUtils.getBoundingBoxForCells(cells);
304      assertEquals(110, (int) bb.getMinLong());
305      assertEquals(-30, (int) bb.getMinLat());
306      assertEquals(130, (int) bb.getMaxLong());
307      assertEquals(-20, (int) bb.getMaxLat());
308    } catch (UnableToGenerateCellIdException e) {
309      fail("Incorrectly thrown - the lat longs are valid: " + e.getMessage());
310    }
