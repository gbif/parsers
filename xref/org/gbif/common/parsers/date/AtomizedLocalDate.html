3   import java.time.temporal.ChronoField;
4   import java.time.temporal.TemporalAccessor;
5   
6   import com.google.common.base.Objects;
7   import org.apache.commons.lang3.builder.HashCodeBuilder;
8   import org.apache.commons.lang3.builder.ToStringBuilder;
9   import org.apache.commons.lang3.builder.ToStringStyle;
10  
11  /**
12   * AtomizedLocalDate is a simple immutable class to hold local date data from a
13   * {@link TemporalAccessor}.
14   *
15   * Thread-Safe, immutable class.
16   *
17   */
18  public class AtomizedLocalDate {
19  
20    // Maximum resolution for a complete Local Date
21    public static final int COMPLETE_LOCAL_DATE_RESOLUTION = 3;
22  
23    private final Integer year;
24    private final Integer month;
25    private final Integer day;
26  
27    private final int resolution;
28  
29    private AtomizedLocalDate(Integer year, Integer month, Integer day){
30      this.year = year;
31      this.month = month;
32      this.day = day;
33  
34      int res = 0;
35      if(year != null){
36        res++;
37      }
38      if(month != null){
39        res++;
40      }
41      if(day != null){
42        res++;
43      }
44      resolution = res;
45    }
46  
47    /**
48     * Build a new instance of {@link AtomizedLocalDate} based on a {@link TemporalAccessor}.
49     * This is done by extracting the {@link ChronoField}.
50     * @param temporalAccessor
51     * @return
52     */
53    public static AtomizedLocalDate fromTemporalAccessor(TemporalAccessor temporalAccessor){
54      if(temporalAccessor == null){
55        return null;
56      }
57  
58      Integer y = null, m = null, d = null;
59      if(temporalAccessor.isSupported(ChronoField.YEAR)){
60        y = temporalAccessor.get(ChronoField.YEAR);
61      }
62  
63      if(temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)){
64        m = temporalAccessor.get(ChronoField.MONTH_OF_YEAR);
65      }
66  
67      if(temporalAccessor.isSupported(ChronoField.DAY_OF_MONTH)){
68        d = temporalAccessor.get(ChronoField.DAY_OF_MONTH);
69      }
70  
71      return new AtomizedLocalDate(y, m, d);
72    }
73  
74    public Integer getYear() {
75      return year;
76    }
77  
78    public Integer getMonth() {
79      return month;
80    }
81  
82    public Integer getDay() {
83      return day;
84    }
85  
86    public int getResolution(){
87      return resolution;
88    }
89  
90    /**
91     * Is the resolution of this local date matches the maximum.
92     * @return
93     */
94    public boolean isComplete() {
95      return COMPLETE_LOCAL_DATE_RESOLUTION == resolution;
96    }
97  
98    @Override
99    public int hashCode() {
100     return new HashCodeBuilder().append(year).append(month).append(day).append(resolution).toHashCode();
101   }
102 
103   @Override
104   public boolean equals(Object obj) {
105 
106     if (this == obj) {
107       return true;
108     }
109     if (!(obj instanceof AtomizedLocalDate)) {
110       return false;
111     }
112 
113     AtomizedLocalDate that = (AtomizedLocalDate) obj;
114     return Objects.equal(this.year, that.year)
115             && Objects.equal(this.month, that.month)
116             && Objects.equal(this.day, that.day)
117             && Objects.equal(this.resolution, that.resolution);
118   }
119 
120   @Override
121   public String toString() {
122     return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE)
123             .append("year", year)
124             .append("month", month)
125             .append("day", day)
126             .append("resolution", resolution)
127             .toString();
128   }
129 }
