15    @Test
16    public void testSuccess() {
17      // check with basic string payload
18      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")));
19      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getPayload());
20      assertEquals(String.class,
21        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getPayload().getClass());
22      assertEquals(ParseResult.CONFIDENCE.DEFINITE,
23        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getConfidence());
24      assertEquals(ParseResult.STATUS.SUCCESS,
25        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getStatus());
26      assertNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new String("Bingo")).getError());
27  
28      // check generics with Date payload
29      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()));
30      assertNotNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getPayload());
31      assertEquals(Date.class, ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getPayload().getClass());
32      assertEquals(ParseResult.CONFIDENCE.DEFINITE,
33        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getConfidence());
34      assertEquals(ParseResult.STATUS.SUCCESS,
35        ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getStatus());
36      assertNull(ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new Date()).getError());
37  
38    }
39  
40    @Test
41    public void testFail() {
42      assertNotNull(ParseResult.fail());
43      assertEquals(ParseResult.STATUS.FAIL, ParseResult.fail().getStatus());
44      assertNull(ParseResult.fail().getConfidence());
45      assertNull(ParseResult.fail().getError());
46      assertNull(ParseResult.fail().getPayload());
47    }

49    @Test
50    public void testUnknownError() {
51      assertNotNull(ParseResult.error());
52      assertEquals(ParseResult.STATUS.ERROR, ParseResult.error().getStatus());
53      assertNull(ParseResult.error().getConfidence());
54      assertNull(ParseResult.error().getError());
55      assertNull(ParseResult.error().getPayload());
56    }

58    @Test
59    public void testError() {
60      assertNotNull(ParseResult.error(new RuntimeException("Bingo")));
61      assertEquals(ParseResult.STATUS.ERROR, ParseResult.error(new RuntimeException("Bingo")).getStatus());
62      assertNull(ParseResult.error(new RuntimeException("Bingo")).getConfidence());
63      assertNotNull(ParseResult.error(new RuntimeException("Bingo")).getError());
64      assertEquals(RuntimeException.class, ParseResult.error(new RuntimeException("Bingo")).getError().getClass());
65      assertEquals("Bingo", ParseResult.error(new RuntimeException("Bingo")).getError().getMessage());
66      assertNull(ParseResult.error(new RuntimeException("Bingo")).getPayload());
67    }
