39  public class MediaParser {
40    private static final Logger LOG = LoggerFactory.getLogger(MediaParser.class);
41    private static final Tika TIKA = new Tika();
42    private static final MimeTypes MIME_TYPES = MimeTypes.getDefaultMimeTypes();
43    private static final String HTML_TYPE = "text/html";
44    // MIME types which we consider as HTML links instead of real media file URIs
45    private static final Set<String> HTML_MIME_TYPES = Collections.unmodifiableSet(
46        new HashSet<>(
47            Arrays.asList("text/x-coldfusion", "text/x-php", "text/asp", "text/aspdotnet", "text/x-cgi",
48                "text/x-jsp", "text/x-perl", HTML_TYPE, MimeTypes.OCTET_STREAM)));

50    // List of exceptions, could be read from a file if it grows. URLs matching this return a media file.
51    private static final Map<Pattern, String> knownUrlPatterns;
52  
53    // Add missing alias types.
54    static {
55      knownUrlPatterns = new HashMap<>();
56      knownUrlPatterns.put(Pattern.compile("http://www\\.unimus\\.no/felles/bilder/web_hent_bilde\\.php\\?id=\\d+&type=jpeg"), "image/jpeg");
57      knownUrlPatterns.put(Pattern.compile("http://www\\.jacq\\.org/image\\.php\\?filename=\\d+&method=europeana"), "image/jpeg");
58      knownUrlPatterns.put(Pattern.compile("https://images\\.ala\\.org\\.au/image/proxyImageThumbnailLarge\\?imageId=[0-9a-f-]{36}"), "image/jpeg");
59      knownUrlPatterns.put(Pattern.compile("http://[a-zA-Z0-9-]+\\.wildlifemonitoring\\.ru/get_photo\\.php\\?id=\\d+"), "image/jpeg");
60      knownUrlPatterns.put(Pattern.compile("http://procyon\\.acadiau\\.ca/ecsmith/cgi-bin/image\\.cgi\\?[0-9A-Z]+,jpeg"), "image/jpeg");
61      knownUrlPatterns.put(Pattern.compile("http://www\\.biologie\\.uni-ulm\\.de/cgi-bin/perl/sound\\.pl\\?sid=T&objid=\\d+"), "audio/vnd.wave");
62      knownUrlPatterns.put(Pattern.compile("https://dofbasen\\.dk/sound_proxy\\.php\\?referer=gbif&mode=o&snd=[0-9_]+.mp3&raw=1"), "audio/mpeg");

81    public MediaObject detectType(MediaObject mo) {
82      if (StringUtils.isEmpty(mo.getFormat())) {
83        // derive from URI
84        mo.setFormat(parseMimeType(mo.getIdentifier()));
85      }
86  
87      // if MIME type is text/html make it a references link instead
88      if (HTML_TYPE.equalsIgnoreCase(mo.getFormat()) && mo.getIdentifier() != null) {
89        // make file URI the references link URL instead
90        mo.setReferences(mo.getIdentifier());
91        mo.setIdentifier(null);
92        mo.setFormat(null);
93      }
94  
95      if (StringUtils.isNotEmpty(mo.getFormat())) {
96        if (mo.getFormat().startsWith("image")) {
97          mo.setType(MediaType.StillImage);
98        } else if (mo.getFormat().startsWith("audio")) {
99          mo.setType(MediaType.Sound);
100       } else if (mo.getFormat().startsWith("video")) {
101         mo.setType(MediaType.MovingImage);
102       } else {
103         LOG.debug("Unsupported media format {}", mo.getFormat());
104       }
105     }
106     return mo;
107   }

109   /**
110    * Parses a MIME type using Apache Tika which can handle the following:
111    * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
112    * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
113    */
114   public String parseMimeType(@Nullable String format) {
115     if (format != null) {
116       format = StringUtils.trimToNull(format.trim().toLowerCase());
117     }
118     if (format == null) {
119       return null;
120     }
121 
122     try {
123       MimeType mime = MIME_TYPES.getRegisteredMimeType(format);
124       if (mime != null) {
125         return mime.getName();
126       }
127 
128     } catch (MimeTypeException e) {
129     }
130 
131     // Failed, but return the input if it's a reasonable MIME type
132     return MimeType.isValid(format) ? format : null;
133   }
134 
135   /**
136    * Parses a MIME type using Apache Tika which can handle the following:
137    * https://github.com/apache/tika/blob/master/tika-core/src/main/resources/org/apache/tika/mime/tika-mimetypes.xml
138    * https://tika.apache.org/1.19.1/formats.html#Full_list_of_Supported_Formats
139    */
140   public String parseMimeType(@Nullable URI uri) {
141     if (uri != null) {
142       String uriString = uri.toString();
143       String mime = TIKA.detect(uriString);
144       if (mime != null && HTML_MIME_TYPES.contains(mime.toLowerCase())) {
145         // We may have something like http://example.org/imageServer?img=test.jpg, so re-run the detection on the last
146         // part of the URL query string.
147         if (uri.getQuery() != null) {
148           mime = TIKA.detect(uri.getQuery());
149           if (mime != null && !HTML_MIME_TYPES.contains(mime.toLowerCase())) {
150             return mime;
151           }
152         }
153 
154         // First check the dictionary for known exceptions.
155         for (Map.Entry<Pattern, String> p : knownUrlPatterns.entrySet()) {
156           if (p.getKey().matcher(uriString).matches()) {
157             return p.getValue();
158           }
159         }
160 
161         // links without any suffix default to OCTET STREAM, see:
162         // http://dev.gbif.org/issues/browse/POR-2066
163         return HTML_TYPE;
164       }
165       return mime;
166     }
167     return null;
168   }
