001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.Continent;
004import org.gbif.common.parsers.core.EnumParser;
005
006/**
007 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/continents.txt.
008 */
009public class ContinentParser extends EnumParser<Continent> {
010
011  private static ContinentParser singletonObject = null;
012
013  private ContinentParser() {
014    super(Continent.class, false);
015    // make sure we have all continents from the enum
016    for (Continent c : Continent.values()) {
017      add(c.name(), c);
018      add(c.getTitle(), c);
019    }
020    // use dict file last
021    init(ContinentParser.class.getResourceAsStream("/dictionaries/parse/continents.tsv"));
022  }
023
024  public static ContinentParser getInstance() {
025    synchronized (ContinentParser.class) {
026      if (singletonObject == null) {
027        singletonObject = new ContinentParser();
028      }
029    }
030    return singletonObject;
031  }
032
033}
