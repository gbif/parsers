12  import static org.junit.Assert.fail;

14  /**
15   * Helper to help running assertions against data held in CSV files.
16   */
17  public class CSVBasedAssertions {
18  
19    private static final String COLUMN_SEPARATOR = ";";
20    private static final String COMMENT_MARKER = "#";
21  
22    /**
23     * Utility function to run assertions received as Function on each rows of an input file.
24     *
25     * @param filepath
26     * @param assertRow
27     */
28    public static void assertTestFile(String filepath, Function<String[], Void> assertRow) {
29      File testInputFile = FileUtils.getClasspathFile(filepath);
30      try (CSVReader csv = CSVReaderFactory.build(testInputFile, COLUMN_SEPARATOR, true)) {
31        while (csv.hasNext()) {
32          String[] row = csv.next();
33          if (row == null || row[0].startsWith(COMMENT_MARKER)) {
34            continue;
35          }
36          assertRow.apply(row);
37        }
38      } catch (IOException e) {
39        fail("Problem reading testFile " + filepath + " " + e.getMessage());
40      }
41    }
42  }
