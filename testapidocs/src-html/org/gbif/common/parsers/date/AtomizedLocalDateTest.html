001package org.gbif.common.parsers.date;
002
003import java.time.LocalDate;
004import java.time.Month;
005import java.time.Year;
006import java.time.YearMonth;
007
008import org.junit.jupiter.api.Test;
009
010import static org.junit.jupiter.api.Assertions.assertEquals;
011
012/**
013 * Unit tests for {@link AtomizedLocalDate}.
014 *
015 */
016public class AtomizedLocalDateTest {
017
018  private static final int YEAR = 1978;
019  private static final Month MONTH_ENUM = Month.DECEMBER;
020  private static final int MONTH = 12;
021  private static final int DAY = 21;
022
023  @Test
024  public void testAtomizedLocalDate(){
025    AtomizedLocalDate date = AtomizedLocalDate.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
026    assertAtomizedLocalDate(date, YEAR, MONTH, DAY);
027
028    date = AtomizedLocalDate.fromTemporalAccessor(YearMonth.of(YEAR, MONTH_ENUM));
029    assertAtomizedLocalDate(date, YEAR, MONTH, null);
030
031    date = AtomizedLocalDate.fromTemporalAccessor(Year.of(YEAR));
032    assertAtomizedLocalDate(date, YEAR, null, null);
033  }
034
035  @Test
036  public void testAtomizedLocalDateEquals(){
037    AtomizedLocalDate date1 = AtomizedLocalDate.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
038    AtomizedLocalDate date2 = AtomizedLocalDate.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
039    assertEquals(date1, date2);
040  }
041
042  private static void assertAtomizedLocalDate(AtomizedLocalDate date, Integer y, Integer m, Integer d){
043    assertEquals(y, date.getYear());
044    assertEquals(m, date.getMonth());
045    assertEquals(d, date.getDay());
046  }
047}
