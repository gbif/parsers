016package org.gbif.common.parsers.date;
017
018import org.gbif.utils.PreconditionUtils;
019
020import java.time.format.DateTimeFormatter;
021import java.time.format.DateTimeParseException;
022import java.time.temporal.TemporalAccessor;
023import java.time.temporal.TemporalQuery;
024import java.util.Objects;
025
026import javax.annotation.Nullable;
027import javax.validation.constraints.NotNull;
028
029/**
030 * Adds some flexibility around {@link DateTimeFormatter} with the {@link DateTimeSeparatorNormalizer} and
031 * simple optimization the support of DateComponentOrdering.
032 * <p>
033 * This class is thread-safe once an instance is created.
034 */
035public class DateTimeParser {
036
037  private final DateTimeFormatter formatter;
038  private final DateTimeSeparatorNormalizer normalizer;
039  private final DateComponentOrdering ordering;
040
041  private final TemporalQuery<?>[] types;
042  private final int minLength;
043
044  /**
045   * Package protected constructor.
046   * Use {@link DateTimeParserBuilder}
047   */
048  DateTimeParser(@NotNull DateTimeFormatter formatter, @Nullable DateTimeSeparatorNormalizer normalizer,
049                 @NotNull DateComponentOrdering ordering, TemporalQuery<?>[] type, int minLength) {
050
051    Objects.requireNonNull(formatter, "DateTimeFormatter can not be null");
052    Objects.requireNonNull(ordering, "DateComponentOrdering can not be null");
053    Objects.requireNonNull(type, "TemporalQuery can not be null");
054    PreconditionUtils.checkArgument(minLength > 0, "minLength must be greater than 0");
055
056    this.formatter = formatter;
057    this.ordering = ordering;
058    this.normalizer = normalizer;
059    this.minLength = minLength;
060    this.types = type;
061  }
062
063  public DateComponentOrdering getOrdering() {
064    return ordering;
065  }
066
067  /**
068   * Parses the provided String as a TemporalAccessor if possible, otherwise returns null.
069   * <p>
070   * This function fully support partial dates and will return the best possible date resolution based
071   * on the {@link DateComponentOrdering} provided.
072   * <p>
073   * This function will not throw DateTimeParseException but returns null in case the input
074   * can not be parsed.
075   *
076   * @return TemporalAccessor or null in case the input can not be parsed.
077   */
078  public TemporalAccessor parse(String input) {
079
080    // return fast if minimum length is not meet
081    if (input.length() < minLength) {
082      return null;
083    }
084
085    if (normalizer != null) {
086      input = normalizer.normalize(input);
087    }
088
089    try {
090      if (types.length > 1) {
091        return formatter.parseBest(input, types);
092      }
093      return (TemporalAccessor) formatter.parse(input, types[0]);
094    } catch (DateTimeParseException dpe) {
095    }
096    return null;
097  }
098
099}
