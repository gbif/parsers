28  import static org.junit.jupiter.api.Assertions.assertNotNull;
29  import static org.junit.jupiter.api.Assertions.assertTrue;

31  /**
32   * Parses a single nomenclatural status out of a given string.
33   * TODO: Improve parser and return a status set instead, see nomStatus.txt file with comma separated values
34   */
35  public class NomStatusParserTest extends ParserTestBase<NomenclaturalStatus> {
36  
37      public NomStatusParserTest() {
38          super(NomStatusParser.getInstance());
39      }
40  
41      /**
42       * This ensures that ALL enum values are at least parsable by the name they
43       * are created with.
44       */
45      @Test
46      public void testCompleteness() {
47          for (NomenclaturalStatus t : NomenclaturalStatus.values()) {
48              assertParseSuccess(t, t.name());
49              assertParseSuccess(t, t.name().toLowerCase());
50              assertParseSuccess(t, t.name().replace("_", "").toLowerCase());
51              if (StringUtils.isNotEmpty(t.getLatinLabel())) {
52                  assertParseSuccess(t, t.getLatinLabel());
53              }
54              if (StringUtils.isNotEmpty(t.getAbbreviatedLabel())) {
55                  assertParseSuccess(t, t.getAbbreviatedLabel());
56              }
57          }
58      }
59  
60      @Test
61      public void testFailures() {
62          assertParseFailure(null);
63          assertParseFailure("");
64          assertParseFailure("Tim");
65      }
66  
67      @Test
68      public void testParse() {
69          // run a few basic tests to check it bootstraps and appears to work
70          assertParseSuccess(NomenclaturalStatus.NUDUM, "nom nud");
71          assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, "later HOmonym");
72          assertParseSuccess(NomenclaturalStatus.NEW_COMBINATION, ParseResult.CONFIDENCE.PROBABLE, "comb nov , rejected name, homotypic syn\tcomb nov , rejected name, homotypic syn");
73          assertParseSuccess(NomenclaturalStatus.CONSERVED, "nom cons , see art 14");
74          assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, ParseResult.CONFIDENCE.PROBABLE, "nom illeg ; (1775), non mill (1768)");
75          assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, "nom illegit , arts 53 1, 53 3");
76          assertParseSuccess(NomenclaturalStatus.INVALID, ParseResult.CONFIDENCE.PROBABLE, "nom inval , art 32 1(b), see art 18 3");
77          assertParseSuccess(NomenclaturalStatus.NEW_SPECIES, "sp nov , heterotypic syn");
78      }
79  
80      @Test
81      public void testFileCoverage() throws IOException {
82          // parses all values in our test file (generated from real occurrence data) and verifies we never get worse at parsing
83          URL fileUrl = this.getClass().getClassLoader().getResource("parse/nom_status.txt");
84          assertNotNull(fileUrl);
85  
86          try (BufferedReader in = new BufferedReader(new InputStreamReader(fileUrl.openStream(), StandardCharsets.UTF_8))) {
87              readLines(in, this);
88          }
89  
90          BatchParseResult result = getResult();
91          System.out.printf("%s out of %s lines failed to parse%n", result.failed, result.total);
92          assertTrue(result.failed > 0);
93          assertTrue(result.failed <= 166); // out of 747
94      }
95  }
