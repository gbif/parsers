014package org.gbif.common.parsers.date;
015
016import java.time.Month;
017
018import org.junit.jupiter.api.Test;
019
020import static org.junit.jupiter.api.Assertions.assertEquals;
021import static org.junit.jupiter.api.Assertions.assertFalse;
022import static org.junit.jupiter.api.Assertions.assertNotNull;
023import static org.junit.jupiter.api.Assertions.assertNull;
024import static org.junit.jupiter.api.Assertions.assertTrue;
025
026/**
027 * Tests related to {@link DatePartsNormalizer}
028 */
029public class DatePartsNormalizerTest {
030
031  private static DatePartsNormalizer NORMALIZER = DatePartsNormalizer.newInstance();
032
033  /**
034   * This happens when integer are exported as float/double by a data publisher.
035   */
036  @Test
037  public void testFloatNormalization(){
038    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975.0", "2.0", "1.0");
039    assertEquals(Integer.valueOf(1), result.getDay());
040    assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
041    assertEquals(Integer.valueOf(1975), result.getYear());
042  }
043
044  @Test
045  public void testNormalizationWithSpaces(){
046    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("1975 ", " 2 ", " 1");
047    assertEquals(Integer.valueOf(1), result.getDay());
048    assertEquals(Month.FEBRUARY.getValue(), result.getMonth().intValue());
049    assertEquals(Integer.valueOf(1975), result.getYear());
050  }
051
052  @Test
053  public void testMonthNameNormalization(){
054    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("January").intValue());
055    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JANUARY").intValue());
056    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("Jan").intValue());
057    assertEquals(Month.JANUARY.getValue(), NORMALIZER.monthNameToNumerical("JAN.").intValue());
058    assertEquals(Month.DECEMBER.getValue(), NORMALIZER.monthNameToNumerical("DEC").intValue());
059    assertEquals(Month.APRIL.getValue(), NORMALIZER.monthNameToNumerical("April").intValue());
060    assertEquals(Month.SEPTEMBER.getValue(), NORMALIZER.monthNameToNumerical("Sept").intValue());
061    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("June").intValue());
062    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("Jun").intValue());
063    assertEquals(Month.JUNE.getValue(), NORMALIZER.monthNameToNumerical("JUNE").intValue());
064    assertEquals(Month.NOVEMBER.getValue(), NORMALIZER.monthNameToNumerical("November").intValue());
065  }
066
067  @Test
068  public void testDiscardedDateParts(){
069
070    DatePartsNormalizer.NormalizedYearMonthDay result = NORMALIZER.normalize("a", "5", "3");
071    assertTrue(result.yDiscarded());
072    assertNull(result.getYear());
073    assertNotNull(result.getMonth());
074
075    result = NORMALIZER.normalize("2000", "a", "3");
076    assertTrue(result.mDiscarded());
077    assertNull(result.getMonth());
078    assertNotNull(result.getYear());
079
080    result = NORMALIZER.normalize("2000", "5", "a");
081    assertTrue(result.dDiscarded());
082    assertNull(result.getDay());
083    assertNotNull(result.getYear());
084
085    // empty String, null and \\N should NOT be considered as "discarded"
086    // they simply represent "not provided"
087    result = NORMALIZER.normalize("2000", "5", "");
088    assertFalse(result.dDiscarded());
089    assertNull(result.getDay());
090
091    result = NORMALIZER.normalize("2000", "5", null);
092    assertFalse(result.dDiscarded());
093    assertNull(result.getDay());
094
095    result = NORMALIZER.normalize("2000", "5", "\\N");
096    assertFalse(result.dDiscarded());
097    assertNull(result.getDay());
098  }
099}
