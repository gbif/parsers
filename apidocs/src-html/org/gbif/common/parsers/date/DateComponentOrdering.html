001package org.gbif.common.parsers.date;
002
003/**
004 * Date component orderings are given to the parser to help to select the right set of DateTimeFormatters
005 */
006public enum DateComponentOrdering {
007  /** Year, month, day, time, zone */
008  YMDTZ,
009  /** Year, month, day, time */
010  YMDT,
011  /** Year, month, day */
012  YMD,
013  /** Day, month, year, time */
014  DMYT,
015  /** Day, month, year */
016  DMY,
017  /** Month, day, year, time */
018  MDYT,
019  /** Month, day, year */
020  MDY,
021  /** Year, month */
022  YM,
023  /** Year, week number */
024  YW,
025  /** Year, day of year */
026  YD,
027  /** Year */
028  Y,
029  /** Chinese, Japanese, Korean: Year, month, day. */
030  HAN,
031  /** The default set of unambiguous, year-month-day-like orderings. */
032  ISO_ETC;
033
034  /** ISO formats */
035  public static DateComponentOrdering[] ISO_FORMATS = new DateComponentOrdering[]{YMDTZ, YMDT, YMD, YM, YW, YD, Y};
036
037  /** Day-Month-Year formats */
038  public static DateComponentOrdering[] DMY_FORMATS = new DateComponentOrdering[]{DMYT, DMY};
039
040  /** Month-Day-Year formats, primarily used in the USA */
041  public static DateComponentOrdering[] MDY_FORMATS = new DateComponentOrdering[]{MDYT, MDY};
042}
