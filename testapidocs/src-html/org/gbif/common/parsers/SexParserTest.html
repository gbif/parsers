001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.Sex;
004
005import org.gbif.common.parsers.core.ParseResult;
006import org.junit.jupiter.api.Test;
007
008import java.io.BufferedReader;
009import java.io.IOException;
010import java.io.InputStreamReader;
011import java.nio.charset.StandardCharsets;
012import java.util.HashSet;
013import java.util.Set;
014
015import static org.junit.jupiter.api.Assertions.assertTrue;
016
017public class SexParserTest extends ParserTestBase<Sex> {
018
019  public SexParserTest() {
020    super(SexParser.getInstance());
021  }
022
023  /**
024   * Makes sure all Sex enum values are parsed ok.
025   */
026  @Test
027  public void testParseAllEnumValues() {
028    for (Sex c : Sex.values()) {
029      assertParseSuccess(c, c.name());
030      assertParseSuccess(c, c.name().toLowerCase());
031    }
032  }
033
034  @Test
035  public void testParseFail() {
036    assertParseFailure("Really great");
037    assertParseFailure("Padua");
038  }
039
040  @Test
041  public void testParseSuccess() {
042    assertParseSuccess(Sex.MALE, "männlich");
043    assertParseSuccess(Sex.MALE, "man");
044    assertParseSuccess(Sex.FEMALE, "women");
045  }
046
047  /**
048   * Parse all unique sex values found in our index and make sure parsing doesn't get worse.
049   * If the test file is updated, values here need to be adjusted!
050   */
051  @Test
052  public void testSexValues() throws IOException {
053    final int CURRENT_TESTS_SUCCESSFUL = 111;
054    final int CURRENT_DISTINCT = 3;
055
056    int failed = 0;
057    int success = 0;
058    Set<Sex> values = new HashSet<>();
059
060    BufferedReader r = new BufferedReader(new InputStreamReader(
061      getClass().getResourceAsStream("/parse/sexes.txt"), StandardCharsets.UTF_8));
062    String line;
063    while ((line=r.readLine()) != null) {
064      ParseResult<Sex> parsed = parser.parse(line);
065      if (parsed.isSuccessful()) {
066        //System.out.println("Succeeded: " + line + " → " + parsed.getPayload().name());
067        success++;
068        values.add(parsed.getPayload());
069      } else {
070        //System.out.println("Failed:    " + line);
071        failed++;
072      }
073    }
074
075    System.out.println(failed + " failed parse results");
076    System.out.println(success + " successful parse results");
077    System.out.println(values.size() + " distinct basis of record values parsed");
078    assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
079    assertTrue(CURRENT_DISTINCT <= values.size());
080  }
081}
