20    @Test
21    public void testParseElevation() throws Exception {
22      assertResult(MeterRangeParser.parseElevation("10", "20", null), true, 15d, 5d);
23      assertResult(MeterRangeParser.parseElevation("10", "20", "1"), true, 15d, 6d);
24      assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
25      assertResult(MeterRangeParser.parseElevation("10", null, null), true, 10d, null);
26      assertResult(MeterRangeParser.parseElevation(null, "10", null), true, 10d, null);
27      assertResult(MeterRangeParser.parseElevation("-800", null, null), true, -800d, null);
28      assertResult(MeterRangeParser.parseElevation("-100", "-50", null), true, -75d, 25d);
29      assertResult(MeterRangeParser.parseElevation("110", "115", null), true, 112.5d, 2.5d);
30      assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
31      assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
32      assertResult(MeterRangeParser.parseElevation(null, "10000", "1"), true, 10000d, 1d);
33      assertResult(MeterRangeParser.parseElevation("4061987", "4061987", null), false, null, null,
34                   OccurrenceIssue.ELEVATION_UNLIKELY);
35  
36      // check units are removed
37      assertResult(MeterRangeParser.parseElevation("1000m", null, "1"), true, 1000d, 1d,
38                   OccurrenceIssue.ELEVATION_NON_NUMERIC);
39      assertResult(MeterRangeParser.parseElevation("3280ft", null, "1"), true, 999.74, 1d,
40                    OccurrenceIssue.ELEVATION_NOT_METRIC, OccurrenceIssue.ELEVATION_NON_NUMERIC);
41  
42      // check out of range
43      assertResult(MeterRangeParser.parseElevation("100000000000", null, "1"), false, null, 1d,
44                   OccurrenceIssue.ELEVATION_UNLIKELY);
45  
46      // nonsense
47      assertResult(MeterRangeParser.parseElevation("booya", "boom", "1"), false, null, 1d,
48                   OccurrenceIssue.ELEVATION_NON_NUMERIC);
49    }

84    private void assertResult(OccurrenceParseResult<?> pr, boolean success, Double elevation, Double precision, OccurrenceIssue ... issue) {
85      if (success) {
86        assertExpected(pr, new DoubleAccuracy(elevation, precision), ParseResult.CONFIDENCE.DEFINITE, issue);
87      } else {
88        assertFailed(pr);
89      }
90    }
91  
92    private void assertExpected(OccurrenceParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
93      assertNotNull(pr);
94      assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
95      assertEquals(c, pr.getConfidence());
96      assertNotNull(pr.getPayload());
97      assertEquals(expected, pr.getPayload());
98      if (issue == null) {
99        assertTrue(pr.getIssues().isEmpty());
100     } else {
101       assertEquals("Issue mismatch, found issues: " + pr.getIssues(), issue.length, pr.getIssues().size());
102       for (OccurrenceIssue iss : issue) {
103         assertTrue(pr.getIssues().contains(iss));
104       }
105     }
106   }
107 
108   private void assertFailed(ParseResult<?> pr) {
109     assertNotNull(pr);
110     assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
111   }
