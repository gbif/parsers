001package org.gbif.common.parsers.utils;
002
003import lombok.extern.slf4j.Slf4j;
004import org.junit.jupiter.api.Test;
005
006import static org.junit.jupiter.api.Assertions.assertArrayEquals;
007import static org.junit.jupiter.api.Assertions.assertNull;
008
009@Slf4j
010public class DelimiterUtilsTest {
011
012  @Test
013  public void testISORange() {
014    assertArrayEquals(
015        new String[] {"1999-01-20", "1999-01-31"},
016        DelimiterUtils.splitISODateRange("1999-01-20/31"));
017    assertArrayEquals(
018        new String[] {"1999-01", "1999-12"}, DelimiterUtils.splitISODateRange("1999-01/12"));
019    assertArrayEquals(
020        new String[] {"1999-1", "1999-2"}, DelimiterUtils.splitISODateRange("1999-1/2"));
021    assertArrayEquals(
022        new String[] {"1999-1-21", "1999-1-6"}, DelimiterUtils.splitISODateRange("1999-1-21/6"));
023    assertArrayEquals(
024        new String[] {"1999-1-02", "1999-1-6"}, DelimiterUtils.splitISODateRange("1999-1-02/6"));
025    assertNull(DelimiterUtils.splitISODateRange("1999-01/31")); // Wrong month
026    assertNull(DelimiterUtils.splitISODateRange("1999-01-20"));
027
028    assertArrayEquals(
029        new String[] {"1999-01-20", "1999-01-31"}, DelimiterUtils.splitPeriod("1999-01-20/31"));
030    assertArrayEquals(
031        new String[] {"1999-01-20", "1999-01-20"}, DelimiterUtils.splitPeriod("1999-01-20"));
032    assertArrayEquals(
033        new String[] {"19990120", "19990230"}, DelimiterUtils.splitPeriod("19990120/19990230"));
034    assertArrayEquals(
035        new String[] {"1999-01", "1999-02"}, DelimiterUtils.splitPeriod("1999-01/02"));
036    assertArrayEquals(
037        new String[] {"1998-9-30", "1998-10-7"}, DelimiterUtils.splitPeriod("1998-9-30/10-7"));
038    assertArrayEquals(new String[] {"", ""}, DelimiterUtils.splitPeriod(null));
039  }
040}
