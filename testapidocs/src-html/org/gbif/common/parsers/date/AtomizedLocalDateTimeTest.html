001package org.gbif.common.parsers.date;
002
003import java.time.LocalDate;
004import java.time.LocalDateTime;
005import java.time.LocalTime;
006import java.time.Month;
007
008import org.junit.jupiter.api.Test;
009
010import static org.junit.jupiter.api.Assertions.assertEquals;
011
012public class AtomizedLocalDateTimeTest {
013
014  private static final int YEAR = 1978;
015  private static final Month MONTH_ENUM = Month.DECEMBER;
016  private static final int MONTH = 12;
017  private static final int DAY = 21;
018
019  private static final int HOUR = 2;
020  private static final int MINUTE = 3;
021  private static final int SECOND = 4;
022  private static final int MILLISECOND = 5;
023  // not a mistake, LocalDateTime.of take nanoseconds and not milliseconds
024  private static final int NANOSECOND = 5000000;
025
026  @Test
027  public void testAtomizedLocalDateTime(){
028    AtomizedLocalDateTime dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDateTime.of(YEAR, MONTH_ENUM, DAY, HOUR, MINUTE, SECOND, NANOSECOND));
029    assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, HOUR, MINUTE, SECOND, MILLISECOND);
030
031    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDateTime.of(YEAR, MONTH_ENUM, DAY, HOUR, MINUTE));
032    assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, HOUR, MINUTE, 0, 0);
033    assertEquals(7, dateTime.getResolution());
034
035    // this will probably not be supported ------
036    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalTime.of(HOUR, MINUTE));
037    assertAtomizedLocalDateTime(dateTime, null, null, null, HOUR, MINUTE, 0, 0);
038    assertEquals(4, dateTime.getResolution());
039
040    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalTime.of(HOUR, MINUTE));
041    assertAtomizedLocalDateTime(dateTime, null, null, null, HOUR, MINUTE, 0, 0);
042    assertEquals(4, dateTime.getResolution());
043    // end of this will probably not be supported ------
044
045    // ensure it works with a LocalDate alone (without time component)
046    dateTime = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
047    assertAtomizedLocalDateTime(dateTime, YEAR, MONTH, DAY, null, null, null, null);
048  }
049
050  @Test
051  public void testAtomizedLocalDateEquals(){
052    AtomizedLocalDateTime date1 = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
053    AtomizedLocalDateTime date2 = AtomizedLocalDateTime.fromTemporalAccessor(LocalDate.of(YEAR, MONTH_ENUM, DAY));
054    assertEquals(date1, date2);
055  }
056
057  private static void assertAtomizedLocalDateTime(AtomizedLocalDateTime dateTime, Integer year, Integer month,
058                                                  Integer day, Integer hour, Integer minute, Integer second,
059                                                  Integer millisecond){
060    assertEquals(year, dateTime.getYear());
061    assertEquals(month, dateTime.getMonth());
062    assertEquals(day, dateTime.getDay());
063    assertEquals(hour, dateTime.getHour());
064    assertEquals(minute, dateTime.getMinute());
065    assertEquals(second, dateTime.getSecond());
066    assertEquals(millisecond, dateTime.getMillisecond());
067  }
068}
