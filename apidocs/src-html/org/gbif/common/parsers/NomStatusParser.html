001package org.gbif.common.parsers;
002
003import org.apache.commons.lang3.StringUtils;
004import org.gbif.api.vocabulary.NomenclaturalStatus;
005import org.gbif.common.parsers.core.EnumParser;
006import org.gbif.common.parsers.core.ParseResult;
007
008import java.io.InputStream;
009import java.util.Collections;
010import java.util.HashMap;
011import java.util.Map;
012import java.util.regex.Pattern;
013
014/**
015 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/nomStatus.txt.
016 */
017public class NomStatusParser extends EnumParser<NomenclaturalStatus> {
018    private Pattern CLEAN_PREFIX = Pattern.compile("\\s*\\.?\\s*");
019    private final Map<String, NomenclaturalStatus> PREFIXES;
020  private static NomStatusParser singletonObject = null;
021
022  {
023    Map<String, NomenclaturalStatus> prefixes = new HashMap<>();
024    prefixes.put("nom illeg", NomenclaturalStatus.ILLEGITIMATE);
025    prefixes.put("nom inval", NomenclaturalStatus.INVALID);
026    prefixes.put("comb nov", NomenclaturalStatus.NEW_COMBINATION);
027    prefixes.put("nom nov", NomenclaturalStatus.REPLACEMENT);
028    prefixes.put("nom nud", NomenclaturalStatus.NUDUM);
029    prefixes.put("nom rej", NomenclaturalStatus.REJECTED);
030    prefixes.put("unavailable", NomenclaturalStatus.INVALID);
031    PREFIXES = Collections.unmodifiableMap(prefixes);
032  }
033
034  private NomStatusParser(InputStream... file) {
035    super(NomenclaturalStatus.class, false, file);
036    // also make sure we have all enum knowledge mapped
037    for (NomenclaturalStatus ns : NomenclaturalStatus.values()) {
038      add(ns.getLatinLabel(), ns);
039      add(ns.getAbbreviatedLabel(), ns);
040    }
041  }
042
043    @Override
044    public ParseResult<NomenclaturalStatus> parse(String input) {
045        ParseResult<NomenclaturalStatus> result = super.parse(input);
046        if (!result.isSuccessful() && StringUtils.isNotEmpty(input)) {
047            String normed = CLEAN_PREFIX.matcher(input).replaceFirst(" ").trim().toLowerCase();
048            if (StringUtils.isNotEmpty(normed)) {
049                // try generic parsing of status prefixes only
050                for (Map.Entry<String, NomenclaturalStatus> entry : PREFIXES.entrySet()) {
051                    if (normed.startsWith(entry.getKey())) {
052                        return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, entry.getValue());
053                    }
054                }
055            }
056        }
057        return result;
058    }
059
060    public static NomStatusParser getInstance()
061    throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
062    synchronized (NomStatusParser.class) {
063      if (singletonObject == null) {
064        singletonObject = new NomStatusParser(NomStatusParser.class.getResourceAsStream("/dictionaries/parse/nomStatus.tsv"));
065      }
066    }
067    return singletonObject;
068  }
069
070
071}
