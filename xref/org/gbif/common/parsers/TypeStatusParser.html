7   import java.util.regex.Matcher;
8   import java.util.regex.Pattern;

10  import org.gbif.common.parsers.core.ParseResult;
11  
12  /**
13   * Singleton implementation of the dictionary that uses the file /dictionaries/parse/typeStatus.txt.
14   */
15  public class TypeStatusParser extends EnumParser<TypeStatus> {
16  
17    private static TypeStatusParser singletonObject = null;
18    private static final Pattern NAME_SEPARATOR = Pattern.compile("^(.+) (OF|FOR) ");
19  
20    private TypeStatusParser() {
21      super(TypeStatus.class, false);
22      init(TypeStatusParser.class.getResourceAsStream("/dictionaries/parse/typeStatus.tsv"));
23    }
24  
25    @Override
26    protected String normalize(String value) {
27      if (StringUtils.isEmpty(value)) {
28        return null;
29      }
30      // uppercase
31      value = value.toUpperCase().trim();
32  
33      // keep only words before the typifiedName if existing, e.g. Holotype for "Holotype of Dianthus fruticosus ssp. amorginus Runemark"
34      Matcher m = NAME_SEPARATOR.matcher(value);
35      if (m.find()) {
36        value = m.group(1);
37      }
38      // remove whitespace and non letters
39      ParseResult<String> ascii = asciiParser.parse(value);
40  
41      // remove all non-letters
42      return ascii.getPayload().chars()
43          .filter(p -> Character.isLetter((char) p))
44          .collect(StringBuilder::new, StringBuilder::appendCodePoint, StringBuilder::append)
45          .toString();
46    }
47  
48  
49    public static TypeStatusParser getInstance()
50      throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
51      synchronized (TypeStatusParser.class) {
52        if (singletonObject == null) {
53          singletonObject = new TypeStatusParser();
54        }
55      }
56      return singletonObject;
57    }
58  }
