001package org.gbif.common.parsers.geospatial;
002
003import org.gbif.api.vocabulary.OccurrenceIssue;
004
005import java.util.EnumSet;
006import java.util.Set;
007
008import com.google.common.base.Objects;
009import com.google.common.collect.Sets;
010
011/**
012 * Simple container class for integer based values with an accurracy (e.g. depth in meters).
013 */
014public class IntPrecisionIssue {
015
016  private final Integer value;
017  private final Integer precision;
018  private final Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
019
020  public IntPrecisionIssue(Integer value, Integer precision, Set<OccurrenceIssue> issues) {
021    this.value = value;
022    this.precision = precision;
023    this.issues.addAll(issues);
024  }
025
026  public IntPrecisionIssue(Integer value, Integer precision, OccurrenceIssue... issue) {
027    this(value, precision, Sets.newHashSet(issue));
028  }
029
030  public Integer getValue() {
031    return value;
032  }
033
034  public Integer getPrecision() {
035    return precision;
036  }
037
038  public Set<OccurrenceIssue> getIssues() {
039    return issues;
040  }
041
042  @Override
043  public String toString() {
044    return Objects.toStringHelper(this)
045      .add("value", value)
046      .add("issues", issues)
047      .add("precision", precision)
048      .toString();
049  }
050
051  @Override
052  public int hashCode() {
053    return Objects.hashCode(value, precision, issues);
054  }
055
056  @Override
057  public boolean equals(Object object) {
058    if (this == object) {
059      return true;
060    }
061    if (!(object instanceof IntPrecisionIssue)) {
062      return false;
063    }
064
065    IntPrecisionIssue that = (IntPrecisionIssue) object;
066    return Objects.equal(this.value, that.value)
067           && Objects.equal(this.issues, that.issues)
068           && Objects.equal(this.precision, that.precision);
069  }
070
071}
