001package org.gbif.common.parsers.date;
002
003import java.util.List;
004import java.util.Map;
005import java.util.regex.Pattern;
006
007import com.google.common.base.MoreObjects;
008import com.google.common.base.Objects;
009import com.google.common.collect.ImmutableList;
010import com.google.common.collect.ImmutableMap;
011import com.google.common.collect.Lists;
012import com.google.common.collect.Maps;
013import org.apache.commons.lang3.StringUtils;
014
015/**
016 * The contract of the {@link TextualMonthDateTokenizer} is to break a string representing a date with a textual
017 * month representation into a list of {@link DateToken}.
018 *
019 * A {@link DateToken} should be interpreted as a possible/candidate for date part. No validation will be performed
020 * by the {@link TextualMonthDateTokenizer}, simply patterns matching.
021 *
022 * The class it-self is Thread-Safe, see nested classes for specific details about them.
023 */
024public class TextualMonthDateTokenizer {
025
026  public enum TokenType {
027    /** Matches 1 or 2 integer(s) (possibly a day) */
028    INT_2,
029    /** Matches 4 integers (possibly a year) */
030    INT_4,
031    /** Matches between 1 and 10 letters including the dot (.) */
032    TEXT
033  }
034
035  private static final Pattern SEPARATOR_PATTERN =  Pattern.compile("[^A-Za-z0-9.]+");
036  private static final Pattern DAY_SUFFIXES_PATTERN =  Pattern.compile("(?<=[0-9]{1,2})(st|nd|rd|th|\\.)",
037          Pattern.CASE_INSENSITIVE);
038
039  private static final Map<TokenType, Pattern> PATTERNS_BY_TYPE = ImmutableMap.of(
040          TokenType.INT_2, Pattern.compile("[0-9]{1,2}"),
041          TokenType.INT_4, Pattern.compile("[0-9]{4}"),
042          TokenType.TEXT, Pattern.compile("[A-Za-z.]{1,10}"));
043
044  /**
045   * Private constructor use static method {@link #newInstance()}
046   */
047  private TextualMonthDateTokenizer() {}
048
049  public static TextualMonthDateTokenizer newInstance(){
050    return new TextualMonthDateTokenizer();
051  }
052
053  /**
054   * Tokenize a string into a {@link DateTokens}.
055   *
056   * @param str
057   * @return {@link DateTokens} instance, or null if str is null or empty
058   */
059  public DateTokens tokenize(String str) {
060    if (StringUtils.isBlank(str)) {
061      return null;
062    }
063
064    str = DAY_SUFFIXES_PATTERN.matcher(str).replaceAll("");
065    DateTokens tokens = new DateTokens();
066
067    String[] parts = SEPARATOR_PATTERN.split(str);
068    for (String part : parts) {
069      for (TokenType tokenType : PATTERNS_BY_TYPE.keySet()) {
070        if (PATTERNS_BY_TYPE.get(tokenType).matcher(part).matches()) {
071          tokens.addToken(new DateToken(part, tokenType));
072          //should always match only on pattern
073          break;
074        }
075      }
076    }
077    return tokens;
078  }
079
080  /**
081   * Contains the result of the tokenization.
082   * DateToken are stored by TokenType on a 1 to 1 assumption.
083   * If a DateToken already exists for the same TokenType it will be replaced and the previous one will be moved to the
084   * discardedTokens list.
085   *
086   * This class is NOT Thread-Safe
087   *
088   */
089  public static class DateTokens {
090    private final Map<TokenType, DateToken> tokens = Maps.newHashMapWithExpectedSize(3);
091    private List<DateToken> discardedTokens = null;
092
093    private void addToken(DateToken dateToken){
094      DateToken prev = tokens.put(dateToken.type, dateToken);
095      if(prev != null){
096        addDiscardedToken(prev);
097      }
098    }
099
100    private void addDiscardedToken(DateToken dateToken){
101      if(discardedTokens == null){
102        discardedTokens = Lists.newArrayList();
103      }
104      discardedTokens.add(dateToken);
105    }
106
107    /**
108     * Checks if some DateToken were discarded during the tokenization.
109     *
110     * @return
111     */
112    public boolean containsDiscardedTokens(){
113      return discardedTokens != null;
114    }
115
116    /**
117     * Size does NOT include discarded token(s).
118     *
119     * @return
120     */
121    public int size(){
122      return tokens.size();
123    }
124
125    public DateToken getToken(TokenType tokenType){
126      return tokens.get(tokenType);
127    }
128
129    public List<DateToken> getDiscardedTokens() {
130      return ImmutableList.copyOf(discardedTokens);
131    }
132
133    @Override
134    public String toString(){
135      return MoreObjects.toStringHelper(this)
136              .add("tokens", tokens)
137              .add("discardedTokens", discardedTokens).toString();
138    }
139
140  }
141
142  /**
143   * Represents a possible candidate for date part. The value of the token represents what was provided and
144   * may or may not be valid.
145   *
146   * This class is Thread-Safe.
147   */
148  public static class DateToken {
149    private final String token;
150    private final TokenType type;
151
152    DateToken(String token, TokenType tokenType){
153      this.token = token;
154      this.type = tokenType;
155    }
156
157    public String getToken() {
158      return token;
159    }
160
161    @Override
162    public int hashCode() {
163      return Objects.hashCode(token, type);
164    }
165
166    @Override
167    public boolean equals(Object object){
168      if (object instanceof DateToken) {
169        DateToken that = (DateToken) object;
170        return Objects.equal(this.token, that.token)
171                && Objects.equal(this.type, that.type);
172      }
173      return false;
174    }
175
176    @Override
177    public String toString(){
178      return MoreObjects.toStringHelper(this)
179              .add("token", token)
180              .add("type", type).toString();
181    }
182  }
183}
