18    private final TemporalParser parser =  CustomizedTextDateParser.getInstance(DMY_FORMATS);
19  
20    /**
21     * Can parse: 1969-02
22     * Cannot parse: 1969/02
23     */
24    @Test
25    public void testTextDateParsing(){
26      ParseResult<TemporalAccessor> parseResult =  parser.parse("1/2/2000");
27      assertEquals(LocalDate.of(2000, Month.FEBRUARY, 1), LocalDate.from(parseResult.getPayload()));
28      parseResult =  parser.parse("2000-2-1");
29      assertEquals(LocalDate.of(2000, Month.FEBRUARY, 1), LocalDate.from(parseResult.getPayload()));
30      parseResult = parser.parse("23-March-1969");
31      assertEquals(LocalDate.of(1969, Month.MARCH, 23), LocalDate.from(parseResult.getPayload()));
32      parseResult = parser.parse("1969-02");
33      assertEquals("1969-02", YearMonth.from(parseResult.getPayload()).toString());
34    }
35  
36    /**
37     * Because TextDateParse does not initiate ThreeTenNumericalDateParse with start year, like 19
38     */
39    @Test
40    public void shouldFail(){
41      ParseResult<TemporalAccessor> parseResult = parser.parse("7/10/08");
42      assertFalse(parseResult.isSuccessful());
43      parseResult = parser.parse("17/10/78");
44      assertFalse(parseResult.isSuccessful());
45      parseResult = parser.parse("1969/02");
46      assertFalse(parseResult.isSuccessful());
47    }
