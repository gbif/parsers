6   import java.time.temporal.TemporalAccessor;

9   
10  /**
11   *
12   * Main interface for date/time parsing based.
13   *
14   */
15  public interface TemporalParser extends Parsable<TemporalAccessor> {
16  
17    /**
18     * Parse a date represented as a single String into a TemporalAccessor.
19     *
20     * @param input
21     * @param hint help to speed up the parsing and possibly return a better confidence
22     * @return result, never null
23     */
24    ParseResult<TemporalAccessor> parse(String input, @Nullable DateFormatHint hint);
25  
26    /**
27     * Parse year, month, day strings as a TemporalAccessor.
28     *
29     * @param year numerical value of a year
30     * @param month value of the mont depending on the implementation, numerical value of a
31     *              month (starting at 1 for January) or possibly text.
32     * @param day numerical value of a day
33     * @return result, never null
34     */
35    ParseResult<TemporalAccessor> parse(@Nullable String year, @Nullable String month, @Nullable String day);
36  
37    /**
38     * Parse year, month, day integers as a TemporalAccessor.
39     *
40     * @param year numerical value of a year
41     * @param month numerical value of a month (starting at 1 for January)
42     * @param day numerical value of a day
43     * @return result, never null
44     */
45    ParseResult<TemporalAccessor> parse(@Nullable Integer year, @Nullable Integer month, @Nullable Integer day);
46  
47  }
