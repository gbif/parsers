023  @Test
024  public void singleDateRangeTest() {
025    TemporalRangeParser trp =
026        TemporalRangeParser.builder()
027            .temporalParser(MultiinputTemporalParser.create(Collections.singletonList(DMY)))
028            .create();
029
030    OccurrenceParseResult<IsoDateInterval> result = trp.parse("1930/1929");
031    assertEquals("1929", result.getPayload().getFrom().toString());
032    assertEquals("1930", result.getPayload().getTo().toString());
033
034    result = trp.parse("1930/1931");
035    assertEquals("1930", result.getPayload().getFrom().toString());
036    assertEquals("1931", result.getPayload().getTo().toString());
037
038    result = trp.parse("1930/1930");
039    assertEquals("1930", result.getPayload().getFrom().toString());
040    assertEquals("1930", result.getPayload().getTo().toString());
041
042    result = trp.parse("1930-01");
043    assertEquals("1930-01", result.getPayload().getFrom().toString());
044    assertEquals("1930-01", result.getPayload().getTo().toString());
045
046    // Does not support
047    result = trp.parse("01/1930");
048    assertFalse(result.isSuccessful());
049    assertEquals(1, result.getIssues().size());
050    assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
051
052    result = trp.parse("1930-01-02/1930-02-01");
053    assertEquals("1930-01-02", result.getPayload().getFrom().toString());
054    assertEquals("1930-02-01", result.getPayload().getTo().toString());
055
056    result = trp.parse("02/01/1930");
057    assertEquals("1930-01-02", result.getPayload().getFrom().toString());
058    assertEquals("1930-01-02", result.getPayload().getTo().toString());
059
060    result = trp.parse("1930-01-02/02-01");
061    assertEquals("1930-01-02", result.getPayload().getFrom().toString());
062    assertEquals("1930-02-01", result.getPayload().getTo().toString());
063
064    result = trp.parse("1930-01-02/15");
065    assertEquals("1930-01-02", result.getPayload().getFrom().toString());
066    assertEquals("1930-01-15", result.getPayload().getTo().toString());
067
068    result = trp.parse("1930-01-02/1930-01-02");
069    assertEquals("1930-01-02", result.getPayload().getFrom().toString());
070    assertEquals("1930-01-02", result.getPayload().getTo().toString());
071  }

073  @Test
074  public void ambiguousDateTest() {
075    // Use a default parser, which cannot understand like: 01/02/2020
076    TemporalRangeParser trp = TemporalRangeParser.builder().create();
077
078    OccurrenceParseResult<IsoDateInterval> result = trp.parse("01/02/1999");
079    assertFalse(result.getIssues().isEmpty());
080    assertEquals(1, result.getIssues().size());
081    assertTrue(result.getIssues().contains(OccurrenceIssue.RECORDED_DATE_INVALID));
082
083    result = trp.parse("1999-01-02");
084    assertTrue(result.getIssues().isEmpty());
085    assertEquals("1999-01-02", result.getPayload().getFrom().toString());
086    assertEquals("1999-01-02", result.getPayload().getTo().toString());
087
088    // Use a DMY_FORMATS parser
089    trp =
090        TemporalRangeParser.builder()
091            .temporalParser(MultiinputTemporalParser.create(Arrays.asList(DMY_FORMATS)))
092            .create();
093
094    result = trp.parse("01/02/1999");
095    assertTrue(result.getIssues().isEmpty());
096    assertEquals("1999-02-01", result.getPayload().getFrom().toString());
097    assertEquals("1999-02-01", result.getPayload().getTo().toString());
098
099    result = trp.parse("1999-01-02");
100    assertTrue(result.getIssues().isEmpty());
101    assertEquals("1999-01-02", result.getPayload().getFrom().toString());
102    assertEquals("1999-01-02", result.getPayload().getTo().toString());
103  }

120  public void testYMDT() {
121    TemporalRangeParser trp =
122      TemporalRangeParser.builder()
123        .temporalParser(MultiinputTemporalParser.create(Collections.singletonList(DMY)))
124        .create();
125    // Should fail
126    OccurrenceParseResult<IsoDateInterval> result = trp.parse("01/03/1930T12:01");
127    assertFalse(result.isSuccessful());
128    assertEquals(1, result.getIssues().size());
129    assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
130
131    trp =
132      TemporalRangeParser.builder()
133        .temporalParser(MultiinputTemporalParser.create(Arrays.asList(DMY, DMYT)))
134        .create();
135    result = trp.parse("01/03/1930T12:01");
136
137    assertEquals("1930-03-01T12:01", result.getPayload().getFrom().toString());
138  }

176  @Test
177  public void mismatchingDatesTest() {
178    TemporalRangeParser trp = TemporalRangeParser.builder().create();
179    OccurrenceParseResult<IsoDateInterval> result;
180
181    result = trp.parse("2000", "06", "06", "", "9", null);
182    assertEquals("2000", result.getPayload().getFrom().toString());
183    assertEquals("2000", result.getPayload().getTo().toString());
184    assertEquals(1, result.getIssues().size());
185    assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
186
187    result = trp.parse("2000", "06", "06", "", "-9", null);
188    assertEquals("2000-06-06", result.getPayload().getFrom().toString());
189    assertEquals("2000-06-06", result.getPayload().getTo().toString());
190    assertEquals(1, result.getIssues().size());
191    assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
192
193    result = trp.parse("2000", "06", "06", "", "15.23", null);
194    assertEquals("2000-06-06", result.getPayload().getFrom().toString());
195    assertEquals("2000-06-06", result.getPayload().getTo().toString());
196    assertEquals(1, result.getIssues().size());
197    assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
198
199    result = trp.parse("2000", null, null, "2000-04-01/2001-06-01", null, null);
200    assertFalse(result.isSuccessful());
201    assertEquals(1, result.getIssues().size());
202    assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
203
204    result = trp.parse(null, null, null, "05/02/78", null, null);
205    assertFalse(result.isSuccessful());
206    assertEquals(1, result.getIssues().size());
207    assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
208  }

247  @Test
248  public void testEqualResolution() {
249    TemporalRangeParser trp = TemporalRangeParser.builder().create();
250
251    // Beginning of range will be parsed as 2000, but end of range as 2000-06.  We must have 2000/2000, not 2000/2000-06.
252    OccurrenceParseResult<IsoDateInterval> result = trp.parse("2000", "06", "06", "2000-04-01T01:02:03/2000-06-01T22:23:24", null, null);
253    assertEquals("2000", result.getPayload().getFrom().toString());
254    assertEquals("2000", result.getPayload().getTo().toString());
255    assertEquals(1, result.getIssues().size());
256    assertEquals(OccurrenceIssue.RECORDED_DATE_MISMATCH, result.getIssues().iterator().next());
257
258    // Event date has different resolutions on each side of the range
259    result = trp.parse(null, null, null, "2009-09/2009-10-05", null, null);
260    assertEquals("2009-09", result.getPayload().getFrom().toString());
261    assertEquals("2009-10", result.getPayload().getTo().toString());
262    // TODO: Could improve by adding an issue in this case.
263    //assertEquals(1, result.getIssues().size());
264    //assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
265
266    // Event date has different resolutions on each side of the range
267    result = trp.parse(null, null, null, "2009-09-18/2009-10-05T17:36+0200", null, null);
268    assertEquals("2009-09-18", result.getPayload().getFrom().toString());
269    assertEquals("2009-10-05", result.getPayload().getTo().toString());
270    // TODO: Could improve by adding an issue in this case.
271    //assertEquals(1, result.getIssues().size());
272    //assertEquals(OccurrenceIssue.RECORDED_DATE_INVALID, result.getIssues().iterator().next());
273
274    // Event date has different resolutions on each side of the range
275    result = trp.parse(null, null, null, "2019-10-07T13:42:25Z/2019-10-07", null, null);
276    assertEquals("2019-10-07", result.getPayload().getFrom().toString());
277    assertEquals("2019-10-07", result.getPayload().getTo().toString());
