001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.Rank;
004import org.gbif.common.parsers.core.EnumParser;
005
006import java.io.InputStream;
007
008/**
009 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/rank.txt.
010 */
011public class RankParser extends EnumParser<Rank> {
012
013  private static RankParser singletonObject = null;
014
015  private RankParser(InputStream... file) {
016    super(Rank.class, false, file);
017    // also make sure we have all enum values mapped
018    for (Rank r : Rank.values()) {
019      add(r.name(), r);
020      add(r.getMarker(), r);
021    }
022  }
023
024  public static RankParser getInstance()
025    throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
026    synchronized (RankParser.class) {
027      if (singletonObject == null) {
028        singletonObject = new RankParser(RankParser.class.getResourceAsStream("/dictionaries/parse/rank.txt"));
029      }
030    }
031    return singletonObject;
032  }
033
034
035}
