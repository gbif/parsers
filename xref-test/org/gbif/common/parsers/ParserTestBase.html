1   package org.gbif.common.parsers;

3   import com.google.common.base.Strings;
4   import com.google.common.io.LineProcessor;
5   import org.gbif.common.parsers.core.Parsable;
6   import org.gbif.common.parsers.core.ParseResult;
7   import org.junit.Before;
8   
9   import java.io.IOException;
10  
11  import static org.junit.Assert.assertEquals;
12  import static org.junit.Assert.assertNotNull;
13  
14  public abstract class ParserTestBase<T> implements LineProcessor<ParserTestBase.BatchParseResult>  {
15    protected final Parsable<T> parser;
16    private BatchParseResult batchResult;
17  
18    public static class BatchParseResult {
19      public int total;
20      public int failed;
21    }
22  
23    @Before
24    public void setupBatch() {
25      batchResult = new BatchParseResult();
26    }
27  
28    public ParserTestBase(Parsable<T> parser) {
29      this.parser = parser;
30    }
31  
32    protected void assertParseFailure(String input) {
33      ParseResult<T> parsed = parser.parse(input);
34      assertEquals("Expected "+input+" to fail but got "+parsed.getPayload()+" instead", ParseResult.STATUS.FAIL, parsed.getStatus());
35    }
36  
37    protected void assertParseSuccess(T expected, String input) {
38      assertParseSuccess(expected, ParseResult.CONFIDENCE.DEFINITE, input);
39    }
40  
41    protected void assertParseSuccess(T expected, ParseResult.CONFIDENCE confidence, String input) {
42      ParseResult<T> parsed = parser.parse(input);
43      assertNotNull(parsed);
44      assertEquals("BAD PARSING OF: "+input, ParseResult.STATUS.SUCCESS, parsed.getStatus());
45      assertNotNull(parsed.getPayload());
46      assertEquals("BAD PARSING OF: "+input, expected, parsed.getPayload());
47      if (confidence != null) {
48        assertEquals(confidence, parsed.getConfidence());
49      }
50    }
51  
52    @Override
53    public boolean processLine(String line) throws IOException {
54      if (Strings.isNullOrEmpty(line)) {
55        return false;
56      }
57      ParseResult<T> parsed = parser.parse(line);
58      batchResult.total++;
59      if (parsed == null || !parsed.isSuccessful() || parsed.getPayload()==null) {
60        batchResult.failed++;
61      }
62      return true;
63    }
64  
65    @Override
66    public ParserTestBase.BatchParseResult getResult() {
67      return batchResult;
68    }
69  }
