001package org.gbif.common.parsers;
002
003import org.gbif.api.util.VocabularyUtils;
004import org.gbif.api.vocabulary.Country;
005import org.gbif.common.parsers.core.EnumParser;
006
007import java.util.regex.Pattern;
008
009import com.google.common.base.CharMatcher;
010import com.google.common.base.Strings;
011
012/**
013 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/countryName.txt.
014 */
015public class CountryParser extends EnumParser<Country> {
016
017  private static CountryParser singletonObject = null;
018
019  private static final CharMatcher LETTER_MATCHER = CharMatcher.JAVA_LETTER.or(CharMatcher.WHITESPACE).precomputed();
020  private static final CharMatcher WHITESPACE_MATCHER = CharMatcher.WHITESPACE.precomputed();
021  // "off Australia"
022  private static final Pattern REMOVE_OFF_PATTERN = Pattern.compile("off ", Pattern.CASE_INSENSITIVE);
023
024  private static final String ISO_3166_2_FORMAT = "ISO_3166-2:";
025  private static final String ISO_3166_3_FORMAT = "ISO_3166-3:";
026
027  private CountryParser() {
028    super(Country.class, false);
029    // also make sure we have all official iso countries mapped
030    for (Country c : Country.OFFICIAL_COUNTRIES) {
031      add(c.name(), c);
032      add(c.getTitle(), c);
033      add(c.getIso2LetterCode(), c);
034      add(c.getIso3LetterCode(), c);
035      add(ISO_3166_2_FORMAT+c.getIso2LetterCode(), c);
036      add(ISO_3166_3_FORMAT+c.getIso3LetterCode(), c);
037    }
038    // and Kosovo (which is not an official code, but should be treated as such by GBIF)
039    add(Country.KOSOVO.name(), Country.KOSOVO);
040    add(Country.KOSOVO.getTitle(), Country.KOSOVO);
041    add(Country.KOSOVO.getIso2LetterCode(), Country.KOSOVO);
042    add(Country.KOSOVO.getIso3LetterCode(), Country.KOSOVO);
043    // use dict file last
044    init(CountryParser.class.getResourceAsStream("/dictionaries/parse/countryName.tsv"));
045  }
046
047  @Override
048  protected String normalize(String value) {
049    value = handleNotAvailable(value);
050    if (value != null) {
051      String cleanedCountry = LETTER_MATCHER.retainFrom(value);
052      cleanedCountry = REMOVE_OFF_PATTERN.matcher(cleanedCountry).replaceFirst("");
053      cleanedCountry = WHITESPACE_MATCHER.trimAndCollapseFrom(cleanedCountry, ' ');
054      cleanedCountry = Strings.emptyToNull(cleanedCountry);
055      return super.normalize(cleanedCountry);
056    }
057    return null;
058  }
059
060  @Override
061  protected Country fromDictFile(String value) {
062    Country c = Country.fromIsoCode(value);
063    if (c == null) {
064      try {
065        c = VocabularyUtils.lookupEnum(value, Country.class);
066      } catch (RuntimeException e) {
067      }
068    }
069    return c;
070  }
071
072  public static CountryParser getInstance() {
073    synchronized (CountryParser.class) {
074      if (singletonObject == null) {
075        singletonObject = new CountryParser();
076      }
077    }
078    return singletonObject;
079  }
080}
