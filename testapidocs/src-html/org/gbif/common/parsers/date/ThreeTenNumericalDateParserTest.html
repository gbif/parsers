051  @Test
052  public void testLocalDateFromFile() {
053    assertTestFile(LOCALDATE_TEST_FILE,
054            new Function<String[], Void>() {
055              @Nullable
056              @Override
057              public Void apply(@Nullable String[] row) {
058                String raw = row[RAW_VAL_IDX];
059                try {
060                  int year = Integer.parseInt(row[YEAR_VAL_IDX]);
061                  int month = Integer.parseInt(row[MONTH_VAL_IDX]);
062                  int day = Integer.parseInt(row[DAY_VAL_IDX]);
063                  ParseResult<TemporalAccessor> result = PARSER.parse(raw);
064                  assertNotNull(raw + " generated null payload", result.getPayload());
065                  assertEquals("Test file rawValue: " + raw, LocalDate.of(year, month, day),
066                          LocalDate.from(result.getPayload()));
067                }
068                catch (NumberFormatException nfEx){
069                  fail("Error while parsing the test input file content." + nfEx.getMessage());
070                }
071                return null;
072              }
073            });
074  }

151  @Test
152  public void testParseAsLocalDateTime(){
153    ThreeTenNumericalDateParser parser = ThreeTenNumericalDateParser.newInstance(Year.of(1900));
154
155    //month first with 2 digits years >_<
156    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("122178").getPayload()));
157    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12/21/78").getPayload()));
158    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12\\21\\78").getPayload()));
159    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12.21.78").getPayload()));
160    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12-21-78").getPayload()));
161    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12_21_78").getPayload()));
162
163    //month/year alone
164    assertEquals(YearMonth.of(1978,12), YearMonth.from(parser.parse("1978-12").getPayload()));
165
166    //year alone
167    assertEquals(Year.of(1978), Year.from(parser.parse("1978").getPayload()));
168   // assertEquals(Year.of(1978), Year.from(parser.parse("78").getPayload()));
169  }

171  @Test
172  public void testParseAsLocalDateByDateParts(){
173    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(PARSER.parse("1978", "12", "21").getPayload()));
174    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(PARSER.parse(1978, 12, 21).getPayload()));
175
176    assertEquals(LocalDate.of(1978, Month.DECEMBER, 1), LocalDate.from(PARSER.parse("1978", "12", "1").getPayload()));
177    assertEquals(YearMonth.of(1978,12), YearMonth.from(PARSER.parse("1978", "12", null).getPayload()));
178    assertEquals(YearMonth.of(1978,12), YearMonth.from(PARSER.parse(1978, 12, null).getPayload()));
179    assertEquals(Year.of(1978), Year.from(PARSER.parse("1978","",null).getPayload()));
180
181    // providing the day without the month should result in an error
182    assertEquals(ParseResult.STATUS.FAIL, PARSER.parse("1978", "", "2").getStatus());
183    assertEquals(ParseResult.STATUS.FAIL, PARSER.parse(1978, null, 2).getStatus());
184  }

201  @Test
202  public void testAmbiguousDates() {
203    ParseResult<TemporalAccessor> result;
204
205    // Ambiguous
206    result = PARSER.parse("1/2/1996");
207    assertNull(result.getPayload());
208    assertEquals(2, result.getAlternativePayloads().size());
209    assertTrue(result.getAlternativePayloads().contains(LocalDate.of(1996, 2, 1)));
210    assertTrue(result.getAlternativePayloads().contains(LocalDate.of(1996, 1, 2)));
211
212    // Not ambiguous
213    result = PARSER.parse("1/1/1996");
214    assertEquals(LocalDate.of(1996, 1, 1), result.getPayload());
215    assertNull(result.getAlternativePayloads());
216
217    result = PARSER.parse("31/1/1996");
218    assertEquals(LocalDate.of(1996, 1, 31), result.getPayload());
219    assertNull(result.getAlternativePayloads());
220
221    // Dots aren't used in America.
222    result = PARSER.parse("4.5.1996");
223    assertEquals(LocalDate.of(1996, 5, 4), result.getPayload());
224    assertNull(result.getAlternativePayloads());
225  }
