036public final class ASCIIParser implements Parsable<String> {
037
038  private static ASCIIParser singletonObject = null;
039
040  private ASCIIParser() {
041  }
042
043  public static ASCIIParser getInstance() {
044    synchronized (ASCIIParser.class) {
045      if (singletonObject == null) {
046        singletonObject = new ASCIIParser();
047      }
048    }
049    return singletonObject;
050  }
051
052  @Override
053  public ParseResult<String> parse(String input) {
054    if (StringUtils.isEmpty(input)) {
055      return ParseResult.fail();
056    } else {
057      StringBuilder sb = new StringBuilder();
058      for (char c : input.toCharArray()) {
059        foldToASCII(c, sb);
060      }
061      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, sb.toString());
062    }
063  }
