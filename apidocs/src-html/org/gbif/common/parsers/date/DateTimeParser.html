001package org.gbif.common.parsers.date;
002
003
004import javax.annotation.Nullable;
005import javax.validation.constraints.NotNull;
006
007import com.google.common.base.Preconditions;
008import org.threeten.bp.LocalDate;
009import org.threeten.bp.LocalDateTime;
010import org.threeten.bp.Year;
011import org.threeten.bp.YearMonth;
012import org.threeten.bp.ZonedDateTime;
013import org.threeten.bp.format.DateTimeFormatter;
014import org.threeten.bp.format.DateTimeParseException;
015import org.threeten.bp.temporal.TemporalAccessor;
016import org.threeten.bp.temporal.TemporalQuery;
017
018/**
019 * Adds some flexibility around {@link DateTimeFormatter} with the {@link DateTimeSeparatorNormalizer} and
020 * simple optimization the support of DateFormatHint.
021 *
022 *
023 * This class is thread-safe once an instance is created.
024 *
025 */
026public class DateTimeParser {
027
028  private final DateTimeFormatter formatter;
029  private final DateTimeSeparatorNormalizer normalizer;
030  private final DateFormatHint hint;
031
032  private final TemporalQuery<?>[] types;
033  private final int minLength;
034
035  /**
036   * Package protected constructor.
037   * Use {@link DateTimeParserBuilder}
038   *
039   * @param formatter
040   * @param normalizer optional, can be null
041   * @param hint
042   * @param minLength
043   */
044  DateTimeParser(@NotNull DateTimeFormatter formatter, @Nullable DateTimeSeparatorNormalizer normalizer,
045                 @NotNull DateFormatHint hint, int minLength){
046
047    Preconditions.checkNotNull(formatter, "DateTimeFormatter can not be null");
048    Preconditions.checkNotNull(hint, "DateFormatHint can not be null");
049    Preconditions.checkArgument(minLength > 0, "minLength must be greater than 0");
050
051    this.formatter = formatter;
052    this.hint = hint;
053    this.normalizer = normalizer;
054    this.minLength = minLength;
055    this.types = getTypesFromHint(hint);
056  }
057
058  /**
059   * The idea is to only use the types that are possible with DateTimeFormatter.parseBest method.
060   *
061   * @param hint
062   * @return
063   */
064  private TemporalQuery<?>[] getTypesFromHint(DateFormatHint hint){
065    switch(hint){
066      case YMDT: return new TemporalQuery<?>[]{ZonedDateTime.FROM, LocalDateTime.FROM, LocalDate.FROM, YearMonth.FROM, Year.FROM};
067      case YMD: return new TemporalQuery<?>[]{LocalDate.FROM, YearMonth.FROM, Year.FROM};
068      case YM: return new TemporalQuery<?>[]{YearMonth.FROM, Year.FROM};
069      case Y: return new TemporalQuery<?>[]{Year.FROM};
070      case DMY:
071      case MDY:
072      case HAN:  return new TemporalQuery<?>[]{LocalDate.FROM, YearMonth.FROM, Year.FROM};
073      case NONE:
074      default: return new TemporalQuery<?>[]{LocalDateTime.FROM, LocalDate.FROM, YearMonth.FROM, Year.FROM};
075    }
076  }
077
078  public DateFormatHint getHint() {
079    return hint;
080  }
081
082  /**
083   * Parses the provided String as a TemporalAccessor if possible, otherwise returns null.
084   *
085   * This function fully support partial dates and will return the best possible date resolution based
086   * on the {@link DateFormatHint} provided.
087   *
088   * This function will not throw DateTimeParseException but returns null in case the input
089   * can not be parsed.
090   *
091   * @param input
092   * @return TemporalAccessor or null in case the input can not be parsed.
093   */
094  public TemporalAccessor parse(String input){
095
096    // return fast if minimum length is not meet
097    if(input.length() < minLength){
098      return null;
099    }
100
101    if(normalizer != null){
102      input = normalizer.normalize(input);
103    }
104
105    try {
106      if(types.length > 1) {
107        return formatter.parseBest(input, types);
108      }
109      return (TemporalAccessor)formatter.parse(input, types[0]);
110    }
111    catch (DateTimeParseException dpe){}
112    return null;
113  }
114
115}
