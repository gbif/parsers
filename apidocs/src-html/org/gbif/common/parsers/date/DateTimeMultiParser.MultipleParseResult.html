001package org.gbif.common.parsers.date;
002
003import java.util.List;
004import javax.annotation.Nullable;
005import javax.validation.constraints.NotNull;
006
007import com.google.common.base.Preconditions;
008import com.google.common.collect.ImmutableList;
009import com.google.common.collect.Lists;
010import org.threeten.bp.temporal.TemporalAccessor;
011
012/**
013 * Supports multiple {@link DateTimeParser} that are considered ambiguous. Two {@link DateTimeParser} are considered
014 * ambiguous when they can possibly produce 2 different {@link TemporalAccessor}.
015 * e.g. "dd/MM/yyyy" and "MM/dd/yyyy"
016 *
017 *
018 * This class will try all the parsers and keep the all the successful results.
019 *
020 * This class is thread-safe once an instance is created.
021 */
022public class DateTimeMultiParser {
023
024  private final DateTimeParser preferred;
025  private final List<DateTimeParser> otherParsers;
026  private final List<DateTimeParser> allParsers;
027
028  /**
029   * Create a new instance of {@link DateTimeMultiParser}.
030   * @param parsers requires more than 1 element in list
031   */
032  DateTimeMultiParser(@NotNull List<DateTimeParser> parsers){
033    this(null, parsers);
034  }
035
036  /**
037   *
038   * Create a new instance of {@link DateTimeMultiParser}.
039   * At least 2 {@link DateTimeParser} must be provided see details on parameters.
040   *
041   * @param preferred the preferred {@link DateTimeParser} or null
042   * @param otherParsers list of {@link DateTimeParser} containing more than 1 element if no
043   *                     preferred {@link DateTimeParser} is provided. Otherwise, the list must contain at least 1 element.
044   */
045  DateTimeMultiParser(@Nullable DateTimeParser preferred, @NotNull List<DateTimeParser> otherParsers){
046
047    Preconditions.checkNotNull(otherParsers, "otherParsers list can not be null");
048    Preconditions.checkArgument(otherParsers.size() > 0, "otherParsers must contain at least 1 element");
049
050    if(preferred == null) {
051      Preconditions.checkArgument(otherParsers.size() > 1, "If no preferred DateTimeParser is provided, " +
052              "the otherParsers list must contain more than 1 element");
053    }
054
055    this.preferred = preferred;
056    this.otherParsers = Lists.newArrayList(otherParsers);
057
058    ImmutableList.Builder immutableListBuilder = new ImmutableList.Builder<DateTimeParser>();
059    if(preferred != null){
060      immutableListBuilder.add(preferred);
061    }
062    immutableListBuilder.addAll(otherParsers);
063
064    this.allParsers = immutableListBuilder.build();
065  }
066
067  /**
068   * Get the list of all parsers: the preferred (if specified in the constructor) + otherParsers.
069   *
070   * @return never null
071   */
072  public List<DateTimeParser> getAllParsers(){
073    return allParsers;
074  }
075
076  /**
077   * Try to parse the input using all the parsers specified in the constructor.
078   *
079   * @param input
080   * @return {@link MultipleParseResult} instance, never null.
081   */
082  public MultipleParseResult parse(String input){
083    int numberParsed = 0;
084    TemporalAccessor lastParsed = null;
085    TemporalAccessor preferredResult = null;
086
087    //lazy initialized assuming it should not be used most of the time
088    List<TemporalAccessor> otherResults = null;
089    for(DateTimeParser currParser : otherParsers){
090      lastParsed = currParser.parse(input);
091      if(lastParsed != null){
092        numberParsed++;
093        if(otherResults == null){
094          otherResults = Lists.newArrayList();
095        }
096        otherResults.add(lastParsed);
097      }
098    }
099
100    //try the preferred DateTimeParser
101    if(this.preferred != null){
102      lastParsed = this.preferred.parse(input);
103      if(lastParsed != null){
104        numberParsed++;
105        preferredResult = lastParsed;
106      }
107    }
108    return new MultipleParseResult(numberParsed, preferredResult, otherResults);
109  }
110
111  /**
112   * Nested class representing the result of a multi-parse.
113   *
114   */
115  public static class MultipleParseResult {
116    private int numberParsed;
117    private TemporalAccessor preferredResult;
118    private List<TemporalAccessor> otherResults;
119
120    public MultipleParseResult(int numberParsed, TemporalAccessor preferredResult, List<TemporalAccessor> otherResults){
121      this.numberParsed = numberParsed;
122      this.preferredResult = preferredResult;
123      this.otherResults = otherResults;
124    }
125
126    public int getNumberParsed() {
127      return numberParsed;
128    }
129
130    public TemporalAccessor getPreferredResult() {
131      return preferredResult;
132    }
133
134    public List<TemporalAccessor> getOtherResults() {
135      return otherResults;
136    }
137
138    /**
139     * Return the preferredResult if available otherwise the first element of otherResults.
140     * If otherResults is empty, null is returned.
141     * @return
142     */
143    public TemporalAccessor getResult() {
144      if(preferredResult != null) {
145        return preferredResult;
146      }
147
148      if(otherResults != null && otherResults.size() > 0){
149        return otherResults.get(0);
150      }
151      return null;
152    }
153  }
154}
